200#define _syscall0(type,name) \
201type name(void) \
202{ \
203 long __res; \
204 if (Z1(__NR_##name)) { \
205 __asm__ __volatile__ ("push %[sysN]; popl %0; int $0x80" \
206 : "=a" (__res) \
207 : [sysN] "g" (__NR_##name)); \
208 } else { \
209 __asm__ __volatile__ ("int $0x80" \
210 : "=a" (__res) \
211 : [sysN] "a" (__NR_##name)); \
212 } \
213 return (type) __res; \
214}

216#define _syscall1(type,name,type1,arg1) \
217type name(type1 arg1) \
218{ \
219 long __res, junk; \
220 if (Z1(__NR_##name)) { \
221 if (Z0(arg1)) { \
222 __asm__ __volatile__ ("xorl %%ebx,%%ebx; push %[sysN]; popl %0; int $0x80" \
223 : "=a" (__res) \
224 : [sysN] "g" (__NR_##name) \
225 : "ebx"); \
226 } else if (Z1(arg1)) { \
227 __asm__ __volatile__ ("push %[a1]; popl %%ebx; push %[sysN]; popl %0; int $0x80" \
228 : "=a" (__res) \
229 : [sysN] "g" (__NR_##name), [a1] "g" ((long)(arg1)) \
230 : "ebx"); \
231 } else { \
232 __asm__ __volatile__ ("push %[sysN]; popl %0; int $0x80" \
233 : "=a" (__res), "=b" (junk) \
234 : [sysN] "g" (__NR_##name),"b" ((long)(arg1))); \
235 } \
236 } else { \
237 __asm__ __volatile__ ("int $0x80" \
238 : "=a" (__res), "=b" (junk) \
239 : "a" (__NR_##name),"b" ((long)(arg1))); \
240 } \
241 return (type) __res; \
242}

244#define _syscall1noreturn(name,type1,arg1) \
245void __attribute__((__noreturn__,__nothrow__)) name(type1 arg1) \
246{ \
247 if (Z1(__NR_##name)) { \
248 if (Z0(arg1)) { \
249 __asm__ __volatile__ ("xorl %%ebx,%%ebx; push %0; popl %%eax; int $0x80" \
250 : : "g" (__NR_##name) ); \
251 } else if (Z1(arg1)) { \
252 __asm__ __volatile__ ("push %[a1]; popl %%ebx; push %[sysN]; popl %%eax; int $0x80" \
253 : : [sysN] "g" (__NR_##name), [a1] "g" ((long)(arg1)) ); \
254 } else { \
255 __asm__ __volatile__ ("push %0; popl %%eax; int $0x80" \
256 : : "g" (__NR_##name),"b" ((long)(arg1))); \
257 } \
258 } else { \
259 __asm__ __volatile__ ("int $0x80" \
260 : : "a" (__NR_##name),"b" ((long)(arg1))); \
261 } \
262 for(;;) ; \
263}

265#define _syscall2(type,name,type1,arg1,type2,arg2) \
266type name(type1 arg1,type2 arg2) \
267{ \
268 long __res, junkb, junkc; \
269 if (Z1(__NR_##name)) { \
270 if (Z0(arg1) && Z0(arg2)) { \

272 : "=a" (__res) \
273 : [sysN] "g" (__NR_##name) \
274 : "ebx", "ecx"); \
275 } else if (Z0(arg1) && Z1(arg2)) { \
276 __asm__ __volatile__ ("push %[a2]; popl %%ecx; xorl %%ebx,%%ebx; push %[sysN]; popl %0; int $0x80" \
277 : "=a" (__res) \

281 __asm__ __volatile__ ("xorl %%ecx,%%ecx; push %[a1]; popl %%ebx; push %[sysN]; popl %0; int $0x80" \
282 : "=a" (__res) \
283 : [sysN] "g" (__NR_##name), [a1] "g" ((long)(arg1)) \
284 : "ebx", "ecx"); \
285 } else if (Z1(arg1) && Z1(arg2)) { \
286 __asm__ __volatile__ ("push %[a2]; popl %%ecx; push %[a1]; popl %%ebx; push %[sysN]; popl %0; int $0x80" \
287 : "=a" (__res) \
288 : [sysN] "g" (__NR_##name), [a1] "g" ((long)(arg1)), [a2] "g" ((long)(arg2)) \

304 : "ebx"); \
305 } else if (Z1(arg2)) { \
306 __asm__ __volatile__ ("push %[sysN]; popl %0; push %[a2]; popl %%ecx; int $0x80" \
307 : "=a" (__res), "=b" (junkb) \
308 : [sysN] "g" (__NR_##name),"b" ((long)(arg1)), [a2] "g" ((long)(arg2)) \
309 : "ecx"); \
310 } else { \
311 __asm__ __volatile__ ("push %[sysN]; popl %0; int $0x80" \
312 : "=a" (__res), "=b" (junkb), "=c" (junkc) \

314 } \

319 } \

376/* Cannot write an inline assembler constraint for a specific register.
377 http://gcc.gnu.org/ml/gcc-help/2007-02/msg00068.html
378 Workaround:
379 const register int num asm("v0") = 4;
380 const register char *const str asm("a0") = string;
381 asm volatile("syscall" : : "r"(num), "r"(str));
382*/
383static void *mmap(
384 void *addr, size_t len,
385 int prot, int flags,
386 int fd, off_t offset
387)
388{
389#define __NR_mmap (90+ 4000)
390 register void * const a0 asm("a0") = addr;
391 register size_t const a1 asm("a1") = len;
392 register int const a2 asm("a2") = prot;
393 register int a3 asm("a3") = flags;
394 register int const t0 asm("t0") = fd;
395 register off_t const t1 asm("t1") = offset;
396 register int v0 asm("v0") = __NR_mmap;
397 __asm__ __volatile__(
398 /*"break\n"*/ /* debug only */
399 "addiu $29,$29,-0x20\n"
400 "\tsw $8,0x10($29)\n"
401 "\tsw $9,0x14($29)\n"
402 "\tsyscall\n"
403 "\taddiu $29,$29, 0x20\n"
404 "\tb sysret\n"
405 "sysgo:"
406 /*"break\n"*/ /* debug only */
407 "\tsyscall\n"
408 "sysret: .set noat\n"
409 "\tsltiu $1,$7,1\n" /* 1: no error; 0: error; $7 == a3 */
410 "\taddiu $1,$1,-1\n" /* 0: no error; -1: error */
411 "\tor $2,$2,$1\n" /* $2 == v0; good result, else -1 for error */
412 ".set at\n"
413 : "+r"(v0), "+r"(a3) /* "+r" ==> both read and write */
414 : "r"(a0), "r"(a1), "r"(a2), "r"(t0), "r"(t1)
415 );
416 return (void *)v0;
417}

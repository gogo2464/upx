44int upx_doctest_check(int argc, char **argv) {
45#if defined(DOCTEST_CONFIG_DISABLE)
46 UNUSED(argc);
47 UNUSED(argv);
48 return 0;
49#else
50 if (is_envvar_true("UPX_DEBUG_DOCTEST_DISABLE", "UPX_DEBUG_DISABLE_DOCTEST"))
51 return 0;
52 bool minimal = true; // don't show summary
53 bool duration = false; // don't show timings
54 bool success = false; // don't show all succeeding tests
55#if DEBUG
56 // default for debug builds: do show the [doctest] summary
57 minimal = false;
58#endif
59 const char *e = upx_getenv("UPX_DEBUG_DOCTEST_VERBOSE");
60 if (e && e[0]) {
61 if (strcmp(e, "0") == 0) {
62 minimal = true;
63 } else if (strcmp(e, "1") == 0) {
64 minimal = false;
65 } else if (strcmp(e, "2") == 0) {
66 minimal = false;
67 duration = true;
68 } else if (strcmp(e, "3") == 0) {
69 minimal = false;
70 duration = true;
71 success = true;
72 }
73 }
74 doctest::Context context;
75 if (minimal)
76 context.setOption("dt-minimal", true);
77 if (duration)
78 context.setOption("dt-duration", true);
79 if (success)
80 context.setOption("dt-success", true);
81 // this requires that main_get_options() understands/ignores doctest "--dt-XXX" options
82 if (argc > 0 && argv != nullptr)
83 context.applyCommandLine(argc, argv);
84 int r = context.run();
85 if (r != 0)
86 return 1;
87 if (context.shouldExit())
88 return 2;
89 return 0;
90#endif // DOCTEST_CONFIG_DISABLE
91}

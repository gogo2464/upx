43tribool PackSys::canPack() {
44 byte buf[128];
45
46 fi->readx(buf, sizeof(buf));
47 if (memcmp(buf, "\xff\xff\xff\xff", 4) != 0)
48 return false;
49 if (!fn_has_ext(fi->getName(), "sys")) // query file name
50 return false;
51 checkAlreadyPacked(buf, sizeof(buf));
52 if (file_size < 1024)
53 throwCantPack("file is too small for dos/sys");
54 if (file_size > 0x10000)
55 throwCantPack("file is too large for dos/sys");
56 return true;
57}

86void PackSys::patchLoader(OutputFile *fo, byte *loader, int lsize, unsigned calls) {
87 const int e_len = getLoaderSectionStart("SYSCUTPO");
88 const int d_len = lsize - e_len;
89 assert(e_len > 0 && e_len < 128);
90 assert(d_len > 0 && d_len < 256);
91
92 if (ph.u_len + d_len + ph.overlap_overhead > 0xfffe)
93 throwNotCompressible();
94
95 // use some fields of the original file
96 linker->defineSymbol("attribute", get_le16(ibuf + 4));
97 linker->defineSymbol("interrupt", get_le16(ibuf + 8));
98
99 unsigned copy_to = ph.u_len + d_len + ph.overlap_overhead;
100
101 linker->defineSymbol("calltrick_calls", calls);
102 linker->defineSymbol("copy_source", ph.c_len + lsize - 1);
103 linker->defineSymbol("copy_destination", copy_to);
104 linker->defineSymbol("neg_e_len", 0 - e_len);
105 linker->defineSymbol("NRV2B160", ph.u_len + ph.overlap_overhead + 1);
106 linker->defineSymbol("original_strategy", get_le16(ibuf + 6));
107
108 relocateLoader();
109 loader = getLoader();
110
111 // some day we could use the relocation stuff for patchPackHeader too..
112 patchPackHeader(loader, e_len);
113 // write loader + compressed file
114 fo->write(loader, e_len); // entry
115 fo->write(obuf, ph.c_len); // compressed
116 fo->write(loader + e_len, d_len); // decompressor
117 NO_printf("%-13s: entry : %8u bytes\n", getName(), e_len);
118 NO_printf("%-13s: compressed : %8u bytes\n", getName(), ph.c_len);
119 NO_printf("%-13s: decompressor : %8u bytes\n", getName(), d_len);
120}

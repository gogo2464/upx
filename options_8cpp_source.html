41void Options::reset() noexcept {
42#define opt ERROR_DO_NOT_USE_opt // self-protect against using the wrong variable
43 Options *const o = this;
44 mem_clear(o);
45 o->crp.reset();
46
47 o->cmd = CMD_NONE;
48 o->method = M_NONE;
49 o->level = -1;
50 o->filter = FT_NONE;
51
52 o->backup = -1;
53 o->overlay = -1;
54 o->preserve_mode = true;
55 o->preserve_ownership = true;
56 o->preserve_timestamp = true;
57 o->verbose = 2;
58
59 o->console = CON_FILE;
60#if (ACC_OS_DOS32) && defined(__DJGPP__)
61 o->console = CON_INIT;
62#elif (USE_SCREEN_WIN32)
63 o->console = CON_INIT;
64#elif 1 && defined(__linux__)
65 o->console = CON_INIT;
66#endif
67 // support NO_COLOR, see https://no-color.org/
68 // "... when present and not an empty string (regardless of its value)"
69 const char *e = upx_getenv("NO_COLOR");
70 if (e && e[0])
71 o->console = CON_FILE;
72
73 // options for various executable formats
74
75 o->o_unix.osabi0 = 3; // 3 == ELFOSABI_LINUX
76
77 o->win32_pe.compress_exports = 1;
78 o->win32_pe.compress_icons = 2;
79 o->win32_pe.compress_resources = -1;
80 for (size_t i = 0; i < TABLESIZE(o->win32_pe.compress_rt); i++)
81 o->win32_pe.compress_rt[i] = -1;
82 o->win32_pe.compress_rt[24] = false; // 24 == RT_MANIFEST
83 o->win32_pe.strip_relocs = -1;
84 o->win32_pe.keep_resource = "";
85#undef opt
86}

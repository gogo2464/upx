36Throwable::Throwable(const char *m, int e, bool w) noexcept : super(),
37 msg(nullptr),
38 err(e),
39 is_warning(w) {
40 if (m != nullptr) {
41 msg = strdup(m);
42 assert_noexcept(msg != nullptr);
43 }
44 NO_fprintf(stderr, "construct exception: %zu %zu %s\n", size_t(stats.counter_current),
45 size_t(stats.counter_total), (const char *) msg);
46 stats.counter_current += 1;
47 stats.counter_total += 1;
48}

50Throwable::Throwable(const Throwable &other) noexcept : super(other),
51 msg(nullptr),
52 err(other.err),
53 is_warning(other.is_warning) {
54 if (other.msg != nullptr) {
55 msg = strdup(other.msg);
56 assert_noexcept(msg != nullptr);
57 }
58 NO_fprintf(stderr, "copy construct exception: %zu %zu %s\n", size_t(stats.counter_current),
59 size_t(stats.counter_total), (const char *) msg);
60 stats.counter_current += 1;
61 stats.counter_total += 1;
62}

64Throwable::~Throwable() noexcept {

67 size_t(stats.counter_total), (const char *) msg);
68 upx::owner_free(msg);

198const char *prettyExceptionName(const char *n) noexcept {
199 if (n == nullptr)
200 return "(null)";
201 while (*n) {
202 if (*n >= '0' && *n <= '9') // Linux ABI
203 n++;
204 else if (*n == ' ')
205 n++;
206 else if (strncmp(n, "class ", 6) == 0) // Visual C++ (MSVC)
207 n += 6;
208 else
209 break;
210 }
211 return n;
212}

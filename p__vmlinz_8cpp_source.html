107int PackVmlinuzI386::readFileHeader()
108{
109 setup_size = 0;
110
111 fi->readx(&h, sizeof(h));
112 if (h.boot_flag != 0xAA55)
113 return 0;
114 const bool hdrs = (memcmp(h.hdrs, "HdrS", 4) == 0);
115
116 setup_size = (1 + (h.setup_sects ? h.setup_sects : 4)) * 0x200;
117 if (setup_size <= 0 || setup_size >= file_size)
118 return 0;
119
120 int format = UPX_F_VMLINUZ_i386;
121 unsigned sys_size = ALIGN_UP((unsigned) file_size_u, 16u) - setup_size;
122
123 const unsigned char *p = (const unsigned char *) &h + 0x1e3;
124
125 if (hdrs && memcmp(p, "\x0d\x0a\x07""ELKS", 7) == 0)
126 {
127 format = UPX_F_ELKS_8086;
128 }
129 else if (hdrs && (h.load_flags & 1) != 0)
130 {
131 format = UPX_F_BVMLINUZ_i386;
132 }
133
134 if (0x204<=h.version) {
135 if ((16u * h.sys_size) != sys_size)
136 return 0;
137 }
138 else { // h.sys_size is only 2 bytes
139 if ((16u * (0xffff & h.sys_size)) != (~(~0u<<20) & sys_size))
140 return 0;
141 }
142
143 // FIXME: add more checks for a valid kernel
144
145 return format;
146}

154// read full kernel into obuf[], gzip-decompress into ibuf[],
155// return decompressed size
156int PackVmlinuzI386::decompressKernel()
157{
158 // read whole kernel image
159 obuf.alloc(file_size);
160 fi->seek(0, SEEK_SET);
161 fi->readx(obuf, file_size);
162
163 {
164 const upx_byte *base = nullptr;
165 unsigned relocated = 0;
166
167 // See startup_32: in linux/arch/i386/boot/compressed/head.S
168 const upx_byte *p;
169 unsigned cpa_0 = 0;
170 unsigned cpa_1 = 0;
171 int j;
172 if (0x205<=h.version) {
173 cpa_0 = h.kernel_alignment;
174 cpa_1 = 0u - cpa_0;
175 } else
176 for ((p = &obuf[setup_size]), (j= 0); j < 0x200; ++j, ++p) {
177 if (0==memcmp("\x89\xeb\x81\xc3", p, 4)
178 && 0==memcmp("\x81\xe3", 8+ p, 2)) {
179 // movl %ebp,%ebx
180 // addl $imm.w,%ebx
181 // andl $imm.w,%ebx
182 cpa_0 = 1+ get_te32( 4+ p);
183 cpa_1 = get_te32(10+ p);
184 break;
185 }
186 }
187 for ((p = &obuf[setup_size]), (j= 0); j < 0x200; ++j, ++p) {
188 if (0==memcmp("\x8d\x83", p, 2) // leal d32(%ebx),%eax
189 && 0==memcmp("\xff\xe0", 6+ p, 2) // jmp *%eax
190 ) {
191 relocated = get_te32(2+ p);
192 }
193 if (0==memcmp("\xE8\x00\x00\x00\x00\x5D", p, 6)) {
194 // "call 1f; 1f: pop %ebp" determines actual execution address.
195 // linux-2.6.21 (spring 2007) and later; upx stub needs work
196 // unless LOAD_PHYSICAL_ADDR is known.
197 // Allowed code is: linux-2.6.23/arch/x86/head_32.S 2008-01-01
198 // call 1f
199 // 1: popl %ebp
200 // subl $1b, %ebp # 32-bit immediate
201 // movl $LOAD_PHYSICAL_ADDR, %ebx
202 //
203 if (0==memcmp("\x81\xed", 6+ p, 2) // subl $imm.w,%ebp
204 && 0==memcmp("\xbb", 12+ p, 1) ) { // movl $imm.w,%ebx
205 physical_start = get_te32(13+ p);
206 } else
207 if (0==memcmp("\x81\xed", 6+ p, 2) // subl $imm.w,%ebp
208 && is_pow2(cpa_0) && (0u-cpa_0)==cpa_1) {
209 base = (5+ p) - get_te32(8+ p);
210 config_physical_align = cpa_0;
211 }
212 else {
213 throwCantPack("Unrecognized relocatable kernel");
214 }
215 }
216 // Find "ljmp $__BOOT_CS,$__PHYSICAL_START" if any.
217 if (0==memcmp("\xEA\x00\x00", p, 3) && 0==(0xf & p[3]) && 0==p[4]) {
218 /* whole megabyte < 16 MiB */
219 physical_start = get_te32(1+ p);
220 break;
221 }
222 }
223 if (base && relocated) {
224 p = base + relocated;
225 for (j = 0; j < 0x200; ++j, ++p) {
226 if (0==memcmp("\x01\x9c\x0b", p, 3) // addl %ebx,d32(%ebx,%ecx)
227 ) {
228 page_offset = 0u - get_te32(3+ p);
229 }
230 if (0==memcmp("\x89\xeb", p, 2) // movl %ebp,%ebx
231 && 0==memcmp("\x81\xeb", 2+ p, 2) // subl $imm32,%ebx
232 ) {
233 physical_start = get_te32(4+ p);
234 }
235 }
236 }
237 }
238
239 checkAlreadyPacked(obuf + setup_size, UPX_MIN(file_size - setup_size, 1024LL));
240
241 int gzoff = setup_size;
242 if (0x208<=h.version) {
243 gzoff += h.payload_offset;
244 }
245 for (; gzoff < file_size; gzoff++)
246 {
247 // find gzip header (2 bytes magic + 1 byte method "deflated")
248 int off = find(obuf + gzoff, file_size - gzoff, "\x1F\x8B\x08", 3);
249 if (off < 0)
250 break;
251 gzoff += off;
252 const int gzlen = (h.version < 0x208) ? (file_size - gzoff) : h.payload_length;
253 if (gzlen < 256)
254 break;
255 // check gzip flag byte
256 unsigned char flags = obuf[gzoff + 3];
257 if ((flags & 0xe0) != 0) // reserved bits set
258 continue;
259 //printf("found gzip header at offset %d\n", gzoff);
260
261 // try to decompress
262 int klen;
263 int fd;
264 upx_off_t fd_pos;
265 for (;;)
266 {
267 klen = -1;
268 fd_pos = -1;
269 // open
270 fi->seek(gzoff, SEEK_SET);
271 fd = fi->dupFd();
272 if (fd < 0)
273 break;
274 gzFile zf = gzdopen(fd, "rb");
275 if (zf == nullptr)
276 break;
277 // estimate gzip-decompressed kernel size & alloc buffer
278 if (ibuf.getSize() == 0)
279 ibuf.alloc(gzlen * 3);
280 // decompress
281 klen = gzread(zf, ibuf, ibuf.getSize());
282 fd_pos = lseek(fd, 0, SEEK_CUR);
283 gzclose(zf);
284 fd = -1;
285 if (klen != (int)ibuf.getSize())
286 break;
287 // realloc and try again
288 unsigned s = ibuf.getSize();
289 ibuf.dealloc();
290 ibuf.alloc(3 * s / 2);
291 }
292 if (fd >= 0)
293 (void) close(fd);
294 if (klen <= 0)
295 continue;
296
297 if (klen <= gzlen)
298 continue;
299
300 if (0x208<=h.version && 0==memcmp("\177ELF", ibuf, 4)) {
301 // Full ELF in theory; for now, try to handle as .bin at physical_start.
302 // Check for PT_LOAD.p_paddr being ascending and adjacent.
303 Elf_LE32_Ehdr const *const ehdr = (Elf_LE32_Ehdr const *)(void const *)ibuf;
304 Elf_LE32_Phdr const *phdr = (Elf_LE32_Phdr const *)(ehdr->e_phoff + (char const *)ehdr);
305 Elf_LE32_Shdr const *shdr = (Elf_LE32_Shdr const *)(ehdr->e_shoff + (char const *)ehdr);
306 unsigned hi_paddr = 0, lo_paddr = 0;
307 unsigned delta_off = 0;
308 for (unsigned j=0; j < ehdr->e_phnum; ++j, ++phdr) {
309 if (phdr->PT_LOAD==phdr->p_type) {
310 unsigned step = (hi_paddr + phdr->p_align - 1) & ~(phdr->p_align - 1);
311 if (0==hi_paddr) { // first PT_LOAD
312 if (physical_start!=phdr->p_paddr) {
313 return 0;
314 }
315 delta_off = phdr->p_paddr - phdr->p_offset;
316 lo_paddr = phdr->p_paddr;
317 hi_paddr = phdr->p_filesz + phdr->p_paddr;
318 }
319 else if (step==phdr->p_paddr
320 && delta_off==(phdr->p_paddr - phdr->p_offset)) {
321 hi_paddr = phdr->p_filesz + phdr->p_paddr;
322 }
323 else {
324 return 0; // Not equivalent to a .bin. Too complex for now.
325 }
326 }
327 }
328 // FIXME: ascending order is only a convention; might need sorting.
329 for (unsigned j=1; j < ehdr->e_shnum; ++j) {
330 if (shdr->SHT_PROGBITS==shdr->sh_type) { // SHT_REL might be intermixed
331 if (shdr->SHF_EXECINSTR & shdr[j].sh_flags) {
332 filter_len += shdr[j].sh_size; // FIXME: include sh_addralign
333 }
334 else {
335 break;
336 }
337 }
338 }
339 memmove(ibuf, (lo_paddr - delta_off) + ibuf, hi_paddr - lo_paddr); // FIXME: set_size
340 // FIXME: .bss ? Apparently handled by head.S
341 }
342
343 if (opt->force > 0)
344 return klen;
345
346 // some checks
347 if (fd_pos != file_size)
348 {
349 NO_printf("fd_pos: %jd, file_size: %lld\n", (intmax_t) fd_pos, file_size);
350
351 // linux-2.6.21.5/arch/i386/boot/compressed/vmlinux.lds
352 // puts .data.compressed ahead of .text, .rodata, etc;
353 // so piggy.o need not be last in bzImage. Alas.
354 //throwCantPack("trailing bytes after kernel image; use option '-f' to force packing");
355 }
356
357
358 // see /usr/src/linux/arch/i386/kernel/head.S
359 // 2.4.x: [cli;] cld; mov $...,%eax
360 if (memcmp(ibuf, "\xFC\xB8", 2) == 0) goto head_ok;
361 if (memcmp(ibuf, "\xFA\xFC\xB8", 3) == 0) goto head_ok;
362 // 2.6.21.5 CONFIG_PARAVIRT mov %cs,%eax; test $3,%eax; jne ...;
363 if (memcmp(ibuf, "\x8c\xc8\xa9\x03\x00\x00\x00\x0f\x85", 9) == 0) goto head_ok;
364 if (memcmp(ibuf, "\x8c\xc8\xa8\x03\x0f\x85", 6) == 0) goto head_ok;
365 // 2.6.x: [cli;] cld; lgdt ...
366 if (memcmp(ibuf, "\xFC\x0F\x01", 3) == 0) goto head_ok;
367 if (memcmp(ibuf, "\xFA\xFC\x0F\x01", 4) == 0) goto head_ok;
368 // 2.6.x+grsecurity+strongswan+openwall+trustix: ljmp $0x10,...
369 if (ibuf[0] == 0xEA && memcmp(ibuf+5, "\x10\x00", 2) == 0) goto head_ok;
370 // x86_64 2.6.x
371 if (0xB8==ibuf[0] // mov $...,%eax
372 && 0x8E==ibuf[5] && 0xD8==ibuf[6] // mov %eax,%ds
373 && 0x0F==ibuf[7] && 0x01==ibuf[8] && 020==(070 & ibuf[9]) // lgdtl
374 && 0xB8==ibuf[14] // mov $...,%eax
375 && 0x0F==ibuf[19] && 0xA2==ibuf[20] // cpuid
376 ) goto head_ok;
377
378 // cmpw $0x207,0x206(%esi) Debian vmlinuz-2.6.24-12-generic
379 if (0==memcmp("\x66\x81\xbe\x06\x02\x00\x00\x07\x02", ibuf, 9)) goto head_ok;
380
381 // testb $0x40,0x211(%esi) Fedora vmlinuz-2.6.25-0.218.rc8.git7.fc9.i686
382 if (0==memcmp("\xf6\x86\x11\x02\x00\x00\x40", ibuf, 7)) goto head_ok;
383
384 // rex.W prefix for x86_64
385 if (0x48==ibuf[0]) throwCantPack("x86_64 bzImage is not yet supported");
386
387 throwCantPack("unrecognized kernel architecture; use option '-f' to force packing");
388 head_ok:
389
390 // FIXME: more checks for special magic bytes in ibuf ???
391 // FIXME: more checks for kernel architecture ???
392
393 return klen;
394 }
395
396 return 0;
397}

401{

404 throwCantPack("kernel decompression failed");
405 //OutputFile::dump("kernel.img", ibuf, klen);

409 memcpy(setup_buf, obuf, setup_size);
410 //OutputFile::dump("setup.img", setup_buf, setup_size);
411
412 obuf.dealloc();
413 obuf.allocForCompression(klen);
414
415 ph.u_len = klen;
416 ph.filter = 0;
417}

451void PackVmlinuzI386::pack(OutputFile *fo)
452{
453 readKernel();
454
455 // prepare filter
456 Filter ft(ph.level);
457 ft.buf_len = ph.u_len;
458 ft.addvalue = physical_start; // saves 4 bytes in unfilter code
459
460 // compress
461 upx_compress_config_t cconf; cconf.reset();
462 // limit stack size needed for runtime decompression
463 cconf.conf_lzma.max_num_probs = 1846 + (768 << 4); // ushort: ~28 KiB stack
464 compressWithFilters(&ft, 512, &cconf, getStrategy(ft));
465
466 const unsigned lsize = getLoaderSize();
467
468 defineDecompressorSymbols();
469 defineFilterSymbols(&ft);
470 linker->defineSymbol("src_for_decompressor", zimage_offset + lsize);
471 linker->defineSymbol("original_entry", physical_start);
472 linker->defineSymbol("stack_offset", stack_offset_during_uncompression);
473 relocateLoader();
474
475 MemBuffer loader(lsize);
476 memcpy(loader, getLoader(), lsize);
477 patchPackHeader(loader, lsize);
478
479 boot_sect_t * const bs = (boot_sect_t *) ((unsigned char *) setup_buf);
480 bs->sys_size = ALIGN_UP(lsize + ph.c_len, 16u) / 16;
481 bs->payload_length = ph.c_len;
482
483 fo->write(setup_buf, setup_buf.getSize());
484 fo->write(loader, lsize);
485 fo->write(obuf, ph.c_len);
486#if 0
487 printf("%-13s: setup : %8ld bytes\n", getName(), (long) setup_buf.getSize());
488 printf("%-13s: loader : %8ld bytes\n", getName(), (long) lsize);
489 printf("%-13s: compressed : %8ld bytes\n", getName(), (long) ph.c_len);
490#endif
491
492 // verify
493 verifyOverlappingDecompression();
494
495 // finally check the compression ratio
496 if (!checkFinalCompressionRatio(fo))
497 throwNotCompressible();
498}

505void PackBvmlinuzI386::buildLoader(const Filter *ft)
506{
507 // prepare loader
508 initLoader(stub_i386_linux_kernel_vmlinuz, sizeof(stub_i386_linux_kernel_vmlinuz));
509 if (0!=page_offset) { // relocatable kernel
510 assert(0==ft->id || 0x40==(0xf0 & ft->id)); // others assume fixed buffer address
511 addLoader("LINUZ000,LINUZ001,LINUZVGA,LINUZ101,LINUZ110",
512 ((0!=config_physical_align) ? "LINUZ120" : "LINUZ130"),
513 "LINUZ140,LZCUTPOI,LINUZ141",
514 (ft->id ? "LINUZ145" : ""),
515 (ph.first_offset_found == 1 ? "LINUZ010" : ""),
516 nullptr);
517 }
518 else {
519 addLoader("LINUZ000,LINUZ001,LINUZVGA,LINUZ005",
520 ph.first_offset_found == 1 ? "LINUZ010" : "",
521 (0x40==(0xf0 & ft->id)) ? "LZCKLLT1" : (ft->id ? "LZCALLT1" : ""),
522 "LBZIMAGE,IDENTSTR",
523 "+40", // align the stuff to 4 byte boundary
524 "UPX1HEAD", // 32 byte
525 "LZCUTPOI",
526 nullptr);
527 // fake alignment for the start of the decompressor
528 //linker->defineSymbol("LZCUTPOI", 0x1000);
529 }
530
531 addLoader(getDecompressorSections(), nullptr);
532
533 if (ft->id)
534 {
535 assert(ft->calls > 0);
536 if (0x40==(0xf0 & ft->id)) {
537 addLoader("LZCKLLT9", nullptr);
538 }
539 else {
540 addLoader("LZCALLT9", nullptr);
541 }
542 addFilter32(ft->id);
543 }
544 if (0!=page_offset) {
545 addLoader("LINUZ150,IDENTSTR,+40,UPX1HEAD", nullptr);
546 unsigned const l_len = getLoaderSize();
547 unsigned const c_len = ALIGN_UP(ph.c_len, 4u);
548 unsigned const e_len = getLoaderSectionStart("LINUZ141") -
549 getLoaderSectionStart("LINUZ110");
550 linker->defineSymbol("compressed_length", c_len);
551 linker->defineSymbol("load_physical_address", physical_start); // FIXME
552 if (0!=config_physical_align) {
553 linker->defineSymbol("neg_config_physical_align", 0u - config_physical_align);
554 }
555 linker->defineSymbol("neg_length_mov", 0u - ALIGN_UP(c_len + l_len, 4u));
556 linker->defineSymbol("neg_page_offset", 0u - page_offset);
557 //linker->defineSymbol("physical_start", physical_start);
558 linker->defineSymbol("unc_length", ph.u_len);
559 linker->defineSymbol("dec_offset", ph.overlap_overhead + e_len);
560 linker->defineSymbol("unc_offset", ph.overlap_overhead + ph.u_len - c_len);
561 }
562 else {
563 addLoader("LINUZ990", nullptr);
564 }
565}

568void PackBvmlinuzI386::pack(OutputFile *fo)
569{
570 readKernel();
571
572 // prepare filter
573 Filter ft(ph.level);
574 ft.buf_len = (filter_len ? filter_len : (ph.u_len * 3)/5);
575 // May 2008: 3/5 is heuristic to cover most .text but avoid non-instructions.
576 // Otherwise "call trick" filter cannot find a free marker byte,
577 // especially when it searches over tables of data.
578 ft.addvalue = 0; // The destination buffer might be relocated at runtime.
579
580 upx_compress_config_t cconf; cconf.reset();
581 // LINUZ001 allows most of low memory as stack for Bvmlinuz
582 cconf.conf_lzma.max_num_probs = (0x90000 - 0x10000)>>1; // ushort: 512 KiB stack
583
584 compressWithFilters(&ft, 512, &cconf, getStrategy(ft));
585
586 // align everything to dword boundary - it is easier to handle
587 unsigned c_len = ph.c_len;
588 memset(obuf + c_len, 0, 4);
589 c_len = ALIGN_UP(c_len, 4u);
590
591 const unsigned lsize = getLoaderSize();
592
593 if (M_IS_LZMA(ph.method)) {
594 const lzma_compress_result_t *res = &ph.compress_result.result_lzma;
595 upx_uint32_t properties = // lc, lp, pb, dummy
596 (res->lit_context_bits << 0) |
597 (res->lit_pos_bits << 8) |
598 (res->pos_bits << 16);
599 if (bele->isBE()) // big endian - bswap32
600 properties = bswap32(properties);
601 linker->defineSymbol("lzma_properties", properties);
602 // -2 for properties
603 if (linker->findSymbol("lzma_c_len", false)) {
604 linker->defineSymbol("lzma_c_len", ph.c_len - 2);
605 }
606 if (linker->findSymbol("lzma_u_len", false)) {
607 linker->defineSymbol("lzma_u_len", ph.u_len);
608 }
609 unsigned const stack = getDecompressorWrkmemSize();
610 linker->defineSymbol("lzma_stack_adjust", 0u - stack);
611 }
612
613 const int e_len = getLoaderSectionStart("LZCUTPOI");
614 assert(e_len > 0);
615
616 if (0==page_offset) { // not relocatable kernel
617 const unsigned d_len4 = ALIGN_UP(lsize - e_len, 4u);
618 const unsigned decompr_pos = ALIGN_UP(ph.u_len + ph.overlap_overhead, 16u);
619 const unsigned copy_size = c_len + d_len4;
620 const unsigned edi = decompr_pos + d_len4 - 4; // copy to

761int PackVmlinuzARMEL::decompressKernel()
762{
763 // read whole kernel image
764 obuf.alloc(file_size);
765 fi->seek(0, SEEK_SET);
766 fi->readx(obuf, file_size);
767
768 //checkAlreadyPacked(obuf + setup_size, UPX_MIN(file_size - setup_size, 1024LL));
769
770 // Find head.S:
771 // bl decompress_kernel # 0xeb......
772 // b call_kernel # 0xea......
773 //LC0: .word LC0 # self!
774 unsigned decompress_kernel = 0;
775 unsigned caller1 = 0;
776 unsigned caller2 = 0;
777 unsigned got_start = 0;
778 unsigned got_end = 0;
779 for (unsigned j = 0; j < 0x400; j+=4) {
780 unsigned w;
781 if (j!=get_te32(j + obuf)) {
782 continue;
783 }
784 if (0xea000000!=(0xff000000& get_te32(j - 4 + obuf))
785 || 0xeb000000!=(0xff000000&(w= get_te32(j - 8 + obuf))) ) {
786 continue;
787 }
788 caller1 = j - 8;
789 decompress_kernel = ((0x00ffffff & w)<<2) + 8+ caller1;
790 for (unsigned k = 12; k<=128; k+=4) {
791 w = get_te32(j - k + obuf);
792 if (0xeb000000==(0xff000000 & w)
793 && decompress_kernel==(((0x00ffffff & w)<<2) + 8+ j - k) ) {
794 caller2 = j - k;
795 break;
796 }
797 }
798 got_start = get_te32(5*4 + j + obuf);
799 got_end = get_te32(6*4 + j + obuf);
800#if 0 /*{*/
801 printf("decompress_kernel=0x%x got_start=0x%x got_end=0x%x\n",
802 decompress_kernel, got_start, got_end);
803#endif /*}*/
804 break;
805 }
806 if (0==decompress_kernel) {
807 return 0;
808 }
809
810 // Find first subroutine that is called by decompress_kernel,
811 // which we will consider to be the start of the gunzip module
812 // and the end of the non-gunzip modules.
813 for (unsigned j = decompress_kernel; j < (unsigned)file_size; j+=4) {
814 unsigned w = get_te32(j + obuf);
815 if (0xeb800000==(0xff800000 & w)) {
816 setup_size = 8+ ((0xff000000 | w)<<2) + j;
817 // Move the GlobalOffsetTable.
818 for (unsigned k = got_start; k < got_end; k+=4) {
819 w = get_te32(k + obuf);
820 // FIXME: must relocate w
821 set_te32(k - got_start + setup_size + obuf, w);
822 }
823 setup_size += got_end - got_start;
824 set_te32(&obuf[caller1], 0xeb000000 |
825 (0x00ffffff & ((setup_size - (8+ caller1))>>2)) );
826 set_te32(&obuf[caller2], 0xeb000000 |
827 (0x00ffffff & ((setup_size - (8+ caller2))>>2)) );
828 break;
829 }
830 }
831
832 for (int gzoff = 0; gzoff < 0x4000; gzoff+=4) {
833 // find gzip header (2 bytes magic + 1 byte method "deflated")
834 int off = find(obuf + gzoff, file_size - gzoff, "\x1F\x8B\x08", 3);
835 if (off < 0 || 0!=(3u & off))
836 break; // not found, or not word-aligned
837 gzoff += off;
838 const int gzlen = file_size - gzoff;
839 if (gzlen < 256)
840 break;
841 // check gzip flag byte
842 unsigned char flags = obuf[gzoff + 3];
843 if ((flags & 0xe0) != 0) // reserved bits set
844 continue;
845 //printf("found gzip header at offset %d\n", gzoff);
846
847 // try to decompress
848 int klen;
849 int fd;
850 upx_off_t fd_pos;
851 for (;;)
852 {
853 klen = -1;
854 fd_pos = -1;
855 // open
856 fi->seek(gzoff, SEEK_SET);
857 fd = fi->dupFd();
858 if (fd < 0)
859 break;
860 gzFile zf = gzdopen(fd, "rb");
861 if (zf == nullptr)
862 break;
863 // estimate gzip-decompressed kernel size & alloc buffer
864 if (ibuf.getSize() == 0)
865 ibuf.alloc(gzlen * 3);
866 // decompress
867 klen = gzread(zf, ibuf, ibuf.getSize());
868 fd_pos = lseek(fd, 0, SEEK_CUR);
869 gzclose(zf);
870 fd = -1;
871 if (klen != (int)ibuf.getSize())
872 break;
873 // realloc and try again
874 unsigned const s = ibuf.getSize();
875 ibuf.dealloc();
876 ibuf.alloc(3 * s / 2);
877 }
878 if (fd >= 0)
879 (void) close(fd);
880 if (klen <= 0)
881 continue;
882
883 if (klen <= gzlen)
884 continue;
885
886 if (opt->force > 0)
887 return klen;
888
889 // some checks
890 if (fd_pos != file_size) {
891 //printf("fd_pos: %ld, file_size: %ld\n", (long)fd_pos, (long)file_size);
892 }
893
894 //head_ok:
895
896 // FIXME: more checks for special magic bytes in ibuf ???
897 // FIXME: more checks for kernel architecture ???
898
899 return klen;
900 }
901
902 return 0;
903}

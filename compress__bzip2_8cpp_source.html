47static int convert_errno_from_bzip2(int r) {
48 switch (r) {
49 case BZ_OK:
50 return UPX_E_OK;
51 case BZ_MEM_ERROR:
52 return UPX_E_OUT_OF_MEMORY;
53 // TODO later: convert to UPX_E_INPUT_OVERRUN, UPX_E_OUTPUT_OVERRUN
54 default:
55 break;
56 }
57 return UPX_E_ERROR;
58}

64int upx_bzip2_compress(const upx_bytep src, unsigned src_len, upx_bytep dst, unsigned *dst_len,
65 upx_callback_t *cb_parm, int method, int level,
66 const upx_compress_config_t *cconf_parm, upx_compress_result_t *cresult) {
67 assert(method == M_BZIP2);
68 assert(level > 0);
69 assert(cresult != nullptr);
70 UNUSED(cb_parm);
71 int r = UPX_E_ERROR;
72 const bzip2_compress_config_t *const lcconf = cconf_parm ? &cconf_parm->conf_bzip2 : nullptr;
73 bzip2_compress_result_t *const res = &cresult->result_bzip2;
74 res->reset();
75
76 int blockSize100k = (src_len + 100000 - 1) / 100000;
77 if (blockSize100k < 1)
78 blockSize100k = 1;
79 if (blockSize100k > 9)
80 blockSize100k = 9;
81 // idea for later: could enhance lcconf to allow setting blockSize100k
82 UNUSED(lcconf);
83 if (level <= 3 && blockSize100k > level)
84 blockSize100k = level;
85
86 char *dest = (char *) dst;
87 char *source = (char *) const_cast<byte *>(src);
88 r = BZ2_bzBuffToBuffCompress(dest, dst_len, source, src_len, blockSize100k, 0, 0);
89 return convert_errno_from_bzip2(r);
90}

112int upx_bzip2_test_overlap(const upx_bytep buf, const upx_bytep tbuf, unsigned src_off,
113 unsigned src_len, unsigned *dst_len, int method,
114 const upx_compress_result_t *cresult) {
115 assert(method == M_BZIP2);
116
117 MemBuffer b(src_off + src_len);
118 memcpy(b + src_off, buf + src_off, src_len);
119 unsigned saved_dst_len = *dst_len;
120 int r = upx_bzip2_decompress(raw_index_bytes(b, src_off, src_len), src_len,
121 raw_bytes(b, *dst_len), dst_len, method, cresult);
122 if (r != UPX_E_OK)
123 return r;
124 if (*dst_len != saved_dst_len)
125 return UPX_E_ERROR;
126 // NOTE: there is a very tiny possibility that decompression has
127 // succeeded but the data is not restored correctly because of
128 // in-place buffer overlapping, so we use an extra memcmp().
129 if (tbuf != nullptr && memcmp(tbuf, b, *dst_len) != 0)
130 return UPX_E_ERROR;
131 return UPX_E_OK;
132}

148static bool check_bzip2(const int method, const int level, const unsigned expected_c_len) {
149 const unsigned u_len = 16384;
150 const unsigned c_extra = 4096;
151 MemBuffer u_buf, c_buf, d_buf;
152 unsigned c_len, d_len;
153 upx_compress_result_t cresult;
154 int r;
155
156 u_buf.alloc(u_len);
157 memset(u_buf, 0, u_len);
158 c_buf.allocForCompression(u_len, c_extra);
159 d_buf.allocForDecompression(u_len);
160
161 c_len = c_buf.getSize() - c_extra;
162 r = upx_bzip2_compress(raw_bytes(u_buf, u_len), u_len, raw_index_bytes(c_buf, c_extra, c_len),
163 &c_len, nullptr, method, level, NULL_cconf, &cresult);
164 if (r != 0 || c_len != expected_c_len)
165 return false;
166
167 d_len = d_buf.getSize();
168 r = upx_bzip2_decompress(raw_index_bytes(c_buf, c_extra, c_len), c_len, raw_bytes(d_buf, d_len),
169 &d_len, method, nullptr);
170 if (r != 0 || d_len != u_len || memcmp(u_buf, d_buf, u_len) != 0)
171 return false;
172
173 d_len = u_len - 1;
174 r = upx_bzip2_decompress(raw_index_bytes(c_buf, c_extra, c_len), c_len, raw_bytes(d_buf, d_len),
175 &d_len, method, nullptr);
176 if (r == 0)
177 return false;
178
179 // TODO: rewrite Packer::findOverlapOverhead() so that we can test it here
180 // unsigned x_len = d_len;
181 // r = upx_bzip2_test_overlap(c_buf, u_buf, c_extra, c_len, &x_len, method, nullptr);
182 return true;
183}

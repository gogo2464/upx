49static inline unsigned f80_call(int filter_id) { return (1 + (0x0f & filter_id)) % 3; }

51static inline unsigned f80_jmp1(int filter_id) { return ((1 + (0x0f & filter_id)) / 3) % 3; }

55void Packer::addFilter32(int filter_id) {
56 assert(filter_id > 0);
57 assert(isValidFilter(filter_id));
58
59 if (filter_id < 0x80) {
60 if (0x50 == (0xF0 & filter_id)) {
61 addLoader("ctok32.00",
62 ((0x50 == filter_id) ? "ctok32.50"
63 : (0x51 == filter_id) ? "ctok32.51"
64 : ""),
65 "ctok32.10");
66 } else if ((filter_id & 0xf) % 3 == 0) {
67 if (filter_id < 0x40) {
68 addLoader("CALLTR00", (filter_id > 0x20) ? "CTCLEVE1" : "", "CALLTR01",
69 (filter_id & 0xf) > 3
70 ? (filter_id > 0x20 ? "CTBSHR01,CTBSWA01" : "CTBROR01,CTBSWA01")
71 : "",
72 "CALLTR02");
73 } else if (0x40 == (0xF0 & filter_id)) {
74 addLoader("ctok32.00");
75 if (9 <= (0xf & filter_id)) {
76 addLoader("ctok32.10");
77 }
78 addLoader("ctok32.20");
79 if (9 <= (0xf & filter_id)) {
80 addLoader("ctok32.30");
81 }
82 addLoader("ctok32.40");
83 }
84 } else
85 addLoader("CALLTR10", (filter_id & 0xf) % 3 == 1 ? "CALLTRE8" : "CALLTRE9", "CALLTR11",
86 (filter_id > 0x20) ? "CTCLEVE2" : "", "CALLTR12",
87 (filter_id & 0xf) > 3
88 ? (filter_id > 0x20 ? "CTBSHR11,CTBSWA11" : "CTBROR11,CTBSWA11")
89 : "",
90 "CALLTR13");
91 }
92 if (0x80 == (filter_id & 0xF0)) {
93 const bool x386 = (opt->cpu_x86 <= opt->CPU_386);
94 const unsigned n_mru = ph.n_mru ? 1 + ph.n_mru : 0;
95 const bool mrupwr2 = (0 != n_mru) && 0 == ((n_mru - 1) & n_mru);
96 const unsigned f_call = f80_call(filter_id);
97 const unsigned f_jmp1 = f80_jmp1(filter_id);
98 const unsigned f_jcc2 = f80_jcc2(filter_id);
99
100 if (NOFILT != f_jcc2) {
101 addLoader("LXJCC010");
102 if (n_mru) {
103 addLoader("LXMRU045");
104 } else {
105 addLoader("LXMRU046");
106 }
107 if (0 == n_mru || MRUFLT != f_jcc2) {
108 addLoader("LXJCC020");
109 } else { // 0 != n_mru
110 addLoader("LXJCC021");
111 }
112 if (NOFILT != f_jcc2) {
113 addLoader("LXJCC023");
114 }
115 }
116 addLoader("LXUNF037");
117 if (x386) {
118 if (n_mru) {
119 addLoader("LXUNF386");
120 }
121 addLoader("LXUNF387");
122 if (n_mru) {
123 addLoader("LXUNF388");
124 }
125 } else {
126 addLoader("LXUNF486");
127 if (n_mru) {
128 addLoader("LXUNF487");
129 }
130 }
131 if (n_mru) {
132 addLoader("LXMRU065");
133 if (256 == n_mru) {
134 addLoader("MRUBYTE3");
135 } else {
136 addLoader("MRUARB30");
137 if (mrupwr2) {
138 addLoader("MRUBITS3");
139 } else {
140 addLoader("MRUARB40");
141 }
142 }
143 addLoader("LXMRU070");
144 if (256 == n_mru) {
145 addLoader("MRUBYTE4");
146 } else if (mrupwr2) {
147 addLoader("MRUBITS4");
148 } else {
149 addLoader("MRUARB50");
150 }
151 addLoader("LXMRU080");
152 if (256 == n_mru) {
153 addLoader("MRUBYTE5");
154 } else {
155 addLoader("MRUARB60");
156 if (mrupwr2) {
157 addLoader("MRUBITS5");
158 } else {
159 addLoader("MRUARB70");
160 }
161 }
162 addLoader("LXMRU090");
163 if (256 == n_mru) {
164 addLoader("MRUBYTE6");
165 } else {
166 addLoader("MRUARB80");
167 if (mrupwr2) {
168 addLoader("MRUBITS6");
169 } else {
170 addLoader("MRUARB90");
171 }
172 }
173 addLoader("LXMRU100");
174 }
175 addLoader("LXUNF040");
176 if (n_mru) {
177 addLoader("LXMRU110");
178 } else {
179 addLoader("LXMRU111");
180 }
181 addLoader("LXUNF041");
182 addLoader("LXUNF042");
183 if (n_mru) {
184 addLoader("LXMRU010");
185 if (NOFILT != f_jmp1 && NOFILT == f_call) {
186 addLoader("LXJMPA00");
187 } else {
188 addLoader("LXCALLB0");
189 }
190 addLoader("LXUNF021");
191 } else {
192 addLoader("LXMRU022");
193 if (NOFILT != f_jmp1 && NOFILT == f_call) {
194 addLoader("LXJMPA01");
195 } else {
196 addLoader("LXCALLB1");
197 }
198 }
199 if (n_mru) {
200 if (256 != n_mru && mrupwr2) {
201 addLoader("MRUBITS1");
202 }
203 addLoader("LXMRU030");
204 if (256 == n_mru) {
205 addLoader("MRUBYTE1");
206 } else {
207 addLoader("MRUARB10");
208 }
209 addLoader("LXMRU040");
210 }
211
212 addLoader("LXUNF030");
213 if (NOFILT != f_jcc2) {
214 addLoader("LXJCC000");
215 }
216 if (NOFILT != f_call || NOFILT != f_jmp1) { // at least one is filtered
217 // shift opcode origin to zero
218 if (0 == n_mru) {
219 addLoader("LXCJ0MRU");
220 } else {
221 addLoader("LXCJ1MRU");
222 }
223
224 // determine if in range
225 if ((NOFILT != f_call) && (NOFILT != f_jmp1)) { // unfilter both
226 addLoader("LXCALJMP");
227 }
228 if ((NOFILT == f_call) ^ (NOFILT == f_jmp1)) { // unfilter just one
229 if (0 == n_mru) {
230 addLoader("LXCALL00");
231 } else {
232 addLoader("LXCALL01");
233 }
234 }
235
236 // determine if mru applies
237 if (0 == n_mru || !((FNOMRU == f_call) || (FNOMRU == f_jmp1))) {
238 // no mru, or no exceptions
239 addLoader("LXCJ2MRU");
240 } else {
241 // mru on one, but not the other
242 addLoader("LXCJ4MRU");
243 if (MRUFLT == f_jmp1) { // JMP only
244 addLoader("LXCJ6MRU");
245 } else if (MRUFLT == f_call) { // CALL only
246 addLoader("LXCJ7MRU");
247 }
248 addLoader("LXCJ8MRU");
249 }
250 }
251 addLoader("LXUNF034");
252 if (n_mru) {
253 addLoader("LXMRU055");
254 if (256 == n_mru) {
255 addLoader("MRUBYTE2");
256 } else if (mrupwr2) {
257 addLoader("MRUBITS2");
258 } else {
259 addLoader("MRUARB20");
260 }
261 addLoader("LXMRU057");
262 }
263 }
264}

274void Packer::defineFilterSymbols(const Filter *ft) {
275 if (ft->id == 0) {
276 linker->defineSymbol("filter_length", 0);
277 linker->defineSymbol("filter_cto", 0);
278 return;
279 }
280 assert(ft->calls > 0);
281 assert(ft->buf_len > 0);
282
283 if (ft->id >= 0x40 && ft->id <= 0x4f) {
284 linker->defineSymbol("filter_length", ft->buf_len);
285 linker->defineSymbol("filter_cto", ft->cto);
286 } else if (ft->id >= 0x50 && ft->id <= 0x5f) {
287 linker->defineSymbol("filter_id", ft->id);
288 linker->defineSymbol("filter_cto", ft->cto);
289 } else if ((ft->id & 0xf) % 3 == 0) {
290 linker->defineSymbol("filter_length", ft->calls);
291 linker->defineSymbol("filter_cto", ft->cto);
292 } else {
293 linker->defineSymbol("filter_length", ft->lastcall - ft->calls * 4);
294 linker->defineSymbol("filter_cto", ft->cto);
295 }
296
297#if 0

299 const int mru = ph.n_mru ? 1 + ph.n_mru : 0;
300 if (mru && mru != 256) {
301 const unsigned is_pwr2 = (0 == ((mru - 1) & mru));

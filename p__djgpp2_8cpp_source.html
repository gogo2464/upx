108static bool is_dlm(InputFile *fi, unsigned coff_offset) {
109 byte buf[4];
110 unsigned off;
111
112 try {
113 fi->seek(coff_offset, SEEK_SET);
114 fi->readx(buf, 4);
115 off = get_le32(buf);
116 if (off > coff_offset + 4)
117 return false;
118 fi->seek(off, SEEK_SET);
119 fi->readx(buf, 4);
120 if (memcmp(buf, "DLMF", 4) == 0)
121 return true;
122 } catch (const IOException &) {
123 }
124 return false;
125}

127static void handle_allegropak(InputFile *fi, OutputFile *fo) {
128 byte b[8];
129 int pfsize = 0;
130
131 try {
132 fi->seek(-8, SEEK_END);
133 fi->readx(b, 8);
134 if (memcmp(b, "slh+", 4) != 0)
135 return;
136 pfsize = get_be32_signed(b + 4);
137 if (pfsize <= 8 || pfsize >= fi->st.st_size)
138 return;
139 fi->seek(-pfsize, SEEK_END);
140 } catch (const IOException &) {
141 return;
142 }
143 MemBuffer buf(0x4000);
144 while (pfsize > 0) {
145 const int len = UPX_MIN(pfsize, (int) buf.getSize());
146 fi->readx(buf, len);
147 fo->write(buf, len);
148 pfsize -= len;
149 }
150}

152int PackDjgpp2::readFileHeader() {
153 byte hdr[0x1c];
154 byte magic[8];
155
156 fi->seek(0, SEEK_SET);
157 fi->readx(hdr, sizeof(hdr));
158 if (get_le16(hdr) == 0x5a4d) // MZ exe signature, stubbed?
159 {
160 coff_offset = 512 * get_le16(hdr + 4);
161 if (get_le16(hdr + 2) != 0)
162 coff_offset += get_le16(hdr + 2) - 512;
163 fi->seek(512, SEEK_SET);
164 fi->readx(magic, 8);
165 if (memcmp("go32stub", magic, 8) != 0)
166 return 0; // not V2 image
167 fi->seek(coff_offset, SEEK_SET);
168 if (fi->read(&coff_hdr, sizeof(coff_hdr)) != sizeof(coff_hdr))
169 throwCantPack("skipping djgpp symlink");
170 } else {
171 fi->seek(coff_offset, SEEK_SET);
172 fi->readx(&coff_hdr, 0xa8);
173 }
174 if (coff_hdr.f_magic != 0x014c) // I386MAGIC
175 return 0;
176 if ((coff_hdr.f_flags & 2) == 0) // F_EXEC - COFF executable
177 return 0;
178 if (coff_hdr.a_magic != 0413) // ZMAGIC - demand load format
179 return 0;
180 // FIXME: check for Linux etc.
181
182 text = &coff_hdr.sh[0];
183 data = &coff_hdr.sh[1];
184 bss = &coff_hdr.sh[2];
185 return UPX_F_DJGPP2_COFF;
186}

200tribool PackDjgpp2::canPack() {
201 if (!readFileHeader())
202 return false;
203 if (is_dlm(fi, coff_offset))
204 throwCantPack("can't handle DLM");
205
206 if (opt->force == 0)
207 if (text->size != coff_hdr.a_tsize || data->size != coff_hdr.a_dsize)
208 throwAlreadyPacked();
209
210 // Check for gap in vaddr between text and data, or between data and bss.
211 if (text->vaddr + text->size != data->vaddr || data->vaddr + data->size != bss->vaddr) {
212 // "Non-standard" layout of text,data,bss: not contiguous in vaddr.
213 // But should be OK if no overlap.
214
215 // Check for no overlap of text and data:
216 // neither by vaddr, nor by image data
217 if (text->vaddr + text->size <= data->vaddr &&
218 data->scnptr - text->scnptr <= data->vaddr - text->vaddr) {
219 // Examples: Quake1; FreePascal(DOS) install.exe (github-issue45)
220 // Hack: enlarge text image data to eliminate the gap.
221 text->size = coff_hdr.a_tsize = data->scnptr - text->scnptr;
222 // But complain if this causes overlap in vaddr
223 if (text->vaddr + text->size > data->vaddr)
224 throwAlreadyPacked();
225 } else
226 throwAlreadyPacked();
227 }
228 // FIXME: check for Linux etc.
229 return true;
230}

236void PackDjgpp2::pack(OutputFile *fo) {
237 handleStub(fo);
238
239 // patch coff header #1: "strip" debug info
240 stripDebug();
241
242 // read file
243 const unsigned size = text->size + data->size;
244 const unsigned tpos = text->scnptr;
245 const unsigned hdrsize = 20 + 28 + mem_size(sizeof(external_scnhdr_t), coff_hdr.f_nscns);
246 const unsigned usize = size + hdrsize;
247 if (hdrsize < sizeof(coff_hdr) || hdrsize > tpos)
248 throwCantPack("coff header error");
249
250 ibuf.alloc(usize);
251 obuf.allocForCompression(usize);
252
253 fi->seek(coff_offset, SEEK_SET);
254 fi->readx(ibuf, hdrsize); // orig. coff header
255 fi->seek(coff_offset + tpos, SEEK_SET);
256 fi->readx(ibuf + hdrsize, size);
257
258 // prepare packheader
259 ph.u_len = usize;
260 // prepare filter
261 Filter ft(ph.level);
262 ft.buf_len = usize - data->size;
263 ft.addvalue = text->vaddr - hdrsize;
264 // compress
265 upx_compress_config_t cconf;
266 cconf.reset();
267 // limit stack size needed for runtime decompression
268 cconf.conf_lzma.max_num_probs = 1846 + (768 << 4); // ushort: ~28 KiB stack
269 compressWithFilters(&ft, 512, &cconf);
270
271 // patch coff header #2
272 const unsigned lsize = getLoaderSize();
273 assert(lsize % 4 == 0);
274 text->size = lsize; // new size of .text
275 data->size = ph.c_len; // new size of .data
276
277 unsigned stack = 1024 + ph.overlap_overhead + getDecompressorWrkmemSize();
278 stack = ALIGN_UP(stack, 16u);
279 if (bss->size < stack) // give it a .bss
280 bss->size = stack;
281
282 text->scnptr = sizeof(coff_hdr);
283 data->scnptr = text->scnptr + text->size;
284 data->vaddr = bss->vaddr + ((data->scnptr + data->size) & 0x1ff) - data->size +
285 ph.overlap_overhead - 0x200;
286 coff_hdr.f_nscns = 3;
287
288 linker->defineSymbol("original_entry", coff_hdr.a_entry);
289 linker->defineSymbol("length_of_bss", ph.overlap_overhead / 4);
290 defineDecompressorSymbols();
291 // Just need no overlap; non-contiguous (gap length > 0)) is OK
292 assert(bss->vaddr >= ((size + 0x1ff) & ~0x1ff) + (text->vaddr & ~0x1ff));
293 linker->defineSymbol("stack_for_lzma", bss->vaddr + bss->size);
294 linker->defineSymbol("start_of_uncompressed", text->vaddr - hdrsize);
295 linker->defineSymbol("start_of_compressed", data->vaddr);
296 defineFilterSymbols(&ft);
297
298 // we should not overwrite our decompressor during unpacking
299 // the original coff header (which is put just before the
300 // beginning of the original .text section)
301 assert(text->vaddr > hdrsize + lsize + sizeof(coff_hdr));
302
303 // patch coff header #3
304 text->vaddr = sizeof(coff_hdr);
305 coff_hdr.a_entry = sizeof(coff_hdr) + getLoaderSection("DJ2MAIN1");
306 bss->vaddr += ph.overlap_overhead;
307 bss->size -= ph.overlap_overhead;
308
309 // because of a feature (bug?) in stub.asm we need some padding
310 memcpy(obuf + data->size, "UPX", 3);
311 data->size = ALIGN_UP(data->size, 4u);
312
313 linker->defineSymbol("DJ2MAIN1", coff_hdr.a_entry);
314 relocateLoader();
315
316 // prepare loader
317 MemBuffer loader(lsize);
318 memcpy(loader, getLoader(), lsize);
319 patchPackHeader(loader, lsize);
320
321 // write coff header, loader and compressed file
322 fo->write(&coff_hdr, sizeof(coff_hdr));
323 fo->write(loader, lsize);
324 if (opt->debug.dump_stub_loader)
325 OutputFile::dump(opt->debug.dump_stub_loader, loader, lsize);
326 fo->write(obuf, data->size);
327#if 0
328 printf("%-13s: coff hdr : %8d bytes\n", getName(), (int) sizeof(coff_hdr));
329 printf("%-13s: loader : %8d bytes\n", getName(), (int) lsize);
330 printf("%-13s: compressed : %8d bytes\n", getName(), (int) data->size);
331#endif
332
333 // verify
334 verifyOverlappingDecompression();
335
336 // handle overlay
337 // FIXME: only Allegro pakfiles are supported
338 handle_allegropak(fi, fo);
339
340 // finally check the compression ratio
341 if (!checkFinalCompressionRatio(fo))
342 throwNotCompressible();
343}

362void PackDjgpp2::unpack(OutputFile *fo) {
363 handleStub(fo);
364
365 ibuf.alloc(ph.c_len);
366 obuf.allocForDecompression(ph.u_len);
367
368 fi->seek(coff_offset + ph.buf_offset + ph.getPackHeaderSize(), SEEK_SET);
369 fi->readx(ibuf, ph.c_len);
370
371 // decompress
372 decompress(ibuf, obuf);
373
374 coff_header_t *const chdr = (coff_header_t *) raw_bytes(obuf, sizeof(coff_header_t));
375 text = &chdr->sh[0];
376 data = &chdr->sh[1];
377 bss = &chdr->sh[2];
378
379 const unsigned hdrsize = 20 + 28 + mem_size(sizeof(external_scnhdr_t), chdr->f_nscns);
380 if (hdrsize < sizeof(coff_hdr) || hdrsize > text->scnptr || hdrsize > ph.u_len)
381 throwCantUnpack("coff header error");
382
383 unsigned addvalue;
384 if (ph.version >= 14)
385 addvalue = text->vaddr - hdrsize;
386 else
387 addvalue = text->vaddr & ~0x1ff; // for old versions
388
389 // unfilter
390 if (ph.filter) {
391 Filter ft(ph.level);
392 ft.init(ph.filter, addvalue);
393 ft.cto = (byte) ph.filter_cto;
394 if (ph.version < 11) {
395 byte ctobuf[4];
396 fi->readx(ctobuf, 4);
397 ft.cto = (byte) (get_le32(ctobuf) >> 24);
398 }
399 ft.unfilter(obuf, ph.u_len - data->size);
400 }
401
402 if (ph.version < 14) {
403 // fixup for the aligning bug in strip 2.8+
404 text->scnptr &= 0x1ff;
405 data->scnptr = text->scnptr + text->size;
406 // write decompressed file
407 if (fo)
408 fo->write(obuf, ph.u_len);
409 } else {
410 // write the header
411 // some padding might be required between the end
412 // of the header and the start of the .text section
413
414 const unsigned padding = text->scnptr - hdrsize;
415 ibuf.clear(0, padding);
416
417 if (fo) {
418 fo->write(obuf, hdrsize);
419 fo->write(ibuf, padding);
420 fo->write(obuf + hdrsize, ph.u_len - hdrsize);
421 }
422 }
423
424 if (fo)
425 handle_allegropak(fi, fo);
426}

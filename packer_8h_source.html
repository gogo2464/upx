300 // try to detect TE16 vs TE32 vs TE64 size mismatches; note that byte is explicitly allowed
301 template <class T>
302 static inline constexpr bool is_te16_type =
303 upx::is_same_any_v<T, byte, upx_uint16_t, BE16, LE16>;
304 template <class T>
305 static inline constexpr bool is_te32_type =
306 upx::is_same_any_v<T, byte, upx_uint32_t, BE32, LE32>;
307 template <class T>
308 static inline constexpr bool is_te64_type =
309 upx::is_same_any_v<T, byte, upx_uint64_t, BE64, LE64>;
310
311 template <class T>
312 using enable_if_te16 = std::enable_if_t<is_te16_type<T>, T>;
313 template <class T>
314 using enable_if_te32 = std::enable_if_t<is_te32_type<T>, T>;
315 template <class T>
316 using enable_if_te64 = std::enable_if_t<is_te64_type<T>, T>;
317
318 template <class T, class = enable_if_te16<T> >
319 inline unsigned get_te16(const T *p) const noexcept {
320 return bele->get16(p);
321 }
322 template <class T, class = enable_if_te32<T> >
323 inline unsigned get_te32(const T *p) const noexcept {
324 return bele->get32(p);
325 }
326 template <class T, class = enable_if_te64<T> >
327 inline unsigned get_te64_32(const T *p) const {
328 upx_uint64_t val = get_te64(p);
329 if (val >> 32)
330 throwCantPack("64-bit value too big %#llx", val);
331 return (unsigned) val;
332 }

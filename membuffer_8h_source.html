30// A MemBuffer allocates memory on the heap, and automatically
31// gets destructed when leaving scope or on exceptions.
32
33
38template <class T>
39class MemBufferBase {
40public:
41 typedef T element_type;
42 typedef typename std::add_lvalue_reference<T>::type reference;
43 typedef typename std::add_pointer<T>::type pointer;
44 typedef unsigned size_type; // limited by UPX_RSIZE_MAX
45 typedef pointer iterator;
46 typedef typename std::add_pointer<const T>::type const_iterator;
47protected:
48 static const size_t element_size = sizeof(element_type);
49
50protected:
51 pointer ptr;
52 size_type size_in_bytes;
53
54public:
55 explicit inline MemBufferBase() noexcept : ptr(nullptr), size_in_bytes(0) {}
56 forceinline ~MemBufferBase() noexcept {}
57
58 // IMPORTANT NOTE: automatic conversion to underlying pointer
59 // HINT: for fully bound-checked pointer use XSPAN_S from xspan.h
60 operator pointer() const noexcept { return ptr; }
61
62 // array access
63 reference operator[](ptrdiff_t i) const may_throw {
64 // NOTE: &array[SIZE] is *not* legal for containers like std::vector and MemBuffer !
65 if very_unlikely (i < 0 || mem_size(element_size, i) + element_size > size_in_bytes)
66 throwCantPack("MemBuffer invalid array index %td (%u bytes)", i, size_in_bytes);
67 return ptr[i];
68 }
69 // dereference
70 reference operator*() const DELETED_FUNCTION;
71 // arrow operator
72 pointer operator->() const DELETED_FUNCTION;
73
74 iterator begin() const may_throw {
75 if very_unlikely (ptr == nullptr)
76 throwCantPack("MemBuffer begin() unexpected NULL ptr");
77 return ptr;
78 }
79 const_iterator cbegin() const may_throw {
80 if very_unlikely (ptr == nullptr)
81 throwCantPack("MemBuffer cbegin() unexpected NULL ptr");
82 return ptr;
83 }
84 iterator end() const may_throw {
85 if very_unlikely (ptr == nullptr)
86 throwCantPack("MemBuffer end() unexpected NULL ptr");
87 return ptr + size_in_bytes / element_size;
88 }
89 const_iterator cend() const may_throw {
90 if very_unlikely (ptr == nullptr)
91 throwCantPack("MemBuffer cend() unexpected NULL ptr");
92 return ptr + size_in_bytes / element_size;
93 }
94
95 // membuffer + n -> pointer
96 template <class U>
97 typename std::enable_if<std::is_integral<U>::value, pointer>::type operator+(U n) const
98 may_throw {
99 size_t bytes = mem_size(element_size, n); // check mem_size
100 return raw_bytes(bytes) + n; // and check bytes
101 }
102private:
103 // membuffer - n -> pointer; not allowed - use raw_bytes() if needed
104 template <class U>
105 typename std::enable_if<std::is_integral<U>::value, pointer>::type operator-(U n) const
106 DELETED_FUNCTION;
107
108public: // raw access
109 pointer raw_ptr() const noexcept { return ptr; }
110 size_type raw_size_in_bytes() const noexcept { return size_in_bytes; }
111
112 pointer raw_bytes(size_t bytes) const may_throw {
113 if (bytes > 0) {
114 if very_unlikely (ptr == nullptr)
115 throwCantPack("MemBuffer raw_bytes unexpected NULL ptr");
116 if very_unlikely (bytes > size_in_bytes)
117 throwCantPack("MemBuffer raw_bytes invalid size");
118 }
119 return ptr;
120 }
121
122private:
123 // disable taking the address => force passing by reference
124 // [I'm not too sure about this design decision, but we can always allow it if needed]
125 UPX_CXX_DISABLE_ADDRESS(MemBufferBase)
126};

156 typedef typename MemBufferBase<T>::element_type element_type;
157 return mbb.raw_bytes(mem_size(sizeof(element_type), index, size_in_bytes)) + index;

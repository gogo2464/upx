45static int convert_errno_from_zlib(int zr) {
46 switch (zr) {
47 case Z_OK:
48 return UPX_E_OK;
49 // positive values
50 case Z_STREAM_END:
51 return UPX_E_ERROR;
52 case Z_NEED_DICT:
53 return UPX_E_ERROR;
54 // negative values
55 case Z_ERRNO:
56 return UPX_E_ERROR;
57 case Z_STREAM_ERROR:
58 return UPX_E_ERROR;
59 case Z_DATA_ERROR:
60 return UPX_E_ERROR;
61 case Z_MEM_ERROR:
62 return UPX_E_OUT_OF_MEMORY;
63 case Z_BUF_ERROR:
64 return UPX_E_OUTPUT_OVERRUN;
65 case Z_VERSION_ERROR:
66 return UPX_E_ERROR;
67 case -7: // UPX extra
68 return UPX_E_INPUT_OVERRUN;
69 default:
70 break;
71 }
72 return UPX_E_ERROR;
73}

79int upx_zlib_compress(const upx_bytep src, unsigned src_len, upx_bytep dst, unsigned *dst_len,
80 upx_callback_t *cb_parm, int method, int level,
81 const upx_compress_config_t *cconf_parm, upx_compress_result_t *cresult) {
82 assert(method == M_DEFLATE);
83 assert(level > 0);
84 assert(cresult != nullptr);
85 UNUSED(cb_parm);
86 int r = UPX_E_ERROR;
87 int zr;
88 const zlib_compress_config_t *const lcconf = cconf_parm ? &cconf_parm->conf_zlib : nullptr;
89 zlib_compress_result_t *const res = &cresult->result_zlib;
90 res->reset();
91
92 if (level == 10)
93 level = 9;
94
95 zlib_compress_config_t::mem_level_t mem_level;
96 zlib_compress_config_t::window_bits_t window_bits;
97 zlib_compress_config_t::strategy_t strategy;
98 // cconf overrides
99 if (lcconf) {
100 upx::oassign(mem_level, lcconf->mem_level);
101 upx::oassign(window_bits, lcconf->window_bits);
102 upx::oassign(strategy, lcconf->strategy);
103 }
104
105 z_stream s;
106 s.zalloc = (alloc_func) nullptr;
107 s.zfree = (free_func) nullptr;
108 s.next_in = src;
109 s.avail_in = src_len;
110 s.next_out = dst;
111 s.avail_out = *dst_len;
112 s.total_in = s.total_out = 0;
113
114 zr = (int) deflateInit2(&s, level, Z_DEFLATED, 0 - (int) window_bits, mem_level, strategy);
115 if (zr != Z_OK)
116 goto error;
117 assert(s.state->level == level);
118 zr = deflate(&s, Z_FINISH);
119 if (zr != Z_STREAM_END)
120 goto error;
121 zr = deflateEnd(&s);
122 if (zr != Z_OK)
123 goto error;
124 r = UPX_E_OK;
125 goto done;
126error:
127 (void) deflateEnd(&s);
128 r = convert_errno_from_zlib(zr);
129 if (r == UPX_E_OK)
130 r = UPX_E_ERROR;
131done:
132 if (r == UPX_E_OK) {
133 if (s.avail_in != 0 || s.total_in != src_len)
134 r = UPX_E_ERROR;
135 }
136 assert(s.total_in <= src_len);
137 assert(s.total_out <= *dst_len);
138 *dst_len = s.total_out;
139 return r;
140}

146int upx_zlib_decompress(const upx_bytep src, unsigned src_len, upx_bytep dst, unsigned *dst_len,
147 int method, const upx_compress_result_t *cresult) {
148 assert(method == M_DEFLATE);
149 UNUSED(method);
150 UNUSED(cresult);
151 int r = UPX_E_ERROR;
152 int zr;
153
154 z_stream s;
155 s.zalloc = (alloc_func) nullptr;
156 s.zfree = (free_func) nullptr;
157 s.next_in = src;
158 s.avail_in = src_len;
159 s.next_out = dst;
160 s.avail_out = *dst_len;
161 s.total_in = s.total_out = 0;
162
163 zr = inflateInit2(&s, -15);
164 if (zr != Z_OK)
165 goto error;
166 zr = inflate(&s, Z_FINISH);
167 if (zr != Z_STREAM_END) {
168 if (zr == Z_BUF_ERROR && s.avail_in == 0)
169 zr = -7; // UPX extra
170 goto error;
171 }
172 zr = inflateEnd(&s);
173 if (zr != Z_OK)
174 goto error;
175 r = UPX_E_OK;
176 goto done;
177error:
178 (void) inflateEnd(&s);
179 r = convert_errno_from_zlib(zr);
180 if (r == UPX_E_OK)
181 r = UPX_E_ERROR;
182done:
183 if (r == UPX_E_OK) {
184 if (s.avail_in != 0 || s.total_in != src_len)
185 r = UPX_E_INPUT_NOT_CONSUMED;
186 }
187 assert(s.total_in <= src_len);
188 assert(s.total_out <= *dst_len);
189 *dst_len = s.total_out;
190 return r;
191}

197int upx_zlib_test_overlap(const upx_bytep buf, const upx_bytep tbuf, unsigned src_off,
198 unsigned src_len, unsigned *dst_len, int method,
199 const upx_compress_result_t *cresult) {
200 assert(method == M_DEFLATE);
201
202 MemBuffer b(src_off + src_len);
203 memcpy(b + src_off, buf + src_off, src_len);
204 unsigned saved_dst_len = *dst_len;
205 int r = upx_zlib_decompress(raw_index_bytes(b, src_off, src_len), src_len,
206 raw_bytes(b, *dst_len), dst_len, method, cresult);
207 if (r != UPX_E_OK)
208 return r;
209 if (*dst_len != saved_dst_len)
210 return UPX_E_ERROR;
211 // NOTE: there is a very tiny possibility that decompression has
212 // succeeded but the data is not restored correctly because of
213 // in-place buffer overlapping, so we use an extra memcmp().
214 if (tbuf != nullptr && memcmp(tbuf, b, *dst_len) != 0)
215 return UPX_E_ERROR;
216 return UPX_E_OK;
217}

249static bool check_zlib(const int method, const int level, const unsigned expected_c_len) {
250 const unsigned u_len = 16384;
251 const unsigned c_extra = 4096;
252 MemBuffer u_buf, c_buf, d_buf;
253 unsigned c_len, d_len;
254 upx_compress_result_t cresult;
255 int r;
256
257 u_buf.alloc(u_len);
258 memset(u_buf, 0, u_len);
259 c_buf.allocForCompression(u_len, c_extra);
260 d_buf.allocForDecompression(u_len);
261
262 c_len = c_buf.getSize() - c_extra;
263 r = upx_zlib_compress(raw_bytes(u_buf, u_len), u_len, raw_index_bytes(c_buf, c_extra, c_len),
264 &c_len, nullptr, method, level, NULL_cconf, &cresult);
265 if (r != 0 || c_len != expected_c_len)
266 return false;
267
268 d_len = d_buf.getSize();
269 r = upx_zlib_decompress(raw_index_bytes(c_buf, c_extra, c_len), c_len, raw_bytes(d_buf, d_len),
270 &d_len, method, nullptr);
271 if (r != 0 || d_len != u_len || memcmp(u_buf, d_buf, u_len) != 0)
272 return false;
273
274 d_len = u_len - 1;
275 r = upx_zlib_decompress(raw_index_bytes(c_buf, c_extra, c_len), c_len, raw_bytes(d_buf, d_len),
276 &d_len, method, nullptr);
277 if (r == 0)
278 return false;
279
280 // TODO: rewrite Packer::findOverlapOverhead() so that we can test it here
281 // unsigned x_len = d_len;
282 // r = upx_zlib_test_overlap(c_buf, u_buf, c_extra, c_len, &x_len, method, nullptr);
283 return true;
284}

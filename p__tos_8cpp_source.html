64void PackTos::LinkerSymbols::LoopInfo::init(unsigned count_, bool allow_dbra) {
65 count = value = count_;
66 if (count == 0)
67 mode = LOOP_NONE;
68 else if (count <= 65536 && allow_dbra) {
69 mode = LOOP_DBRA;
70 value -= 1;
71 value &= 0xffff;
72 } else if (count <= 65536) {
73 mode = LOOP_SUBQ_W;
74 value &= 0xffff;
75 } else
76 mode = LOOP_SUBQ_L;
77}

79unsigned PackTos::getDecomprOffset(int method, int small) const {
80 UNUSED(small);
81 if (M_IS_NRV2B(method))
82 return 2; // FIXME: do not hardcode this value
83 else if (M_IS_NRV2D(method))
84 return 2; // FIXME: do not hardcode this value
85 else if (M_IS_NRV2E(method))
86 return 2; // FIXME: do not hardcode this value
87 else if (M_IS_LZMA(method))
88 return linker->getSectionSize("__mulsi3");
89 else
90 throwBadLoader();
91 return 0;
92}

94void PackTos::buildLoader(const Filter *ft) {
95 assert(ft->id == 0);
96
97 initLoader(stub_m68k_atari_tos, sizeof(stub_m68k_atari_tos));
98 // linker->dumpSymbols();
99
100 //
101 // part 1a
102 //
103
104 addLoader("entry");
105
106 if (symbols.up21_a6 <= 32767)
107 addLoader("set_up21_a6.w");
108 else if (symbols.up21_d4 <= 32767)
109 addLoader("set_up21_d4.w");
110 else
111 addLoader("set_up21_d4.l");
112
113 assert(symbols.loop1.count || symbols.loop2.count);
114 if (symbols.loop1.count) {
115 if (symbols.loop1.value <= 127)
116 addLoader("loop1_set_count.b");
117 else if (symbols.loop1.value <= 65535)
118 addLoader("loop1_set_count.w");
119 else
120 addLoader("loop1_set_count.l");
121 addLoader("loop1_label");
122 addLoader(opt->small ? "loop1.small" : "loop1.fast");
123 if (symbols.loop1.mode == symbols.LOOP_SUBQ_L)
124 addLoader("loop1_subql");
125 else if (symbols.loop1.mode == symbols.LOOP_SUBQ_W)
126 addLoader("loop1_subqw");
127 else if (symbols.loop1.mode == symbols.LOOP_DBRA)
128 addLoader("loop1_dbra");
129 else
130 throwBadLoader();
131 }
132 if (symbols.loop2.count) {
133 assert(symbols.loop2.mode == symbols.LOOP_DBRA);
134 addLoader(opt->small ? "loop2.small" : "loop2.fast");
135 }
136
137 addLoader("copy_to_stack");
138
139 if (M_IS_NRV2B(ph.method))
140 addLoader("nrv2b.init");
141 else if (M_IS_NRV2D(ph.method))
142 addLoader("nrv2d.init");
143 else if (M_IS_NRV2E(ph.method))
144 addLoader("nrv2e.init");
145 else if (M_IS_LZMA(ph.method))
146 addLoader("lzma.init");
147 else
148 throwBadLoader();
149
150 symbols.up31_d4 = symbols.up31_base_d4 + getDecomprOffset(ph.method, opt->small);
151 symbols.up31_a6 = symbols.up31_base_a6 + getDecomprOffset(ph.method, opt->small);
152 if (symbols.up31_a6 <= 32767)
153 addLoader("jmp_decompressor_a6.w");
154 else if (symbols.up31_d4 <= 32767)
155 addLoader("jmp_decompressor_d4.w");
156 else if (symbols.up31_a6 <= 65534)
157 addLoader("jmp_decompressor_a6.w2");
158 else
159 addLoader("jmp_decompressor_d4.l");
160
161 //
162 // part 1b
163 //
164
165 addLoader("code_on_stack");
166
167 addLoader("clear_dirty_bss");
168 addLoader("loop3_label");
169 addLoader(opt->small ? "loop3.small" : "loop3.fast");
170 if (symbols.loop3.mode == symbols.LOOP_SUBQ_L)
171 addLoader("loop3_subql");
172 else if (symbols.loop3.mode == symbols.LOOP_SUBQ_W)
173 addLoader("loop3_subqw");
174 else if (symbols.loop3.mode == symbols.LOOP_DBRA)
175 addLoader("loop3_dbra");
176 else
177 throwBadLoader();
178
179 addLoader("flush_cache");
180 addLoader("restore_stack");
181#if 0
182 addLoader("clear_dirty_stack");
183#endif
184 addLoader("start_program");
185
186 addLoader("IDENTSTR,+40D,UPX1HEAD,CUTPOINT");
187
188 //
189 // part 2
190 //
191
192 if (M_IS_NRV2B(ph.method)) {
193 addLoader(opt->small ? "nrv2b_8.small" : "nrv2b_8.fast");
194 } else if (M_IS_NRV2D(ph.method)) {
195 addLoader(opt->small ? "nrv2d_8.small" : "nrv2d_8.fast");
196 } else if (M_IS_NRV2E(ph.method)) {
197 addLoader(opt->small ? "nrv2e_8.small" : "nrv2e_8.fast");
198 } else if (M_IS_LZMA(ph.method)) {
199 addLoader("__mulsi3");
200 addLoader(opt->small ? "lzma.small" : "lzma.fast");
201 addLoader("lzma.finish");
202 } else
203 throwBadLoader();
204
205 if (symbols.need_reloc)
206 addLoader("reloc");
207
208 assert(symbols.loop3.count);
209 if (symbols.loop3.value <= 127)
210 addLoader("loop3_set_count.b");
211 else if (symbols.loop3.value <= 65535)
212 addLoader("loop3_set_count.w");
213 else
214 addLoader("loop3_set_count.l");
215
216 addLoader("jmp_stack");
217}

223/* flags for curproc->memflags */
224/* also used for program headers fh_flag */
225#define F_FASTLOAD 0x01 // don't zero heap
226#define F_ALTLOAD 0x02 // OK to load in alternate ram
227#define F_ALTALLOC 0x04 // OK to malloc from alt. ram
228#define F_SMALLTPA 0x08
229// used in MagiC: TPA can be allocated as specified in the program header
230// rather than the biggest free memory block
231#define F_MEMFLAGS 0xf0 // reserved for future use
232#define F_SHTEXT 0x800 // program's text may be shared
233
234#define F_MINALT 0xf0000000 // used to decide which type of RAM to load in
235
236#define F_ALLOCZERO 0x2000 // zero mem, for bugged (GEM...) programs
237
238/* Bit in Mxalloc's arg for "don't auto-free this memory" */
239#define F_KEEP 0x4000
240
241#define F_OS_SPECIAL 0x8000 // mark as a special process
242
243/* flags for curproc->memflags (that is, fh_flag) and also Mxalloc mode. */
244/* (Actually, when users call Mxalloc, they add 0x10 to what you see here) */
245#define F_PROTMODE 0xf0 // protection mode bits
246#define F_PROT_P 0x00 // no read or write
247#define F_PROT_G 0x10 // any access OK
248#define F_PROT_S 0x20 // any super access OK
249#define F_PROT_PR 0x30 // any read OK, no write
250#define F_PROT_I 0x40 // invalid page

258int PackTos::readFileHeader() {
259 fi->seek(0, SEEK_SET);
260 fi->readx(&ih, FH_SIZE);
261 if (ih.fh_magic != 0x601a)
262 return 0;
263 if (0ull + FH_SIZE + ih.fh_text + ih.fh_data + ih.fh_sym > file_size_u)
264 return 0;
265 return UPX_F_ATARI_TOS;
266}

268bool PackTos::checkFileHeader() {
269 const unsigned f = ih.fh_flag;
270 // printf("flags: 0x%x, text: %d, data: %d, bss: %d, sym: %d\n", f, (int) ih.fh_text,
271 // (int) ih.fh_data, (int) ih.fh_bss, (int) ih.fh_sym);
272 if ((ih.fh_text & 1) || (ih.fh_data & 1))
273 throwCantPack("odd size values in text/data");
274 if (f & F_OS_SPECIAL)
275 throwCantPack("I won't pack F_OS_SPECIAL programs");
276 if ((f & F_PROTMODE) > F_PROT_I)
277 throwCantPack("invalid protection mode");
278 if ((f & F_PROTMODE) != F_PROT_P) {
279 if (opt->force < 1)
280 throwCantPack("no private memory protection; use option '-f' to force packing");
281 }
282 if (f & F_SHTEXT) {
283 if (opt->force < 1)
284 throwCantPack("shared text segment; use option '-f' to force packing");
285 }
286#if 0
287 // fh_reserved seems to be unused
288 if (ih.fh_reserved != 0) {
289 if (opt->force < 1)
290 throwCantPack("reserved header field set; use option '-f' to force packing");
291 }
292#endif
293 return true;
294}

300// Check relocations for errors to make sure our loader can handle them
301static bool check_relocs(const byte *relocs, unsigned rsize, unsigned image_size,
302 unsigned *relocnum, unsigned *relocsize, unsigned *overlay) {
303 assert(rsize >= 4);
304 assert(image_size >= 4);
305 unsigned fixup = get_be32(relocs);
306 if (fixup == 0 || fixup >= image_size)
307 return false;
308 unsigned last_fixup = fixup;
309 unsigned i = 4;
310
311 *relocnum = 1;
312 for (;;) {
313 if (fixup & 1) // must be word-aligned
314 return false;
315 if (fixup + 4 > image_size) // out of bounds
316 return false;
317 if (i >= rsize) // premature EOF in relocs
318 return false;
319 unsigned c = relocs[i++];
320 if (c == 0) // EOF end marker
321 break;
322 else if (c == 1) // increase fixup, no reloc
323 fixup += 254;
324 else if (c & 1) // must be word-aligned
325 return false;
326 else // next reloc is here
327 {
328 fixup += c;
329 if (fixup - last_fixup < 4) // overlapping relocation
330 return false;
331 last_fixup = fixup;
332 *relocnum += 1;
333 }
334 }
335
336 *relocsize = i;
337 *overlay = rsize - i;
338 return true;
339}

345tribool PackTos::canPack() {
346 if (!readFileHeader())
347 return false;
348
349 byte buf[768];
350 fi->readx(buf, sizeof(buf));
351 checkAlreadyPacked(buf, sizeof(buf));
352
353 if (!checkFileHeader())
354 throwCantPack("unsupported header flags");
355 if (file_size < 1024)
356 throwCantPack("program is too small for atari/tos");
357
358 return true;
359}

372void PackTos::pack(OutputFile *fo) {
373 unsigned t;
374 unsigned relocnum = 0;
375 unsigned relocsize = 0;
376 unsigned overlay = 0;
377
378 const unsigned i_text = ih.fh_text;
379 const unsigned i_data = ih.fh_data;
380 const unsigned i_sym = ih.fh_sym;
381 const unsigned i_bss = ih.fh_bss;
382
383 symbols.reset();
384 symbols.need_reloc = false;
385 // prepare symbols for buildLoader() - worst case
386 symbols.loop1.init(65536 + 1);
387 symbols.loop2.init((160 - 1) / 4);
388 symbols.loop3.init(65536 + 1);
389 symbols.up21_d4 = 65536 + 1;
390 symbols.up21_a6 = 65536 + 1;
391 symbols.up31_base_d4 = 65536 + 1;
392 symbols.up31_base_a6 = 65536 + 1;
393
394 // read file
395 const unsigned isize = file_size_u - i_sym;
396 ibuf.alloc(isize);
397 fi->seek(FH_SIZE, SEEK_SET);
398 // read text + data
399 t = i_text + i_data;
400 fi->readx(ibuf, t);
401 // skip symbols
402 if (i_sym && opt->exact)
403 throwCantPackExact();
404 fi->seek(i_sym, SEEK_CUR);
405 // read relocations + overlay
406 overlay = file_size_u - (FH_SIZE + i_text + i_data + i_sym);
407 fi->readx(ibuf + t, overlay);
408
409#if TESTING
410 printf("text: %d, data: %d, sym: %d, bss: %d, flags=0x%x\n", i_text, i_data, i_sym, i_bss,
411 (int) ih.fh_flag);
412 printf("xx1 reloc: %d, overlay: %d, fixup: %d\n", relocsize, overlay,
413 overlay >= 4 ? (int) get_be32(ibuf + t) : -1);
414#endif
415
416 // Check relocs (see load_and_reloc() in freemint/sys/memory.c).
417 // Must work around TOS bugs and lots of broken programs.
418 if (overlay < 4) {
419 // Bug workaround: Whatever this is, silently keep it in
420 // the (unused) relocations for byte-identical unpacking.
421 relocsize = overlay;
422 overlay = 0;
423 } else if (get_be32(ibuf + t) == 0) {
424 // Bug workaround - check the empty fixup before testing fh_reloc.
425 relocsize = 4;
426 overlay -= 4;
427 } else if (ih.fh_reloc != 0)
428 relocsize = 0;
429 else {
430 if (!check_relocs(ibuf + t, overlay, t, &relocnum, &relocsize, &overlay))
431 throwCantPack("bad relocation table");
432 symbols.need_reloc = true;
433 }
434
435#if TESTING
436 printf("xx2: %d relocs: %d, overlay: %d, t: %d\n", relocnum, relocsize, overlay, t);
437#endif
438
439 checkOverlay(overlay);
440
441 // Append original fileheader.
442 t += relocsize;
443 ih.fh_sym = 0; // we stripped all symbols
444 memcpy(ibuf + t, &ih, FH_SIZE);
445 t += FH_SIZE;
446#if TESTING
447 printf("xx3 reloc: %d, overlay: %d, t: %d\n", relocsize, overlay, t);
448#endif
449 assert(t <= isize);
450
451 // Now the data in ibuf[0..t] looks like this:
452 // text + data + relocs + original file header
453 // After compression this will become the first part of the
454 // data segment. The second part will be the decompressor.
455
456 // alloc buffer (4096 is for decompressor and the various alignments)
457 obuf.allocForCompression(t, 4096);
458
459 // prepare packheader
460 ph.u_len = t;
461 // prepare filter
462 Filter ft(ph.level);
463 // compress (max_match = 65535)
464 upx_compress_config_t cconf;
465 cconf.reset();
466 cconf.conf_ucl.max_match = 65535;
467 cconf.conf_lzma.max_num_probs = 1846 + (768 << 4); // ushort: ~28 KiB stack
468 compressWithFilters(&ft, 512, &cconf);
469
470 //
471 // multipass buildLoader()
472 //
473
474 // save initial loader
475 const unsigned initial_lsize = getLoaderSize();
476 unsigned last_lsize = initial_lsize;
477 MemBuffer last_loader(last_lsize);
478 memcpy(last_loader, getLoader(), last_lsize);
479
480 unsigned o_text, o_data, o_bss;
481 unsigned e_len, d_len, d_off;
482 for (;;) {
483 // The decompressed data will now get placed at this offset:
484 unsigned offset = (ph.u_len + ph.overlap_overhead) - ph.c_len;
485
486 // get loader
487 const unsigned lsize = getLoaderSize();
488 e_len = getLoaderSectionStart("CUTPOINT");
489 d_len = lsize - e_len;
490 assert((e_len & 3) == 0 && (d_len & 1) == 0);
491
492 // compute section sizes
493 o_text = e_len;
494 o_data = ph.c_len;
495 o_bss = i_bss;
496
497 // word align len of compressed data
498 while (o_data & 1) {
499 obuf[o_data++] = 0;
500 offset++;
501 }
502
503 // append decompressor (part 2 of loader)
504 d_off = o_data;
506 o_data += d_len;
507
508 // dword align the len of the final data segment
509 while (o_data & 3) {
510 obuf[o_data++] = 0;
511 offset++;
512 }
513 // dword align offset
514 while (offset & 3)
515 offset++;
516
517 // new bss
518 if (i_text + i_data + i_bss > o_text + o_data + o_bss)
519 o_bss = (i_text + i_data + i_bss) - (o_text + o_data);
520
521 // dirty bss
522 unsigned dirty_bss = (o_data + offset) - (i_text + i_data);
523 // printf("real dirty_bss: %d\n", dirty_bss);
524 // dword align (or 16 - for speedup when clearing the dirty bss)
525 const unsigned dirty_bss_align = opt->small ? 4 : 16;
526 while (dirty_bss & (dirty_bss_align - 1))
527 dirty_bss++;
528 // adjust bss, assert room for some stack
529 unsigned stack = 512 + getDecompressorWrkmemSize();
530 if (dirty_bss + stack > o_bss)
531 o_bss = dirty_bss + stack;
532
533 // dword align the len of the final bss segment
534 while (o_bss & 3)
535 o_bss++;
536
537 // update symbols for buildLoader()
538 if (opt->small) {
539 symbols.loop1.init(o_data / 4);
540 symbols.loop2.init(0);
541 } else {
542 symbols.loop1.init(o_data / 160);
543 symbols.loop2.init((o_data % 160) / 4);
544 }
545 symbols.loop3.init(dirty_bss / dirty_bss_align);
546
547 symbols.up21_d4 = o_data + offset;
548 symbols.up31_base_d4 = d_off + offset;
549 symbols.up21_a6 = symbols.up21_d4 - (i_text + i_data);
550 symbols.up31_base_a6 = symbols.up31_base_d4 - (i_text + i_data);
551 assert((int) symbols.up21_a6 > 0);
552 assert((int) symbols.up31_base_a6 > 0);
553
554 const unsigned c = linker->getSymbolOffset("code_on_stack");
555 unsigned d;
556 d = linker->getSymbolOffset("flush_cache_rts") - c;
557 symbols.flush_cache_rts_offset = d;
558 d = linker->getSymbolOffset("clear_dirty_stack_loop") - c;
559 symbols.clear_dirty_stack_len = (d + 3) / 4 + 32 - 1;
560 d = linker->getSymbolOffset("code_on_stack_end") - c;
561 symbols.copy_to_stack_len = d / 2 - 1;
562
563 // now re-build loader
564 buildLoader(&ft);
565 unsigned new_lsize = getLoaderSize();
566 // printf("buildLoader %d %d\n", new_lsize, initial_lsize);
567 assert(new_lsize <= initial_lsize);
568 if (new_lsize == last_lsize && memcmp(getLoader(), last_loader, last_lsize) == 0)
569 break;
570 last_lsize = new_lsize;
571 memcpy(last_loader, getLoader(), last_lsize);
572 }
573
574 //
575 // define symbols and reloc
576 //
577
578 defineDecompressorSymbols();
579
580 linker->defineSymbol("loop1_count", symbols.loop1.value);
581 linker->defineSymbol("loop2_count", symbols.loop2.value);
582 linker->defineSymbol("loop3_count", symbols.loop3.value);
583
584 linker->defineSymbol("orig_p_tlen", i_text);
585 linker->defineSymbol("orig_p_dlen", i_data);
586 linker->defineSymbol("orig_p_blen", i_bss);
587
588 if (symbols.up21_a6 <= 32767)
589 linker->defineSymbol("up21_a6", symbols.up21_a6);
590 else
591 linker->defineSymbol("up21_d4", symbols.up21_d4);
592
593 if (symbols.up31_a6 <= 32767)
594 linker->defineSymbol("up31_a6", symbols.up31_a6);
595 else if (symbols.up31_d4 <= 32767)
596 linker->defineSymbol("up31_d4", symbols.up31_d4);
597 else if (symbols.up31_a6 <= 65534)
598 linker->defineSymbol("up31_a6", symbols.up31_a6 - 32767);
599 else
600 linker->defineSymbol("up31_d4", symbols.up31_d4);
601#if 0
602 printf("relocsize = %d\n", relocsize);
603 printf("upx21(d4) = %d\n", symbols.up21_d4);
604 printf("upx21(a6) = %d\n", symbols.up21_a6);
605 printf("upx31(d4) = %d\n", symbols.up31_d4);
606 printf("upx31(a6) = %d\n", symbols.up31_a6);
607#endif
608
609 linker->defineSymbol("flush_cache_rts_offset", symbols.flush_cache_rts_offset);
610 linker->defineSymbol("copy_to_stack_len", symbols.copy_to_stack_len);
611 linker->defineSymbol("clear_dirty_stack_len", symbols.clear_dirty_stack_len);
612
613 relocateLoader();
614
615 //
616 // write
617 //
618
619 // set new file_hdr
620 memcpy(&oh, &ih, FH_SIZE);
621 if (opt->atari_tos.split_segments) {
622 oh.fh_text = o_text;
623 oh.fh_data = o_data;
624 } else {
625 // put everything into the text segment
626 oh.fh_text = o_text + o_data;
627 oh.fh_data = 0;
628 }
629 oh.fh_bss = o_bss;
630 oh.fh_sym = 0;
631 oh.fh_reserved = 0;
632 // only keep the following flags:
633 oh.fh_flag = ih.fh_flag & (F_FASTLOAD | F_ALTALLOC | F_SMALLTPA | F_ALLOCZERO | F_KEEP);
634 // add an empty relocation fixup to workaround a bug in some TOS versions
635 oh.fh_reloc = 0;
636
637#if TESTING
638 printf("old text: %6d, data: %6d, bss: %6d, reloc: %d, overlay: %d\n", i_text, i_data, i_bss,
639 relocsize, overlay);
640 printf("new text: %6d, data: %6d, bss: %6d, flag=0x%x\n", o_text, o_data, o_bss,
641 (int) oh.fh_flag);
642 linker->dumpSymbols();
643#endif
644
645 // prepare loader
646 MemBuffer loader(o_text);
647 memcpy(loader, getLoader(), o_text);
648 patchPackHeader(loader, o_text);
649
650 // write new file header, loader and compressed file
651 fo->write(&oh, FH_SIZE);
652 fo->write(loader, o_text); // entry
653 if (opt->debug.dump_stub_loader)
654 OutputFile::dump(opt->debug.dump_stub_loader, loader, o_text);
655 memcpy(obuf + d_off, getLoader() + e_len, d_len); // copy decompressor
656 fo->write(obuf, o_data); // compressed + decompressor
657
658 // write empty relocation fixup
659 fo->write("\x00\x00\x00\x00", 4);
660
661 // verify
662 verifyOverlappingDecompression();
663
664 // copy the overlay

697 obuf.allocForDecompression(ph.u_len);
698

704

709 overlay -= 4; // this is our empty fixup

711
712 fo->write(obuf + ph.u_len - FH_SIZE, FH_SIZE); // orig. file_hdr
713 fo->write(obuf, ph.u_len - FH_SIZE); // orig. text+data+relocs
714
715 // copy any overlay
716 copyOverlay(fo, overlay, obuf);
717 }

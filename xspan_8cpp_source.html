73void xspan_check_range(const void *ptr, const void *base, ptrdiff_t size_in_bytes) {
74 xspan_stats.check_range_counter += 1;
75 if very_unlikely (ptr == nullptr)
76 xspan_fail_range_nullptr();
77 if very_unlikely (base == nullptr)
78 xspan_fail_range_nullbase();
79 upx_sptraddr_t off = ptraddr_diff(ptr, base);
80 if very_unlikely (off < 0 || off > size_in_bytes || size_in_bytes > UPX_RSIZE_MAX)
81 xspan_fail_range_range();
82 NO_fprintf(stderr, "xspan_check_range done\n");
83}

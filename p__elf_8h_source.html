48// The ELF file header. This appears at the start of every ELF file.
49template <class TElfITypes>
50packed_struct(Ehdr) {
51 typedef typename TElfITypes::Half Half;
52 typedef typename TElfITypes::Word Word;
53 typedef typename TElfITypes::Addr Addr;
54 typedef typename TElfITypes::Off Off;
55
56 unsigned char e_ident[16]; /* Magic number and other info */
57 Half e_type; /* Object file type */
58 Half e_machine; /* Architecture */
59 Word e_version; /* Object file version */
60 Addr e_entry; /* Entry point virtual address */
61 Off e_phoff; /* Program header table file offset */
62 Off e_shoff; /* Section header table file offset */
63 Word e_flags; /* Processor-specific flags */
64 Half e_ehsize; /* ELF header size in bytes */
65 Half e_phentsize; /* Program header table entry size */
66 Half e_phnum; /* Program header table entry count */
67 Half e_shentsize; /* Section header table entry size */
68 Half e_shnum; /* Section header table entry count */
69 Half e_shstrndx; /* Section header string table index */
70
71#define WANT_EHDR_ENUM 1
72#include "p_elf_enum.h"
73};

165template <class TElfITypes>
166packed_struct(Sym) {
167 typedef typename TElfITypes::Word Word;
168 typedef typename TElfITypes::Addr Addr;
169 typedef typename TElfITypes::Section Section;
170
171 Word st_name; /* symbol name (index into string table) */
172 Addr st_value; /* symbol value */
173 Word st_size; /* symbol size */
174 unsigned char st_info; /* symbol type and binding */
175 unsigned char st_other; /* symbol visibility */
176 Section st_shndx; /* section index */
177
178#define WANT_SYM_ENUM 1
179#include "p_elf_enum.h"
180
181 static constexpr unsigned get_st_bind(unsigned x) noexcept { return 0xf & (x >> 4); }
182 static constexpr unsigned get_st_type(unsigned x) noexcept { return 0xf & x; }
183 static constexpr unsigned char make_st_info(unsigned bind, unsigned type) noexcept {
184 return (unsigned char) (((bind << 4) + (0xf & type)) & 0xff);
185 }

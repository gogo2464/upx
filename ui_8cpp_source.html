128static const char *mkline(upx_uint64_t fu_len, upx_uint64_t fc_len, upx_uint64_t u_len,
129 upx_uint64_t c_len, const char *format_name, const char *filename,
130 bool decompress = false) {
131 static char buf[2048]; // static! // TODO later: check if affected by WITH_THREADS
132 char r[7 + 1];
133 char fn[15 + 1];
134 const char *f;
135
136 // Large ratios can happen because of overlays that are
137 // appended after a program is packed.
138 unsigned ratio = get_ratio(fu_len, fc_len);
139 if (ratio >= 1000 * 1000)
140 strcpy(r, "overlay");
141 else
142 upx_safe_snprintf(r, sizeof(r), "%3u.%02u%%", ratio / 10000, (ratio % 10000) / 100);
143 if (decompress)
144 f = "%10lld <-%10lld %7s %15s %s";
145 else
146 f = "%10lld ->%10lld %7s %15s %s";
147 center_string(fn, sizeof(fn), format_name);
148 assert(strlen(fn) == 15);
149 upx_safe_snprintf(buf, sizeof(buf), f, (long long) fu_len, (long long) fc_len, r, fn, filename);
150 UNUSED(u_len);
151 UNUSED(c_len);
152 return buf;
153}

161 upx_std_call_once(init_done, init_global_constants);

179 s->mode = M_MSG;
180 else if (s->screen == nullptr)
181 s->mode = M_CB_TERM;
182 else
183 s->mode = M_CB_SCREEN;

208void UiPacker::startCallback(unsigned u_len, unsigned step, int pass, int total_passes) {
209 s->u_len = u_len;
210 s->step = step;
211 s->next_update = step;
212
213 s->pass = pass;
214 s->total_passes = total_passes;
215 // printf("startCallback %d %d\n", s->pass, s->total_passes);
216
217 s->bar_len = 64;
218 s->pos = -2;
219 s->spin_counter = 0;
220 s->bar_pos = 1; // because of the leading '\r'
221 s->pass_digits = 0;
222
223 cb.reset();
224
225 if (s->pass < 0) // no callback wanted
226 return;
227
228 if (s->mode <= M_INFO)
229 return;
230 if (s->mode == M_MSG) {
231 if (pass <= 1) {
232 printInfo(0);
233 fflush(stdout);
234 printSetNl(2);
235 }
236 return;
237 }
238
239 cb.nprogress = progress_callback;
240 cb.user = this; // parameter for static function UiPacker::progress_callback()
241
242 if (s->mode == M_CB_TERM) {
243 const char *fname = fn_basename(pb->fi->getName());
244 int l = (int) strlen(fname);
245 if (l > 0 && l <= 30) {
246 strcpy(&s->msg_buf[s->bar_pos], fname);
247 s->bar_pos += l;
248 s->msg_buf[s->bar_pos++] = ' ';
249 s->msg_buf[s->bar_pos++] = ' ';
250 s->bar_len -= l + 2;
251 }
252 }
253
254 // set pass
255 if (total_passes > 1) {
256 int buflen, l;
257 do {
258 s->pass_digits++;
259 total_passes /= 10;
260 } while (total_passes > 0);
261 buflen = sizeof(s->msg_buf) - s->bar_pos;
262 l = upx_safe_snprintf(&s->msg_buf[s->bar_pos], buflen, "%*d/%*d ", s->pass_digits, s->pass,
263 s->pass_digits, s->total_passes);
264 if (l > 0 && s->bar_len - l > 10) {
265 s->bar_len -= l;
266 s->bar_pos += l;
267 }
268 }
269
270#if (UI_USE_SCREEN)
271 if (s->mode == M_CB_SCREEN) {
272 if (!s->screen_init_done) {
273 s->screen_init_done = 1;
274 if (s->screen->hideCursor)
275 s->cursor_shape = s->screen->hideCursor(s->screen);
276 s->s_fg = s->screen->getFg(s->screen);
277 s->s_bg = s->screen->getBg(s->screen);
278 s->screen->getCursor(s->screen, &s->s_cx, &s->s_cy);
279 s->scroll_up = s->screen->getScrollCounter(s->screen);
280 printInfo(1);
281 s->screen->getCursor(s->screen, &s->b_cx, &s->b_cy);
282 s->scroll_up = s->screen->getScrollCounter(s->screen) - s->scroll_up;
283 }
284 }
285#endif /* UI_USE_SCREEN */
286}

309void UiPacker::endCallback(bool done) {
310 if (s->pass < 0) // no callback wanted
311 return;
312
313 if (s->mode == M_CB_TERM) {
314 if (done)
315 printClearLine(stdout);
316 else
317 printSetNl(2);
318 }
319
320// restore screen
321#if (UI_USE_SCREEN)
322 if (s->mode == M_CB_SCREEN) {
323 if (done) {
324 int cx, cy, sy;
325 assert(s->screen_init_done);
326 s->screen_init_done = 0;
327 assert(s->s_cx == 0 && s->b_cx == 0);
328 s->screen->getCursor(s->screen, &cx, &cy);
329 sy = UPX_MAX(0, s->s_cy - s->scroll_up);
330 while (cy >= sy)
331 s->screen->clearLine(s->screen, cy--);
332 s->screen->setCursor(s->screen, s->s_cx, sy);
333 s->screen->setFg(s->screen, s->s_fg);
334 s->screen->setBg(s->screen, s->s_bg);
335 if (s->cursor_shape > 0)
336 s->screen->setCursorShape(s->screen, s->cursor_shape);
337 } else {
338 // not needed:
339 // s->screen->clearLine(s->screen, s->b_cy);
340 // s->screen->setCursor(s->screen, s->b_cx, s->b_cy);
341 }
342 }
343#endif /* UI_USE_SCREEN */
344
345 cb.reset();
346#if 0
347 printf("callback: pass %d, step %6d, updates %6d\n",
348 s->pass, s->step, s->spin_counter);
349#endif
350}

363void UiPacker::doCallback(unsigned isize, unsigned osize) {
364 int i;
365 static const char spinner[] = "|/-\\";
366
367 if (s->pass < 0) // no callback wanted
368 return;
369
370 if (s->u_len == 0 || isize > s->u_len)
371 return;
372 // check if we should update the display
373 if (s->step > 0 && isize > 0 && isize < s->u_len) {
374 if (isize < s->next_update)
375 return;
376 s->next_update += s->step;
377 }
378
379 // compute progress position
380 int pos = -1;
381 if (isize >= s->u_len)
382 pos = s->bar_len;
383 else if (isize > 0) {
384 pos = get_ratio(s->u_len, isize) * s->bar_len / 1000000;
385 assert(pos >= 0);
386 assert(pos <= s->bar_len);
387 }
388
389#if 0
390 printf("%6d %6d %6d %6d %3d %3d\n", isize, osize, s->step, s->next_update, pos, s->pos);
391 return;
392#endif
393
394 if (pos < s->pos)
395 return;
396 if (pos < 0 && pos == s->pos)
397 return;
398
399 // fill the progress bar
400 char *m = &s->msg_buf[s->bar_pos];
401 *m++ = progress_filler[2];
402 for (i = 0; i < s->bar_len; i++)
403 *m++ = progress_filler[i <= pos ? 1 : 0];
404 *m++ = progress_filler[3];
405
406 // compute current compression ratio
407 unsigned ratio = 1000000;
408 if (osize > 0)
409 ratio = get_ratio(isize, osize);
410
411 int buflen = (int) (&s->msg_buf[sizeof(s->msg_buf)] - m);
412 upx_safe_snprintf(m, buflen, " %3d.%1d%% %c ", ratio / 10000, (ratio % 10000) / 1000,
413 spinner[s->spin_counter & 3]);
414 assert(strlen(s->msg_buf) < 1 + 80);
415
416 s->pos = pos;
417 s->spin_counter++;
418
419 if (s->mode == M_CB_TERM) {
420 const char *msg = &s->msg_buf[0];
421 int fg = con_fg(stdout, FG_CYAN);
422 con_fprintf(stdout, "%s", msg); // avoid backslash interpretation
423 (void) con_fg(stdout, fg);
424 fflush(stdout);
425 printSetNl(1);
426 UNUSED(fg);
427 return;
428 }
429
430#if (UI_USE_SCREEN)
431 if (s->mode == M_CB_SCREEN) {
432 const char *msg = &s->msg_buf[1];
433#if 0
434 s->screen->putString(s->screen,msg,s->b_cx,s->b_cy);
435#else
436 // FIXME: this doesn't honor '--mono' etc.
437 int attr = FG_CYAN | s->s_bg;
438 s->screen->putStringAttr(s->screen, msg, attr, s->b_cx, s->b_cy);
439#endif
440 s->screen->refresh(s->screen);
441 }
442#endif /* UI_USE_SCREEN */
443}

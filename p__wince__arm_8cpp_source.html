51const int *PackWinCeArm::getCompressionMethods(int method, int level) const {
52 static const int m_all[] = {M_NRV2B_8, M_NRV2E_8, M_LZMA, M_END};
53 static const int m_lzma[] = {M_LZMA, M_END};
54 static const int m_nrv2b[] = {M_NRV2B_8, M_END};
55 static const int m_nrv2e[] = {M_NRV2E_8, M_END};
56
57 if (!use_thumb_stub)
58 return getDefaultCompressionMethods_8(method, level);
59
60 if (method == M_ALL)
61 return m_all;
62 if (M_IS_LZMA(method))
63 return m_lzma;
64 if (M_IS_NRV2B(method))
65 return m_nrv2b;
66 if (M_IS_NRV2E(method))
67 return m_nrv2e;
68 return m_nrv2e;
69}

80void PackWinCeArm::processImports2(unsigned myimport, unsigned iat_off) // pass 2

114tribool PackWinCeArm::canPack() {
115 if (!readFileHeader())
116 return false;
117 checkMachine(ih.cpu);
118 if (ih.cpu != IMAGE_FILE_MACHINE_ARM && ih.cpu != IMAGE_FILE_MACHINE_THUMB)
119 return false;
120 use_thumb_stub |= ih.cpu == IMAGE_FILE_MACHINE_THUMB || (ih.entry & 1) == 1;
121 // HACK FIXME later: don't misuse opt->cpu_x86, need an extra option to force thumb stub
122 use_thumb_stub |= (opt->cpu_x86 == opt->CPU_8086);
123 return true;
124}

126void PackWinCeArm::buildLoader(const Filter *ft) {
127 const byte *loader = use_thumb_stub ? stub_arm_v4t_wince_pe : stub_arm_v4a_wince_pe;
128 unsigned size = use_thumb_stub ? sizeof(stub_arm_v4t_wince_pe) : sizeof(stub_arm_v4a_wince_pe);
129
130 // prepare loader
131 initLoader(loader, size);
132
133 if (isdll)
134 addLoader("DllStart");
135 addLoader("ExeStart");
136
137 if (ph.method == M_NRV2E_8)
138 addLoader("Call2E");
139 else if (ph.method == M_NRV2B_8)
140 addLoader("Call2B");
141 else if (ph.method == M_NRV2D_8)
142 addLoader("Call2D");
143 else if (M_IS_LZMA(ph.method))
144 addLoader("+40C,CallLZMA");
145
146 if (ft->id == 0x50)
147 addLoader("+40C,Unfilter_0x50");
148
149 if (sorelocs)
150 addLoader("+40C,Relocs");
151
152 addLoader("+40C,Imports");
153 addLoader("ProcessEnd");
154
155 if (ph.method == M_NRV2E_8)
156 addLoader(".ucl_nrv2e_decompress_8");
157 else if (ph.method == M_NRV2B_8)
158 addLoader(".ucl_nrv2b_decompress_8");
159 else if (ph.method == M_NRV2D_8)
160 addLoader(".ucl_nrv2d_decompress_8");
161 else if (M_IS_LZMA(ph.method))
162 addLoader("+40C,LZMA_DECODE,LZMA_DEC10");
163
164 addLoader("IDENTSTR,UPX1HEAD");
165}

167bool PackWinCeArm::needForceOption() const {
168 // return true if we need `--force` to pack this file
169 bool r = false;
170 r |= (ih.opthdrsize != 0xe0);
171 r |= ((ih.flags & IMAGE_FILE_EXECUTABLE_IMAGE) == 0);
172 r |= (ih.entry == 0 /*&& !isdll*/);
173 r |= (ih.ddirsentries != 16);
176 return r;
177}

179void PackWinCeArm::callCompressWithFilters(Filter &ft, int filter_strategy, unsigned ih_codebase) {
180 // limit stack size needed for runtime decompression

182 cconf.reset();
183 cconf.conf_lzma.max_num_probs = 1846 + (768 << 4); // ushort: ~28 KiB stack
184 compressWithFilters(&ft, 2048, &cconf, filter_strategy, ih_codebase, rvamin, 0, nullptr, 0);

187void PackWinCeArm::addNewRelocations(Reloc &rel, unsigned upxsection) {
188 static const char *const symbols_to_relocate[] = {"ONAM", "BIMP", "BREL", "FIBE",
189 "FIBS", "ENTR", "DST0", "SRC0"};
190 for (unsigned s2r = 0; s2r < TABLESIZE(symbols_to_relocate); s2r++) {
191 unsigned off = linker->getSymbolOffset(symbols_to_relocate[s2r]);
192 if (off != 0xdeaddead)
193 rel.add_reloc(off + upxsection, IMAGE_REL_BASED_HIGHLOW);
194 }

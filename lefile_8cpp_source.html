146void LeFile::readImage() {
147 soimage = mem_size(mps, pages); // assert size
148 if (!soimage) // late detection, but protect against .alloc(0)
149 throwCantPack("no soimage");
150 mb_iimage.alloc(soimage);
151 mb_iimage.clear();
152 iimage = mb_iimage; // => now a SPAN_S
153
154 unsigned ic, jc;
155 for (ic = jc = 0; ic < pages; ic++) {
156 if ((ipm_entries[ic].type & 0xC0) == 0) {
157 fif->seek(ih.data_pages_offset + exe_offset +
158 (ipm_entries[ic].m * 0x100 + ipm_entries[ic].l - 1) * mps,
159 SEEK_SET);
160 unsigned bytes = ic != pages - 1 ? mps : ih.bytes_on_last_page;
161 fif->readx(iimage + jc, bytes);
162 }
163 jc += mps;
164 }
165}

186bool LeFile::readFileHeader() {
187#define H(x) get_le16(header + (2 * (x)))
188 byte header[0x40];
189 le_offset = exe_offset = 0;
190 int ic;
191
192 for (ic = 0; ic < 20; ic++) {
193 fif->seek(le_offset, SEEK_SET);
194 fif->readx(header, sizeof(header));
195
196 if (memcmp(header, "MZ", 2) == 0) // normal dos exe
197 {
198 exe_offset = le_offset;
199 if (H(0x18 / 2) >= 0x40 && memcmp(header + 0x19, "TIPPACH", 7)) // new format exe
200 le_offset += H(0x3c / 2) + H(0x3e / 2) * 65536;
201 else {
202 le_offset += H(2) * 512 + H(1);
203 if (H(1))
204 le_offset -= 512;
205 else if (H(2) == 0)
206 return false;
207 }
208 } else if (memcmp(header, "BW", 2) == 0) // used in dos4gw.exe
209 le_offset += H(2) * 512 + H(1);
210 else if (memcmp(header, "LE", 2) == 0)
211 break; // success
212 else if (memcmp(header, "PMW1", 4) == 0)
213 throwCantPack("already packed with PMWLITE");
214 else
215 return false;
216 }
217 if (ic == 20)
218 return false;
219 fif->seek(le_offset, SEEK_SET);
220 fif->readx(&ih, sizeof(ih));
221 if (mps < 512 || mps > 2097152 || (mps & (mps - 1)) != 0)
222 throwCantPack("LE file header invalid page size %u", (unsigned) mps);
223 if (ih.bytes_on_last_page > mps || pages == 0)
224 throwCantPack("bad LE file header");
225 if (!mem_size_valid(mps, pages) || exe_offset > le_offset)
226 throwCantPack("bad LE file header");
227 return true;
228#undef H
229}

231void LeFile::writeFile(OutputFile *f, bool le) {
232 fof = f;
233 memcpy(&oh, &ih,
234 (charptr) &oh.memory_pages - (charptr) &oh); // copy some members of the orig. header
235 oh.memory_page_size = mps;
236 oh.object_table_offset = sizeof(oh);
237 oh.object_table_entries = soobject_table;
238 oh.object_pagemap_offset = oh.object_table_offset + soobject_table * sizeof(*iobject_table);
239 oh.resident_names_offset = oh.object_pagemap_offset + sopm_entries * sizeof(*ipm_entries);
240 oh.entry_table_offset = oh.resident_names_offset + sores_names;
241 oh.fixup_page_table_offset = oh.entry_table_offset + soentries;
242 oh.fixup_record_table_offset = oh.fixup_page_table_offset + sofpage_table * 4;
243 oh.imported_modules_name_table_offset = oh.fixup_record_table_offset + sofixups - FIXUP_EXTRA;
244 oh.imported_procedures_name_table_offset = oh.imported_modules_name_table_offset;
245 oh.data_pages_offset =
246 oh.fixup_record_table_offset + sofixups + (le ? 0 : le_offset - exe_offset);
247 if (ih.non_resident_name_table_length) {
248 oh.non_resident_name_table_offset = oh.data_pages_offset + soimage;
249 oh.non_resident_name_table_length = sononres_names;
250 }
251 oh.fixup_size = sofixups + 4 * sofpage_table;
252 oh.loader_size = oh.fixup_size + oh.fixup_page_table_offset - sizeof(oh);
253
254 fof->write(&oh, sizeof(oh));
255 writeObjectTable();
256 writePageMap();
257 writeResidentNames();
258 writeEntryTable();
259 writeFixupPageTable();
260 writeFixups();
261 writeImage();
262 writeNonResidentNames();
263}

265void LeFile::countFixups(unsigned *counts) const {
266 const unsigned o = objects;
267 memset(counts, 0, mem_size(sizeof(unsigned), o + 2));
268 // counts[0..objects-1] - # of 32-bit offset relocations in for that objects
269 // counts[objects] - # of selector fixups
270 // counts[objects+1] - # of self-relative fixups
271
272 const byte *fix = ifixups;
273 const unsigned sfixups = get_le32(ifpage_table + pages);
274 unsigned ll;
275
276 while (ptr_udiff_bytes(fix, ifixups) < sfixups) {
277 if ((fix[1] & ~0x10) != 0)
278 throwCantPack("unsupported fixup record");
279 switch (*fix) {
280 case 2: // selector fixup
281 counts[o] += 9;
282 fix += 5;
283 break;
284 case 0x12: // alias selector
285 throwCantPack("16-bit selector alias fixup not yet supported");
286 case 5: // 16-bit offset
287 fix += (fix[1] & 0x10) ? 9 : 7;
288 break;
289 case 6: // 16:32 pointer
290 counts[o] += 9;
291 // fall through
292 case 7: // 32-bit offset
293 counts[fix[4] - 1] += 4;
294 fix += (fix[1] & 0x10) ? 9 : 7;
295 break;
296 case 0x27: // 32-bit offset list
297 ll = fix[2];
298 counts[fix[3] - 1] += ll * 4;
299 fix += (fix[1] & 0x10) ? 6 : 4;
300 fix += ll * 2;
301 break;
302 case 8: // 32-bit self relative fixup
303 counts[o + 1] += 4;
304 fix += (fix[1] & 0x10) ? 9 : 7;
305 break;
306 default:
307 throwCantPack("unsupported fixup record");
308 }
309 }
310 counts[o]++; // extra space for 'ret'
311 counts[o + 1] += 4; // extra space for 0xFFFFFFFF
312}

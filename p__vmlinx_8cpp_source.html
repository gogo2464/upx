79template <class T>
80int PackVmlinuxBase<T>::getStrategy(Filter &/*ft*/)
81{
82 // Called just before reading and compressing each block.
83 // Might want to adjust blocksize, etc.
84
85 // If user specified the filter, then use it (-2==strategy).
86 // Else try the first two filters, and pick the better (2==strategy).
87 return (opt->no_filter ? -3 : ((opt->filter > 0) ? -2 : 2));
88}

90template <class T>
91int __acc_cdecl_qsort
92PackVmlinuxBase<T>::compare_Phdr(void const *aa, void const *bb)
93{
94 Phdr const *const a = (Phdr const *)aa;
95 Phdr const *const b = (Phdr const *)bb;
96 unsigned const xa = a->p_type - Phdr::PT_LOAD;
97 unsigned const xb = b->p_type - Phdr::PT_LOAD;
98 if (xa < xb) return -1; // PT_LOAD first
99 if (xa > xb) return 1;
100 if (a->p_paddr < b->p_paddr) return -1; // ascending by .p_paddr
101 if (a->p_paddr > b->p_paddr) return 1;
102 // What could remain?
103 // try to make sort order deterministic and just compare more fields
104#define CMP(field) \
105 if (a->field != b->field) return a->field < b->field ? -1 : 1
106 CMP(p_offset);
107 CMP(p_vaddr);
108 CMP(p_filesz);
109 CMP(p_memsz);
110 CMP(p_flags);
111 CMP(p_align);
112#undef CMP
113 return 0;
114}

116template <class T>
117typename T::Shdr const *PackVmlinuxBase<T>::getElfSections()
118{
119 unsigned const e_shnum = ehdri.e_shnum;
120 if (ehdri.e_shentsize != sizeof(*shdri)
121 || file_size_u < ehdri.e_shoff
122 || file_size_u < ehdri.e_shoff + mem_size(ehdri.e_shentsize, e_shnum)) {
123 infoWarning("bad ElfXX_Shdrs");
124 return nullptr;
125 }
126 shdri = new Shdr[(unsigned) e_shnum];
127 fi->seek(ehdri.e_shoff, SEEK_SET);
128 fi->readx(shdri, e_shnum * sizeof(*shdri));
129 unsigned const e_shstrndx = ehdri.e_shstrndx;
130 if (e_shnum <= e_shstrndx) {
131 infoWarning("bad .e_shstrndx %#x", e_shstrndx);
132 return nullptr;
133 }
134 Shdr const *p = &shdri[e_shstrndx];
135 if (Shdr::SHT_STRTAB==p->sh_type
136 && p->sh_offset <= (file_size_u - sizeof(*shdri))
137 && p->sh_size <= (file_size_u - p->sh_offset)
138 && p->sh_name <= (file_size_u - p->sh_offset)
139 && 10 <= (file_size_u - p->sh_name)
140 // 10 == (1+ strlen(".shstrtab"))
141 ) {
142 if (p->sh_size <= p->sh_name) {
143 infoWarning("bad .shstrtab _Shdr[%u]", (unsigned)ehdri.e_shstrndx);
144 return nullptr;
145 }
146 shstrtab = new char[1+ p->sh_size];
147 fi->seek(p->sh_offset, SEEK_SET);
148 fi->readx(shstrtab, p->sh_size);
149 shstrtab[p->sh_size] = '\0';
150 if (0==strcmp(".shstrtab", shstrtab + p->sh_name)) {
151 return p;
152 }
153 }
154 return nullptr;
155}

157template <class T>
158tribool PackVmlinuxBase<T>::canPack()
159{
160 fi->seek(0, SEEK_SET);
161 fi->readx(&ehdri, sizeof(ehdri));
162
163 // now check the ELF header
164 if (memcmp(&ehdri, "\x7f\x45\x4c\x46", 4)
165 || ehdri.e_ident[Ehdr::EI_CLASS] != my_elfclass
166 || ehdri.e_ident[Ehdr::EI_DATA] != my_elfdata
167 || ehdri.e_ident[Ehdr::EI_VERSION] != Ehdr::EV_CURRENT
168 || !memcmp(&ehdri.e_ident[8], "FreeBSD", 7) // branded
169 || ehdri.e_machine != my_e_machine
170 || ehdri.e_version != 1 // version
171 || ehdri.e_ehsize != sizeof(ehdri) // different <elf.h> ?
172 ) {
173 return false;
174 }
175
176 // additional requirements for vmlinux
177 if (ehdri.e_type != Ehdr::ET_EXEC
178 || ehdri.e_phoff != sizeof(ehdri) // Phdr not contiguous with Ehdr
179 || ehdri.e_phentsize!=sizeof(Phdr)
180 || !is_valid_e_entry(ehdri.e_entry)
181 ) {
182 return false;
183 }
184
185 // A Linux kernel must have a __ksymtab section. [??]
186 Shdr const *p, *const shstrsec = getElfSections();
187 if (nullptr==shstrsec) {
188 return false;
189 }
190 {
191 int j;
192 for (p = shdri, j= ehdri.e_shnum; --j>=0; ++p) {
193 if (Shdr::SHT_PROGBITS==p->sh_type
194 && p->sh_name < shstrsec->sh_size
195 && 0==strcmp("__ksymtab", p->sh_name + shstrtab)) {
196 break;
197 }
198 }
199 if (j < 0) {
200 return false;
201 }
202 }
203
204 phdri = new Phdr[(unsigned) ehdri.e_phnum];
205 fi->seek(ehdri.e_phoff, SEEK_SET);
206 fi->readx(phdri, ehdri.e_phnum * sizeof(*phdri));
207
208 // Put PT_LOAD together at the beginning, ascending by .p_paddr.
209 upx_qsort(phdri, ehdri.e_phnum, sizeof(*phdri), compare_Phdr);
210
211 // Find convex hull of physical addresses, and count the PT_LOAD.
212 // Ignore ".bss": .p_filesz < .p_memsz
213 unsigned phys_lo= ~0u, phys_hi= 0u;
214 for (unsigned j = 0; j < ehdri.e_phnum; ++j) {
215 if (Phdr::PT_LOAD==phdri[j].p_type) {
216 // Check for general sanity (not necessarily required.)
217 if (0xfff & (phdri[j].p_offset | phdri[j].p_paddr
218 | phdri[j].p_align | phdri[j].p_vaddr) ) {
219 return false;
220 }
221 if (phys_lo > phdri[j].p_paddr) {
222 phys_lo = phdri[j].p_paddr;
223 }
224 if (phys_hi < (phdri[j].p_filesz + phdri[j].p_paddr)) {
225 phys_hi = (phdri[j].p_filesz + phdri[j].p_paddr);
226 }
227 ++n_ptload;
228 }
229 }
230 paddr_min = phys_lo;
231 sz_ptload = phys_hi - phys_lo;
232 return 0 < n_ptload;
233}

237template <class T>
238void PackVmlinuxBase<T>::pack(OutputFile *fo)
239{
240 unsigned fo_off = 0;
241 Ehdr ehdro;
242 TE32 tmp_u32;
243
244 // nullptr
245 // .text(PT_LOADs) .note(1st page) .note(rest)
246 // .shstrtab .symtab .strtab
247 Shdr shdro[1+3+3];
248 memset(shdro, 0, sizeof(shdro));
249
250 ibuf.alloc(file_size);
251 obuf.allocForCompression(file_size);
252
253 // .e_ident, .e_machine, .e_version, .e_flags
254 memcpy(&ehdro, &ehdri, sizeof(ehdro));
255 ehdro.e_type = Ehdr::ET_REL;
256 ehdro.e_entry = 0;
257 ehdro.e_phoff = 0;
258 ehdro.e_shoff = sizeof(ehdro);
259 ehdro.e_phentsize = 0;
260 ehdro.e_phnum = 0;
261 ehdro.e_shnum = 1+3+3;
262 ehdro.e_shstrndx = 4;
263 fo->write(&ehdro, sizeof(ehdro)); fo_off+= sizeof(ehdro);
264 fo->write(shdro, sizeof(shdro)); fo_off+= sizeof(shdro);
265
266// Notice overlap [containment] of physical PT_LOAD[2] into PTLOAD[1]
267// in this vmlinux for x86_64 from Fedora Core 6 on 2007-01-07:
268//Program Headers:
269// Type Offset VirtAddr PhysAddr
270// FileSiz MemSiz Flags Align
271// LOAD 0x0000000000200000 0xffffffff80200000 0x0000000000200000
272// 0x000000000034bce8 0x000000000034bce8 R E 200000
273// LOAD 0x000000000054c000 0xffffffff8054c000 0x000000000054c000
274// 0x00000000000ed004 0x00000000001702a4 RWE 200000
275// LOAD 0x0000000000800000 0xffffffffff600000 0x00000000005f5000
276// 0x0000000000000c08 0x0000000000000c08 RWE 200000
277// NOTE 0x0000000000000000 0x0000000000000000 0x0000000000000000
278// 0x0000000000000000 0x0000000000000000 R 8
279// Therefore we must "compose" the convex hull to be loaded.
280
281 ph.u_len = sz_ptload;
282 memset(ibuf, 0, sz_ptload);
283 for (unsigned j = 0; j < ehdri.e_phnum; ++j) {
284 if (Phdr::PT_LOAD==phdri[j].p_type) {
285 fi->seek(phdri[j].p_offset, SEEK_SET);
286 fi->readx(ibuf + ((unsigned) phdri[j].p_paddr - paddr_min), phdri[j].p_filesz);
287 }
288 }
289 checkAlreadyPacked(ibuf + (ph.u_len - 1024), 1024);
290
291 // prepare filter
292 ph.filter = 0;
293 Filter ft(ph.level);
294 ft.buf_len = ph.u_len;
295 ft.addvalue = 0; // we are independent of actual runtime address; see ckt32
296
297 upx_compress_config_t cconf; cconf.reset();
298 // limit stack size needed for runtime decompression
299 cconf.conf_lzma.max_num_probs = 1846 + (768 << 4); // ushort: ~28 KiB stack
300
301 unsigned ppc32_extra = 0;
302 if (Ehdr::EM_PPC==my_e_machine) {
303 // output layout:
304 // .long UPX_MAGIC_LE32
305 // .long L20 - L10
306 // L10:
307 // b_info for Ehdr; compressed Ehdr; .balign 4 // one block only
308 // b_info for LOAD; compressed LOAD; .balign 4 // possibly many blocks
309 // // This allows per-block filters!
310 // L20:
311 // b f_decompress
312 // +4: f_unfilter(char *buf, unsigned len, unsigned cto8, unsigned ftid)
313 // // Code for multiple filters can "daisy chain" on ftid.
314 // f_decompress(char const *src, unsigned src_len,
315 // char *dst, unsigned *dst_len, int method)
316 unsigned tmp;
317 tmp = UPX_MAGIC_LE32; fo->write(&tmp, sizeof(tmp)); fo_off += sizeof(tmp);
318 tmp = 0; fo->write(&tmp, sizeof(tmp)); fo_off += sizeof(tmp);
319 ppc32_extra += 2*sizeof(tmp);
320 unsigned const len_unc = sizeof(ehdri) + sizeof(Phdr) * ehdri.e_phnum;
321 MemBuffer unc_hdr(len_unc);
322 MemBuffer cpr_hdr; cpr_hdr.allocForCompression(len_unc);
323 memcpy(&unc_hdr[0], &ehdri, sizeof(ehdri));
324 memcpy(&unc_hdr[sizeof(ehdri)], phdri, sizeof(Phdr) * ehdri.e_phnum);
325 unsigned len_cpr = 0;
326 int const r = upx_compress(unc_hdr, len_unc, cpr_hdr, &len_cpr,
327 nullptr, ph.method, 10, nullptr, nullptr );
328 if (UPX_E_OK!=r || len_unc<=len_cpr) // FIXME: allow no compression
329 throwInternalError("Ehdr compression failed");
330
331 packed_struct(b_info) { // 12-byte header before each compressed block
332 NE32 sz_unc; // uncompressed_size
333 NE32 sz_cpr; // compressed_size
334 unsigned char b_method; // compression algorithm
335 unsigned char b_ftid; // filter id
336 unsigned char b_cto8; // filter parameter
337 unsigned char b_unused; // FIXME: !=0 for partial-block unfilter
338 // NE32 f_offset, f_len; // only if partial-block unfilter
339 };
340
341 struct b_info hdr_info;
342 set_be32(&hdr_info.sz_unc, len_unc);
343 set_be32(&hdr_info.sz_cpr, len_cpr);
344 hdr_info.b_method = ph.method;
345 hdr_info.b_ftid = 0;
346 hdr_info.b_cto8 = 0;
347 hdr_info.b_unused = 0;
348 fo->write(&hdr_info, sizeof(hdr_info)); fo_off += sizeof(hdr_info);
349 unsigned const frag = (3& (0u-len_cpr));
350 ppc32_extra += sizeof(hdr_info) + len_cpr + frag;
351 fo_off += len_cpr + frag;
352 memset(&cpr_hdr[len_cpr], 0, frag); // valgrind only
353 fo->write(cpr_hdr, len_cpr + frag);
354
355 // Partial filter: .text and following contiguous SHF_EXECINSTR
356 upx_bytep f_ptr = ibuf;
357 unsigned f_len = 0;
358 Shdr const *shdr = 1+ shdri; // skip empty shdr[0]
359 if (ft.buf_len==0 // not specified yet FIXME: was set near construction
360 && (Shdr::SHF_ALLOC & shdr->sh_flags)
361 && (Shdr::SHF_EXECINSTR & shdr->sh_flags)) {
362 // shdr[1] is instructions (probably .text)
363 f_ptr = ibuf + (unsigned) (shdr->sh_offset - phdri[0].p_offset);
364 f_len = shdr->sh_size;
365 ++shdr;
366 for (int j= ehdri.e_shnum - 2; --j>=0; ++shdr) {
367 unsigned prev_end = shdr[-1].sh_size + shdr[-1].sh_offset;
368 prev_end += ~(0u-shdr[0].sh_addralign) & (0u-prev_end); // align_up
369 if ((Shdr::SHF_ALLOC & shdr->sh_flags)
370 && (Shdr::SHF_EXECINSTR & shdr->sh_flags)
371 && shdr[0].sh_offset==prev_end) {
372 f_len += shdr->sh_size;
373 }
374 else {
375 break;
376 }
377 }
378 }
379 else { // ft.buf_len already specified, or Shdr[1] not instructions
380 f_ptr = ibuf;
381 f_len = ph.u_len;
382 }
383 compressWithFilters(ibuf, ph.u_len, obuf,
384 f_ptr, f_len, // filter range
385 nullptr, 0, // hdr_ptr, hdr_len
386 &ft, 512, &cconf, getStrategy(ft));
387
388 set_be32(&hdr_info.sz_unc, ph.u_len);
389 set_be32(&hdr_info.sz_cpr, ph.c_len);
390 hdr_info.b_ftid = ft.id;
391 hdr_info.b_cto8 = ft.cto;
392 if (ph.u_len!=f_len) {
393 hdr_info.b_unused = 1; // flag for partial filter
394 }
395 fo->write(&hdr_info, sizeof(hdr_info)); fo_off += sizeof(hdr_info);
396 ppc32_extra += sizeof(hdr_info);
397 if (ph.u_len!=f_len) {
398 set_be32(&hdr_info.sz_unc, f_ptr - (upx_bytep) ibuf);
399 set_be32(&hdr_info.sz_cpr, f_len);
400 fo->write(&hdr_info, 2*sizeof(unsigned)); fo_off += 2*sizeof(unsigned);
401 ppc32_extra += 2*sizeof(unsigned);
402 }
403 }
404 else {
405 compressWithFilters(&ft, 512, &cconf, getStrategy(ft));
406 }
407 unsigned const txt_c_len = ph.c_len;
408
409 const unsigned lsize = getLoaderSize();
410
411 defineDecompressorSymbols();
412 defineFilterSymbols(&ft);
413 relocateLoader();
414
415 MemBuffer loader(lsize);
416 memcpy(loader, getLoader(), lsize);
417 patchPackHeader(loader, lsize);
418
419#define shstrtab local_shstrtab // avoid -Wshadow warning
420 char const shstrtab[]= "\0.text\0.note\0.shstrtab\0.symtab\0.strtab";
421 char const *p = shstrtab;
422 while (0!=*p++) ;
423 shdro[1].sh_name = ptr_diff(p, shstrtab);
424 shdro[1].sh_type = Shdr::SHT_PROGBITS;
425 shdro[1].sh_flags = Shdr::SHF_ALLOC | Shdr::SHF_EXECINSTR;
426 shdro[1].sh_offset = fo_off - ppc32_extra;
427 shdro[1].sh_size = ppc32_extra + txt_c_len + lsize; // plus more ...
428 shdro[1].sh_addralign = 1; // default
429
430 fo_off += write_vmlinux_head(fo, &shdro[1]);
431 fo->write(obuf, txt_c_len); fo_off += txt_c_len;
432 unsigned const a = (shdro[1].sh_addralign -1) & (0u-(ppc32_extra + txt_c_len));
433 if (0!=a) { // align
434 fo_off += a;
435 shdro[1].sh_size += a;
436 fo->seek(a, SEEK_CUR);
437 }
438 fo->write(loader, lsize); fo_off += lsize;
439
440#if 0
441 printf("%-13s: compressed : %8u bytes\n", getName(), txt_c_len);
442 printf("%-13s: decompressor : %8u bytes\n", getName(), lsize);
443#endif
444 verifyOverlappingDecompression();
445
446 // .note with 1st page --------------------------------
447 ph.u_len = phdri[0].p_offset;
448 fi->seek(0, SEEK_SET);
449 fi->readx(ibuf, ph.u_len);
450 compress(ibuf, ph.u_len, obuf, &cconf);
451
452 while (0!=*p++) ;
453 shdro[2].sh_name = ptr_diff(p, shstrtab);
454 shdro[2].sh_type = Shdr::SHT_NOTE;
455 shdro[2].sh_offset = fo_off;
456 shdro[2].sh_size = sizeof(ph.u_len) + ph.c_len;
457 shdro[2].sh_addralign = 1;
458 tmp_u32 = ph.u_len; fo->write(&tmp_u32, 4);
459 fo->write(obuf, ph.c_len); fo_off += shdro[2].sh_size;
460
461 // .note with rest --------------------------------
462 ph.u_len = file_size - (sz_ptload + phdri[0].p_offset);
463 fi->seek(sz_ptload + phdri[0].p_offset, SEEK_SET);
464 fi->readx(ibuf, ph.u_len);
465
466 // Temporarily decrease ph.level by about (1+ log2(sz_rest / sz_ptload))
467 // to avoid spending unreasonable effort compressing large symbol tables
468 // that are discarded 99.9% of the time anyway.
469 int const old_level = ph.level;
470 for (unsigned v = ((ph.u_len>>3) + ph.u_len) / sz_ptload; 0 < v; v>>=1) {
471 if (0== --ph.level) {
472 ph.level = 1;
473 }
474 }
475 compress(ibuf, ph.u_len, obuf, &cconf);
476 ph.level = old_level;
477
478 // while (0!=*p++) ; // name is the same
479 shdro[3].sh_name = ptr_diff(p, shstrtab);
480 shdro[3].sh_type = Shdr::SHT_NOTE;
481 shdro[3].sh_offset = fo_off;
482 shdro[3].sh_size = sizeof(ph.u_len) + ph.c_len;
483 shdro[3].sh_addralign = 1;
484 tmp_u32 = ph.u_len; fo->write(&tmp_u32, 4);
485 fo->write(obuf, ph.c_len); fo_off += shdro[3].sh_size;
486
487 while (0!=*p++) ;
488 shdro[4].sh_name = ptr_diff(p, shstrtab);
489 shdro[4].sh_type = Shdr::SHT_STRTAB;
490 shdro[4].sh_offset = fo_off;
491 shdro[4].sh_size = sizeof(shstrtab); // already includes terminating '\0'
492 shdro[4].sh_addralign = 1;
493 fo->write(shstrtab, shdro[4].sh_size); fo_off += shdro[4].sh_size;
494
495 fo_off = ~3 & (3+ fo_off); fo->seek(fo_off, SEEK_SET);
496 while (0!=*p++) ;
497 shdro[5].sh_name = ptr_diff(p, shstrtab);
498 shdro[5].sh_type = Shdr::SHT_SYMTAB;
499 shdro[5].sh_offset = fo_off;
500 shdro[5].sh_size = ((Ehdr::EM_PPC==my_e_machine) + 5)*sizeof(Sym);
501 //shdro[5].sh_flags = Shdr::SHF_INFO_LINK;
502 shdro[5].sh_link = 6; // to .strtab for symbols
503 shdro[5].sh_info = 1+3; // number of non-global symbols [binutils/bfd/elf.c]
504 shdro[5].sh_addralign = 4;
505 shdro[5].sh_entsize = sizeof(Sym);
506
507 Sym sec_sym;
508
509 // Symbol 0; no references, but bfd demands it.
510 memset(&sec_sym, 0, sizeof(sec_sym));
511 fo->write(&sec_sym, sizeof(sec_sym)); fo_off += sizeof(sec_sym);
512
513 // Each section before .shstrtab needs a symbol.
514 sec_sym.st_info = sec_sym.make_st_info(Sym::STB_LOCAL, Sym::STT_SECTION);
515 sec_sym.st_other = Sym::STV_DEFAULT;
516 sec_sym.st_shndx = 1; // .text
517 fo->write(&sec_sym, sizeof(sec_sym)); fo_off += sizeof(sec_sym);
518 sec_sym.st_shndx = 2; // .note
519 fo->write(&sec_sym, sizeof(sec_sym)); fo_off += sizeof(sec_sym);
520 sec_sym.st_shndx = 3; // .note
521 fo->write(&sec_sym, sizeof(sec_sym)); fo_off += sizeof(sec_sym);
522
523 // the symbol we care about
524 Sym unc_ker;
525 unc_ker.st_name = 1; // 1 byte into strtab
526 unc_ker.st_value = 0;
527 unc_ker.st_size = ppc32_extra + txt_c_len;
528 unc_ker.st_info = unc_ker.make_st_info(Sym::STB_GLOBAL, Sym::STT_FUNC);
529 unc_ker.st_other = Sym::STV_DEFAULT;
530 unc_ker.st_shndx = 1; // .text
531 fo->write(&unc_ker, sizeof(unc_ker)); fo_off += sizeof(unc_ker);
532
533 unsigned const lablen = strlen(my_boot_label);
534 if (Ehdr::EM_PPC==my_e_machine) {
535 unc_ker.st_name += 1+ lablen;
536 unc_ker.st_value = unc_ker.st_size;
537 unc_ker.st_size = 0;
538 fo->write(&unc_ker, sizeof(unc_ker)); fo_off += sizeof(unc_ker);
539 }
540 while (0!=*p++) ;
541 shdro[6].sh_name = ptr_diff(p, shstrtab);
542 shdro[6].sh_type = Shdr::SHT_STRTAB;
543 shdro[6].sh_offset = fo_off;
544 shdro[6].sh_size = 2+ lablen + (Ehdr::EM_PPC==my_e_machine)*(1+ 12); // '\0' before and after
545 shdro[6].sh_addralign = 1;
546 fo->seek(1, SEEK_CUR); // the '\0' before
547 fo->write(my_boot_label, 1+ lablen); // include the '\0' terminator
548 if (Ehdr::EM_PPC==my_e_machine) {
549 fo->write("_vmlinux_end", 1+ 12); fo_off += 1+ 12;
550 }
551 fo_off += 2+ lablen;
552
553 fo->seek(0, SEEK_SET);
554 fo->write(&ehdro, sizeof(ehdro));
555 fo->write(&shdro[0], sizeof(shdro));
556 if (Ehdr::EM_PPC==my_e_machine) {
557 fo->seek(sizeof(unsigned), SEEK_CUR);
558 set_be32(&ppc32_extra, ppc32_extra - 2*sizeof(unsigned) + txt_c_len);
559 fo->write(&ppc32_extra, sizeof(ppc32_extra));
560 }
561
562 if (!checkFinalCompressionRatio(fo))
563 throwNotCompressible();
564#undef shstrtab
565}

567template <class T>
568tribool PackVmlinuxBase<T>::canUnpack()
569{
570 fi->seek(0, SEEK_SET);
571 fi->readx(&ehdri, sizeof(ehdri));
572
573 // now check the ELF header
574 if (memcmp(&ehdri, "\x7f\x45\x4c\x46", 4)
575 || ehdri.e_ident[Ehdr::EI_CLASS] != my_elfclass
576 || ehdri.e_ident[Ehdr::EI_DATA] != my_elfdata
577 || ehdri.e_ident[Ehdr::EI_VERSION] != Ehdr::EV_CURRENT
578 || !memcmp(&ehdri.e_ident[8], "FreeBSD", 7) // branded
579 || ehdri.e_machine != my_e_machine
580 || ehdri.e_version != 1 // version
581 || ehdri.e_ehsize != sizeof(ehdri) // different <elf.h> ?
582 )
583 return false;
584
585 if (ehdri.e_type != Ehdr::ET_REL
586 //i386 fails || ehdri.e_shoff != sizeof(ehdri) // Shdr not contiguous with Ehdr
587 || ehdri.e_shentsize!=sizeof(Shdr)
588 || ehdri.e_shnum < 4
589 || (unsigned)file_size < (ehdri.e_shnum * sizeof(Shdr) + ehdri.e_shoff)
590 )
591 return false;
592
593 // find the .shstrtab section
594 Shdr const *const shstrsec = getElfSections();
595 if (nullptr==shstrsec) {
596 return false;
597 }
598
599 // check for .text .note .note and sane (.sh_size + .sh_offset)
600 p_note0 = p_note1 = p_text = nullptr;
601 int j;
602 Shdr *p;
603 for (p= shdri, j= ehdri.e_shnum; --j>=0; ++p) {
604 if ((unsigned)file_size < (p->sh_size + p->sh_offset)
605 || (5+ p->sh_name) < p->sh_name // wrap: ignore malformed
606 || shstrsec->sh_size < (5+ p->sh_name) ) {
607 continue;
608 }
609 if (0==strcmp(".text", shstrtab + p->sh_name)) {
610 p_text = p;
611 }
612 if (0==strcmp(".note", shstrtab + p->sh_name)) {
613 if (nullptr==p_note0) {
614 p_note0 = p;
615 } else
616 if (nullptr==p_note1) {
617 p_note1 = p;
618 }
619 }
620 }
621 if (nullptr==p_text || nullptr==p_note0 || nullptr==p_note1) {
622 return false;
623 }
624
625 char buf[1024];
626 fi->seek(p_text->sh_offset + p_text->sh_size - sizeof(buf), SEEK_SET);
627 fi->readx(buf, sizeof(buf));
628 if (!getPackHeader(buf, sizeof(buf)))
629 return -1; // format is known, but definitely is not packed
630
631 return true;
632}

634template <class T>
635void PackVmlinuxBase<T>::unpack(OutputFile *fo)
636{
637 TE32 word;
638 PackHeader const ph_tmp(ph);
639
640 fi->seek(p_note0->sh_offset, SEEK_SET);
641 fi->readx(&word, sizeof(word));
642 ph.u_len = word;
643 ph.c_len = p_note0->sh_size - sizeof(word);
644 ibuf.alloc(ph.c_len);
645 fi->readx(ibuf, ph.c_len);
646 obuf.allocForDecompression(ph.u_len);
647 decompress(ibuf, obuf, false);
648 fo->write(obuf, ph.u_len);
649 obuf.dealloc();
650 ibuf.dealloc();
651
652 ph = ph_tmp;
653 if (!has_valid_vmlinux_head()) {
654 throwCantUnpack(".text corrupted");
655 }
656 ibuf.alloc(ph.c_len);
657 fi->readx(ibuf, ph.c_len);
658 obuf.allocForDecompression(ph.u_len);
659 decompress(ibuf, obuf);
660
661 Filter ft(ph.level);
662 ft.init(ph.filter, 0);
663 ft.cto = (unsigned char) ph.filter_cto;
664 ft.unfilter(obuf, ph.u_len);
665 fo->write(obuf, ph.u_len);
666 obuf.dealloc();
667 ibuf.dealloc();
668
669 fi->seek(p_note1->sh_offset, SEEK_SET);
670 fi->readx(&word, sizeof(word));
671 ph.u_len = word;
672 ph.c_len = p_note1->sh_size - sizeof(word);
673 ibuf.alloc(ph.c_len);
674 fi->readx(ibuf, p_note1->sh_size - sizeof(ph.u_len));
675 obuf.allocForDecompression(ph.u_len);
676 decompress(ibuf, obuf, false);
677 fo->write(obuf, ph.u_len);
678 obuf.dealloc();
679 ibuf.dealloc();
680
681 ph = ph_tmp;
682}

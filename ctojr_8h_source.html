45static unsigned f80_call(Filter const *f) { return (1 + (0x0f & f->id)) % 3; }

47static unsigned f80_jmp1(Filter const *f) { return ((1 + (0x0f & f->id)) / 3) % 3; }

49static unsigned f80_jcc2(Filter const *f) { return f80_jmp1(f); }

51#define N_MRU 32 // does not have to be a power of 2
52
53// Adaptively remember recent destinations.
54static void update_mru(const unsigned jc, // destination address
55 const int kh, // mru[kh] is slot where found
56 unsigned mru[N_MRU], // circular buffer of most recent destinations
57 int &hand, // mru[hand] is most recent destination
58 int &tail // mru[tail] is beyond oldest destination ("cold cache" startup)
59) {
60 if (0 > --hand) {
61 hand = N_MRU - 1;
62 }
63 const unsigned t = mru[hand]; // entry which will be overwritten by jc
64 if (0 != t) { // have seen at least N_MRU destinations
65 mru[kh] = t;
66 } else { // "cold cache": keep active region contiguous
67 if (0 > --tail) {
68 tail = N_MRU - 1;
69 }
70 const unsigned t2 = mru[tail];
71 mru[tail] = 0;
72 mru[kh] = t2;
73 }
74 mru[hand] = jc;
75}
76#endif //}

78static int F(Filter *f) {
79#ifdef U
80 // filter
81 byte *const b = f->buf;
82#else
83 // scan
84 const byte *b = f->buf;
85#endif
86 const unsigned size = f->buf_len;
87
88 unsigned ic, jc, kc;
89 unsigned calls = 0, noncalls = 0, noncalls2 = 0;
90 unsigned lastnoncall = size, lastcall = 0;
91 unsigned wtally[3];
92 memset(wtally, 0, sizeof(wtally));
93
94#ifdef U //{
95 const unsigned f_call = f80_call(f);
96 const unsigned f_jmp1 = f80_jmp1(f);
97 const unsigned f_jcc2 = f80_jcc2(f);
98
99 int hand = 0, tail = 0;
100 unsigned mru[N_MRU];
101 memset(&mru[0], 0, sizeof(mru));
102 assert(N_MRU <= 256);
103 f->n_mru = (MRUFLT == f_call || MRUFLT == f_jmp1 || MRUFLT == f_jcc2) ? N_MRU : 0;
104#endif //}
105
106 // FIXME: We must fit into 8 MiB because we steal one bit.
107 // find a 16 MiB large empty address space
108 {
109 int which;
110 unsigned char buf[256];
111 memset(buf, 0, 256);
112
113 for (ic = 0; ic < size - 5; ic++)
114 if (CONDF(which, b, ic, lastcall) && get_le32(b + ic + 1) + ic + 1 >= size) {
115 buf[b[ic + 1]] |= 1;
116 }
117 UNUSED(which);
118
119 if (getcto(f, buf) < 0)
120 return -1;
121 }
122 const unsigned char cto8 = f->cto;
123#ifdef U
124 const unsigned cto = (unsigned) f->cto << 24;
125#endif
126
127 for (ic = 0; ic < size - 5; ic++) {
128 int which;
129 int f_on = 0;
130 if (!CONDF(which, b, ic, lastcall))
131 continue;
132 ++wtally[which];
133 jc = get_le32(b + ic + 1) + ic + 1;
134 // try to detect 'real' calls only
135 if (jc < size) {
136#ifdef U
137 if (2 == which && NOFILT != f_jcc2) { // 6-byte Jcc <disp32>
138 // Prefix 0x0f is constant, but opcode condition 0x80..0x8f
139 // varies. Because we store the destination (or its mru index)
140 // in be32 big endian format, the low-addressed bytes
141 // will tend to be constant. Swap prefix and opcode
142 // so that constants are together for better compression.
143 unsigned char const t = b[ic - 1];
144 b[ic - 1] = b[ic];
145 b[ic] = t;
146 }
147 // FIXME [?]: Extend to 8 bytes if "ADD ESP, byte 4*n" follows CALL.
148 // This will require two related cto's (consecutive, or otherwise).
149 if ((0 == which && MRUFLT == f_call) || (1 == which && MRUFLT == f_jmp1) ||
150 (2 == which && MRUFLT == f_jcc2)) {
151 f_on = 1;
152 // Recode the destination: narrower mru indices
153 // should compress better than wider addresses.
154 // (But not when offset of match is unlimited?)
155 int k;
156 for (k = 0; k < N_MRU; ++k) {
157 int kh = hand + k;
158 if (N_MRU <= kh) {
159 kh -= N_MRU;
160 }
161 if (mru[kh] == jc) { // destination was seen recently
162 set_be32(b + ic + 1, ((k << 1) | 0) + cto);
163 update_mru(jc, kh, mru, hand, tail);
164 break;
165 }
166 }
167 if (k == N_MRU) { // loop failed; jc is not in mru[]
168 set_be32(b + ic + 1, ((jc << 1) | 1) + cto);
169 // Adaptively remember recent destinations.
170 if (0 > --hand) {
171 hand = N_MRU - 1;
172 }
173 mru[hand] = jc;
174 }
175 } else if ((0 == which && NOFILT != f_call) || (1 == which && NOFILT != f_jmp1) ||
176 (2 == which && NOFILT != f_jcc2)) {
177 f_on = 1;
178 set_be32(b + ic + 1, jc + cto);
179 }
180#endif
181 if (f_on) {
182 if (ic - lastnoncall < 5) {
183 // check the last 4 bytes before this call
184 for (kc = 4; kc; kc--)
185 if (CONDF(which, b, ic - kc, lastcall) && b[ic - kc + 1] == cto8)
186 break;
187 if (kc) {
188#ifdef U
189 // restore original
190 if (2 == which) {
191 // Unswap prefix and opcode for 6-byte Jcc <disp32>
192 unsigned char const t = b[ic - 1];
193 b[ic - 1] = b[ic];
194 b[ic] = t;
195 }
196 set_le32(b + ic + 1, jc - ic - 1);
197#endif
198 if (b[ic + 1] == cto8)
199 return 1; // fail - buffer not restored
200 lastnoncall = ic;
201 noncalls2++;
202 continue;
203 }
204 }
205 calls++;
206 ic += 4;
207 lastcall = ic + 1;
208 }
209 } else {
210 assert(b[ic + 1] != cto8); // this should not happen
211 lastnoncall = ic;
212 noncalls++;
213 }
214 }
215
216 f->calls = calls;
217 f->noncalls = noncalls;
218 f->lastcall = lastcall;
219
220#if 0 || defined(TESTING)
221 printf("\ncalls=%d noncalls=%d noncalls2=%d text_size=%x calltrickoffset=%x\n",
222 calls,noncalls,noncalls2,size,cto8);
223 printf("CALL/JMP/JCC %d %d %d\n",wtally[0],wtally[1],wtally[2]);
224#endif
225 UNUSED(noncalls2);
226 return 0;
227}

234static int U(Filter *f) {
235 unsigned ic, jc;
236
237 byte *const b = f->buf;
238 const unsigned size5 = f->buf_len - 5;
239 const unsigned cto = (unsigned) f->cto << 24;
240 unsigned lastcall = 0;
241 int hand = 0, tail = 0;
242 const unsigned f_call = f80_call(f);
243 const unsigned f_jmp1 = f80_jmp1(f);
244 const unsigned f_jcc2 = f80_jcc2(f);
245
246 unsigned mru[N_MRU];
247 memset(&mru[0], 0, sizeof(mru));
248
249 for (ic = 0; ic < size5; ic++) {
250 int which;
251 if (CONDU(which, b, ic, lastcall)) {
252 unsigned f_on = 0;
253 jc = get_be32(b + ic + 1) - cto;
254 if (b[ic + 1] == f->cto) {
255 if ((0 == which && MRUFLT == f_call) || (1 == which && MRUFLT == f_jmp1) ||
256 (2 == which && MRUFLT == f_jcc2)) {
257 f_on = 1;
258 if (1 & jc) { // 1st time at this destination
259 jc >>= 1;
260 if (0 > --hand) {
261 hand = N_MRU - 1;
262 }
263 mru[hand] = jc;
264 } else { // not 1st time at this destination
265 jc >>= 1;
266 if (N_MRU <= jc) {
267 throwCompressedDataViolation();
268 }
269 int kh = jc + hand;
270 if (N_MRU <= kh) {
271 kh -= N_MRU;
272 }
273 jc = mru[kh];
274 update_mru(jc, kh, mru, hand, tail);
275 }
276 set_le32(b + ic + 1, jc - ic - 1);
277 } else if ((0 == which && NOFILT != f_call) || (1 == which && NOFILT != f_jmp1) ||
278 (2 == which && NOFILT != f_jcc2)) {
279 f_on = 1;
280 set_le32(b + ic + 1, jc - ic - 1);
281 }
282 if (2 == which && NOFILT != f_jcc2) {
283 // Unswap prefix and opcode for 6-byte Jcc <disp32>
284 unsigned char const t = b[ic - 1];
285 b[ic - 1] = b[ic];
286 b[ic] = t;
287 }
288
289 if (f_on) {
290 f->calls++;
291 ic += 4;
292 f->lastcall = lastcall = ic + 1;
293 }
294 } else
295 f->noncalls++;
296 }
297 }
298 return 0;
299}

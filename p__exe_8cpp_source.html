66int PackExe::fillExeHeader(struct exe_header_t *eh) const {
67#define oh (*eh)
68 // fill new exe header
69 int flag = 0;
70 if (!opt->dos_exe.no_reloc && !M_IS_LZMA(ph.method))
71 flag |= USEJUMP;
72 if (ih.relocs == 0)
73 flag |= NORELOC;
74
75 mem_clear(&oh);
76 oh.ident = 'M' + 'Z' * 256;
77 oh.headsize16 = 2;
78
79 unsigned minsp = 0x200;
80 if (M_IS_LZMA(ph.method))
81 minsp = stack_for_lzma;
82 minsp = ALIGN_UP(minsp, 16u);
83 assert(minsp < 0xff00);
84 if (oh.sp > minsp)
85 minsp = oh.sp;
86 if (minsp < 0xff00 - 2)
87 minsp = ALIGN_UP(minsp, 2u);
88 oh.sp = minsp;
89
90 unsigned destpara = (ph.u_len + ph.overlap_overhead - ph.c_len + 31) / 16;
91 oh.ss = ph.c_len / 16 + destpara;
92 if (ih.ss * 16 + ih.sp < 0x100000 && ih.ss > oh.ss && ih.sp > 0x200)
93 oh.ss = ih.ss;
94 if (oh.ss * 16 + 0x50 < ih.ss * 16 + ih.sp && oh.ss * 16 + 0x200 > ih.ss * 16 + ih.sp)
95 oh.ss += 0x20;
96
97 if (oh.ss != ih.ss)
98 flag |= SS;
99 if (oh.sp != ih.sp || M_IS_LZMA(ph.method))
100 flag |= SP;
101 return flag;
102#undef oh
103}

105void PackExe::addLoaderEpilogue(int flag) {
106 addLoader("EXEMAIN5");
107 if (relocsize)
108 addLoader(ph.u_len <= DI_LIMIT || (ph.u_len & 0x7fff) >= relocsize ? "EXENOADJ"
109 : "EXEADJUS",
110 "EXERELO1", has_9a ? "EXEREL9A" : "", "EXERELO2",
111 ih_exesize > 0xFE00 ? "EXEREBIG" : "", "EXERELO3");
112 addLoader("EXEMAIN8", device_driver ? "DEVICEEND" : "", (flag & SS) ? "EXESTACK" : "",
113 (flag & SP) ? "EXESTASP" : "", (flag & USEJUMP) ? "EXEJUMPF" : "");
114 if (!(flag & USEJUMP))
115 addLoader(ih.cs ? "EXERCSPO" : "", "EXERETIP");
116
117 linker->defineSymbol("original_cs", ih.cs);
118 linker->defineSymbol("original_ip", ih.ip);
119 linker->defineSymbol("original_sp", ih.sp);
120 linker->defineSymbol("original_ss", ih.ss);
121 linker->defineSymbol(
122 "reloc_size",
123 (ph.u_len <= DI_LIMIT || (ph.u_len & 0x7fff) >= relocsize ? 0 : MAXRELOCSIZE) - relocsize);
124}

126void PackExe::buildLoader(const Filter *) {
127 // get flag
128 exe_header_t dummy_oh;
129 int flag = fillExeHeader(&dummy_oh);
130
131 initLoader(stub_i086_dos16_exe, sizeof(stub_i086_dos16_exe));
132
133 if (M_IS_LZMA(ph.method)) {
134 addLoader("LZMA_DEC00", opt->small ? "LZMA_DEC10" : "LZMA_DEC20", "LZMA_DEC30",
135 use_clear_dirty_stack ? "LZMA_DEC31" : "", "LZMA_DEC32",
136 ph.u_len > 0xffff ? "LZMA_DEC33" : "");
137
138 addLoaderEpilogue(flag);
139 defineDecompressorSymbols();
140 const unsigned lsize0 = getLoaderSize();
141
142 // Lzma decompression code starts at ss:0x10, and its size is
143 // lsize bytes. It also needs getDecompressorWrkmemSize() bytes
144 // during uncompression. It also uses some stack, so 0x100
145 // more bytes are allocated
146 stack_for_lzma = 0x10 + lsize0 + getDecompressorWrkmemSize() + 0x100;
147 stack_for_lzma = ALIGN_UP(stack_for_lzma, 16u);
148
149 unsigned clear_dirty_stack_low = 0x10 + lsize0;
150 clear_dirty_stack_low = ALIGN_UP(clear_dirty_stack_low, 2u);
151 if (use_clear_dirty_stack)
152 linker->defineSymbol("clear_dirty_stack_low", clear_dirty_stack_low);
153
154 relocateLoader();
155 const unsigned lsize = getLoaderSize();
156 assert(lsize0 == lsize);
157 MemBuffer loader(lsize);
158 memcpy(loader, getLoader(), lsize);
159
160 MemBuffer compressed_lzma;
161 compressed_lzma.allocForCompression(lsize);
162 unsigned c_len_lzma = MemBuffer::getSizeForCompression(lsize);
163 int r = upx_compress(loader, lsize, compressed_lzma, &c_len_lzma, nullptr, M_NRV2B_LE16, 9,
164 nullptr, nullptr);
165 assert(r == UPX_E_OK);
166 assert(c_len_lzma < lsize);
167
168 info("lzma+relocator code compressed: %u -> %u", lsize, c_len_lzma);
169 // reinit the loader
170 initLoader(stub_i086_dos16_exe, sizeof(stub_i086_dos16_exe));
171 // prepare loader
172 if (device_driver)
173 addLoader("DEVICEENTRY,LZMADEVICE,DEVICEENTRY2");
174
175 linker->addSection("COMPRESSED_LZMA", compressed_lzma, c_len_lzma, 0);
176 addLoader("LZMAENTRY,NRV2B160,NRVDDONE,NRVDECO1,NRVGTD00,NRVDECO2");
177
178 } else if (device_driver)
179 addLoader("DEVICEENTRY,DEVICEENTRY2");
180
181 addLoader("EXEENTRY", M_IS_LZMA(ph.method) && device_driver ? "LONGSUB" : "SHORTSUB",
182 "JNCDOCOPY", relocsize ? "EXERELPU" : "", "EXEMAIN4",
183 M_IS_LZMA(ph.method) ? "" : "EXEMAIN4B", "EXEMAIN4C",
184 M_IS_LZMA(ph.method) ? "COMPRESSED_LZMA_START,COMPRESSED_LZMA" : "",
185 "+G5DXXXX,UPX1HEAD,EXECUTPO");
186 if (ph.method == M_NRV2B_8)
187 addLoader("NRV2B16S", // decompressor
188 ph.u_len > DI_LIMIT ? "N2B64K01" : "", "NRV2BEX1",
189 opt->cpu_x86 == opt->CPU_8086 ? "N2BX8601" : "N2B28601", "NRV2BEX2",
190 opt->cpu_x86 == opt->CPU_8086 ? "N2BX8602" : "N2B28602", "NRV2BEX3",
191 ph.c_len > 0xffff ? "N2B64K02" : "", "NRV2BEX9");
192 else if (ph.method == M_NRV2D_8)
193 addLoader("NRV2D16S", ph.u_len > DI_LIMIT ? "N2D64K01" : "", "NRV2DEX1",
194 opt->cpu_x86 == opt->CPU_8086 ? "N2DX8601" : "N2D28601", "NRV2DEX2",
195 opt->cpu_x86 == opt->CPU_8086 ? "N2DX8602" : "N2D28602", "NRV2DEX3",
196 ph.c_len > 0xffff ? "N2D64K02" : "", "NRV2DEX9");
197 else if (ph.method == M_NRV2E_8)
198 addLoader("NRV2E16S", ph.u_len > DI_LIMIT ? "N2E64K01" : "", "NRV2EEX1",
199 opt->cpu_x86 == opt->CPU_8086 ? "N2EX8601" : "N2E28601", "NRV2EEX2",
200 opt->cpu_x86 == opt->CPU_8086 ? "N2EX8602" : "N2E28602", "NRV2EEX3",
201 ph.c_len > 0xffff ? "N2E64K02" : "", "NRV2EEX9");
202 else if M_IS_LZMA (ph.method)
203 return;
204 else
205 throwInternalError("unknown compression method");
206
207 addLoaderEpilogue(flag);
208}

214int PackExe::readFileHeader() {
215 ih_exesize = ih_imagesize = ih_overlay = 0;
216 fi->readx(&ih, sizeof(ih));
217 if (ih.ident != 'M' + 'Z' * 256 && ih.ident != 'Z' + 'M' * 256)
218 return 0;
219 ih_exesize = ih.m512 + ih.p512 * 512 - (ih.m512 ? 512 : 0);
220 if (ih_exesize == 0)
221 ih_exesize = file_size;
222 ih_imagesize = ih_exesize - ih.headsize16 * 16;
223 ih_overlay = file_size - ih_exesize;
224 if (file_size_u < sizeof(ih) || ((ih.m512 | ih.p512) && ih.m512 + ih.p512 * 512u < sizeof(ih)))
225 throwCantPack("illegal exe header");
226 if (ih_exesize > file_size_u || ih_imagesize < 4 || ih_imagesize > ih_exesize)
227 throwCantPack("exe header corrupted");
228 NO_printf("dos/exe header: %d %d %d\n", ih_exesize, ih_imagesize, ih_overlay);
229 return UPX_F_DOS_EXE;
230}

232tribool PackExe::canPack() {
233 if (fn_has_ext(fi->getName(), "sys")) // dos/sys
234 return false;
235 if (!readFileHeader())
236 return false;
237 if (file_size < 1024 || ih_imagesize < 512)
238 throwCantPack("file is too small for dos/exe");
239 fi->seek(0x3c, SEEK_SET);
240 LE32 offs;
241 fi->readx(&offs, sizeof(offs));
242 if (ih.relocoffs >= 0x40 && offs) {
243 if (opt->dos_exe.force_stub)
244 opt->overlay = opt->COPY_OVERLAY;
245 else
246 throwCantPack("dos/exe: can't pack new-exe");
247 }
248 return true;
249}

255static unsigned optimize_relocs(SPAN_S(byte) image, const unsigned image_size,
256 SPAN_S(const byte) relocs, const unsigned relocnum,
257 SPAN_S(byte) crel, bool *has_9a) {
258#if WITH_XSPAN >= 2
259 ptr_check_no_overlap(image.data(image_size), image_size, relocs.data(), relocs.size_bytes(),
260 crel.data(), crel.size_bytes());
261#endif
262 if (opt->exact)
263 throwCantPackExact();
264
265 SPAN_S_VAR(byte, const crel_start, crel);
266 unsigned seg_high = 0;
267#if 0
268 unsigned seg_low = 0xffffffff;
269 unsigned off_low = 0xffffffff;
270 unsigned off_high = 0;
271 unsigned linear_low = 0xffffffff;
272 unsigned linear_high = 0;
273#endif
274
275 // pass 1 - find 0x9a bounds in image
276 for (unsigned i = 0; i < relocnum; i++) {
277 unsigned addr = get_le32(relocs + 4 * i);
278 if (addr >= image_size - 1)
279 throwCantPack("unexpected relocation 1");
280 if (addr >= 3 && image[addr - 3] == 0x9a) {
281 unsigned seg = get_le16(image + addr);
282 if (seg > seg_high)
283 seg_high = seg;
284#if 0
285 if (seg < seg_low)
286 seg_low = seg;
287 unsigned off = get_le16(image + addr - 2);
288 if (off < off_low)
289 off_low = off;
290 if (off > off_high)
291 off_high = off;
292 unsigned l = (seg << 4) + off;
293 if (l < linear_low)
294 linear_low = l;
295 if (l > linear_high)
296 linear_high = l;
297#endif
298 }
299 }
300 // printf("%d %d\n", seg_low, seg_high);
301 // printf("%d %d\n", off_low, off_high);
302 // printf("%d %d\n", linear_low, linear_high);
303
304 // pass 2 - reloc
305
306 crel += 4; // to be filled in later
307
308 unsigned ones = 0;
309 unsigned es = 0;
310 for (unsigned i = 0; i < relocnum;) {
311 unsigned addr = get_le32(relocs + 4 * i);
312 unsigned di = addr & 0x0f;
313 set_le16(crel + 0, di);
314 set_le16(crel + 2, (addr >> 4) - es);
315 crel += 4;
316 es = addr >> 4;
317
318 for (++i; i < relocnum; i++) {
319 unsigned t;
320 addr = get_le32(relocs + 4 * i);
321 NO_printf("%x\n", es * 16 + di);
322 if ((addr - es * 16 > 0xfffe) || (i == relocnum - 1 && addr - es * 16 > 0xff00)) {
323 // segment change
324 t = 1 + (0xffff - di) / 254;
325 memset(crel, 1, t);
326 crel += t;
327 ones += t - 1; // -1 is used to help the assembly stuff
328 break;
329 }
330 unsigned offs = addr - es * 16;
331 if (offs >= 3 && image[es * 16 + offs - 3] == 0x9a && offs > di + 3) {
332 for (t = di; t < offs - 3; t++)
333 if (image[es * 16 + t] == 0x9a && get_le16(image + es * 16 + t + 3) <= seg_high)
334 break;
335 if (t == offs - 3) {
336 // code 0: search for 0x9a
337 *crel++ = 0;
338 di = offs;
339 *has_9a = true;
340 continue;
341 }
342 }
343 t = offs - di;
344 if ((int) t < 2)
345 throwCantPack("unexpected relocation 2");
346 while (t >= 256) {
347 // code 1: add 254, don't reloc
348 *crel++ = 1;
349 t -= 254;
350 ones++;
351 }
352 *crel++ = (byte) t;
353 di = offs;
354 }
355 }
356 *crel++ = 1;
357 ones++;
358 set_le16(crel_start, ones);
359 set_le16(crel_start + 2, seg_high);
360
361 // OutputFile::dump("x.rel", crel_start, ptr_udiff_bytes(crel, crel_start));
362 return ptr_udiff_bytes(crel, crel_start);
363}

369void PackExe::pack(OutputFile *fo) {
370 unsigned ic;
371
372 const unsigned relocnum = ih.relocs;
373 if (relocnum > MAXRELOCSIZE) // early check
374 throwCantPack("too many relocations");
375 checkOverlay(ih_overlay);
376
377 // read image
378 // image + space for optimized relocs + safety/alignments
379 ibuf.alloc(ih_imagesize + 4 * relocnum + 1024);
380 fi->seek(ih.headsize16 * 16, SEEK_SET);
381 fi->readx(ibuf, ih_imagesize);
382
383 checkAlreadyPacked(ibuf, UPX_MIN(ih_imagesize, 127u));
384
385 device_driver = get_le32(ibuf) == 0xffffffffu;
386
387 // relocations
388 relocsize = 0;
389 has_9a = false;
390 if (relocnum) {
391 MemBuffer mb_relocs(4 * relocnum);
392 SPAN_S_VAR(byte, relocs, mb_relocs);
393 fi->seek(ih.relocoffs, SEEK_SET);
394 fi->readx(relocs, 4 * relocnum);
395
396 // dos/exe runs in real-mode, so convert to linear addresses
397 for (ic = 0; ic < relocnum; ic++) {
398 unsigned jc = get_le32(relocs + 4 * ic);
399 set_le32(relocs + 4 * ic, ((jc >> 16) * 16 + (jc & 0xffff)) & 0xfffff);
400 }
401 upx_qsort(raw_bytes(relocs, 4 * relocnum), relocnum, 4, le32_compare);
402
403 SPAN_S_VAR(byte, image, ibuf + 0, ih_imagesize);
404 SPAN_S_VAR(byte, crel, ibuf + ih_imagesize, ibuf);
405 relocsize = optimize_relocs(image, ih_imagesize, relocs, relocnum, crel, &has_9a);
406 set_le16(crel + relocsize, relocsize + 2);
407 relocsize += 2;
408 assert(relocsize >= 11);
409 if (relocsize > MAXRELOCSIZE) // optimize_relocs did not help
410 throwCantPack("too many relocations");
411#if TESTING && 0
412 unsigned rout_len = MemBuffer::getSizeForCompression(relocsize);
413 MemBuffer rout(rout_len);
414 ucl_nrv2b_99_compress(raw_bytes(crel, relocsize), relocsize, rout, &rout_len, nullptr, 9,
415 nullptr, nullptr);
416 printf("dos/exe reloc compress: %d -> %d\n", relocsize, rout_len);
417#endif
418 }
419
420 // prepare packheader
421 ph.u_len = ih_imagesize + relocsize;
422 obuf.allocForCompression(ph.u_len);
423 // prepare filter
424 Filter ft(ph.level);
425 // compress (max_match = 8192)
426 upx_compress_config_t cconf;
427 cconf.reset();
428 cconf.conf_ucl.max_match = MAXMATCH;
429 cconf.conf_lzma.max_num_probs = 1846 + (768 << 4); // ushort: ~28 KiB stack
430 compressWithFilters(&ft, 32, &cconf);
431
432 if (M_IS_NRV2B(ph.method) || M_IS_NRV2D(ph.method) || M_IS_NRV2E(ph.method))
433 if (ph.max_run_found + ph.max_match_found > 0x8000)
434 throwCantPack("decompressor limit exceeded, send a bugreport");
435
436#if TESTING
437 if (opt->debug.debug_level) {
438 printf("image+relocs %d -> %d\n", ih_imagesize + relocsize, ph.c_len);
439 printf("offsets: %d - %d\nmatches: %d - %d\nruns: %d - %d\n", 0 /*ph.min_offset_found*/,
440 ph.max_offset_found, 0 /*ph.min_match_found*/, ph.max_match_found,
441 0 /*ph.min_run_found*/, ph.max_run_found);
442 }
443#endif
444
445 int flag = fillExeHeader(&oh);
446
447 const unsigned lsize = getLoaderSize();
448 MemBuffer loader(lsize);
449 memcpy(loader, getLoader(), lsize);
450 // OutputFile::dump("xxloader.dat", loader, lsize);
451
452 // patch loader
453 const unsigned packedsize = ph.c_len;
454 const unsigned e_len = getLoaderSectionStart("EXECUTPO");
455 const unsigned d_len = lsize - e_len;
456 assert((e_len & 15) == 0);
457
458 const unsigned copysize = (1 + packedsize + d_len) & ~1;
459 const unsigned firstcopy = copysize % 0x10000 ? copysize % 0x10000 : 0x10000;
460
461 // set oh.min & oh.max
462 ic = ih.min * 16 + ih_imagesize;
463 if (ic < oh.ss * 16u + oh.sp)
464 ic = oh.ss * 16u + oh.sp;
465 oh.min = (ic - (packedsize + lsize)) / 16;
466 ic = oh.min + (ih.max - ih.min);
467 oh.max = ic < 0xffff && ih.max != 0xffff ? ic : 0xffff;
468
469 // set extra info
470 byte extra_info[9];
471 unsigned eisize = 0;
472 if (oh.ss != ih.ss) {
473 set_le16(extra_info + eisize, ih.ss);
474 eisize += 2;
475 assert((flag & SS) != 0); // set in fillExeHeader()
476 }
477 if (oh.sp != ih.sp) {
478 set_le16(extra_info + eisize, ih.sp);
479 eisize += 2;
480 assert((flag & SP) != 0); // set in fillExeHeader()
481 }
482 if (ih.min != oh.min) {
483 set_le16(extra_info + eisize, ih.min);
484 eisize += 2;
485 flag |= MINMEM;
486 }
487 if (ih.max != oh.max) {
488 set_le16(extra_info + eisize, ih.max);
489 eisize += 2;
490 flag |= MAXMEM;
491 }
492 extra_info[eisize++] = (byte) flag;
493
494 if (M_IS_NRV2B(ph.method) || M_IS_NRV2D(ph.method) || M_IS_NRV2E(ph.method))
495 linker->defineSymbol("bx_magic", 0x7FFF + 0x10 * ((packedsize & 15) + 1));
496
497 unsigned decompressor_entry = 1 + (packedsize & 15);
498 if (M_IS_LZMA(ph.method))
499 decompressor_entry = 0x10;
500 linker->defineSymbol("decompressor_entry", decompressor_entry);
501
502 // patch loader
503 if (flag & USEJUMP) {
504 // I use a relocation entry to set the original cs
505 unsigned n = getLoaderSectionStart("EXEJUMPF") + 1;
506 n += packedsize + 2;
507 oh.relocs = 1;
508 oh.firstreloc = (n & 0xf) + ((n >> 4) << 16);
509 } else {
510 oh.relocs = 0;
511 oh.firstreloc = ih.cs * 0x10000 + ih.ip;
512 }
513
514 oh.relocoffs = offsetof(exe_header_t, firstreloc);
515
516 linker->defineSymbol("destination_segment", oh.ss - ph.c_len / 16 - e_len / 16);
517 linker->defineSymbol("source_segment", e_len / 16 + (copysize - firstcopy) / 16);
518 linker->defineSymbol("copy_offset", firstcopy - 2);
519 linker->defineSymbol("words_to_copy", firstcopy / 2);
520
521 linker->defineSymbol("exe_stack_sp", oh.sp);
522 linker->defineSymbol("exe_stack_ss", oh.ss);
523 linker->defineSymbol("interrupt", get_le16(ibuf + 8));
524 linker->defineSymbol("attribute", get_le16(ibuf + 4));
525 linker->defineSymbol("orig_strategy", get_le16(ibuf + 6));
526
527 const unsigned outputlen = sizeof(oh) + e_len + packedsize + d_len + eisize;
528 oh.m512 = outputlen & 511;
529 oh.p512 = (outputlen + 511) >> 9;
530
531 const char *exeentry = M_IS_LZMA(ph.method) ? "LZMAENTRY" : "EXEENTRY";
532 oh.ip = device_driver ? getLoaderSection(exeentry) - 2 : 0;
533
534 defineDecompressorSymbols();
535 relocateLoader();
536 memcpy(loader, getLoader(), lsize);
537 patchPackHeader(loader, e_len);
538
539 NO_fprintf(stderr, "\ne_len=%x d_len=%x c_len=%x oo=%x ulen=%x copysize=%x imagesize=%x", e_len,
540 d_len, packedsize, ph.overlap_overhead, ph.u_len, copysize, ih_imagesize);
541
542 // write header + write loader + compressed file
543#if TESTING
544 if (opt->debug.debug_level)
545 printf("\n%d %d %d %d\n", (int) sizeof(oh), e_len, packedsize, d_len);
546#endif
547 fo->write(&oh, sizeof(oh)); // program header
548 fo->write(loader, e_len); // entry code
549 fo->write(obuf, packedsize); // compressed data
550 fo->write(loader + e_len, d_len); // decompressor code
551 fo->write(extra_info, eisize); // extra info for unpacking
552 assert(eisize <= 9);
553 NO_printf("%-13s: program hdr : %8u bytes\n", getName(), usizeof(oh));
554 NO_printf("%-13s: entry : %8u bytes\n", getName(), e_len);
555 NO_printf("%-13s: compressed : %8u bytes\n", getName(), packedsize);
556 NO_printf("%-13s: decompressor : %8u bytes\n", getName(), d_len);
557 NO_printf("%-13s: extra info : %8u bytes\n", getName(), eisize);
558
559 // verify
560 verifyOverlappingDecompression();
561
562 // copy the overlay
563 copyOverlay(fo, ih_overlay, obuf);
564 NO_fprintf(stderr, "dos/exe %x %x\n", relocsize, ph.u_len);
565
566 // finally check the compression ratio
567 if (!checkFinalCompressionRatio(fo))
568 throwNotCompressible();
569}

588void PackExe::unpack(OutputFile *fo) {
589 ibuf.alloc(file_size);
590 obuf.allocForDecompression(ph.u_len);
591
592 // read the file
593 fi->seek(ih.headsize16 * 16, SEEK_SET);
594 fi->readx(ibuf, ih_imagesize);
595
596 // get compressed data offset
597 unsigned e_len = ph.buf_offset + ph.getPackHeaderSize();
598 if (ih_imagesize <= e_len + ph.c_len)
599 throwCantUnpack("file damaged");
600
601 checkOverlay(ih_overlay);
602
603 // decompress
604 decompress(ibuf + e_len, obuf);
605
606 unsigned imagesize = ih_imagesize;
607 imagesize -= 1;
608 const byte flag = ibuf[imagesize];
609
610 // relocations
611 unsigned relocnum = 0;
612 SPAN_S_VAR(const byte, relocstart, obuf + ph.u_len, obuf);
613 MemBuffer mb_relocs;
614 SPAN_0_VAR(byte, relocs, nullptr);
615 if (!(flag & NORELOC)) {
616 mb_relocs.alloc(4 * MAXRELOCSIZE);
617 relocs = mb_relocs; // => now a SPAN_S
618
619 relocsize = get_le16(obuf + ph.u_len - 2);
620 ph.u_len -= 2;
621 if (relocsize < 11 || relocsize > MAXRELOCSIZE || relocsize >= imagesize)
622 throwCantUnpack("bad relocations");
623 relocstart -= relocsize;
624
625 // unoptimize_relocs
626 unsigned ones = get_le16(relocstart);
627 const unsigned seg_high = get_le16(relocstart + 2);
628 SPAN_S_VAR(const byte, p, relocstart + 4);
629 unsigned es = 0;
630 while (ones) {
631 unsigned di = get_le16(p);
632 es += get_le16(p + 2);
633 bool dorel = true;
634 for (p += 4; ones && di < 0x10000; p++) {
635 if (dorel) {
636 set_le16(relocs + (4 * relocnum + 0), di);
637 set_le16(relocs + (4 * relocnum + 2), es);
638 NO_printf("dos/exe unreloc %4d %6x\n", relocnum, es * 16 + di);
639 relocnum++;
640 }
641 dorel = true;
642 if (*p == 0) {
643 SPAN_S_VAR(const byte, q, obuf + (es * 16 + di), obuf);
644 while (!(*q == 0x9a && get_le16(q + 3) <= seg_high))
645 q++;
646 di = ptr_udiff_bytes(q, obuf + (es * 16)) + 3;
647 } else if (*p == 1) {
648 di += 254;
649 if (di < 0x10000)
650 ones--;
651 dorel = false;
652 } else
653 di += *p;
654 }
655 }
656 }
657
658 // fill new exe header
659 mem_clear(&oh);
660 oh.ident = 'M' + 'Z' * 256;
661
662 if (relocnum) {
663 oh.relocs = relocnum;
664 while (relocnum & 3) // paragraph align
665 set_le32(relocs + (4 * relocnum++), 0);
666 }
667
668 unsigned outputlen = sizeof(oh) + 4 * relocnum + ptr_udiff_bytes(relocstart, obuf);
669 oh.m512 = outputlen & 511;
670 oh.p512 = (outputlen + 511) >> 9;
671 oh.headsize16 = 2 + relocnum / 4;
672
673 oh.max = ih.max;
674 oh.min = ih.min;
675 oh.sp = ih.sp;
676 oh.ss = ih.ss;
677
678 if (flag & MAXMEM) {
679 imagesize -= 2;
680 oh.max = get_le16(ibuf + imagesize);
681 }
682 if (flag & MINMEM) {
683 imagesize -= 2;
684 oh.min = get_le16(ibuf + imagesize);
685 }
686 if (flag & SP) {
687 imagesize -= 2;
688 oh.sp = get_le16(ibuf + imagesize);
689 }
690 if (flag & SS) {
691 imagesize -= 2;
692 oh.ss = get_le16(ibuf + imagesize);
693 }
694
695 unsigned ip = (flag & USEJUMP) ? get_le32(ibuf + imagesize - 4) : (unsigned) ih.firstreloc;
696 oh.ip = ip & 0xffff;
697 oh.cs = ip >> 16;
698
699 oh.relocoffs = sizeof(oh);
700 oh.firstreloc = 0;
701 if (!fo)
702 return;
703
704 // write header + relocations + uncompressed file
705 fo->write(&oh, sizeof(oh));
706 if (relocnum)
707 fo->write(relocs, 4 * relocnum);
708 fo->write(obuf, ptr_udiff_bytes(relocstart, obuf));
709
710 // copy the overlay
711 copyOverlay(fo, ih_overlay, obuf);
712}

36// default: for any regular pointer, raw_bytes() is just the pointer itself
37template <class T>
38inline
39 typename std::enable_if<std::is_pointer<T>::value && !upx::is_bounded_array<T>::value, T>::type
40 raw_bytes(T ptr, size_t size_in_bytes) may_throw {
41 if (size_in_bytes > 0) {
42 if very_unlikely (ptr == nullptr)
43 throwCantPack("raw_bytes unexpected NULL ptr");
44 if very_unlikely (__acc_cte(VALGRIND_CHECK_MEM_IS_ADDRESSABLE(ptr, size_in_bytes) != 0))
45 throwCantPack("raw_bytes valgrind-check-mem");
46 }
47 return ptr;
48}

50// default: for any regular pointer, raw_index_bytes() is just "pointer + index"
51// NOTE: index == number of elements, *NOT* size in bytes!
52template <class T>
53inline typename std::enable_if<std::is_pointer<T>::value && !upx::is_bounded_array<T>::value &&
54 !std::is_void<typename std::remove_pointer<T>::type>::value,
55 T>::type
56raw_index_bytes(T ptr, size_t index, size_t size_in_bytes) may_throw {
57 typedef typename std::remove_pointer<T>::type element_type;
58 if very_unlikely (ptr == nullptr)
59 throwCantPack("raw_index_bytes unexpected NULL ptr");
60 size_in_bytes = mem_size(sizeof(element_type), index, size_in_bytes); // assert size
61 if very_unlikely (__acc_cte(VALGRIND_CHECK_MEM_IS_ADDRESSABLE(ptr, size_in_bytes) != 0))
62 throwCantPack("raw_index_bytes valgrind-check-mem");
63 UNUSED(size_in_bytes);
64 return ptr + index;
65}

67// same for bounded arrays
68template <class T, size_t N>
69inline T *raw_bytes(T (&array)[N], size_t size_in_bytes) may_throw {
70 typedef T element_type;
71 if very_unlikely (size_in_bytes > mem_size(sizeof(element_type), N))
72 throwCantPack("raw_bytes out of range");
73 return array;
74}

76template <class T, size_t N>
77inline T *raw_index_bytes(T (&array)[N], size_t index, size_t size_in_bytes) may_throw {
78 typedef T element_type;
79 return raw_bytes(array, mem_size(sizeof(element_type), index, size_in_bytes)) + index;
80}

90static bool set_eec(int ec, int *eec) {
91 if (ec == EXIT_FATAL) {
92 *eec = EXIT_ERROR;
93 return true;
94 }
95 if (ec < 0 || ec == EXIT_ERROR) {
96 *eec = EXIT_ERROR;
97 } else if (ec == EXIT_WARN) {
98 if (!opt->ignorewarn)
99 if (*eec == EXIT_OK)
100 *eec = ec;
101 } else if (ec == EXIT_OK) {
102 /* do nothing */
103 } else {
104 assert(0);
105 }
106 return false;
107}

166static void check_and_update_options(int i, int argc) {
167 assert(i <= argc);
168
169 if (opt->cmd != CMD_COMPRESS) {
170 // invalidate compression options
171 opt->method = 0;
172 opt->level = 0;
173 opt->exact = 0;
174 opt->small = 0;
175 opt->crp.reset();
176 }
177
178 // set default overlay action
179 if (!(opt->cmd == CMD_COMPRESS || opt->cmd == CMD_DECOMPRESS))
180 opt->overlay = opt->COPY_OVERLAY;
181 else if (opt->overlay < 0)
182 opt->overlay = opt->COPY_OVERLAY;
183
184 check_not_both(opt->exact, opt->overlay == opt->STRIP_OVERLAY, "--exact", "--overlay=strip");
185
186 // set default backup option
187 if (opt->backup < 0)
188 opt->backup = 0;
189 if (!(opt->cmd == CMD_COMPRESS || opt->cmd == CMD_DECOMPRESS))
190 opt->backup = 1;
191
192 check_not_both(opt->to_stdout, opt->output_name != nullptr, "--stdout", "-o");
193 if (opt->to_stdout && opt->cmd == CMD_COMPRESS) {
194 fprintf(stderr, "%s: cannot use '--stdout' when compressing\n", argv0);
195 e_usage();
196 }
197 if (opt->to_stdout || opt->output_name) {
198 if (i + 1 != argc) {
199 fprintf(stderr, "%s: need exactly one argument when using '%s'\n", argv0,
200 opt->to_stdout ? "--stdout" : "-o");
201 e_usage();
202 }
203 }
204 check_not_both(opt->force_overwrite, opt->preserve_link, "--force-overwrite", "--link");
205 check_not_both(opt->to_stdout, opt->preserve_link, "--stdout", "--link");
206
207#if defined(__unix__)
208 static_assert(HAVE_LSTAT);
209#else
210 // preserve_link is currently silently ignored on non-Unix platforms
211 // (we may revisit this decision later if there is some actual use-case)
212 opt->preserve_link = false;
213#endif
214}

237static bool set_method(int m, int l) {
238 if (m > 0) {
239 if (!Packer::isValidCompressionMethod(m))
240 return false;
241 // something like "--brute --lzma" should not disable "--brute"
242 if (!opt->all_methods)
243 opt->method = m;
244 }
245 if (l > 0)
246 opt->level = l;
247 set_cmd(CMD_COMPRESS);
248 return true;
249}

273static char *prepare_shortopts(char *buf, const char *n, const struct mfx_option *longopts) {
274 char *o = buf;
275
276 for (; n && *n; n++)
277 if (*n != ' ')
278 *o++ = *n;
279 *o = 0;
280 for (; longopts && longopts->name; longopts++) {
281 int v = longopts->val;
282#if !defined(NDEBUG)
283 assert(longopts->name[0] != '\0');
284 assert(longopts->name[0] != '-');
285 if (longopts->has_arg & 0x20)
286 assert((longopts->has_arg & 0xf) == 1);
287#endif
288#if 0
289 static char vopts[1024];
290 if (v > 0 && v < 1024)
291 {
292 if (vopts[v] && strchr(buf,v) == nullptr)
293 printf("warning: duplicate option %d ('%c')!\n", v, v & 127);
294 vopts[v] = 1;
295 }
296#endif
297 if (v > 0 && v < 256 && strchr(buf, v) == nullptr) {
298 *o++ = (char) v;
299 if ((longopts->has_arg & 0xf) >= 1)
300 *o++ = ':';
301 if ((longopts->has_arg & 0xf) >= 2)
302 *o++ = ':';
303 *o = 0;
304 }
305 if (longopts->has_arg & 0x20)
306 assert((longopts->has_arg & 0xf) == 1);
307 }
308 return buf;
309}

311template <class T>
312static int getoptvar(T *var, const T min_value, const T max_value, const char *arg_fatal) {
313 const char *p = mfx_optarg;
314 char *endptr = nullptr;
315 int r = 0;
316 long long n;
317 T v;
318
319 if (!p || !p[0]) {
320 r = -1;
321 goto error;
322 }
323 // avoid interpretation as octal value
324 while (p[0] == '0' && isdigit(p[1]))
325 p++;
326 n = strtoll(p, &endptr, 0);
327 if (*endptr != '\0') {
328 r = -2;
329 goto error;
330 }
331 v = (T) n;
332 if ((long long) v != n || v < min_value || v > max_value) {
333 r = -3;
334 goto error;
335 }
336 *var = v;
337 goto done;
338error:
339 if (arg_fatal != nullptr)
340 e_optval(arg_fatal);
341done:
342 return r;
343}
344
345template <class T, T default_value, T min_value, T max_value>
346static int getoptvar(upx::OptVar<T, default_value, min_value, max_value> *var,
347 const char *arg_fatal) {
348 T v = default_value;
349 int r = getoptvar(&v, min_value, max_value, arg_fatal);
350 if (r == 0)
351 *var = v;
352 return r;
353}

771 // printf("compress_icons: %d\n", opt->win32_pe.compress_icons);

773 case 632:

776 int value = 0;
777 getoptvar(&value, 0, 1, arg);
778 opt->win32_pe.compress_resources = bool(value);
779 }
780 // printf("compress_resources: %d\n", opt->win32_pe.compress_resources);
781 break;
782 case 633:
783 // opt->win32_pe.strip_loadconf - OBSOLETE - IGNORED
784 break;
785 case 634:
786 opt->win32_pe.strip_relocs = 1;
787 if (mfx_optarg && mfx_optarg[0])
788 getoptvar(&opt->win32_pe.strip_relocs, 0, 1, arg);
789 // printf("strip_relocs: %d\n", opt->win32_pe.strip_relocs);
790 break;
791 case 635:
792 if (!mfx_optarg || !mfx_optarg[0])
793 e_optarg(arg);
794 opt->win32_pe.keep_resource = mfx_optarg;
795 break;
796
797#if !defined(DOCTEST_CONFIG_DISABLE)
798 case 999: // [doctest] --dt-XXX option; ignored here, see upx_doctest_check()
799 break;
800#endif
801
802 case '\0':
803 return -1;
804 case ':':
805 return -2;
806 default:
807 fprintf(stderr, "%s: internal error in getopt (%d)\n", argv0, optc);
808 return -3;
809 }
810
811 UNUSED(i);
812 return 0;
813}

1195int upx_main(int argc, char *argv[]) may_throw {
1196 int i;
1197 static char default_argv0[] = "upx";
1198 assert(argc >= 1); // sanity check
1199 if (!argv[0] || !argv[0][0])
1200 argv[0] = default_argv0;
1201 argv0 = argv[0];
1202
1203 upx_compiler_sanity_check();
1204 int dt_res = upx_doctest_check(argc, argv);
1205 if (dt_res != 0) {
1206 if (dt_res == 2)
1207 fprintf(stderr, "%s: doctest requested program exit; Stop.\n", argv0);
1208 else
1209 fprintf(stderr, "%s: internal error: doctest check failed\n", argv0);
1210 e_exit(EXIT_INIT);
1211 }
1212
1213 // Allow serial re-use of upx_main() as a subroutine
1214 exit_code = EXIT_OK;
1215 opt->reset();
1216
1217#if (ACC_OS_CYGWIN || ACC_OS_DOS16 || ACC_OS_DOS32 || ACC_OS_EMX || ACC_OS_TOS || ACC_OS_WIN16 || \
1218 ACC_OS_WIN32 || ACC_OS_WIN64)
1219 {
1220 char *prog = fn_basename(argv0);
1221 char *p;
1222 bool allupper = true;
1223 for (p = prog; *p; p++)
1224 if (islower((uchar) *p))
1225 allupper = false;
1226 if (allupper)
1227 fn_strlwr(prog);
1228 if (p - prog > 4) {
1229 p -= 4;
1230 if (fn_strcmp(p, ".exe") == 0 || fn_strcmp(p, ".ttp") == 0)
1231 *p = 0;
1232 }
1233 progname = prog;
1234 }
1235#else
1236 progname = fn_basename(argv0);
1237#endif
1238 while (progname[0] == '.' && progname[1] == '/' && progname[2])
1239 progname += 2;
1240
1241 set_term(stderr);
1242
1243#if (WITH_BZIP2)
1244 assert(upx_bzip2_init() == 0);
1245#endif
1246 assert(upx_lzma_init() == 0);
1247#if (WITH_NRV)
1248 assert(upx_nrv_init() == 0);
1249#endif
1250 assert(upx_ucl_init() == 0);
1251#if (WITH_ZLIB)
1252 assert(upx_zlib_init() == 0);
1253#endif
1254#if (WITH_ZSTD)
1255 assert(upx_zstd_init() == 0);
1256#endif
1257

1259 first_options(argc, argv);
1260 if (!opt->no_env)
1261 main_get_envoptions();
1262 i = main_get_options(argc, argv);
1263 assert(i <= argc);
1264
1265 set_term(nullptr);
1266 switch (opt->cmd) {
1267 case CMD_NONE:

1292 break;
1293 case CMD_VERSION:

1295 e_exit(EXIT_OK);
1296 break;
1297 default:
1298 assert(false); // should not happen
1299 break;
1300 }

1305 set_term(stderr);

1308 if (num_files < 1) {

1312 e_usage();
1313 }

1318 assert(exit_code != 0);

1320 }

1330 con_fprintf(

1332 fg = con_fg(f, fg);

1334 }
1335 }

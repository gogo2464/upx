62PackMaster::PackMaster(InputFile *f, Options *o) noexcept : fi(f) {
63 // replace global options with local options
64 if (o != nullptr) {
65#if WITH_THREADS
66 // TODO later: check for possible "noexcept" violation here
67 std::lock_guard<std::mutex> lock(opt_lock_mutex);
68#endif

72 }
73}

85 }
86}

92static noinline tribool try_can_pack(PackerBase *pb, void *user) may_throw {
93 InputFile *f = (InputFile *) user;
94 try {
95 pb->initPackHeader();
96 f->seek(0, SEEK_SET);
97 tribool r = pb->canPack();
98 if (r) {
99 if (opt->cmd == CMD_COMPRESS)
100 pb->updatePackHeader();
101 f->seek(0, SEEK_SET);
102 return true; // success
103 }
104 if (r.isThird()) // aka "-1"
105 return r; // canPack() says the format is recognized and we should fail early
106 } catch (const IOException &) {
107 // ignored
108 }
109 return false;
110}

112static noinline tribool try_can_unpack(PackerBase *pb, void *user) may_throw {
113 InputFile *f = (InputFile *) user;
114 try {
115 pb->initPackHeader();
116 f->seek(0, SEEK_SET);
117 tribool r = pb->canUnpack();
118 if (r) {
119 f->seek(0, SEEK_SET);
120 return true; // success
121 }
122 if (r.isThird()) // aka "-1"
123 return r; // canUnpack() says the format is recognized and we should fail early
124 } catch (const IOException &) {
125 // ignored
126 }
127 return false;
128}

134/*static*/
135PackerBase *PackMaster::visitAllPackers(visit_func_t func, InputFile *f, const Options *o,
136 void *user) may_throw {
137#define VISIT(Klass) \
138 do { \
139 static_assert(std::is_class_v<Klass>); \
140 static_assert(std::is_nothrow_destructible_v<Klass>); \
141 auto pb = std::unique_ptr<PackerBase>(new Klass(f)); \
142 if (o->debug.debug_level) \
143 fprintf(stderr, "visitAllPackers: (ver=%d, fmt=%3d) %s\n", pb->getVersion(), \
144 pb->getFormat(), #Klass); \
145 pb->assertPacker(); \
146 tribool r = func(pb.get(), user); \
147 if (r) \
148 return pb.release(); /* success */ \
149 if (r.isThird()) \
150 return nullptr; /* stop and fail early */ \
151 } while (0)
152
153 // NOTE: order of tries is important !!!
154
155 //
156 // .exe
157 //
158 if (!o->dos_exe.force_stub) {
159 // dos32
160 VISIT(PackDjgpp2);
161 VISIT(PackTmt);
162 VISIT(PackWcle);
163 // Windows
164 // VISIT(PackW64PeArm64EC); // NOT YET IMPLEMENTED
165 // VISIT(PackW64PeArm64); // NOT YET IMPLEMENTED
166 VISIT(PackW64PeAmd64);
167 VISIT(PackW32PeI386);
168 VISIT(PackWinCeArm);
169 }
170 VISIT(PackExe); // dos/exe
171
172 //
173 // linux kernel
174 //
175 VISIT(PackVmlinuxARMEL);
176 VISIT(PackVmlinuxARMEB);
177 VISIT(PackVmlinuxPPC32);
178 VISIT(PackVmlinuxPPC64LE);
179 VISIT(PackVmlinuxAMD64);
180 VISIT(PackVmlinuxI386);
181#if (WITH_ZLIB)
182 VISIT(PackVmlinuzI386);
183 VISIT(PackBvmlinuzI386);
184 VISIT(PackVmlinuzARMEL);
185#endif
186
187 //
188 // linux
189 //
190 if (!o->o_unix.force_execve) {
191 if (o->o_unix.use_ptinterp) {
192 VISIT(PackLinuxElf32x86interp);
193 }
194 VISIT(PackFreeBSDElf32x86);
195 VISIT(PackNetBSDElf32x86);
196 VISIT(PackOpenBSDElf32x86);
197 VISIT(PackLinuxElf32x86);
198 VISIT(PackLinuxElf64amd);
199 VISIT(PackLinuxElf32armLe);
200 VISIT(PackLinuxElf32armBe);
201 VISIT(PackLinuxElf64arm);
202 VISIT(PackLinuxElf32ppc);
203 VISIT(PackLinuxElf64ppc);
204 VISIT(PackLinuxElf64ppcle);
205 VISIT(PackLinuxElf32mipsel);
206 VISIT(PackLinuxElf32mipseb);
207 VISIT(PackLinuxI386sh);
208 }
209 VISIT(PackBSDI386);
210 VISIT(PackMachFat); // cafebabe conflict
211 VISIT(PackLinuxI386); // cafebabe conflict
212
213 // Mach (Darwin / macOS)
214 VISIT(PackDylibAMD64);
215 VISIT(PackMachPPC32); // TODO: this works with upx 3.91..3.94 but got broken in 3.95; FIXME
216 VISIT(PackMachI386);
217 VISIT(PackMachAMD64);
218 VISIT(PackMachARMEL);
219 VISIT(PackMachARM64EL);
220
221 // 2010-03-12 omit these because PackMachBase<T>::pack4dylib (p_mach.cpp)
222 // does not understand what the Darwin (Apple Mac OS X) dynamic loader
223 // assumes about .dylib file structure.
224 // VISIT(PackDylibI386);
225 // VISIT(PackDylibPPC32);
226
227 //
228 // misc
229 //
230 VISIT(PackTos); // atari/tos
231 VISIT(PackPs1); // ps1/exe
232 VISIT(PackSys); // dos/sys
233 VISIT(PackCom); // dos/com
234
235 return nullptr;
236#undef VISIT
237}

239/*static*/ PackerBase *PackMaster::getPacker(InputFile *f) may_throw {
240 PackerBase *pb = visitAllPackers(try_can_pack, f, opt, f);
241 if (!pb)
242 throwUnknownExecutableFormat();
243 return pb;
244}

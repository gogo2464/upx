85 // prepare loader
86 initLoader(stub_i386_dos32_watcom_le, sizeof(stub_i386_dos32_watcom_le));
87 addLoader("IDENTSTR,WCLEMAIN", ph.first_offset_found == 1 ? "WCLEMAIN02" : "",

89
90 // fake alignment for the start of the decompressor
91 linker->defineSymbol("WCLECUTP", 0x1000);
92
93 addLoader(getDecompressorSections(), "WCLEMAI2");
94 if (ft->id) {
95 assert(ft->calls > 0);
96 addLoader(ft->addvalue ? "WCCTTPOS" : "WCCTTNUL");
97 addFilter32(ft->id);
98 }
99#if 1
100 // FIXME: if (has_relocation)
101 { addLoader("WCRELOC1,RELOC320", big_relocs ? "REL32BIG" : "", "RELOC32J"); }
102#endif
103 addLoader(has_extra_code ? "WCRELSEL" : "", "WCLEMAI4");
104}

128void PackWcle::encodeEntryTable() {
129 unsigned count, object, n;
130 byte *p = ientries;
131 n = 0;
132 while (*p) {
133 count = *p;
134 n += count;
135 if (p[1] == 0) // unused bundle
136 p += 2;
137 else if (p[1] == 3) // 32-bit bundle
138 {
139 object = get_le16(p + 2) - 1;
140 set_le16(p + 2, 1);
141 p += 4;
142 for (; count; count--, p += 5)
143 set_le32(p + 1, IOT(object, my_base_address) + get_le32(p + 1));
144 } else
145 throwCantPack("unsupported bundle type in entry table");
146 }
147
148 // if (Opt_debug) printf("%d entries encoded.\n",n);
149 UNUSED(n);
150
151 soentries = ptr_udiff_bytes(p, ientries) + 1;
152 oentries = ientries;
153 ientries = nullptr;
154}

156void PackWcle::readObjectTable() {
157 LeFile::readObjectTable();
158
159 // temporary copy of the object descriptors
160 if (!objects) // late detection, but protect against .alloc(0)
161 throwCantPack("no objects");
162 iobject_desc.alloc(objects * sizeof(*iobject_table));
163 memcpy(iobject_desc, iobject_table, objects * sizeof(*iobject_table));
164
165 unsigned ic, jc, virtual_size;
166
167 for (ic = jc = virtual_size = 0; ic < objects; ic++) {
168 jc += IOT(ic, npages);
169 IOT(ic, my_base_address) = virtual_size;
170 virtual_size += (IOT(ic, virtual_size) + mps - 1) & ~(mps - 1);
171 }
172 if (pages != jc)
173 throwCantPack("bad page number");
174}

176void PackWcle::encodeObjectTable() {
177 unsigned ic, jc;
178
179 soobject_table = 2;
180 oobject_table = New(le_object_table_entry_t, soobject_table);
181 memset(oobject_table, 0, mem_size(sizeof(*oobject_table), soobject_table));
182
183 // object #1:
184 OOT(0, base_address) = IOT(0, base_address);
185
186 ic = IOT(objects - 1, my_base_address) + IOT(objects - 1, virtual_size);
187 jc = mem_size(mps, pages, sofixups, 1024);
188 if (ic < jc)
189 ic = jc;
190
191 unsigned csection = (ic + ph.overlap_overhead + mps - 1) & ~(mps - 1);
192
193 OOT(0, virtual_size) = csection + mps;
194 OOT(0, flags) = LEOF_READ | LEOF_EXEC | LEOF_HUGE32 | LEOF_PRELOAD;
195 OOT(0, pagemap_index) = 1;
196 OOT(0, npages) = opages;
197
198 // object #2: stack
199 OOT(1, base_address) = (OOT(0, base_address) + OOT(0, virtual_size) + mps - 1) & ~(mps - 1);
200 OOT(1, virtual_size) = mps + getDecompressorWrkmemSize();
201 OOT(1, flags) = LEOF_READ | LEOF_HUGE32 | LEOF_WRITE;
202 OOT(1, pagemap_index) = 1;
203
204 oh.init_cs_object = 1;
205 oh.init_eip_offset = neweip;
206 oh.init_ss_object = 2;
207 oh.init_esp_offset = OOT(1, virtual_size);
208 oh.automatic_data_object = 2;
209}

228void PackWcle::encodeFixups() {
229 sofixups = 1 * 7 + FIXUP_EXTRA;
230 ofixups = New(byte, sofixups);
231 memset(ofixups, 0, sofixups);
232 ofixups[0] = 7;
233 set_le16(ofixups + 2, (LE_STUB_EDI + neweip) & (mps - 1));
234 ofixups[4] = 1;

237void PackWcle::preprocessFixups() {
238 big_relocs = 0;
239
240 unsigned ic, jc;
241
242 Array(unsigned, counts, objects + 2);
243 countFixups(counts);
244
245 for (ic = jc = 0; ic < objects; ic++)
246 jc += counts[ic];
247
248 if (jc == 0) {
249 // FIXME: implement this
250 throwCantPack("files without relocations are not supported");
251 }
252
253 MemBuffer mb_relocs(jc);
254 ByteArray(srf, counts[objects + 0] + 1);
255 ByteArray(slf, counts[objects + 1] + 1);
256
257 SPAN_S_VAR(byte, relocs, mb_relocs);
258 SPAN_S_VAR(byte, selector_fixups, srf_membuf);
259 SPAN_S_VAR(byte, selfrel_fixups, slf_membuf);
260 unsigned relocnum = 0;
261
262 byte *fix = ifixups;
263 for (ic = jc = 0; ic < pages; ic++) {
264 while (ptr_udiff_bytes(fix, ifixups) < get_le32(ifpage_table + (ic + 1))) {
265 const int fixp2 = get_le16_signed(fix + 2);
266 unsigned value;
267
268 switch (*fix) {
269 case 2: // selector fixup
270 if (fixp2 < 0) {
271 // cross page selector fixup
272 dputc('S', stdout);
273 fix += 5;
274 break;
275 }
276 dputc('s', stdout);
277 memcpy(selector_fixups, "\x8C\xCB\x66\x89\x9D",
278 5); // mov bx, cs ; mov [xxx+ebp], bx
279 if (IOT(fix[4] - 1, flags) & LEOF_WRITE)
280 selector_fixups[1] = 0xDB; // ds
281 set_le32(selector_fixups + 5, jc + fixp2);
282 selector_fixups += 9;
283 fix += 5;
284 break;
285 case 5: // 16-bit offset
286 if ((unsigned) fixp2 < 4096 && IOT(fix[4] - 1, my_base_address) == jc)
287 dputc('6', stdout);
288 else
289 throwCantPack("unsupported 16-bit offset relocation");
290 fix += (fix[1] & 0x10) ? 9 : 7;
291 break;
292 case 6: // 16:32 pointer
293 if (fixp2 < 0) {
294 // cross page pointer fixup
295 dputc('P', stdout);
296 fix += (fix[1] & 0x10) ? 9 : 7;
297 break;
298 }
299 dputc('p', stdout);
300 memcpy(iimage + jc + fixp2, fix + 5, (fix[1] & 0x10) ? 4 : 2);
301 set_le32(relocs + 4 * relocnum++, jc + fixp2);
302 set_le32(iimage + jc + fixp2,
303 get_le32(iimage + jc + fixp2) + IOT(fix[4] - 1, my_base_address));
304
305 memcpy(selector_fixups, "\x8C\xCA\x66\x89\x95", 5);
306 if (IOT(fix[4] - 1, flags) & LEOF_WRITE)
307 selector_fixups[1] = 0xDA; // ds
308 set_le32(selector_fixups + 5, jc + fixp2 + 4);
309 selector_fixups += 9;
310 fix += (fix[1] & 0x10) ? 9 : 7;
311 break;
312 case 7: // 32-bit offset
313 if (fixp2 < 0) {
314 fix += (fix[1] & 0x10) ? 9 : 7;
315 break;
316 }
317 // if (memcmp(iimage+jc+fixp2,fix+5,(fix[1] & 0x10) ? 4 : 2))
318 // throwCantPack("illegal fixup offset");
319
320 // work around a pmwunlite bug: remove duplicated fixups
321 // FIXME: fix the other cases too
322 if (relocnum == 0 || get_le32(relocs + 4 * relocnum - 4) != jc + fixp2) {
323 set_le32(relocs + 4 * relocnum++, jc + fixp2);
324 set_le32(iimage + jc + fixp2,
325 get_le32(iimage + jc + fixp2) + IOT(fix[4] - 1, my_base_address));
326 }
327 fix += (fix[1] & 0x10) ? 9 : 7;
328 break;
329 case 8: // 32-bit self relative fixup
330 if (fixp2 < 0) {
331 // cross page self relative fixup
332 dputc('R', stdout);
333 fix += (fix[1] & 0x10) ? 9 : 7;
334 break;
335 }
336 value = get_le32(fix + 5);
337 if (fix[1] == 0)
338 value &= 0xffff;
339 set_le32(iimage + jc + fixp2,
340 (value + IOT(fix[4] - 1, my_base_address)) - jc - fixp2 - 4);
341 set_le32(selfrel_fixups, jc + fixp2);
342 selfrel_fixups += 4;
343 dputc('r', stdout);
344 fix += (fix[1] & 0x10) ? 9 : 7;
345 break;
346 default:
347 throwCantPack("unsupported fixup record");
348 }
349 }
350 jc += mps;
351 }
352
353 // resize ifixups if it's too small
354 if (sofixups < 4 * relocnum + 8192) {
355 delete[] ifixups;
356 sofixups = 4 * relocnum + 8192;
357 ifixups = New(byte, sofixups);
358 }
359 SPAN_S_VAR(byte, orelocs, ifixups, sofixups);
360 fix =
361 ifixups + optimizeReloc(relocnum, relocs, orelocs, iimage, soimage, 32, true, &big_relocs);
362 has_extra_code = ptr_udiff_bytes(selector_fixups, srf) != 0;
363 // FIXME: this could be removed if has_extra_code = false
364 // but then we'll need a flag
365 *selector_fixups++ = 0xC3; // ret
366 memcpy(fix, srf, ptr_udiff_bytes(selector_fixups, srf)); // copy selector fixup code
367 fix += ptr_udiff_bytes(selector_fixups, srf);
368
369 memcpy(fix, slf, ptr_udiff_bytes(selfrel_fixups, slf)); // copy self-relative fixup positions
370 fix += ptr_udiff_bytes(selfrel_fixups, slf);
371 set_le32(fix, 0xFFFFFFFFUL);
372 fix += 4;
373
374 sofixups = ptr_udiff_bytes(fix, ifixups);
375}

377#define RESERVED 0x1000
378void PackWcle::encodeImage(Filter *ft) {
379 // concatenate image & preprocessed fixups
380 unsigned isize = soimage + sofixups;
381 ibuf.alloc(isize);
382 memcpy(ibuf, iimage, soimage);
383 memcpy(ibuf + soimage, ifixups, sofixups);
384
385 delete[] ifixups;
386 ifixups = nullptr;
387
388 mb_oimage.allocForCompression(isize, RESERVED + 512);
389 oimage = mb_oimage; // => now a SPAN_S
390 // prepare packheader
391 ph.u_len = isize;
392 // prepare filter [already done]
393 // compress
394 upx_compress_config_t cconf;
395 cconf.reset();
396 cconf.conf_lzma.max_num_probs = 1846 + (768 << 4); // ushort: ~28 KiB stack
397 compressWithFilters(ibuf, isize, raw_bytes(oimage + RESERVED, mb_oimage.getSize() - RESERVED),
398 ibuf + ft->addvalue, ft->buf_len, nullptr, 0, ft, 512, &cconf, 0);
399
400 ibuf.dealloc();
401 soimage = ph.c_len;
402 while (soimage & 3)
403 oimage[RESERVED + soimage++] = 0;
404}

406void PackWcle::pack(OutputFile *fo) {
407 handleStub(fo);
408
409 if (ih.byte_order || ih.word_order || ih.exe_format_level || ih.cpu_type < 2 ||
410 ih.cpu_type > 5 || ih.target_os != 1 || ih.module_type != 0x200 ||
411 ih.object_iterate_data_map_offset || ih.resource_entries || ih.module_directives_entries ||
412 ih.imported_modules_count || ih.object_table_entries > 255)
413 throwCantPack("watcom/le: unexpected value in header");
414
415 readObjectTable();
416 readPageMap();
417 readResidentNames();
418 readEntryTable();
419 readFixupPageTable();
420 readFixups();
421 readImage();
422 readNonResidentNames();
423
424 // if (find_le32(iimage,20,get_le32("UPX ")) >= 0)
425 if (find_le32(raw_bytes(iimage, soimage), UPX_MIN(soimage, 256u), UPX_MAGIC_LE32) >= 0)
426 throwAlreadyPacked();
427
428 if (ih.init_ss_object != objects)
429 throwCantPack("the stack is not in the last object");
430
431 preprocessFixups();
432
433 const unsigned text_size = mem_size(mps, IOT(ih.init_cs_object - 1, npages));
434 const unsigned text_vaddr = IOT(ih.init_cs_object - 1, my_base_address);
435
436 // attach some useful data at the end of preprocessed fixups
437 ifixups[sofixups++] = (byte) (ih.automatic_data_object & 0xff);
438 unsigned ic = objects * sizeof(*iobject_table);
439 memcpy(ifixups + sofixups, iobject_desc, ic);
440 iobject_desc.dealloc();
441
442 sofixups += ic;
443 set_le32(ifixups + sofixups,
444 ih.init_esp_offset + IOT(ih.init_ss_object - 1, my_base_address)); // old stack pointer
445 set_le32(ifixups + sofixups + 4, ih.init_eip_offset + text_vaddr); // real entry point
446 set_le32(ifixups + sofixups + 8,
447 mem_size(mps, pages)); // virtual address of unpacked relocations
448 ifixups[sofixups + 12] = (byte) (unsigned) objects;
449 sofixups += 13;
450
451 // prepare filter
452 Filter ft(ph.level);
453 ft.buf_len = text_size;
454 ft.addvalue = text_vaddr;
455 // compress
456 encodeImage(&ft);
457
458 const unsigned lsize = getLoaderSize();
459 neweip = getLoaderSection("WCLEMAIN");
460 int e_len = getLoaderSectionStart("WCLECUTP");
461 const unsigned d_len = lsize - e_len;
462 assert(e_len > 0 && e_len < RESERVED);
463
464 memmove(oimage + e_len, oimage + RESERVED, soimage);
465 soimage += lsize;
466
467 opages = (soimage + mps - 1) / mps;
468 oh.bytes_on_last_page = soimage % mps;
469
470 encodeObjectTable();
471 encodeFixups();
472 encodeFixupPageTable();
473 encodePageMap();
474 encodeEntryTable();
475
476 encodeResidentNames();
477 encodeNonResidentNames();
478
479 // patch loader
480 ic = (OOT(0, virtual_size) - d_len) & ~15;
481 assert(ic > ((ph.u_len + ph.overlap_overhead + 31) & ~15));
482
483 linker->defineSymbol("WCLECUTP", ic);
484
485 linker->defineSymbol("original_entry", ih.init_eip_offset + text_vaddr);
486 linker->defineSymbol("original_stack",
487 ih.init_esp_offset + IOT(ih.init_ss_object - 1, my_base_address));
488 linker->defineSymbol("start_of_relocs", mem_size(mps, pages));
489 defineDecompressorSymbols();
490 defineFilterSymbols(&ft);
491 linker->defineSymbol("filter_buffer_start", text_vaddr);
492
493 unsigned jpos = (((ph.c_len + 3) & ~3) + d_len + 3) / 4;
494 linker->defineSymbol("words_to_copy", jpos);
495 linker->defineSymbol("copy_dest", ((ic + d_len + 3) & ~3) - 4);
496 linker->defineSymbol("copy_source", e_len + jpos * 4 - 4);
497
498 relocateLoader();
499
500 MemBuffer loader(lsize);
501 memcpy(loader, getLoader(), lsize);
502 patchPackHeader(loader, lsize);
503
504 memcpy(oimage, loader, e_len);
505 memcpy(oimage + soimage - d_len, loader + e_len, d_len);
506
507 writeFile(fo, opt->watcom_le.le);
508
509 // verify
510 verifyOverlappingDecompression(mb_oimage + e_len, mb_oimage.getSize() - e_len);
511
512 // copy the overlay
513 const unsigned overlaystart = ih.data_pages_offset + exe_offset + getImageSize();
514 const unsigned overlay = file_size - overlaystart - ih.non_resident_name_table_length;
515 checkOverlay(overlay);
516 copyOverlay(fo, overlay, mb_oimage);
517
518 // finally check the compression ratio
519 if (!checkFinalCompressionRatio(fo))
520 throwNotCompressible();
521}

527void PackWcle::decodeFixups() {
528 mb_iimage.dealloc();
529 iimage = nullptr;
530
531 SPAN_S_VAR(const byte, p, oimage + soimage);
532 MemBuffer mb_relocs;
533 const unsigned fixupn = unoptimizeReloc(p, mb_relocs, oimage, soimage, 32, true);
534
535 MemBuffer wrkmem(8 * fixupn + 8);
536 unsigned ic, jc, o, r;
537 for (ic = 0; ic < fixupn; ic++) {
538 jc = get_le32(mb_relocs + 4 * ic);
539 set_le32(wrkmem + ic * 8, jc);
540 o = soobject_table;
541 r = get_le32(oimage + jc);
542 virt2rela(oobject_table, &o, &r);
543 set_le32(wrkmem + ic * 8 + 4, OOT(o - 1, my_base_address));
544 set_le32(oimage + jc, r);
545 }
546 set_le32(wrkmem + ic * 8, 0xFFFFFFFF); // end of 32-bit offset fixups
547 mb_relocs.dealloc(); // done
548
549 // selector fixups then self-relative fixups
550 SPAN_S_VAR(const byte, selector_fixups, p);
551
552 // Find selfrel_fixups by skipping over selector_fixups.
553 SPAN_S_VAR(const byte, q, selector_fixups);
554 // The code is a subroutine that ends in RET (0xC3).
555 while (*q != 0xC3) {
556 // Defend against tampered selector_fixups; see PackWcle::preprocessFixups().
557 // selector_fixups[] is x386 code with 9-byte blocks of 2 instructions each:
558 // "\x8C\xCB\x66\x89\x9D" // mov bx, cs ; mov [xxx+ebp], bx
559 // "\x8C\xCA\x66\x89\x95"
560 // and where byte [+1] also can be '\xDA' or '\xDB'.
561 if (0x8C != q[0] || 0x66 != q[2] || 0x89 != q[3]) { // Unexpected; tampering?
562 // Try to recover by looking for the RET.
563 const byte *q2 = (const byte *) memchr(q, 0xC3, 9);
564 if (q2) { // Assume recovery
565 q = q2;
566 break;
567 }
568 }
569 // Guard against run-away.
570 static byte const blank[9] = {0};
571 // catastrophic worst case or no-good early warning
572 if (ptr_diff_bytes(oimage + ph.u_len - sizeof(blank), raw_bytes(q, 0)) < 0 ||
573 !memcmp(blank, q, sizeof(blank))) {
574 char msg[50];
575 snprintf(msg, sizeof(msg), "bad selector_fixups %d",
576 ptr_diff_bytes(q, selector_fixups));
577 throwCantPack(msg);
578 }
579 q += 9;
580 }
581 unsigned selectlen = ptr_udiff_bytes(q, selector_fixups) / 9;
582 SPAN_S_VAR(const byte, selfrel_fixups, q + 1); // Skip the 0xC3
583
584 const unsigned fbytes = fixupn * 9 + 1000 + selectlen * 5;
585 ofixups = New(byte, fbytes);
586 SPAN_S_VAR(byte, fp, ofixups, fbytes, ofixups);
587
588 for (ic = 1, jc = 0; ic <= opages; ic++) {
589 // self relative fixups
590 while ((r = get_le32(selfrel_fixups)) / mps == ic - 1) {
591 fp[0] = 8;
592 set_le16(fp + 2, r & (mps - 1));
593 o = 4 + get_le32(oimage + r);
594 set_le32(oimage + r, 0);
595 r += o;
596 o = soobject_table;
597 virt2rela(oobject_table, &o, &r);
598 fp[4] = (byte) o;
599 set_le32(fp + 5, r);
600 fp[1] = (byte) (r > 0xFFFF ? 0x10 : 0);
601 fp += fp[1] ? 9 : 7;
602 selfrel_fixups += 4;
603 dputc('r', stdout);
604 }
605 // selector fixups
606 while (selectlen && (r = get_le32(selector_fixups + 5)) / mps == ic - 1) {
607 fp[0] = 2;
608 fp[1] = 0;
609 set_le16(fp + 2, r & (mps - 1));
610 unsigned x = selector_fixups[1] > 0xD0 ? oh.init_ss_object : oh.init_cs_object;
611 fp[4] = (byte) x;
612 fp += 5;
613 selector_fixups += 9;
614 selectlen--;
615 dputc('s', stdout);
616 }
617 // 32 bit offset fixups
618 while (get_le32(wrkmem + 4 * jc) < mem_size(mps, ic)) {
619 if (jc > 1 &&
620 ((get_le32(wrkmem + 4 * (jc - 2)) + 3) & (mps - 1)) < 3) // cross page fixup?
621 {
622 r = get_le32(oimage + get_le32(wrkmem + 4 * (jc - 2)));
623 fp[0] = 7;
624 fp[1] = (byte) (r > 0xFFFF ? 0x10 : 0);
625 set_le16(fp + 2, get_le32(wrkmem + 4 * (jc - 2)) | ~3);
626 set_le32(fp + 5, r);
627 o = soobject_table;
628 r = get_le32(wrkmem + 4 * (jc - 1));
629 virt2rela(oobject_table, &o, &r);
630 fp[4] = (byte) o;
631 fp += fp[1] ? 9 : 7;
632 dputc('0', stdout);
633 }
634 o = soobject_table;
635 r = get_le32(wrkmem + 4 * (jc + 1));
636 virt2rela(oobject_table, &o, &r);
637 r = get_le32(oimage + get_le32(wrkmem + 4 * jc));
638 fp[0] = 7;
639 fp[1] = (byte) (r > 0xFFFF ? 0x10 : 0);
640 set_le16(fp + 2, get_le32(wrkmem + 4 * jc) & (mps - 1));
641 fp[4] = (byte) o;
642 set_le32(fp + 5, r);
643 fp += fp[1] ? 9 : 7;
644 jc += 2;
645 }
646 set_le32(ofpage_table + ic, ptr_udiff_bytes(fp, ofixups));
647 }
648 for (ic = 0; ic < FIXUP_EXTRA; ic++)
649 *fp++ = 0;
650 sofixups = ptr_udiff_bytes(fp, ofixups);
651}

659void PackWcle::decodeObjectTable() {
660 soobject_table = oimage[ph.u_len - 1];
661 oobject_table = New(le_object_table_entry_t, soobject_table);
662 unsigned jc, ic = soobject_table * sizeof(*oobject_table);
663
664 const unsigned extradata = ph.version == 10 ? 17 : 13;
665 memcpy(oobject_table, oimage + ph.u_len - extradata - ic, ic);
666 if (ph.version >= 12)
667 oh.automatic_data_object = oimage[ph.u_len - ic - 14];
668
669 for (ic = jc = 0; ic < soobject_table; ic++) {
670 OOT(ic, my_base_address) = jc;
671 jc += (OOT(ic, virtual_size) + mps - 1) & ~(mps - 1);
672 }
673
674 // restore original cs:eip & ss:esp
675 ic = soobject_table;
676 jc = get_le32(oimage + ph.u_len - (ph.version < 11 ? 13 : 9));
677 virt2rela(oobject_table, &ic, &jc);
678 oh.init_cs_object = ic;
679 oh.init_eip_offset = jc;
680
681 ic = soobject_table;
682 if (ph.version < 10)
683 jc = ih.init_esp_offset;
684 else
685 jc = get_le32(oimage + ph.u_len - (ph.version == 10 ? 17 : 13));
686 virt2rela(oobject_table, &ic, &jc);
687 oh.init_ss_object = ic;
688 oh.init_esp_offset = jc;
689}

693 oimage = mb_oimage; // => now a SPAN_S
694
695 decompress(iimage + ph.buf_offset + ph.getPackHeaderSize(), oimage);
696 soimage = get_le32(oimage + ph.u_len - 5);

698 oh.memory_page_size = mps;

701void PackWcle::decodeEntryTable() {
702 unsigned count, object, n, r;
703 SPAN_S_VAR(byte, p, ientries, soentries);
704 n = 0;
705 while (*p) {
706 count = *p;
707 n += count;
708 if (p[1] == 0) // unused bundle
709 p += 2;
710 else if (p[1] == 3) // 32-bit offset bundle
711 {
712 object = get_le16(p + 2);
713 if (object != 1)
714 throwCantUnpack("corrupted entry found");
715 object = soobject_table;
716 r = get_le32(p + 5);
717 virt2rela(oobject_table, &object, &r);
718 set_le16(p + 2, object--);
719 p += 4;
720 for (; count; count--, p += 5)
721 set_le32(p + 1, get_le32(p + 1) - OOT(object, my_base_address));
722 } else
723 throwCantUnpack("unsupported bundle type in entry table");
724 }
725
726 // if (Opt_debug) printf("\n%d entries decoded.\n",n);
727 UNUSED(n);
728
729 soentries = ptr_udiff_bytes(p, ientries) + 1;
730 oentries = ientries;
731 ientries = nullptr;
732}

795

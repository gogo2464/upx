36/*static*/
37unsigned Packer::optimizeReloc(unsigned relocnum, SPAN_P(byte) relocs, SPAN_S(byte) out,
38 SPAN_P(byte) image, unsigned image_size, int bits, bool bswap,
39 int *big) {
40 assert(bits == 32 || bits == 64);
41 mem_size_assert(1, image_size);
42#if WITH_XSPAN >= 2
43 ptr_check_no_overlap(relocs.data(), relocs.size_bytes(), image.data(image_size), image_size,
44 out.data(), out.size_bytes());
45#endif
46 SPAN_S_VAR(byte, fix, out);
47
48 *big = 0;
49 if (opt->exact)
50 throwCantPackExact();
51 if (relocnum == 0)
52 return 0;
53 upx_qsort(raw_bytes(relocs, 4 * relocnum), relocnum, 4, le32_compare);
54 if (0) {
55 printf("optimizeReloc: u_reloc %9u checksum=0x%08x\n", 4 * relocnum,
56 upx_adler32(relocs, 4 * relocnum));
57 printf("optimizeReloc: u_image %9u checksum=0x%08x\n", image_size,
58 upx_adler32(image, image_size));
59 }
60
61 unsigned pc = (unsigned) -4;
62 for (unsigned i = 0; i < relocnum; i++) {
63 unsigned delta = get_le32(relocs + i * 4) - pc;
64 if (delta == 0)
65 continue;
66 else if ((int) delta < 4)
67 throwCantPack("overlapping fixups");
68 else if (delta < 0xf0)
69 *fix++ = (byte) delta;
70 else if (delta < 0x100000) {
71 *fix++ = (byte) (0xf0 + (delta >> 16));
72 *fix++ = (byte) delta;
73 *fix++ = (byte) (delta >> 8);
74 } else {
75 *big = 1;
76 *fix++ = 0xf0;
77 *fix++ = 0;
78 *fix++ = 0;
79 set_le32(fix, delta);
80 fix += 4;
81 }
82 pc += delta;
83 if (pc + 4 > image_size)
84 throwCantPack("bad reloc[%#x] = %#x", i, pc);
85 if (bswap) {
86 if (bits == 32)
87 set_be32(image + pc, get_le32(image + pc));
88 else
89 set_be64(image + pc, get_le64(image + pc));
90 }
91 }
92 *fix++ = 0; // end marker
93 const unsigned bytes = ptr_udiff_bytes(fix, out);
94 if (0) {
95 printf("optimizeReloc: c_reloc %9u checksum=0x%08x\n", bytes, upx_adler32(out, bytes));
96 printf("optimizeReloc: c_image %9u checksum=0x%08x\n", image_size,
97 upx_adler32(image, image_size));
98 }
99 return bytes;
100}

109unsigned Packer::unoptimizeReloc(SPAN_S(const byte) & in, MemBuffer &out, SPAN_P(byte) image,
110 unsigned image_size, int bits, bool bswap) {
111 assert(bits == 32 || bits == 64);
112 mem_size_assert(1, image_size);
113#if WITH_XSPAN >= 2
114 ptr_check_no_overlap(in.data(), in.size_bytes(), image.data(image_size), image_size);
115#endif
116 SPAN_S_VAR(const byte, fix, in);
117
118 // count
119 unsigned relocnum = 0;
120 for (fix = in; *fix; fix++, relocnum++) {
121 if (*fix >= 0xf0) {
122 if (*fix == 0xf0 && get_le16(fix + 1) == 0)
123 fix += 4;
124 fix += 2;
125 }
126 }
127 NO_fprintf(stderr, "relocnum=%x\n", relocnum);
128 if (0) {
129 const unsigned bytes = ptr_udiff_bytes(fix + 1, in);
130 printf("unoptimizeReloc: c_reloc %9u checksum=0x%08x\n", bytes, upx_adler32(in, bytes));
131 printf("unoptimizeReloc: c_image %9u checksum=0x%08x\n", image_size,
132 upx_adler32(image, image_size));
133 }
134
135 out.alloc(mem_size(4, relocnum + 1)); // one extra entry
136 SPAN_S_VAR(LE32, relocs, out);
137
138 fix = in;
139 unsigned pc = (unsigned) -4;
140 for (unsigned i = 0; i < relocnum; i++) {
141 unsigned delta;
142 if (*fix < 0xf0)
143 delta = *fix++;
144 else {
145 delta = (*fix & 0x0f) * 0x10000 + get_le16(fix + 1);
146 fix += 3;
147 if (delta == 0) {
148 delta = get_le32(fix);
149 fix += 4;
150 }
151 }
152 if ((int) delta < 4)
153 throwCantUnpack("overlapping fixups");
154 pc += delta;
155 if (pc + 4 > image_size)
156 throwCantUnpack("bad reloc[%#x] = %#x", i, pc);
157 *relocs++ = pc;
158 if (bswap) {
159 if (bits == 32)
160 set_be32(image + pc, get_le32(image + pc));
161 else
162 set_be64(image + pc, get_le64(image + pc));
163 }
164 }
165 in = fix + 1; // advance
166 assert(relocnum == ptr_udiff_bytes(relocs, raw_bytes(out, 0)) / 4);
167 if (0) {
168 printf("unoptimizeReloc: u_reloc %9u checksum=0x%08x\n", 4 * relocnum,
169 upx_adler32(out, 4 * relocnum));
170 printf("unoptimizeReloc: u_image %9u checksum=0x%08x\n", image_size,
171 upx_adler32(image, image_size));
172 }
173 return relocnum;
174}

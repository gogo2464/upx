156int PeFile::readFileHeader() {
157 struct alignas(1) ExeHeader final {
158 LE16 mz;
159 LE16 m512;
160 LE16 p512;
161 char _[18];
162 LE16 relocoffs;
163 char __[34];
164 LE32 nexepos;
165 };
166
167 COMPILE_TIME_ASSERT(sizeof(ExeHeader) == 64)
168 COMPILE_TIME_ASSERT_ALIGNED1(ExeHeader)
169 COMPILE_TIME_ASSERT(sizeof(((ExeHeader *) nullptr)->_) == 18)
170 COMPILE_TIME_ASSERT(sizeof(((ExeHeader *) nullptr)->__) == 34)
171
172 ExeHeader h;
173 int ic;
174 pe_offset = 0;
175
176 for (ic = 0; ic < 20; ic++) {
177 fi->seek(pe_offset, SEEK_SET);
178 fi->readx(&h, sizeof(h));
179
180 if (h.mz == 'M' + 'Z' * 256) // dos exe
181 {
182 if (h.nexepos && h.nexepos < sizeof(ExeHeader)) {
183 // Overlapping MZ and PE headers by 'leanify', etc.
184 char buf[64];
185 snprintf(buf, sizeof(buf), "PE and MZ header overlap: %#x < %#x",
186 (unsigned) h.nexepos, (unsigned) sizeof(ExeHeader));
187 throwCantPack(buf);
188 }
189 const unsigned delta = (h.relocoffs >= 0x40)
190 ? h.nexepos // new format exe
191 : (h.p512 * 512 + h.m512 - h.m512 ? 512 : h.nexepos);
192
193 if ((pe_offset + delta) < delta // wrap-around
194 || (pe_offset + delta) > file_size_u) {
195 char buf[64];
196 snprintf(buf, sizeof(buf), "bad PE delta %#x at offset %#x", delta, pe_offset);
197 throwCantPack(buf);
198 }
199 pe_offset += delta;
200 } else if (get_le32((const byte *) &h) == 'P' + 'E' * 256)
201 break;
202 else
203 return 0;
204 }
205 if (ic == 20)
206 return 0;
207 fi->seek(pe_offset, SEEK_SET);
208 readPeHeader();
209 return getFormat();
210}

301static constexpr size_t RELOC_ENTRY_SIZE = 5; // encoded size in bytes; actual encoding is private
302static void reloc_entry_encode(SPAN_P(byte) buf, unsigned pos, unsigned reloc_type) {
303 if (reloc_type == 0 || reloc_type >= 16)
304 throwCantPack("bad reloc_type %#x %u", pos, reloc_type);
305 set_ne32(buf, pos);
306 buf[4] = (upx_uint8_t) reloc_type;
307}
308static void reloc_entry_decode(SPAN_P(const byte) buf, unsigned *pos, unsigned *reloc_type) {
309 *pos = get_ne32(buf);
310 *reloc_type = buf[4];
311 assert(*reloc_type > 0 && *reloc_type < 16);
312}
313static int __acc_cdecl_qsort reloc_entry_compare(const void *a, const void *b) {
314 const unsigned pos1 = get_ne32(a);
315 const unsigned pos2 = get_ne32(b);
316 if (pos1 != pos2)
317 return pos1 < pos2 ? -1 : 1;
318 const unsigned reloc_type1 = ((const upx_uint8_t *) a)[4];
319 const unsigned reloc_type2 = ((const upx_uint8_t *) b)[4];
320 if (reloc_type1 != reloc_type2)
321 return reloc_type1 < reloc_type2 ? -1 : 1;
322 return 0;
323}

358// explicitly check values so that we get better error messages (instead of a cryptic SPAN failure)
359bool PeFile::Reloc::readFromRelocationBlock(byte *next_rb) { // set rb
360 assert(!start_did_alloc);
361 const unsigned off = ptr_udiff_bytes(next_rb, start);
362 assert((off & 1) == 0);
363 rb.reset();
364 if (off >= start_size_in_bytes) { // permissive: use ">=" instead of strict "=="
365 if (off > start_size_in_bytes) {
366 // MAYBE TODO later: could add a warning here?
367 }
368 return false; // EOF
369 }
370 if (start_size_in_bytes - off < 8)
371 throwCantPack("relocs overflow");
372 const unsigned sob = get_le32(start_buf + (off + 4)); // size_of_block
373#if 1
374 // ignore a dubious single empty relocation block with sob == 0
375 if (sob == 0 && (off == 0 && start_size_in_bytes == 8))
376 return false; // EOF
377#endif
378 if (ALWAYS_CHECK_STRICT_RELOCS) {
379 if (sob < 8)
380 throwCantPack("bad reloc size_of_block %u", sob);
381 if (start_size_in_bytes - off < sob)
382 throwCantPack("overflow reloc size_of_block %u", sob);
383 if ((sob & 1) != 0)
384 throwCantPack("odd reloc size_of_block %u", sob);
385 } else if (!opt->force) {
386 if (sob < 8)
387 throwCantPack("bad reloc size_of_block %u (try --force)", sob);
388 if (start_size_in_bytes - off < sob)
389 throwCantPack("overflow reloc size_of_block %u (try --force)", sob);
390 if ((sob & 1) != 0)
391 throwCantPack("odd reloc size_of_block %u (try --force)", sob);
392 }
393 // success
394 rb.rel = (BaseReloc *) next_rb; // SPAN checked
395 rb.rel1 = (LE16 *) (next_rb + 8); // SPAN checked
396 rb.count = sob < 8 ? 0 : (sob - 8) / sizeof(LE16);
397 return true;
398}

400bool PeFile::Reloc::next(unsigned &result_pos, unsigned &result_reloc_type) {
401 assert(!start_did_alloc);
402 for (;;) {
403 // search current block
404 while (rb.count > 0) {
405 rb.count -= 1;
406 const unsigned value = *rb.rel1++;
407 result_pos = rb.rel->virtual_address + (value & 0xfff);
408 result_reloc_type = (value >> 12) & 0xf;
409 NO_printf("Reloc::next %#x %d\n", result_pos, result_reloc_type);
410 if (result_reloc_type != IMAGE_REL_BASED_IGNORE)
411 return true; // success
412 }
413 // advance to next block
414 byte *next_rb = (rb.rel == nullptr) ? start : (byte *) raw_bytes(rb.rel1, 0);
415 if (!readFromRelocationBlock(next_rb)) {
416 rb.reset(); // rewind
417 return false; // EOF
418 }
419 }
420}

432void PeFile::Reloc::finish(byte *(&result_ptr), unsigned &result_size) {
433 assert(start_did_alloc);
434 // sort in-place relocs
435 upx_qsort(
436 raw_index_bytes(start_buf, RELOC_INPLACE_OFFSET, mem_size(RELOC_ENTRY_SIZE, counts[0])),
437 counts[0], RELOC_ENTRY_SIZE, reloc_entry_compare);
438
439 auto finish_block = [](SPAN_S(BaseReloc) rel) -> byte * {
440 unsigned sob = rel->size_of_block;
441 assert(sob >= 10 && (sob & 1) == 0);
442 auto end = SPAN_TYPE_CAST(byte, rel) + sob;
443 while ((sob & 3) != 0) { // UPX: we want align by 4 here
444 *end++ = 0; // clear byte (i.e. write IMAGE_REL_BASED_IGNORE)
445 sob += 1;
446 }
447 rel->size_of_block = sob;
448 return raw_bytes(end, 0);
449 };
450
451 rb.reset();
452 unsigned prev_pos = 0;
453 unsigned current_page = 0;
454 for (unsigned ic = 0; ic < counts[0]; ic++) {
455 const auto entry_ptr = start_buf + mem_size(RELOC_ENTRY_SIZE, ic, RELOC_INPLACE_OFFSET);
456 unsigned pos, reloc_type;
457 reloc_entry_decode(entry_ptr, &pos, &reloc_type);
458 if (ic > 0 && pos == prev_pos) {
459 if (ALWAYS_CHECK_STRICT_RELOCS)
460 throwCantPack("duplicate relocs");
461 else if (!opt->force)
462 throwCantPack("duplicate relocs (try --force)");
463 }
464 prev_pos = pos;
465 if (ic == 0 || pos - current_page >= 0x1000) {
466 current_page = pos & ~0xfff; // page start
467 // prepare next block for writing
468 byte *next_rb = (rb.rel == nullptr) ? start : finish_block(rb.rel);
469 rb.rel = (BaseReloc *) next_rb;
470 rb.rel1 = (LE16 *) (next_rb + 8);
471 rb.rel->virtual_address = current_page;
472 rb.rel->size_of_block = 8;
473 }
474 // check for in-place overflow
475 if (ptr_diff_bytes(rb.rel1, entry_ptr) >= 0) {
476 // info: if this is indeed a valid file we must increase RELOC_INPLACE_OFFSET
477 throwCantPack("too many inplace relocs");
478 }
479 // write LE16 IMAGE_BASE_RELOCATION relocation
480 *rb.rel1++ = (reloc_type << 12) | (pos & 0xfff);
481 rb.rel->size_of_block += 2;
482 }
483 result_size = 0; // result_size can be 0 in 64-bit mode
484 if (rb.rel != nullptr)
485 result_size = ptr_udiff_bytes(finish_block(rb.rel), start);
486 assert((result_size & 3) == 0);
487 // transfer ownership
488 assert(start_did_alloc);
489 result_ptr = start;
490 start_did_alloc = false;
491#if DEBUG || 1 // safety, as we are really finished
492 ptr_invalidate_and_poison(start);
493 start_size_in_bytes = 0;
494 SPAN_INVALIDATE(start_buf);
495 SPAN_INVALIDATE(rb.rel);
496 SPAN_INVALIDATE(rb.rel1);
497 rb.count = 0xdeaddead;
498#endif
499}

501void PeFile32::processRelocs() // pass1
502{
503 big_relocs = 0;
504
505 const unsigned skip1 = IDADDR(PEDIR_BASERELOC);
506 const unsigned take1 = IDSIZE(PEDIR_BASERELOC);
507 Reloc rel(ibuf.subref("bad reloc %#x", skip1, take1), take1);
508 const unsigned *const counts = rel.getcounts();
509 unsigned relocnum = 0;
510
511 for (unsigned ic = 1; ic < 16; ic++)
512 relocnum += counts[ic];
513 for (unsigned ic = 0; ic < 16; ic++)
514 NO_printf("reloc counts[%u] %u\n", ic, counts[ic]);
515
516 if (opt->win32_pe.strip_relocs || relocnum == 0) {
517 if (IDSIZE(PEDIR_BASERELOC)) {
518 ibuf.fill(IDADDR(PEDIR_BASERELOC), IDSIZE(PEDIR_BASERELOC), FILLVAL);
519 ih.objects = tryremove(IDADDR(PEDIR_BASERELOC), ih.objects);
520 }
521 mb_orelocs.alloc(1);
522 mb_orelocs.clear();
523 orelocs = mb_orelocs; // => orelocs now is a SPAN_S
524 sorelocs = 0;
525 return;
526 }
527
528 for (unsigned ic = IMAGE_REL_BASED_HIGHADJ; ic < 16; ic++)
529 if (counts[ic])
530 infoWarning("skipping unsupported relocation type %d (%d)", ic, counts[ic]);
531
532 LE32 *fix[4];
533 auto fix_deleter = upx::ArrayDeleter(fix, 0); // don't leak memory
534 for (unsigned ic = 0; ic <= IMAGE_REL_BASED_HIGHLOW; ic++) {
535 fix[ic] = New(LE32, counts[ic]);
536 fix_deleter.count += 1;
537 }
538
539 unsigned xcounts[4];
540 memset(xcounts, 0, sizeof(xcounts));
541
542 // prepare sorting
543 unsigned pos, reloc_type;
544 while (rel.next(pos, reloc_type)) {
545 // FIXME add check for relocations which try to modify the
546 // PE header or other relocation records
547 if (pos >= ih.imagesize)
548 continue; // skip out-of-bounds record
549 if (reloc_type <= IMAGE_REL_BASED_HIGHLOW)
550 fix[reloc_type][xcounts[reloc_type]++] = pos - rvamin;
551 }
552
553 // remove duplicated records
554 for (unsigned ic = 1; ic <= IMAGE_REL_BASED_HIGHLOW; ic++) {
555 upx_qsort(fix[ic], xcounts[ic], 4, le32_compare);
556 unsigned prev = ~0u;
557 unsigned jc = 0;
558 for (unsigned kc = 0; kc < xcounts[ic]; kc++)
559 if (fix[ic][kc] != prev)
560 prev = fix[ic][jc++] = fix[ic][kc];
561
562 NO_printf("reloc xcounts[%u] %u->%u\n", ic, xcounts[ic], jc);
563 xcounts[ic] = jc;
564 }
565
566 // preprocess "type 3" relocation records
567 for (unsigned ic = 0; ic < xcounts[IMAGE_REL_BASED_HIGHLOW]; ic++) {
568 pos = fix[3][ic] + rvamin;
569 unsigned w = get_le32(ibuf.subref("bad reloc type 3 %#x", pos, sizeof(LE32)));
570 set_le32(ibuf + pos, w - ih.imagebase - rvamin);
571 }
572
573 ibuf.fill(IDADDR(PEDIR_BASERELOC), IDSIZE(PEDIR_BASERELOC), FILLVAL);
574 mb_orelocs.alloc(mem_size(4, relocnum, 8192)); // 8192 - safety
575 orelocs = mb_orelocs; // => orelocs now is a SPAN_S
576 sorelocs = optimizeReloc(xcounts[3], (byte *) fix[3], orelocs, ibuf + rvamin, ibufgood - rvamin,
577 32, true, &big_relocs);
578
579 // Malware that hides behind UPX often has PE header info that is
580 // deliberately corrupt. Sometimes it is even tuned to cause us trouble!
581 // Use an extra check to avoid AccessViolation (SIGSEGV) when appending
582 // the relocs into one array.
583 if ((4 * relocnum + 8192) <
584 (sorelocs + 4 * (2 + xcounts[IMAGE_REL_BASED_LOW] + xcounts[IMAGE_REL_BASED_HIGH])))
585 throwCantUnpack("Invalid relocs");
586
587 // append relocs type "LOW" then "HIGH"
588 for (unsigned ic = IMAGE_REL_BASED_LOW; ic >= IMAGE_REL_BASED_HIGH; ic--) {
589 memcpy(orelocs + sorelocs, fix[ic], 4 * xcounts[ic]);
590 sorelocs += 4 * xcounts[ic];
591 set_le32(orelocs + sorelocs, 0);
592 if (xcounts[ic]) {
593 sorelocs += 4;
594 big_relocs |= 2 * ic;
595 }
596 }

696/*
697 ImportLinker: 32 and 64 bit import table building.
698 Import entries (dll name + proc name/ordinal pairs) can be
699 added in arbitrary order.
700
701 Internally it works by creating sections with special names,
702 and adding relocation entries between those sections. The special
703 names ensure that when the import table is built in the memory
704 from those sections, a correct table can be generated simply by
705 sorting the sections by name, and adding all of them to the output
706 in the sorted order.
707 */

709class PeFile::ImportLinker final : public ElfLinkerAMD64 {
710 // temporary string owner, deletes on destruction
711 struct TStr final : private upx::noncopyable {
712 explicit TStr(char *str) noexcept : s(str) {}
713 ~TStr() noexcept { delete[] s; } // delete!
714 operator char *() noexcept { return s; }
715 operator const char *() const noexcept { return s; }
716 private:
717 char *s;
718 };

720 // encoding of dll and proc names are required, so that our special
721 // control characters in the name of sections can work as intended
722 static void encode_name(SPAN_P(const char) name, SPAN_S(char) buf) {
723 while (*name) {
724 *buf++ = 'a' + ((*name >> 4) & 0xf);
725 *buf++ = 'a' + (*name & 0xf);
726 name++;
727 }
728 *buf = 0;
729 }

731 static char *name_for_dll(const char *dll, char first_char) {
732 assert(dll != nullptr);
733 const unsigned l = strlen(dll);
734 assert(l > 0);
735 const unsigned new_size = 1 + 3 * l + 1;
736 char *const new_name = New(char, new_size);
737 SPAN_S_VAR(char, const name, new_name, new_size);
738 name[0] = first_char;
739 SPAN_S_VAR(char, n, name + (1 + 2 * l));
740 do {
741 *n++ = tolower((uchar) *dll);
742 } while (*dll++);
743 encode_name(new_name + (1 + 2 * l), name + 1);
744 return new_name;
745 }

747 static char *name_for_proc(const char *dll, const char *proc, char first_char, char separator) {
748 const unsigned new_size = 1 + 2 * strlen(dll) + 1 + 2 * strlen(proc) + 1 + 1;
749 TStr dll_name(name_for_dll(dll, first_char));
750 char *const new_name = New(char, new_size);
751 SPAN_S_VAR(char, const name, new_name, new_size);
752 upx_safe_snprintf(new_name, new_size, "%s%c", (const char *) dll_name, separator);
753 encode_name(proc, name + strlen(name));
754 return new_name;
755 }

775 void add_import(const char *dll, const char *proc, unsigned ordinal) {
776 TStr sdll(name_for_dll(dll, dll_name_id));
777 TStr desc_name(name_for_dll(dll, descriptor_id));
778
779 char tsep = thunk_separator;
780 if (findSection(sdll, false) == nullptr) {
781 tsep = thunk_separator_first;
782 addSection(sdll, dll, strlen(dll) + 1, 0); // name of the dll
783 addSymbol(sdll, sdll, 0);
784
785 addSection(desc_name, zeros, sizeof(zeros), 0); // descriptor
786 addRelocation(desc_name, offsetof(import_desc, dllname), "R_X86_64_32", sdll, 0);
787 }
788 TStr thunk(proc == nullptr ? name_for_dll(dll, thunk_id)
789 : name_for_proc(dll, proc, thunk_id, tsep));
790
791 if (findSection(thunk, false) != nullptr)
792 return; // we already have this dll/proc
793 addSection(thunk, zeros, thunk_size, 0);
794 addSymbol(thunk, thunk, 0);
795 if (tsep == thunk_separator_first) {
796 addRelocation(desc_name, offsetof(import_desc, iat), "R_X86_64_32", thunk, 0);
797
798 TStr last_thunk(name_for_proc(dll, "X", thunk_id, thunk_separator_last));
799 addSection(last_thunk, zeros, thunk_size, 0);
800 }
801
802 const char *reltype = thunk_size == 4 ? "R_X86_64_32" : "R_X86_64_64";
803 if (ordinal != 0u) {
804 addRelocation(thunk, 0, reltype, "*UND*", ordinal | (1ull << (thunk_size * 8 - 1)));
805 } else if (proc != nullptr) {
806 TStr proc_name(name_for_proc(dll, proc, proc_name_id, procname_separator));
807 addSection(proc_name, zeros, 2, 1); // 2 bytes of word aligned "hint"
808 addSymbol(proc_name, proc_name, 0);
809 addRelocation(thunk, 0, reltype, proc_name, 0);
810
811 strcat(proc_name, "X");
812 addSection(proc_name, proc, strlen(proc), 0); // the name of the symbol
813 } else
814 infoWarning("empty import: %s", dll);
815 }

824 return rc;
825 // What could remain?
826 // make sort order deterministic
827 return a->sort_id < b->sort_id ? -1 : 1;

1375template <typename LEXX>
1376void PeFile::processTls1(Interval *iv, typename tls_traits<LEXX>::cb_value_t imagebase,
1377 unsigned imagesize) { // pass 1
1378 typedef typename tls_traits<LEXX>::tls tls;
1379 typedef typename tls_traits<LEXX>::cb_value_t cb_value_t;
1380 constexpr unsigned cb_size = tls_traits<LEXX>::cb_size;
1381
1382 COMPILE_TIME_ASSERT(sizeof(tls) == tls_traits<LEXX>::sotls)
1383 COMPILE_TIME_ASSERT_ALIGNED1(tls)
1384
1385 if (isefi && IDSIZE(PEDIR_TLS))
1386 throwCantPack("TLS not supported on EFI");
1387
1388 const unsigned take = ALIGN_UP(IDSIZE(PEDIR_TLS), 4u);
1389 sotls = take;
1390 if (!sotls)
1391 return;
1392 const unsigned skip = IDADDR(PEDIR_TLS);
1393 const tls *const tlsp = (const tls *) ibuf.subref("bad tls %#x", skip, sizeof(tls));
1394
1395 // note: TLS callbacks are not implemented in Windows 95/98/ME
1396 if (tlsp->callbacks) {
1397 if (tlsp->callbacks < imagebase)
1398 throwCantPack("invalid TLS callback");
1399 else if (tlsp->callbacks - imagebase + 4 >= imagesize)
1400 throwCantPack("invalid TLS callback");
1401 cb_value_t v =
1402 *(LEXX *) ibuf.subref("bad TLS %#x", (tlsp->callbacks - imagebase), sizeof(LEXX));
1403
1404 if (v != 0) {
1405 // count number of callbacks, just for information string - Stefan Widmann
1406 unsigned num_callbacks = 0;
1407 unsigned callback_offset = 0;
1408 while (*(LEXX *) ibuf.subref(
1409 "bad TLS %#x", tlsp->callbacks - imagebase + callback_offset, sizeof(LEXX))) {
1410 // increment number of callbacks
1411 num_callbacks++;
1412 callback_offset += cb_size;
1413 }
1414 info("TLS: %u callback(s) found, adding TLS callback handler", num_callbacks);
1415 // set flag to include necessary sections in loader
1416 use_tls_callbacks = true;
1417 // define linker symbols
1418 tlscb_ptr = tlsp->callbacks;
1419 }
1420 }
1421
1422 const unsigned tlsdatastart = tlsp->datastart - imagebase;
1423 const unsigned tlsdataend = tlsp->dataend - imagebase;
1424
1425 // now some ugly stuff: find the relocation entries in the tls data area
1426 const unsigned skip2 = IDADDR(PEDIR_BASERELOC);
1427 const unsigned take2 = IDSIZE(PEDIR_BASERELOC);
1428 Reloc rel(ibuf.subref("bad tls reloc %#x", skip2, take2), take2);
1429 unsigned pos, type;
1430 while (rel.next(pos, type))
1431 if (pos >= tlsdatastart && pos < tlsdataend)
1432 iv->add_interval(pos, type);
1433
1434 sotls = sizeof(tls) + tlsdataend - tlsdatastart;
1435 // if TLS callbacks are used, we need two more {D|Q}WORDS at the end of the TLS
1436 // ... and those dwords should be correctly aligned
1437 if (use_tls_callbacks)
1438 sotls = ALIGN_UP(sotls, cb_size) + 2 * cb_size;
1439 const unsigned aligned_sotls = ALIGN_UP(sotls, usizeof(LEXX));
1440
1441 // the PE loader wants this stuff uncompressed
1442 mb_otls.alloc(aligned_sotls);
1443 mb_otls.clear();
1444 otls = mb_otls; // => otls now is a SPAN_S
1445 const unsigned skip1 = IDADDR(PEDIR_TLS);
1446 const unsigned take1 = sizeof(tls);
1447 memcpy(otls, ibuf.subref("bad tls %#x", skip1, take1), take1);
1448 // WARNING: this can access data in BSS
1449 const unsigned take3 = sotls - sizeof(tls);
1450 memcpy(otls + sizeof(tls), ibuf.subref("bad tls %#x", tlsdatastart, take3), take3);
1451 tlsindex = tlsp->tlsindex - imagebase;
1452 // NEW: subtract two dwords if TLS callbacks are used - Stefan Widmann
1453 info("TLS: %u bytes tls data and %u relocations added",
1454 sotls - (unsigned) sizeof(tls) - (use_tls_callbacks ? 2 * cb_size : 0), iv->ivnum);
1455
1456 // makes sure tls index is zero after decompression
1457 if (tlsindex && tlsindex < imagesize)
1458 set_le32(ibuf.subref("bad tlsindex %#x", tlsindex, sizeof(unsigned)), 0);
1459}

1461template <typename LEXX>
1462void PeFile::processTls2(Reloc *const rel, const Interval *const iv, unsigned newaddr,
1463 typename tls_traits<LEXX>::cb_value_t imagebase) { // pass 2
1464 typedef typename tls_traits<LEXX>::tls tls;
1465 typedef typename tls_traits<LEXX>::cb_value_t cb_value_t;
1466 constexpr unsigned cb_size = tls_traits<LEXX>::cb_size;
1467 constexpr unsigned reloc_type = tls_traits<LEXX>::reloc_type;
1468 static_assert(reloc_type > IMAGE_REL_BASED_IGNORE && reloc_type < 16);
1469 constexpr int tls_handler_offset_reloc = tls_traits<LEXX>::tls_handler_offset_reloc;
1470
1471 if (sotls == 0)
1472 return;
1473 // add new relocation entries
1474
1475 if (tls_handler_offset > 0 && tls_handler_offset_reloc > 0)
1476 rel->add_reloc(tls_handler_offset + tls_handler_offset_reloc, reloc_type);
1477
1478 unsigned ic;
1479 // NEW: if TLS callbacks are used, relocate the VA of the callback chain, too - Stefan Widmann
1480 for (ic = 0; ic < (use_tls_callbacks ? 4 * cb_size : 3 * cb_size); ic += cb_size)
1481 rel->add_reloc(newaddr + ic, reloc_type);
1482
1483 SPAN_S_VAR(tls, const tlsp, mb_otls);
1484 // now the relocation entries in the tls data area
1485 for (ic = 0; ic < iv->ivnum; ic += 4) {
1486 SPAN_S_VAR(byte, const pp,
1487 otls + (iv->ivarr[ic].start - (tlsp->datastart - imagebase) + sizeof(tls)));
1488 LEXX *const p = (LEXX *) raw_bytes(pp, sizeof(LEXX));
1489 cb_value_t kc = *p;
1490 if (kc < tlsp->dataend && kc >= tlsp->datastart) {
1491 kc += newaddr + sizeof(tls) - tlsp->datastart;
1492 *p = kc + imagebase;
1493 rel->add_reloc(kc, iv->ivarr[ic].len);
1494 } else
1495 rel->add_reloc(kc - imagebase, iv->ivarr[ic].len);
1496 }
1497
1498 const unsigned tls_data_size = tlsp->dataend - tlsp->datastart;
1499 tlsp->datastart = newaddr + sizeof(tls) + imagebase;
1500 tlsp->dataend = tlsp->datastart + tls_data_size;
1501
1502 // NEW: if we have TLS callbacks to handle, we create a pointer to the new callback chain -
1503 // Stefan Widmann
1504 tlsp->callbacks = (use_tls_callbacks ? newaddr + sotls + imagebase - 2 * cb_size : 0);
1505
1506 if (use_tls_callbacks) {
1507 // set handler offset
1508 SPAN_S_VAR(byte, pp, otls);
1509 pp = otls + (sotls - 2 * cb_size);
1510 *(LEXX *) raw_bytes(pp, sizeof(LEXX)) = tls_handler_offset + imagebase;
1511 pp = otls + (sotls - 1 * cb_size);
1512 *(LEXX *) raw_bytes(pp, sizeof(LEXX)) = 0; // end of one-item list
1513 // add relocation for TLS handler offset
1514 rel->add_reloc(newaddr + sotls - 2 * cb_size, reloc_type);
1515 }
1516}

1690PeFile::Resource::upx_rnode *PeFile::Resource::convert(const void *rnode, upx_rnode *parent,
1691 unsigned level) {
1692 if (level == 3) {
1693 const res_data *node = ACC_STATIC_CAST(const res_data *, rnode);
1694 ibufcheck(node, sizeof(*node));
1695 upx_rleaf *leaf = new upx_rleaf;
1696 leaf->id = 0;
1697 leaf->name = nullptr;
1698 leaf->parent = parent;
1699 leaf->next = head;
1700 leaf->newoffset = 0;
1701 leaf->data = *node;
1702
1703 head = leaf; // append node to a linked list for traversal
1704 dsize += sizeof(res_data);
1705 return leaf;
1706 }
1707
1708 const res_dir *node = ACC_STATIC_CAST(const res_dir *, rnode);
1709 ibufcheck(node, sizeof(*node));
1710 int ic = node->identr + node->namedentr;
1711 if (ic == 0)
1712 return nullptr;
1713
1714 upx_rbranch *branch = new upx_rbranch;
1715 branch->id = 0;
1716 branch->name = nullptr;
1717 branch->parent = parent;
1718 branch->nc = ic;
1719 branch->children = New(upx_rnode *, ic);
1720 branch->data = *node;
1721
1722 for (const res_dir_entry *rde = node->entries + ic - 1; --ic >= 0; rde--) {
1723 upx_rnode *child = convert(start + (rde->child & 0x7fffffff), branch, level + 1);
1724 xcheck(child);
1725 branch->children[ic] = child;
1726 child->id = rde->tnl;
1727 if (child->id & 0x80000000) {
1728 const byte *p = start + (child->id & 0x7fffffff);
1729 ibufcheck(p, 2);
1730 const unsigned len = 2 + 2 * get_le16(p);
1731 ibufcheck(p, len);
1732 child->name = New(byte, len);
1733 memcpy(child->name, p, len); // copy unicode string
1734 ssize += len; // size of unicode strings
1735 }
1736 }
1737 dsize += node->Sizeof();
1738 return branch;

1746
1747 res_data *l = (res_data *) (newstart + bpos);
1748 const upx_rleaf *leaf = (const upx_rleaf *) node;
1749 *l = leaf->data;
1750 if (leaf->newoffset)
1751 l->offset = leaf->newoffset;
1752 bpos += sizeof(*l);
1753 return;
1754 }
1755 if (bpos + sizeof(res_dir) > dirsize())
1756 throwCantUnpack("corrupted resources");
1757
1758 res_dir *const b = (res_dir *) (newstart + bpos);
1759 const upx_rbranch *branch = (const upx_rbranch *) node;
1760 *b = branch->data;
1761 bpos += b->Sizeof();
1762 res_dir_entry *be = b->entries;
1763 for (unsigned ic = 0; ic < branch->nc; ic++, be++) {
1764 xcheck(branch->children[ic]);
1765 be->tnl = branch->children[ic]->id;
1766 be->child = bpos + ((level < 2) ? 0x80000000 : 0);
1767
1768 const byte *p;
1769 if ((p = branch->children[ic]->name) != nullptr) {
1770 be->tnl = spos + 0x80000000;
1771 if (spos + get_le16(p) * 2 + 2 > dirsize())
1772 throwCantUnpack("corrupted resources");
1773 memcpy(newstart + spos, p, get_le16(p) * 2 + 2);
1774 spos += get_le16(p) * 2 + 2;
1775 }
1776
1777 build(branch->children[ic], bpos, spos, level + 1);
1778 }
1779}

1879static bool match(unsigned itype, const byte *ntype, unsigned iname, const byte *nname,
1880 const char *keep) {
1881 // format of string keep: type1[/name1],type2[/name2], ....
1882 // typex and namex can be string or number
1883 // hopefully resource names do not have '/' or ',' characters inside
1884
1885 struct Helper final {
1886 static bool match(unsigned num, const byte *unistr, const char *mkeep) {
1887 if (!unistr)
1888 return (unsigned) atoi(mkeep) == num;
1889 unsigned ic;
1890 for (ic = 0; ic < get_le16(unistr); ic++)
1891 if (unistr[2 + ic * 2] != (byte) mkeep[ic])
1892 return false;
1893 return mkeep[ic] == 0 || mkeep[ic] == ',' || mkeep[ic] == '/';
1894 }
1895 };
1896
1897 // FIXME this comparison is not too exact
1898 for (;;) {
1899 const char *delim1 = strchr(keep, '/');
1900 const char *delim2 = strchr(keep, ',');
1901 if (Helper::match(itype, ntype, keep)) {
1902 if (!delim1)
1903 return true;
1904 if (delim2 && delim2 < delim1)
1905 return true;
1906 if (Helper::match(iname, nname, delim1 + 1))
1907 return true;
1908 }
1909 if (delim2 == nullptr)
1910 break;
1911 keep = delim2 + 1;
1912 }
1913 return false;
1914}

2751void PeFile::rebuildRelocs(SPAN_S(byte) & extra_info, unsigned bits, unsigned flags,
2752 upx_uint64_t imagebase) {
2753 assert(bits == 32 || bits == 64);
2754 if (!ODADDR(PEDIR_BASERELOC) || !ODSIZE(PEDIR_BASERELOC) ||
2755 (flags & IMAGE_FILE_RELOCS_STRIPPED))
2756 return;
2757
2758 if (ODSIZE(PEDIR_BASERELOC) == 8) // some tricky dlls use this
2759 {
2760 omemcpy(obuf + (ODADDR(PEDIR_BASERELOC) - rvamin), "\x0\x0\x0\x0\x8\x0\x0\x0", 8);
2761 return;
2762 }
2763
2764 const unsigned orig_crelocs = mem_size(1, get_le32(extra_info));
2765 const byte big = extra_info[4];
2766 extra_info += 5;
2767
2768 SPAN_S_VAR(const byte, rdata, obuf + orig_crelocs, obuf);
2769 MemBuffer mb_wrkmem;
2770 unsigned relocnum = unoptimizeReloc(rdata, mb_wrkmem, obuf, orig_crelocs, bits, true);
2771
2772 // 16-bit relocations
2773 unsigned r16 = 0;
2774 if (big & 6) { // count 16-bit relocations
2775 SPAN_S_VAR(const LE32, q, SPAN_TYPE_CAST(const LE32, rdata));
2776 while (*q++)
2777 r16++;
2778 if ((big & 6) == 6)
2779 while (*++q)
2780 r16++;
2781 }
2782 Reloc rel(relocnum + r16);
2783 if (big & 6) { // add 16-bit relocations
2784 SPAN_S_VAR(const LE32, q, SPAN_TYPE_CAST(const LE32, rdata));
2785 while (*q)
2786 rel.add_reloc(*q++ + rvamin, (big & 4) ? IMAGE_REL_BASED_LOW : IMAGE_REL_BASED_HIGH);
2787 if ((big & 6) == 6)
2788 while (*++q)
2789 rel.add_reloc(*q + rvamin, IMAGE_REL_BASED_HIGH);
2790 // rdata = (const byte *) raw_bytes(q, 0); // advance rdata
2791 }
2792
2793 SPAN_S_VAR(byte, const wrkmem, mb_wrkmem);
2794 for (unsigned ic = 0; ic < relocnum; ic++) {
2795 OPTR_VAR(byte, const p, obuf + get_le32(wrkmem + 4 * ic));
2796 if (bits == 32)
2797 set_le32(p, get_le32(p) + imagebase + rvamin);
2798 else
2799 set_le64(p, get_le64(p) + imagebase + rvamin);
2800 rel.add_reloc(rvamin + get_le32(wrkmem + 4 * ic),
2801 bits == 32 ? IMAGE_REL_BASED_HIGHLOW : IMAGE_REL_BASED_DIR64);
2802 }
2803 rel.finish(oxrelocs, soxrelocs);
2804
2805 omemcpy(obuf + (ODADDR(PEDIR_BASERELOC) - rvamin), oxrelocs, soxrelocs);
2806 delete[] oxrelocs;
2807 oxrelocs = nullptr;
2808 mb_wrkmem.dealloc();
2809
2810 ODSIZE(PEDIR_BASERELOC) = soxrelocs;
2811}

40void printClearLine(FILE *f) noexcept {
41 static char clear_line_msg[1 + 79 + 1 + 1];
42 if (!clear_line_msg[0]) {
43 char *msg = clear_line_msg;
44 msg[0] = '\r';
45 memset(msg + 1, ' ', 79);
46 msg[80] = '\r';
47 msg[81] = 0;
48 }
49
50 fflush(stdout);
51 fflush(stderr);
52 if (f == nullptr)
53 f = stdout;
54 con_fprintf(f, "%s", clear_line_msg);
55 fflush(f);
56 printSetNl(0);
57}

66static void pr_error(const char *iname, const char *msg, bool is_warning) noexcept {
67 fflush(stdout);
68 fflush(stderr);
69 char buf[1024];
70 buf[0] = 0;
71 if (pr_need_nl == 2)
72 printClearLine(stdout);
73 else if (pr_need_nl) {
74 buf[0] = '\n';
75 buf[1] = 0;
76 printSetNl(0);
77 }
78
79 // This hack is needed, otherwise error messages may get lost
80 // when the cursor is not yet at the bottom of the screen.
81 // At least I can use some colors then...
82 bool c = acc_isatty(STDERR_FILENO) ? 1 : 0;
83
84 int fg = con_fg(stderr, FG_BRTRED);
85 upx_safe_snprintf(buf + strlen(buf), sizeof(buf) - strlen(buf), "%s: ", progname);
86 pr_print(c, buf);
87 //(void)con_fg(stderr,FG_RED);
88 upx_safe_snprintf(buf, sizeof(buf), "%s: ", iname);
89 pr_print(c, buf);
90 //(void)con_fg(stderr,FG_BRTRED);
91 pr_print(c, msg);
92 pr_print(c, "\n");
93 fflush(stdout);
94 fflush(stderr);
95 fg = con_fg(stderr, fg);
96
97 UNUSED(is_warning);
98 UNUSED(fg);
99}

101void printErr(const char *iname, const Throwable &e) noexcept {
102 char buf[1024];
103 size_t l;
104
105 upx_safe_snprintf(buf, sizeof(buf), "%s", prettyExceptionName(typeid(e).name()));
106 l = strlen(buf);
107 if (l < sizeof(buf) && e.getMsg())
108 upx_safe_snprintf(buf + l, sizeof(buf) - l, ": %s", e.getMsg());
109 l = strlen(buf);
110 if (l < sizeof(buf) && e.getErrno()) {
111 upx_safe_snprintf(buf + l, sizeof(buf) - l, ": %s", strerror(e.getErrno()));
112#if 1
113 // some compilers (e.g. Borland C++) put a trailing '\n'
114 // into the strerror() result
115 l = strlen(buf);
116 while (l-- > 0 && (buf[l] == '\n' || buf[l] == ' '))
117 buf[l] = 0;
118#endif
119 }
120 pr_error(iname, buf, e.isWarning());
121}

161static void info_print(const char *msg) {
162 if (opt->info_mode <= 0)
163 return;
164 FILE *f = opt->to_stdout ? stderr : stdout;
165 if (pr_need_nl) {
166 printClearLine(f);
167 con_fprintf(f, "%s\n", msg);
168 } else if (pr_need_nl)
169 con_fprintf(f, "\n%s\n", msg);
170 else
171 con_fprintf(f, "%s\n", msg);
172 fflush(f);
173 printSetNl(0);
174}

190void info(const char *format, ...) {

192 return;

195 const int n = 4 * info_header;
196 memset(buf, ' ', n);
197 va_start(args, format);
198 upx_safe_vsnprintf(buf + n, sizeof(buf) - n, format, args);
199 va_end(args);

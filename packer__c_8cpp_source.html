239unsigned Packer::getDecompressorWrkmemSize() const {
240 unsigned size = 0;
241 if (M_IS_LZMA(ph.method)) {
242 const lzma_compress_result_t *res = &ph.compress_result.result_lzma;
243 // FIXME - this is for i386 only
244 size = 8 + 4 + ALIGN_UP(2 * res->num_probs, 4u);
245 size = ALIGN_UP(size, 16u);
246 }
247 assert((int) size >= 0);
248 return size;
249}

251void Packer::defineDecompressorSymbols() {
252 if (callsManyTimes(ph.format)) {
253 // ELF calls the decompressor many times; the parameters change!
254 return;
255 }
256 if (M_IS_LZMA(ph.method)) {
257 const lzma_compress_result_t *res = &ph.compress_result.result_lzma;
258 upx_uint32_t properties = // lc, lp, pb, dummy
259 (res->lit_context_bits << 0) | (res->lit_pos_bits << 8) | (res->pos_bits << 16);
260 if (bele->isBE()) // big endian - bswap32
261 properties = bswap32(properties);
262
263 linker->defineSymbol("lzma_properties", properties);
264 // len - 2 because of properties
265 if (linker->findSymbol("lzma_c_len", false)) {
266 linker->defineSymbol("lzma_c_len", ph.c_len - 2);
267 }
268 if (linker->findSymbol("lzma_u_len", false)) {
269 linker->defineSymbol("lzma_u_len", ph.u_len);
270 }
271 unsigned stack = getDecompressorWrkmemSize();
272 linker->defineSymbol("lzma_stack_adjust", 0u - stack);
273
274 if (ph.format == UPX_F_DOS_EXE) {
275 linker->defineSymbol("lzma_properties_hi", properties >> 16); // pb
276 linker->defineSymbol("lzma_c_len_hi", (ph.c_len - 2) >> 16);
277 linker->defineSymbol("lzma_u_len_hi", ph.u_len >> 16);
278 linker->defineSymbol("lzma_u_len_segment", (ph.u_len & 0xf0000) >> 4);
279 }
280 }

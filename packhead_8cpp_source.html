52static upx_uint8_t get_packheader_checksum(SPAN_S(const byte) buf, int blen) {
53 assert(blen >= 4);
54 assert(get_le32(buf) == UPX_MAGIC_LE32);
55 buf += 4;
56 blen -= 4;
57 unsigned c = 0;
58 while (blen-- > 0)
59 c += *buf++;
60 c %= 251;
61 return (upx_uint8_t) c;
62}

68// Returns the size that will be generated based on version and format,
69// not necessarily the size that actually is present in spoofed input.
70int PackHeader::getPackHeaderSize() const {
71 if (format < 0 || version < 0)
72 throwInternalError("getPackHeaderSize");
73
74 int n = 0;
75 if (version <= 3) // Note: covers (version <= 0)
76 n = 24;
77 else if (version <= 9) {
78 if (format == UPX_F_DOS_COM || format == UPX_F_DOS_SYS)
79 n = 20;
80 else if (format == UPX_F_DOS_EXE || format == UPX_F_DOS_EXEH)
81 n = 25;
82 else
83 n = 28;
84 } else {
85 if (format == UPX_F_DOS_COM || format == UPX_F_DOS_SYS)
86 n = 22;
87 else if (format == UPX_F_DOS_EXE || format == UPX_F_DOS_EXEH)
88 n = 27;
89 else
90 n = 32;
91 }
92 if (n < 20)
93 throwCantUnpack("unknown header version");
94 return n;
95}

101void PackHeader::putPackHeader(SPAN_S(byte) p) const {
102 // NOTE: It is the caller's responsibility to ensure the buffer p has
103 // sufficient space for the header.
104 assert(get_le32(p) == UPX_MAGIC_LE32);
105 if (get_le32(p + 4) != UPX_MAGIC2_LE32) {
106 NO_fprintf(stderr, "MAGIC2_LE32: %x %x\n", get_le32(p + 4), UPX_MAGIC2_LE32);
107 throwBadLoader();
108 }
109
110 int size = 0;
111 upx_uint8_t old_chksum = 0;
112
113 // the new variable length header
114 if (format < 128) { // little endian
115 if (format == UPX_F_DOS_COM || format == UPX_F_DOS_SYS) {
116 size = 22;
117 old_chksum = get_packheader_checksum(p, size - 1);
118 set_le16(p + 16, u_len);
119 set_le16(p + 18, c_len);
120 p[20] = (byte) filter;
121 } else if (format == UPX_F_DOS_EXE) {
122 size = 27;
123 old_chksum = get_packheader_checksum(p, size - 1);
124 set_le24(p + 16, u_len);
125 set_le24(p + 19, c_len);
126 set_le24(p + 22, u_file_size);
127 p[25] = (byte) filter;
128 } else if (format == UPX_F_DOS_EXEH) {
129 throwInternalError("invalid format");
130 } else {
131 size = 32;
132 old_chksum = get_packheader_checksum(p, size - 1);
133 set_le32(p + 16, u_len);
134 set_le32(p + 20, c_len);
135 set_le32(p + 24, u_file_size);
136 p[28] = (byte) filter;
137 p[29] = (byte) filter_cto;
138 assert(n_mru == 0 || (n_mru >= 2 && n_mru <= 256));
139 p[30] = (byte) (n_mru ? n_mru - 1 : 0);
140 }
141 set_le32(p + 8, u_adler);
142 set_le32(p + 12, c_adler);
143 } else { // big endian
144 size = 32;
145 old_chksum = get_packheader_checksum(p, size - 1);
146 set_be32(p + 8, u_len);
147 set_be32(p + 12, c_len);
148 set_be32(p + 16, u_adler);
149 set_be32(p + 20, c_adler);
150 set_be32(p + 24, u_file_size);
151 p[28] = (byte) filter;
152 p[29] = (byte) filter_cto;
153 assert(n_mru == 0 || (n_mru >= 2 && n_mru <= 256));
154 p[30] = (byte) (n_mru ? n_mru - 1 : 0);
155 }
156
157 p[4] = (byte) version;
158 p[5] = (byte) format;
159 p[6] = (byte) method;
160 p[7] = (byte) level;
161
162 // header_checksum
163 assert(size == getPackHeaderSize());
164 // check old header_checksum
165 if (p[size - 1] != 0) {
166 if (p[size - 1] != old_chksum) {
167 // printf("old_checksum: %d %d\n", p[size - 1], old_chksum);
168 throwBadLoader();
169 }
170 }
171 // store new header_checksum
172 p[size - 1] = get_packheader_checksum(p, size - 1);
173}

179bool PackHeader::decodePackHeaderFromBuf(SPAN_S(const byte) buf, int blen) {
180 int boff = find_le32(raw_bytes(buf, blen), blen, UPX_MAGIC_LE32);
181 if (boff < 0)
182 return false;
183 blen -= boff; // bytes remaining in buf
184 if (blen < 20)
185 throwCantUnpack("header corrupted 1");
186
187 SPAN_S_VAR(const byte, const p, buf + boff);
188
189 version = p[4];
190 format = p[5];
191 method = p[6];
192 level = p[7];
193 filter_cto = 0;
194
195 if (opt->debug.debug_level) {
196 fprintf(stderr, " decodePackHeaderFromBuf version=%d format=%d method=%d level=%d\n",
197 version, format, method, level);
198 }
199 if (!((format >= 1 && format <= UPX_F_W64PE_ARM64EC) ||
200 (format >= 129 && format <= UPX_F_DYLIB_PPC64))) {
201 throwCantUnpack("unknown format %d", format);
202 }
203
204 //
205 // decode the new variable length header
206 //
207
208 int off_filter = 0;
209 if (format < 128) {
210 u_adler = get_le32(p + 8);
211 c_adler = get_le32(p + 12);
212 if (format == UPX_F_DOS_COM || format == UPX_F_DOS_SYS) {
213 u_len = get_le16(p + 16);
214 c_len = get_le16(p + 18);
215 u_file_size = u_len;
216 off_filter = 20;
217 } else if (format == UPX_F_DOS_EXE || format == UPX_F_DOS_EXEH) {
218 if (blen < 25)
219 throwCantUnpack("header corrupted 6");
220 u_len = get_le24(p + 16);
221 c_len = get_le24(p + 19);
222 u_file_size = get_le24(p + 22);
223 off_filter = 25;
224 } else {
225 if (blen < 31)
226 throwCantUnpack("header corrupted 7");
227 u_len = get_le32(p + 16);
228 c_len = get_le32(p + 20);
229 u_file_size = get_le32(p + 24);
230 off_filter = 28;
231 filter_cto = p[29];
232 n_mru = p[30] ? 1 + p[30] : 0;
233 }
234 } else {
235 if (blen < 31)
236 throwCantUnpack("header corrupted 8");
237 u_len = get_be32(p + 8);
238 c_len = get_be32(p + 12);
239 u_adler = get_be32(p + 16);
240 c_adler = get_be32(p + 20);
241 u_file_size = get_be32(p + 24);
242 off_filter = 28;
243 filter_cto = p[29];
244 n_mru = p[30] ? 1 + p[30] : 0;
245 }
246
247 if (version >= 10) {
248 if (blen < off_filter + 1)
249 throwCantUnpack("header corrupted 9");
250 filter = p[off_filter];
251 } else if ((level & 128) == 0)
252 filter = 0;
253 else {
254 // convert old flags to new filter id
255 level &= 127;
256 if (format == UPX_F_DOS_COM || format == UPX_F_DOS_SYS)
257 filter = 0x06;
258 else
259 filter = 0x26;
260 }
261 level &= 15;
262
263 //
264 // now some checks
265 //
266
267 if (version == 0xff)
268 throwCantUnpack("cannot unpack UPX ;-)");
269 // check header_checksum
270 if (version >= 10) {
271 int size = getPackHeaderSize(); // expected; based on format and version
272 if (size > blen || p[size - 1] != get_packheader_checksum(p, size - 1))
273 throwCantUnpack("header corrupted 3");
274 }
275 if (c_len < 2 || u_len < 2 || !mem_size_valid_bytes(c_len) || !mem_size_valid_bytes(u_len))
276 throwCantUnpack("header corrupted 4");
277
278 //
279 // success
280 //
281
282 this->buf_offset = boff;
283 return true;
284}

290bool ph_is_forced_method(int method) noexcept // predicate
291{
292 return (method >> 24) == -0x80;
293}
294
295int ph_force_method(int method) noexcept // mark as forced
296{
297 return method | (0x80u << 24);
298}
299
300int ph_forced_method(int method) noexcept // extract the forced method
301{
302 if (ph_is_forced_method(method))
303 method &= ~(0x80u << 24);
304 assert_noexcept(method > 0);
305 return method;
306}

322void ph_decompress(PackHeader &ph, SPAN_P(const byte) in, SPAN_P(byte) out, bool verify_checksum,
323 Filter *ft) {
324 // verify checksum of compressed data
325 if (verify_checksum) {
326 unsigned adler = upx_adler32(raw_bytes(in, ph.c_len), ph.c_len, ph.saved_c_adler);
327 if (adler != ph.c_adler)
328 throwChecksumError();
329 }
330
331 // decompress
332 if (ph.u_len < ph.c_len)
333 throwCantUnpack("header corrupted");
334 unsigned new_len = ph.u_len;
335 int r = upx_decompress(raw_bytes(in, ph.c_len), ph.c_len, raw_bytes(out, ph.u_len), &new_len,
336 ph_forced_method(ph.method), &ph.compress_result);
337 if (r == UPX_E_OUT_OF_MEMORY)
338 throwOutOfMemoryException();
339 if (r != UPX_E_OK || new_len != ph.u_len)
340 throwCompressedDataViolation();
341
342 // verify checksum of decompressed data
343 if (verify_checksum) {
344 if (ft)
345 ft->unfilter(out, ph.u_len);
346 unsigned adler = upx_adler32(raw_bytes(out, ph.u_len), ph.u_len, ph.saved_u_adler);
347 if (adler != ph.u_adler)
348 throwChecksumError();
349 }
350}

356bool ph_testOverlappingDecompression(const PackHeader &ph, const byte *buf, const byte *tbuf,
357 unsigned overlap_overhead) {
358 if (ph.c_len >= ph.u_len)
359 return false;
360
361 assert((int) overlap_overhead >= 0);
362 assert((int) (ph.u_len + overlap_overhead) >= 0);
363 const int method = ph_forced_method(ph.method);
364
365 // Because upx_test_overlap() does not use the asm_fast decompressor
366 // we must account for extra 3 bytes that asm_fast does use,
367 // or else we may fail at runtime decompression.
368 unsigned extra = 0;
369 if (M_IS_NRV2B(method) || M_IS_NRV2D(method) || M_IS_NRV2E(method))
370 extra = 3;
371 if (overlap_overhead <= 4 + extra) // don't waste time here
372 return false;
373 overlap_overhead -= extra;
374
375 unsigned src_off = ph.u_len + overlap_overhead - ph.c_len;
376 unsigned new_len = ph.u_len;
377 int r = upx_test_overlap(buf - src_off, tbuf, src_off, ph.c_len, &new_len, method,
378 &ph.compress_result);
379 if (r == UPX_E_OUT_OF_MEMORY)
380 throwOutOfMemoryException();
381 return (r == UPX_E_OK && new_len == ph.u_len);
382}

93static void sb_push(screen_t *this, const Cell *line, int len) {
94 memcpy(this->data->sb_buf[this->data->sb_sp], line, len);
95 sb_add(this, &this->data->sb_sp, 1);
96 if (this->data->sb_sp == this->data->sb_base)
97 sb_add(this, &this->data->sb_base, 1);
98}

100static const Cell *sb_pop(screen_t *this) {
101 if (this->data->sb_sp == this->data->sb_base)
102 return nullptr;
103 sb_add(this, &this->data->sb_sp, -1);
104 return this->data->sb_buf[this->data->sb_sp];
105}

213static void setCursorShape(screen_t *this, int shape) {
214 __dpmi_regs r;
215
216 memset(&r, 0, sizeof(r)); /* just in case... */
217 r.x.ax = 0x0103;
218#if 1
219 if (this)
220 r.h.al = getMode(this); /* required for buggy BIOSes */
221#endif
222 r.x.cx = shape & 0x7f1f;
223 __dpmi_int(0x10, &r);
224}

232static int init(screen_t *this, int fd) {
233 int mode;
234 int cols, rows;
235 int attr;
236
237#if 0
238 /* force linkage of conio.o */
239 (void) _conio_kbhit();
240#endif
241
242 if (!this || !this->data)
243 return -1;
244
245 this->data->mode = -1;
246#if USE_SCROLLBACK
247 this->data->sb_size = 32;
248 this->data->sb_base = 0;
249 this->data->sb_sp = 0;
250#endif
251 if (fd < 0 || !acc_isatty(fd))
252 return -1;
253 if (getPage(this) != 0)
254 return -1;
255
256#if 1 && (ACC_OS_DOS32) && defined(__DJGPP__)
257 /* check for Windows NT/2000/XP */
258 if (_get_dos_version(1) == 0x0532)
259 return -1;
260#endif
261
262 cols = ScreenCols();
263 rows = ScreenRows();
264 mode = getMode(this);
265 if (mode > 0x13) {
266 /* assume this is some SVGA/VESA text mode */
267 __dpmi_regs r;
268
269 memset(&r, 0, sizeof(r)); /* just in case... */
270 r.x.ax = 0x4f03; /* VESA - get current video mode */
271 __dpmi_int(0x10, &r);
272 if (r.h.ah == 0)
273 mode = r.x.bx;
274 } else {
275 if (mode != 2 && mode != 3 && mode != 7)
276 return -1;
277 }
278 ScreenGetCursor(&this->data->cursor_y, &this->data->cursor_x);
279 getChar(this, nullptr, &attr, this->data->cursor_x, this->data->cursor_y);
280 this->data->init_attr = attr;
281 if (mode != 7) {
282 /* Does it normally blink when bg has its 3rd bit set? */
283 int b_mask = (_farpeekb(dossel, 0x465) & 0x20) ? 0x70 : 0xf0;
284 attr = attr & (mask_fg | b_mask);
285 }
286 this->data->mode = mode;
287 this->data->cols = cols;
288 this->data->rows = rows;
289 this->data->attr = attr;
290 this->data->empty_attr = attr;
291 this->data->empty_cell = make_cell(this, ' ', attr);
292
293 ae.cursor_shape = getCursorShape(this);
294
295 return 0;
296}

303static void clearLine(screen_t *this, int y) {
304 if (y >= 0 && y < this->data->rows) {
305 unsigned sp = co80 + y * this->data->cols * 2;
306 Cell a = this->data->empty_cell;
307 int i = this->data->cols;
308
309 _farsetsel(dossel);
310 do {
311 _farnspokew(sp, a);
312 sp += 2;
313 } while (--i);
314 }
315}

324static int scrollUp(screen_t *this, int lines) {
325 int sr = this->data->rows;
326 int sc = this->data->cols;
327 int y;
328
329 if (lines <= 0 || lines > sr)
330 return 0;
331
332#if USE_SCROLLBACK
333 /* copy to scrollback buffer */
334 for (y = 0; y < lines; y++) {
335 Cell buf[sc];
336 movedata(dossel, co80 + y * this->data->cols * 2, _my_ds(), (unsigned) buf, sizeof(buf));
337 sb_push(this, buf, sizeof(buf));
338 }
339#endif
340
341 /* move screen up */
342 if (lines < sr)
343 movedata(dossel, co80 + lines * sc * 2, dossel, co80, (sr - lines) * sc * 2);
344
345 /* fill in blank lines at bottom */
346 for (y = sr - lines; y < sr; y++)
347 clearLine(this, y);
348
349 this->data->scroll_counter += lines;
350 return lines;
351}
352
353static int scrollDown(screen_t *this, int lines) {
354 int sr = this->data->rows;
355 int sc = this->data->cols;
356 int y;
357
358 if (lines <= 0 || lines > sr)
359 return 0;
360
361 /* move screen down */
362 if (lines < sr) {
363 /* !@#% movedata can't handle overlapping regions... */
364 /* movedata(dossel,co80,dossel,co80+lines*sc*2,(sr-lines)*sc*2); */
365 Cell buf[(sr - lines) * sc];
366 movedata(dossel, co80, _my_ds(), (unsigned) buf, sizeof(buf));
367 movedata(_my_ds(), (unsigned) buf, dossel, co80 + lines * sc * 2, sizeof(buf));
368 }
369
370 /* copy top lines from scrollback buffer */
371 for (y = lines; --y >= 0;) {
372#if USE_SCROLLBACK
373 const Cell *buf = sb_pop(this);
374 if (buf == nullptr)
375 clearLine(this, y);
376 else
377 updateLineN(this, buf, y, sc * 2);
378#else
379 clearLine(this, y);
380#endif
381 }
382
383 this->data->scroll_counter -= lines;
384 return lines;
385}

394static int intro(screen_t *this, void (*show_frames)(screen_t *)) {
395 int shape;
396 upx_uint16_t old_flags = __djgpp_hwint_flags;
397
398 if ((this->data->init_attr & mask_bg) != BG_BLACK)
399 return 0;
400
401 __djgpp_hwint_flags |= 3;
402 while (kbhit())
403 (void) getkey();
404
405 shape = hideCursor(this);
406 show_frames(this);
407 setCursorShape(this, shape);
408
409 while (kbhit())
410 (void) getkey();
411 __djgpp_hwint_flags = old_flags;
412
413 return 1;
414}

56bool mem_size_valid(upx_uint64_t element_size, upx_uint64_t n, upx_uint64_t extra1,
57 upx_uint64_t extra2) noexcept {
58 assert_noexcept(element_size > 0);
59 if very_unlikely (element_size == 0 || element_size > UPX_RSIZE_MAX)
60 return false;
61 if very_unlikely (n > UPX_RSIZE_MAX)
62 return false;
63 if very_unlikely (extra1 > UPX_RSIZE_MAX)
64 return false;
65 if very_unlikely (extra2 > UPX_RSIZE_MAX)
66 return false;
67 upx_uint64_t bytes = element_size * n + extra1 + extra2; // cannot overflow
68 if very_unlikely (bytes > UPX_RSIZE_MAX)
69 return false;
70 return true;
71}

73upx_rsize_t mem_size(upx_uint64_t element_size, upx_uint64_t n, upx_uint64_t extra1,
74 upx_uint64_t extra2) {
75 assert(element_size > 0);
76 if very_unlikely (element_size == 0 || element_size > UPX_RSIZE_MAX)
77 throwCantPack("mem_size 1; take care");
78 if very_unlikely (n > UPX_RSIZE_MAX)
79 throwCantPack("mem_size 2; take care");
80 if very_unlikely (extra1 > UPX_RSIZE_MAX)
81 throwCantPack("mem_size 3; take care");
82 if very_unlikely (extra2 > UPX_RSIZE_MAX)
83 throwCantPack("mem_size 4; take care");
84 upx_uint64_t bytes = element_size * n + extra1 + extra2; // cannot overflow
85 if very_unlikely (bytes > UPX_RSIZE_MAX)
86 throwCantPack("mem_size 5; take care");
87 return ACC_ICONV(upx_rsize_t, bytes);
88}

109int ptr_diff_bytes(const void *a, const void *b) may_throw {
110 if very_unlikely (a == nullptr)
111 throwCantPack("ptr_diff_bytes null 1; take care");
112 if very_unlikely (b == nullptr)
113 throwCantPack("ptr_diff_bytes null 2; take care");
114 upx_sptraddr_t d = ptraddr_diff(a, b);
115 if (a >= b) {
116 if very_unlikely (!mem_size_valid_bytes(d))
117 throwCantPack("ptr_diff_bytes-1; take care");
118 } else {
119 if very_unlikely (!mem_size_valid_bytes(0ll - d))
120 throwCantPack("ptr_diff_bytes-2; take care");
121 }
122 assert_noexcept(d == ((const charptr) a - (const charptr) b));
123 return ACC_ICONV(int, d);
124}

127 int d = ptr_diff_bytes(a, b);
128 if very_unlikely (d < 0)
129 throwCantPack("ptr_udiff_bytes; take care");

146// check that 2 buffers do not overlap; will throw on error
147void ptraddr_check_no_overlap(upx_ptraddr_t a, size_t a_size, upx_ptraddr_t b, size_t b_size) {
148 if very_unlikely (a == 0 || b == 0)
149 throwCantPack("ptr_check_no_overlap-nullptr");
150 upx_ptraddr_t a_end = a + mem_size(1, a_size);
151 upx_ptraddr_t b_end = b + mem_size(1, b_size);
152 if very_unlikely (a_end < a || b_end < b) // wrap-around
153 throwCantPack("ptr_check_no_overlap-overflow");
154 // simple, but a little bit mind bending:
155 // same as (!(a >= b_end || b >= a_end))
156 // same as (!(a_end <= b || b_end <= a))
157 if very_unlikely (a < b_end && b < a_end)
158 throwCantPack("ptr_check_no_overlap-ab");
159}

161// check that 3 buffers do not overlap; will throw on error
162void ptraddr_check_no_overlap(upx_ptraddr_t a, size_t a_size, upx_ptraddr_t b, size_t b_size,
163 upx_ptraddr_t c, size_t c_size) {
164 if very_unlikely (a == 0 || b == 0 || c == 0)
165 throwCantPack("ptr_check_no_overlap-nullptr");
166 upx_ptraddr_t a_end = a + mem_size(1, a_size);
167 upx_ptraddr_t b_end = b + mem_size(1, b_size);
168 upx_ptraddr_t c_end = c + mem_size(1, c_size);
169 if very_unlikely (a_end < a || b_end < b || c_end < c) // wrap-around
170 throwCantPack("ptr_check_no_overlap-overflow");
171 if very_unlikely (a < b_end && b < a_end)
172 throwCantPack("ptr_check_no_overlap-ab");
173 if very_unlikely (a < c_end && c < a_end)
174 throwCantPack("ptr_check_no_overlap-ac");
175 if very_unlikely (b < c_end && c < b_end)
176 throwCantPack("ptr_check_no_overlap-bc");
177}

293// simple unoptimized memswap()
294// TODO later: CHERI clang-14 bug/miscompilation with upx_memswap(); or
295// maybe caused by tagged-memory issues ???
296void upx_memswap(void *aa, void *bb, size_t bytes) noexcept {
297 if (aa != bb && bytes != 0) {
298 byte *a = (byte *) aa;
299 byte *b = (byte *) bb;
300 do {
301 // strange clang-analyzer-15 false positive when compiling in Debug mode
302 // clang-analyzer-core.uninitialized.Assign
303 byte tmp = *a; // NOLINT(*core.uninitialized.Assign) // bogus clang-analyzer warning
304 *a++ = *b;
305 *b++ = tmp;
306 } while (--bytes != 0);
307 }
308}

310// much better memswap(), optimized for our use case in sort functions below
311static inline void memswap_no_overlap(byte *a, byte *b, size_t bytes) noexcept {
312#if defined(__clang__) && (__clang_major__ < 15) && !defined(__CHERI__)
313 // work around a clang < 15 ICE (Internal Compiler Error)
314 // @COMPILER_BUG @CLANG_BUG
315 upx_memswap(a, b, bytes);
316#else // clang bug
317 upx_alignas_max byte tmp_buf[16];
318#define SWAP(n) \
319 do { \
320 upx_memcpy_inline(tmp_buf, a, n); \
321 upx_memcpy_inline(a, b, n); \
322 upx_memcpy_inline(b, tmp_buf, n); \
323 a += n; \
324 b += n; \
325 } while (0)
326
327 for (; bytes >= 16; bytes -= 16)
328 SWAP(16);
329 if (bytes & 8)
330 SWAP(8);
331 if (bytes & 4)
332 SWAP(4);
333 if (bytes & 2)
334 SWAP(2);
335 if (bytes & 1) {
336 byte tmp = *a;
337 *a = *b;
338 *b = tmp;
339 }
340#undef SWAP
341#endif // clang bug
342}

344// extremely simple (and beautiful) stable sort: Gnomesort
345// WARNING: O(n^2) and thus very inefficient for large n
346void upx_gnomesort(void *array, size_t n, size_t element_size, upx_compare_func_t compare) {
347 for (size_t i = 1; i < n; i++) {
348 byte *a = (byte *) array + element_size * i; // a := &array[i]
349 if (i != 0 && compare(a - element_size, a) > 0) { // if a[-1] > a[0] then
350 memswap_no_overlap(a - element_size, a, element_size); // swap elements a[-1] <=> a[0]
351 i -= 2; // and decrease i
352 }
353 }
354}

356// simple Shell sort using Knuth's gap; NOT stable; uses memswap()
357// cannot compete with modern sort algorithms, but not too bad as a generic fallback
358void upx_shellsort_memswap(void *array, size_t n, size_t element_size, upx_compare_func_t compare) {
359 mem_size_assert(element_size, n); // check size
360 size_t gap = 0; // 0, 1, 4, 13, 40, 121, 364, 1093, ...
361 while (gap * 3 + 1 < n) // cannot overflow because of size check above
362 gap = gap * 3 + 1;
363 for (; gap > 0; gap = (gap - 1) / 3) {
364 const size_t gap_bytes = element_size * gap;
365 byte *p = (byte *) array + gap_bytes;
366 for (size_t i = gap; i < n; i += gap, p += gap_bytes) // invariant: p == &array[i]
367 for (byte *a = p; a != array && compare(a - gap_bytes, a) > 0; a -= gap_bytes)
368 memswap_no_overlap(a - gap_bytes, a, element_size);
369 }
370}

372// simple Shell sort using Knuth's gap; NOT stable; uses memcpy()
373// should be faster than memswap() version in theory, but benchmarks are inconsistent
374void upx_shellsort_memcpy(void *array, size_t n, size_t element_size, upx_compare_func_t compare) {
375 mem_size_assert(element_size, n); // check size
376 constexpr size_t MAX_INLINE_ELEMENT_SIZE = 256;
377 upx_alignas_max byte tmp_buf[MAX_INLINE_ELEMENT_SIZE]; // buffer for one element
378 byte *tmp = tmp_buf;
379 if (element_size > MAX_INLINE_ELEMENT_SIZE) {
380 tmp = (byte *) ::malloc(element_size);
381 assert(tmp != nullptr);
382 }
383 size_t gap = 0; // 0, 1, 4, 13, 40, 121, 364, 1093, ...
384 while (gap * 3 + 1 < n) // cannot overflow because of size check above
385 gap = gap * 3 + 1;
386 for (; gap > 0; gap = (gap - 1) / 3) {
387 const size_t gap_bytes = element_size * gap;
388 byte *p = (byte *) array + gap_bytes;
389 for (size_t i = gap; i < n; i += gap, p += gap_bytes) // invariant: p == &array[i]
390 if (compare(p - gap_bytes, p) > 0) {
391 byte *a = p;
392 memcpy(tmp, a, element_size);
393 do {
394 memcpy(a, a - gap_bytes, element_size);
395 a -= gap_bytes;
396 } while (a != array && compare(a - gap_bytes, tmp) > 0);
397 memcpy(a, tmp, element_size);
398 }
399 }
400 if (element_size > MAX_INLINE_ELEMENT_SIZE)
401 ::free(tmp);
402}

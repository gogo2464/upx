61tribool PackCom::canPack() {
62 byte buf[128];
63
64 fi->readx(buf, sizeof(buf));
65 if (memcmp(buf, "MZ", 2) == 0 || memcmp(buf, "ZM", 2) == 0) // .exe
66 return false;
67 if (memcmp(buf, "\xff\xff\xff\xff", 4) == 0) // .sys
68 return false;
69 if (!fn_has_ext(fi->getName(), "com")) // query file name
70 return false;
71 checkAlreadyPacked(buf, sizeof(buf));
72 if (file_size < 1024)
73 throwCantPack("file is too small for dos/com");
74 if (file_size > 0xFF00)
75 throwCantPack("file is too large for dos/com");
76 return true;
77}

83void PackCom::addFilter16(int filter_id) {
84 assert(filter_id > 0);
85 assert(isValidFilter(filter_id));
86
87 if (filter_id % 3 == 0) {
88 // clang-format off
89 addLoader("CALLTR16",
90 filter_id < 4 ? "CT16SUB0" : "",
91 filter_id < 4 ? "" : (opt->cpu_x86 == opt->CPU_8086 ? "CT16I086" : "CT16I286,CT16SUB0"),
92 "CALLTRI2",
93 getFormat() == UPX_F_DOS_COM ? "CORETURN" : "");
94 // clang-format on
95 } else {
96 // clang-format off
97 addLoader(filter_id % 3 == 1 ? "CT16E800" : "CT16E900",
98 "CALLTRI5",
99 getFormat() == UPX_F_DOS_COM ? "CT16JEND" : "CT16JUL2",
100 filter_id < 4 ? "CT16SUB1" : "",
101 filter_id < 4 ? "" : (opt->cpu_x86 == opt->CPU_8086 ? "CT16I087" : "CT16I287,CT16SUB1"),
102 "CALLTRI6");
103 // clang-format on
104 }

115 ft->id ? "NRVDDONE" : "NRVDRETU",

126void PackCom::patchLoader(OutputFile *fo, byte *loader, int lsize, unsigned calls) {
127 const int e_len = getLoaderSectionStart("COMCUTPO");
128 const int d_len = lsize - e_len;
129 assert(e_len > 0 && e_len < 128);
130 assert(d_len > 0 && d_len < 256);
131
132 const unsigned upper_end = ph.u_len + ph.overlap_overhead + d_len + 0x100;
133 unsigned stacksize = 0x60;
134 if (upper_end + stacksize > 0xfffe)
135 stacksize = 0x56;
136 if (upper_end + stacksize > 0xfffe)
137 throwCantPack("file is too large for dos/com");
138
139 linker->defineSymbol("calltrick_calls", calls);
140 linker->defineSymbol("sp_limit", upper_end + stacksize);
141 linker->defineSymbol("bytes_to_copy", ph.c_len + lsize);
142 linker->defineSymbol("copy_source", ph.c_len + lsize + 0x100);
143 linker->defineSymbol("copy_destination", upper_end);
144 linker->defineSymbol("neg_e_len", 0 - e_len);
145 linker->defineSymbol("NRV2B160", ph.u_len + ph.overlap_overhead);
146
147 relocateLoader();
148 loader = getLoader();
149
150 // some day we could use the relocation stuff for patchPackHeader too..
151 patchPackHeader(loader, e_len);
152 // write loader + compressed file
153 fo->write(loader, e_len); // entry
154 fo->write(obuf, ph.c_len); // compressed
155 fo->write(loader + e_len, d_len); // decompressor
156 NO_printf("%-13s: entry : %8u bytes\n", getName(), e_len);
157 NO_printf("%-13s: compressed : %8u bytes\n", getName(), ph.c_len);
158 NO_printf("%-13s: decompressor : %8u bytes\n", getName(), d_len);
159}

165void PackCom::pack(OutputFile *fo) {
166 // read file
167 ibuf.alloc(file_size);
168 obuf.allocForCompression(file_size);
169 fi->seek(0, SEEK_SET);
170 fi->readx(ibuf, file_size);
171
172 // prepare packheader
173 ph.u_len = file_size;
174 // prepare filter
175 Filter ft(ph.level);
176 ft.addvalue = getCallTrickOffset();
177 // compress
178 const unsigned overlap_range = ph.u_len < 0xFE00 - ft.addvalue ? 32 : 0;
179 compressWithFilters(&ft, overlap_range, NULL_cconf);
180
181 const int lsize = getLoaderSize();
182 MemBuffer loader(lsize);
183 memcpy(loader, getLoader(), lsize);
184
185 const unsigned calls = ft.id % 3 ? ft.lastcall - 2 * ft.calls : ft.calls;
186 patchLoader(fo, loader, lsize, calls);
187
188 // verify
189 verifyOverlappingDecompression();
190
191 // finally check the compression ratio
192 if (!checkFinalCompressionRatio(fo))
193 throwNotCompressible();
194}

212void PackCom::unpack(OutputFile *fo) {
213 ibuf.alloc(file_size);
214 obuf.allocForDecompression(ph.u_len);
215
216 // read whole file
217 fi->seek(0, SEEK_SET);
218 fi->readx(ibuf, file_size);
219
220 // get compressed data offset
221 unsigned e_len = ph.buf_offset + ph.getPackHeaderSize();
222 if (file_size_u <= e_len + ph.c_len)
223 throwCantUnpack("file damaged");
224
225 // decompress
226 decompress(ibuf + e_len, obuf);
227
228 // unfilter
229 Filter ft(ph.level);
230 ft.init(ph.filter, getCallTrickOffset());
231 ft.unfilter(obuf, ph.u_len);
232
233 // write decompressed file
234 if (fo)
235 fo->write(obuf, ph.u_len);
236}

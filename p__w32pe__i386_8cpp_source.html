87void PackW32PeI386::buildLoader(const Filter *ft) {
88 // recompute tlsindex (see pack() below)
89 unsigned tmp_tlsindex = tlsindex;
90 const unsigned oam1 = ih.objectalign - 1;
91 const unsigned newvsize = (ph.u_len + rvamin + ph.overlap_overhead + oam1) & ~oam1;
92 if (tlsindex && ((newvsize - ph.c_len - 1024 + oam1) & ~oam1) > tlsindex + 4)
93 tmp_tlsindex = 0;
94
95 // prepare loader
96 initLoader(stub_i386_win32_pe, sizeof(stub_i386_win32_pe), 2);
97 if (isdll)
98 addLoader("PEISDLL1");
99 addLoader("PEMAIN01", use_stub_relocs ? "PESOCREL" : "PESOCPIC", "PESOUNC0",
100 icondir_count > 1 ? (icondir_count == 2 ? "PEICONS1" : "PEICONS2") : "",
101 tmp_tlsindex ? "PETLSHAK" : "", "PEMAIN02",
102 ph.first_offset_found == 1 ? "PEMAIN03" : "", getDecompressorSections(),
103 // multipass ? "PEMULTIP" : "",
104 "PEMAIN10");
105 addLoader(tmp_tlsindex ? "PETLSHAK2" : "");
106 if (ft->id) {
107 const unsigned texv = ih.codebase - rvamin;
108 assert(ft->calls > 0);
109 addLoader(texv ? "PECTTPOS" : "PECTTNUL");
110 addFilter32(ft->id);
111 }
112 if (soimport)
113 addLoader("PEIMPORT", importbyordinal ? "PEIBYORD" : "", kernel32ordinal ? "PEK32ORD" : "",
114 importbyordinal ? "PEIMORD1" : "", "PEIMPOR2", isdll ? "PEIERDLL" : "PEIEREXE",
115 "PEIMDONE");
116 if (sorelocs) {
117 addLoader(soimport == 0 || soimport + cimports != crelocs ? "PERELOC1" : "PERELOC2",
118 "PERELOC3,RELOC320", big_relocs ? "REL32BIG" : "", "RELOC32J");
119 // FIXME: the following should be moved out of the above if
120 addLoader(big_relocs & 6 ? "PERLOHI0" : "", big_relocs & 4 ? "PERELLO0" : "",
121 big_relocs & 2 ? "PERELHI0" : "");
122 }
123 if (use_dep_hack)
124 addLoader("PEDEPHAK");
125
126 // NEW: TLS callback support PART 1, the callback handler installation - Stefan Widmann
127 if (use_tls_callbacks)
128 addLoader("PETLSC");
129
130 addLoader("PEMAIN20");
131 if (use_clear_dirty_stack)
132 addLoader("CLEARSTACK");
133 addLoader("PEMAIN21");
134 // NEW: last loader sections split up to insert TLS callback handler - Stefan Widmann
135 addLoader(ih.entry || !ilinker ? "PEDOJUMP" : "PERETURN");
136
137 // NEW: TLS callback support PART 2, the callback handler - Stefan Widmann
138 if (use_tls_callbacks)
139 addLoader("PETLSC2");
140
141 addLoader("IDENTSTR,UPX1HEAD");
142}

144bool PackW32PeI386::needForceOption() const {
145 // return true if we need `--force` to pack this file
146 bool r = false;
147 r |= (ih.opthdrsize != 0xe0);
148 r |= ((ih.flags & IMAGE_FILE_EXECUTABLE_IMAGE) == 0);
149 r |= ((ih.flags & IMAGE_FILE_32BIT_MACHINE) == 0); // 32 bit machine flag must be set
150 r |= (ih.coffmagic != 0x10b); // COFF magic is 0x10B in PE files
151 r |= (ih.entry == 0 && !isdll);
152 r |= (ih.ddirsentries != 16);
153 r |= (IDSIZE(PEDIR_EXCEPTION) != 0); // is this used on i386?
155 return r;
156}

158void PackW32PeI386::defineSymbols(unsigned ncsection, unsigned upxsection, unsigned sizeof_oh,
159 unsigned ic, unsigned s1addr) {
160 const unsigned myimport = ncsection + soresources - rvamin;
161
162 // patch loader
163 linker->defineSymbol("original_entry", ih.entry);
164 if (use_dep_hack) {
165 // This works around a "protection" introduced in MSVCRT80, which
166 // works like this:
167 // When the compiler detects that it would link in some code from its
168 // C runtime library which references some data in a read only
169 // section then it compiles in a runtime check whether that data is
170 // still in a read only section by looking at the pe header of the
171 // file. If this check fails the runtime does "interesting" things
172 // like not running the floating point initialization code - the result
173 // is a R6002 runtime error.
174 // These supposed to be read only addresses are covered by the sections
175 // UPX0 & UPX1 in the compressed files, so we have to patch the PE header
176 // in the memory. And the page on which the PE header is stored is read
177 // only so we must make it rw, fix the flags (i.e. clear
178 // IMAGE_SCN_MEM_WRITE of osection[x].flags), and make it ro again.
179
180 // rva of the most significant byte of member "flags" in section "UPX0"
181 const unsigned swri = pe_offset + sizeof_oh + sizeof(pe_section_t) - 1;
182 // make sure we only touch the minimum number of pages
183 const unsigned addr = 0u - rvamin + swri;
184 linker->defineSymbol("swri", addr & 0xfff); // page offset
185 // check whether osection[0].flags and osection[1].flags
186 // are on the same page
187 linker->defineSymbol(
188 "vp_size", ((addr & 0xfff) + 0x28 >= 0x1000) ? 0x2000 : 0x1000); // 2 pages or 1 page
189 linker->defineSymbol("vp_base", addr & ~0xfff); // page mask
190 linker->defineSymbol("VirtualProtect",
191 0u - rvamin + ilinkerGetAddress("kernel32.dll", "VirtualProtect"));
192 }
193 linker->defineSymbol("reloc_delt", 0u - (unsigned) ih.imagebase - rvamin);
194 linker->defineSymbol("start_of_relocs", crelocs);
195
196 if (ilinker) {
197 if (!isdll)
198 linker->defineSymbol("ExitProcess",
199 0u - rvamin + ilinkerGetAddress("kernel32.dll", "ExitProcess"));
200 linker->defineSymbol("GetProcAddress",
201 0u - rvamin + ilinkerGetAddress("kernel32.dll", "GetProcAddress"));
202 linker->defineSymbol("kernel32_ordinals", myimport);
203 linker->defineSymbol("LoadLibraryA",
204 0u - rvamin + ilinkerGetAddress("kernel32.dll", "LoadLibraryA"));
205 linker->defineSymbol("start_of_imports", myimport);
206 linker->defineSymbol("compressed_imports", cimports);
207 }
208
209 defineDecompressorSymbols();
210 linker->defineSymbol("filter_buffer_start", ih.codebase - rvamin);
211
212 // in case of overlapping decompression, this hack is needed,
213 // because windoze zeroes the word pointed by tlsindex before
214 // it starts programs
215 linker->defineSymbol("tls_value",
216 (tlsindex + 4 > s1addr) ? get_le32(obuf + tlsindex - s1addr - ic) : 0);
217 linker->defineSymbol("tls_address", tlsindex - rvamin);
218
219 linker->defineSymbol("icon_delta", icondir_count - 1);
220 linker->defineSymbol("icon_offset", ncsection + icondir_offset - rvamin);
221
222 const unsigned esi0 = s1addr + ic;
223 linker->defineSymbol("start_of_uncompressed", 0u - esi0 + rvamin);
224 linker->defineSymbol("start_of_compressed", use_stub_relocs ? esi0 + ih.imagebase : esi0);
225
226 if (use_tls_callbacks) {
227 // esi is ih.imagebase + rvamin
228 linker->defineSymbol("tls_callbacks_ptr", tlscb_ptr);
229 linker->defineSymbol("tls_module_base", 0u - rvamin);
230 }
231
232 linker->defineSymbol(isdll ? "PEISDLL1" : "PEMAIN01", upxsection);
233 // linker->dumpSymbols();
234}

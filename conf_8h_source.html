436template <class T>
437inline void mem_clear(T *object) noexcept {
438 static_assert(std::is_class_v<T>); // UPX convention
439 static_assert(std::is_standard_layout_v<T>);
440 static_assert(std::is_trivially_copyable_v<T>);
441 constexpr size_t size = sizeof(*object);
442 static_assert(size >= 1 && size <= UPX_RSIZE_MAX_MEM);
443 memset((void *) object, 0, size);
444}

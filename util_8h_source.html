50inline upx_rsize_t mem_size(upx_uint64_t element_size, upx_uint64_t n) may_throw {
51 upx_uint64_t bytes = element_size * n;
52 if very_unlikely (element_size == 0 || element_size > UPX_RSIZE_MAX || n > UPX_RSIZE_MAX ||
53 bytes > UPX_RSIZE_MAX)
54 return mem_size(element_size, n, 0, 0); // this will throw
55 return ACC_ICONV(upx_rsize_t, bytes);
56}

156// invalidate and poison a pointer: point to a non-null invalid address
157// - resulting pointer should crash on dereference
158// - this should be efficient, so no mmap() guard page etc.
159// - this should play nice with runtime checkers like ASAN, MSAN, UBSAN, valgrind, etc.
160// - this should play nice with static analyzers like clang-tidy etc.
161// NOTE: this is clearly UB (Undefined Behaviour), and stricter compilers or
162// architectures may need a more advanced/costly implementation in the future
163template <class T>
164inline void ptr_invalidate_and_poison(T *(&ptr)) noexcept {
165#if defined(__CHERI__) && defined(__CHERI_PURE_CAPABILITY__)
166 // TODO later: examine __builtin_cheri_bounds_set()
167 ptr = upx::ptr_static_cast<T *>(&ptr); // FIXME: HACK: point to address of self
168#else
169 // pretend there is some memory-mapped I/O at address 0x00000200
170 ptr = (T *) (void *) 512; // NOLINT(performance-no-int-to-ptr)
171#endif
172}

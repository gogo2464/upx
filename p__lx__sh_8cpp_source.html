65void
66PackLinuxI386sh::buildLoader(Filter const *ft)
67{
68 unsigned const sz_fold = sizeof(stub_i386_linux_elf_shell_fold);
69 MemBuffer buf(sz_fold);
70 memcpy(buf, stub_i386_linux_elf_shell_fold, sz_fold);
71
72 checkPatch(nullptr, 0, 0, 0); // reset
73 patch_le32(buf,sz_fold,"UPX3",l_shname);
74 patch_le32(buf,sz_fold,"UPX2",o_shname);
75
76 // get fresh filter
77 Filter fold_ft = *ft;
78 fold_ft.init(ft->id, ft->addvalue);
79 int preferred_ctos[2] = { ft->cto, -1 };
80 fold_ft.preferred_ctos = preferred_ctos;
81
82 // filter
83 optimizeFilter(&fold_ft, buf, sz_fold);
84 unsigned fold_hdrlen = sizeof(l_info) + sizeof(Elf32_Ehdr) +
85 sizeof(Elf32_Phdr) * get_te16(&((Elf32_Ehdr const *)(void *)buf)->e_phnum);
86 if (0 == get_le32(buf + fold_hdrlen)) {
87 // inconsistent SIZEOF_HEADERS in *.lds (ld, binutils)
88 fold_hdrlen = upx::umax(0x80u, fold_hdrlen);
89 }
90 bool success = fold_ft.filter(buf + fold_hdrlen, sz_fold - fold_hdrlen);
91 UNUSED(success);
92
93 buildLinuxLoader(
94 stub_i386_linux_elf_shell_entry, sizeof(stub_i386_linux_elf_shell_entry),
95 buf, sz_fold, ft );
96}

101bool PackLinuxI386sh::getShellName(char *buf)
102{
103 exetype = -1;
104 l_shname = (int) strcspn(buf, " \t\n\v\f\r");
105 buf[l_shname] = 0;
106 static char const *const shname[] = { // known shells that accept "-c" arg
107 "ash", "bash", "bsh", "csh", "ksh", "pdksh", "sh", "tcsh", "zsh",
108 "python", "python2", "python3",
109 nullptr
110 };
111 const char *bname = strrchr(buf, '/');
112 if (bname == nullptr)
113 return false;
114 for (int j = 0; nullptr != shname[j]; ++j) {
115 if (0 == strcmp(shname[j], bname + 1)) {
116 bool const s = bool(super::canPack());
117 if (s) {
118 opt->o_unix.blocksize = blocksize = file_size;
119 }
120 unsigned size = fi->st_size();
121 if (size > (125<<10)) { // 128KB but allow 3KB for environment
122 printWarn(fi->getName(), "Likely E2BIG for size=%u", size);
123 return false;
124 }
125 return s;
126 }
127 }
128 return false;
129}

132tribool PackLinuxI386sh::canPack()
133{
134#if defined(__linux__) //{
135 // only compress i386sh scripts when running under Linux
136 char buf[512];
137
138 fi->readx(buf, sizeof(buf));
139 fi->seek(0, SEEK_SET);
140 buf[sizeof(buf) - 1] = 0;
141 if (!memcmp(buf, "#!/", 3)) { // #!/bin/sh
142 o_shname = 2;
143 return getShellName(&buf[o_shname]);
144 }
145 else if (!memcmp(buf, "#! /", 4)) { // #! /bin/sh
146 o_shname = 3;
147 return getShellName(&buf[o_shname]);
148 }
149#endif //}
150 return false;
151}

86// similar to BoundedPtr, except checks only at creation
87// skip == offset, take == size_in_bytes
88void *MemBuffer::subref_impl(const char *errfmt, size_t skip, size_t take) {
89 debug_set(debug.last_return_address_subref, upx_return_address());
90 // check overrun and wrap-around
91 if (skip + take > size_in_bytes || skip + take < skip) {
92 char buf[100];
93 // printf is using unsigned formatting
94 if (!errfmt || !errfmt[0])
95 errfmt = "bad subref %#x %#x";
96 upx_safe_snprintf(buf, sizeof(buf), errfmt, (unsigned) skip, (unsigned) take);
97 throwCantPack(buf);
98 }
99 return ptr + skip;
100}

102/*static*/
103unsigned MemBuffer::getSizeForCompression(unsigned uncompressed_size, unsigned extra) {
104 if (uncompressed_size == 0)
105 throwCantPack("invalid uncompressed_size");
106 const size_t z = uncompressed_size; // fewer keystrokes and display columns
107 size_t bytes = mem_size(1, z); // check size
108 // All literal: 1 bit overhead per literal byte; from UCL documentation
109 bytes = upx::umax(bytes, z + z / 8 + 256);
110 // zstd: ZSTD_COMPRESSBOUND
111 bytes = upx::umax(bytes, z + (z >> 8) + ((z < (128 << 10)) ? (((128 << 10) - z) >> 11) : 0));
112 // add extra and 256 safety for various rounding/alignments
113 bytes = mem_size(1, bytes, extra, 256);
114 return ACC_ICONV(unsigned, bytes);
115}

151#define PTR_BITS32(p) ((upx_uint32_t) (ptr_get_address(p) & 0xffffffff))
152#define MAGIC1(p) ((PTR_BITS32(p) ^ 0xfefdbeeb) | 1)
153#define MAGIC2(p) ((PTR_BITS32(p) ^ 0xfefdbeeb ^ 0x88224411) | 1)

155void MemBuffer::checkState() const may_throw {
156 if (!ptr)
157 throwInternalError("block not allocated");
158 assert(size_in_bytes > 0);
159 if (use_simple_mcheck()) {
160 const byte *p = (const byte *) ptr;
161 if (get_ne32(p - 4) != MAGIC1(p))
162 throwInternalError("memory clobbered before allocated block 1");
163 if (get_ne32(p - 8) != size_in_bytes)
164 throwInternalError("memory clobbered before allocated block 2");
165 if (get_ne32(p + size_in_bytes) != MAGIC2(p))
166 throwInternalError("memory clobbered past end of allocated block");
167 }
168}

170void MemBuffer::alloc(upx_uint64_t bytes) may_throw {
171 // INFO: we don't automatically free a used buffer
172 assert(ptr == nullptr);
173 assert(size_in_bytes == 0);
174 //
175 assert(bytes > 0);
176 debug_set(debug.last_return_address_alloc, upx_return_address());
177 size_t malloc_bytes = mem_size(1, bytes); // check size
178 if (use_simple_mcheck())
179 malloc_bytes += 32;
180 byte *p = (byte *) ::malloc(malloc_bytes);
181 NO_printf("MemBuffer::alloc %llu: %p\n", bytes, p);
182 if (!p)
183 throwOutOfMemoryException();
184 size_in_bytes = ACC_ICONV(unsigned, bytes);
185 if (use_simple_mcheck()) {
186 p += 16;
187 // store magic constants to detect buffer overruns
188 set_ne32(p - 8, size_in_bytes);
189 set_ne32(p - 4, MAGIC1(p));
190 set_ne32(p + size_in_bytes + 0, MAGIC2(p));
191 set_ne32(p + size_in_bytes + 4, stats.global_alloc_counter);
192 }
193 ptr = upx::ptr_static_cast<pointer>(p);
194#if !defined(__SANITIZE_MEMORY__) && DEBUG
195 memset(ptr, 0xfb, size_in_bytes);
196 (void) VALGRIND_MAKE_MEM_UNDEFINED(ptr, size_in_bytes);
197#endif
198 stats.global_alloc_counter += 1;
199 stats.global_total_bytes += size_in_bytes;
200 stats.global_total_active_bytes += size_in_bytes;
201#if DEBUG || 1
202 checkState();
203#endif
204}

206void MemBuffer::dealloc() noexcept {
207 if (ptr != nullptr) {
208 debug_set(debug.last_return_address_dealloc, upx_return_address());
209#if DEBUG || 1
210 // info: calling checkState() here violates "noexcept", so we need a try block
211 bool shall_check = true;
212 // bool shall_check = (std::uncaught_exceptions() == 0); // only if not unwinding
213 // TODO later: add a priority() method to class Throwable
214 if (shall_check) {
215 try {
216 checkState();
217 } catch (const Throwable &e) {
218 printErr("unknown", e);
219 std::terminate();
220 } catch (...) {
221 std::terminate();
222 }
223 }
224#endif
225 stats.global_dealloc_counter += 1;
226 stats.global_total_active_bytes -= size_in_bytes;
227 if (use_simple_mcheck()) {
228 byte *p = (byte *) ptr;
229 // clear magic constants
230 set_ne32(p - 8, 0);
231 set_ne32(p - 4, 0);
232 set_ne32(p + size_in_bytes, 0);
233 set_ne32(p + size_in_bytes + 4, 0);
234 //
235 ::free(p - 16); // NOLINT(clang-analyzer-unix.Malloc) // see NOTE above
236 } else {
237 ::free(ptr); // NOLINT(clang-analyzer-unix.Malloc) // see NOTE above
238 }
239 ptr = nullptr;
240 size_in_bytes = 0;
241 } else {
242 assert_noexcept(size_in_bytes == 0);
243 }
244}

32#define CTSW16(f, cond1, cond2, addvalue, get, set) \
33 byte *b = f->buf; \
34 byte *b_end = b + f->buf_len - 3; \
35 do { \
36 if (cond1) { \
37 b += 1; \
38 unsigned a = (unsigned) (b - f->buf); \
39 f->lastcall = a; \
40 set(b, get(b) + (addvalue)); \
41 f->calls++; \
42 b += 2 - 1; \
43 } else if (cond2) { \
44 b += 1; \
45 unsigned a = (unsigned) (b - f->buf); \
46 f->lastcall = a; \
47 set(b, get(b)); \
48 f->calls++; \
49 b += 2 - 1; \
50 } \
51 } while (++b < b_end); \
52 if (f->lastcall) \
53 f->lastcall += 2; \
54 return 0;

86bool FileBase::do_sopen() {
87 if (_shflags < 0)
88 _fd = ::open(_name, _flags, _mode);
89 else {
90#if (ACC_OS_DOS32) && defined(__DJGPP__)
91 _fd = ::open(_name, _flags | _shflags, _mode);
92#elif (ACC_ARCH_M68K && ACC_OS_TOS && ACC_CC_GNUC) && defined(__MINT__)
93 _fd = ::open(_name, _flags | (_shflags & O_SHMODE), _mode);
94#elif defined(SH_DENYRW)
95 _fd = ::sopen(_name, _flags, _shflags, _mode);
96#else
97 throwInternalError("bad usage of do_sopen()");
98#endif
99 }
100 if (_fd < 0)
101 return false;
102 st.st_size = 0;
103 if (::fstat(_fd, &st) != 0)
104 throwIOException(_name, errno);
105 _length = st.st_size;
106 return true;
107}

128// Return value of ::seek is the resulting file offset (same as ::tell())
129upx_off_t FileBase::seek(upx_off_t off, int whence) {
130 if (!isOpen())
131 throwIOException("bad seek 1");
132 mem_size_assert(1, off >= 0 ? off : -off); // sanity check
133 if (whence == SEEK_SET) {
134 if (off < 0)
135 throwIOException("bad seek 2");
136 off += _offset;
137 } else if (whence == SEEK_END) {
138 if (off > 0)
139 throwIOException("bad seek 3");
140 off += _offset + _length;
141 whence = SEEK_SET;
142 } else if (whence == SEEK_CUR) {
143 } else
144 throwInternalError("bad seek: whence");
145 upx_off_t l = ::lseek(_fd, off, whence);
146 if (l < 0)
147 throwIOException("seek error", errno);
148 return l - _offset;
149}

274void OutputFile::write(SPAN_0(const void) buf, upx_int64_t blen) {
275 if (!isOpen() || blen < 0)
276 throwIOException("bad write");
277 // allow nullptr if blen == 0
278 if (blen == 0)
279 return;
280 int len = (int) mem_size(1, blen); // sanity check
281 errno = 0;
282#if WITH_XSPAN >= 2
283 NO_fprintf(stderr, "write %p %zd (%p) %d\n", buf.raw_ptr(), buf.raw_size_in_bytes(),
284 buf.raw_base(), len);
285#endif
286 long l = acc_safe_hwrite(_fd, raw_bytes(buf, len), len);
287 if (l != len)
288 throwIOException("write error", errno);
289 bytes_written += len;
290#if TESTING && 0
291 static upx_std_atomic(bool) dumping;
292 if (!dumping) {
293 dumping = true;
294 char fn[64];
295 static int part = 0;
296 snprintf(fn, sizeof(fn), "upx-dump-%04d.data", part++);
297 OutputFile::dump(fn, buf, len);
298 dumping = false;
299 }
300#endif
301}

321 mem_size_assert(1, off >= 0 ? off : -off); // sanity check
322 assert(!opt->to_stdout);
323 switch (whence) {
324 case SEEK_SET:
325 if (bytes_written < off)

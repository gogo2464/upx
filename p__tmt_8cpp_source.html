59unsigned PackTmt::findOverlapOverhead(const byte *buf, const byte *tbuf, unsigned range,
60 unsigned upper_limit) const {
61 // make sure the decompressor will be paragraph aligned
62 unsigned o = super::findOverlapOverhead(buf, tbuf, range, upper_limit);
63 o = ((o + 0x20) & ~0xf) - (ph.u_len & 0xf);
64 return o;
65}

89int PackTmt::readFileHeader() {
90#define H(x) get_le16(h + 2 * (x))
91#define H4(x) get_le32(h + (x))
92 byte h[0x40];
93 int ic;
94 unsigned exe_offset = 0;
95 adam_offset = 0;
96
97 for (ic = 0; ic < 20; ic++) {
98 fi->seek(adam_offset, SEEK_SET);
99 fi->readx(h, sizeof(h));
100
101 if (memcmp(h, "MZ", 2) == 0) // dos/exe
102 {
103 exe_offset = adam_offset;
104 adam_offset += H(2) * 512 + H(1);
105 if (H(1))
106 adam_offset -= 512;
107 if (H(0x18 / 2) == 0x40 && H4(0x3c))
108 adam_offset = H4(0x3c);
109 } else if (memcmp(h, "BW", 2) == 0) {
110 adam_offset += H(2) * 512 + H(1);
111 } else if (memcmp(h, "PMW1", 4) == 0) {
112 fi->seek(adam_offset + H4(0x18), SEEK_SET);
113 adam_offset += H4(0x24);
114 unsigned objs = H4(0x1c);
115 while (objs--) {
116 fi->readx(h, 0x18);
117 adam_offset += H4(4);
118 }
119 } else if (memcmp(h, "LE", 2) == 0) {
120 // + (memory_pages-1)*memory_page_size+bytes_on_last_page
121 unsigned offs = exe_offset + (H4(0x14) - 1) * H4(0x28) + H4(0x2c);
122 fi->seek(adam_offset + 0x80, SEEK_SET);
123 fi->readx(h, 4);
124 // + data_pages_offset
125 adam_offset = offs + H4(0);
126 } else if (memcmp(h, "Adam", 4) == 0) {
127 break;
128 } else
129 return 0;
130 }
131 if (ic == 20)
132 return 0;
133
134 fi->seek(adam_offset, SEEK_SET);
135 fi->readx(&ih, sizeof(ih));
136
137 // TODO: could add more checks for the values in 'ih'
138 const unsigned imagesize = ih.imagesize;
139 const unsigned entry = ih.entry;
140 const unsigned rsize = ih.relocsize;
141 if (imagesize < sizeof(ih) || imagesize >= file_size_u || entry >= file_size_u ||
142 rsize >= file_size_u) {
143 throwCantPack("%s: bad header: imagesize=%#x entry=%#x relocsize=%#x", getName(), imagesize,
144 entry, rsize);
145 return 0;
146 }
147
148 return UPX_F_TMT_ADAM;
149#undef H4
150#undef H
151}

163void PackTmt::pack(OutputFile *fo) {
164 big_relocs = 0;
165
166 Packer::handleStub(fi, fo, adam_offset);
167
168 const unsigned usize = ih.imagesize;
169 const unsigned rsize = ih.relocsize;
170 const unsigned relocnum = rsize / 4;
171
172 ibuf.alloc(usize + rsize + 128);
173 obuf.allocForCompression(usize + rsize + 128);
174
175 fi->seek(adam_offset + sizeof(ih), SEEK_SET);
176 fi->readx(ibuf, usize);
177
178 if (find_le32(ibuf, UPX_MIN(128u, usize), get_le32("UPX ")) >= 0)
179 throwAlreadyPacked();
180 if (rsize == 0)
181 throwCantPack("file is already compressed with another packer");
182
183 MemBuffer mb_relocs(rsize);
184 SPAN_S_VAR(byte, relocs, mb_relocs);
185 fi->readx(relocs, rsize);
186
187 const unsigned overlay = file_size - fi->tell();
188 checkOverlay(overlay);
189
190 for (unsigned ic = 0; ic < relocnum; ic++)
191 set_le32(relocs + 4 * ic, get_le32(relocs + 4 * ic) - 4);
192
193 MemBuffer mb_orelocs(4 * relocnum + 8192); // relocations + extra_info
194 SPAN_S_VAR(byte, orelocs, mb_orelocs);
195 unsigned orelocsize =
196 optimizeReloc(relocnum, relocs, orelocs, ibuf, usize, 32, true, &big_relocs);
197 mb_relocs.dealloc(); // done
198#if 1
199 // write duplicate end marker; why is this here - historical oversight ???
200 orelocs[orelocsize++] = 0;
201#endif
202 // extra_info
203 set_le32(orelocs + orelocsize, ih.entry); // save original entry point
204 orelocsize += 4;
205 set_le32(orelocs + orelocsize, orelocsize + 4); // save orelocsize
206 orelocsize += 4;
207 memcpy(raw_index_bytes(ibuf, usize, orelocsize), orelocs, orelocsize);
208 mb_orelocs.dealloc(); // done
209
210 // prepare packheader
211 ph.u_len = usize + orelocsize;
212 // prepare filter
213 Filter ft(ph.level);
214 ft.buf_len = usize;
215 // compress
216 upx_compress_config_t cconf;
217 cconf.reset();
218 // limit stack size needed for runtime decompression
219 cconf.conf_lzma.max_num_probs = 1846 + (768 << 4); // ushort: ~28 KiB stack
220 compressWithFilters(&ft, 512, &cconf);
221
222 const unsigned lsize = getLoaderSize();
223 const unsigned s_point = getLoaderSection("TMTMAIN1");
224 int e_len = getLoaderSectionStart("TMTCUTPO");
225 const unsigned d_len = lsize - e_len;
226 assert(e_len > 0 && s_point > 0);
227
228 // patch loader
229 linker->defineSymbol("original_entry", ih.entry);
230 defineDecompressorSymbols();
231 defineFilterSymbols(&ft);
232
233 linker->defineSymbol("bytes_to_copy", ph.c_len + d_len);
234 linker->defineSymbol("copy_dest", 0u - (ph.u_len + ph.overlap_overhead + d_len - 1));
235 linker->defineSymbol("copy_source", ph.c_len + lsize - 1);
236 linker->defineSymbol("TMTCUTPO", ph.u_len + ph.overlap_overhead);
237 relocateLoader();
238
239 MemBuffer loader(lsize);
240 memcpy(loader, getLoader(), lsize);
241 patchPackHeader(loader, e_len);
242
243 memcpy(&oh, &ih, sizeof(oh));
244 oh.imagesize = ph.c_len + lsize; // new size
245 oh.entry = s_point; // new entry point
246 oh.relocsize = 4;
247
248 // write loader + compressed file
249 fo->write(&oh, sizeof(oh));
250 fo->write(loader, e_len);
251 fo->write(obuf, ph.c_len);
252 fo->write(loader + lsize - d_len, d_len); // decompressor
253 byte rel_entry[4];
254 set_le32(rel_entry, 5 + s_point);
255 fo->write(rel_entry, sizeof(rel_entry));
256
257 // verify
258 verifyOverlappingDecompression();
259
260 // copy the overlay
261 copyOverlay(fo, overlay, obuf);
262
263 // finally check the compression ratio
264 if (!checkFinalCompressionRatio(fo))
265 throwNotCompressible();
266}

275void PackTmt::unpack(OutputFile *fo) {
276 Packer::handleStub(fi, fo, adam_offset);
277
278 ibuf.alloc(ph.c_len);
279 obuf.allocForDecompression(ph.u_len);
280
281 fi->seek(adam_offset + ph.buf_offset + ph.getPackHeaderSize(), SEEK_SET);
282 fi->readx(ibuf, ph.c_len);
283
284 // decompress
285 decompress(ibuf, obuf);
286
287 // read extra_info
288 const unsigned orig_entry = mem_size(1, get_le32(obuf + ph.u_len - 8));
289 const unsigned orelocsize = mem_size(1, get_le32(obuf + ph.u_len - 4));
290 const unsigned osize = mem_size(1, ph.u_len - orelocsize);
291
292 // unfilter
293 if (ph.filter) {
294 Filter ft(ph.level);
295 ft.init(ph.filter, 0);
296 ft.cto = (byte) ph.filter_cto;
297 if (ph.version < 11)
298 ft.cto = (byte) (get_le32(obuf + ph.u_len - 12) >> 24);
299 ft.unfilter(obuf, osize);
300 }
301
302 // decode relocations
303 SPAN_S_VAR(const byte, orelocs, raw_index_bytes(obuf, osize, orelocsize), orelocsize);
304 SPAN_S_VAR(byte, reloc_image, raw_index_bytes(obuf, 0, osize), osize);
305 MemBuffer mb_relocs;
306 const unsigned relocnum = unoptimizeReloc(orelocs, mb_relocs, reloc_image, osize, 32, true);
307 SPAN_S_VAR(byte, relocs, mb_relocs);
308 for (unsigned ic = 0; ic < relocnum; ic++)
309 set_le32(relocs + 4 * ic, get_le32(relocs + 4 * ic) + 4);
310
311 memcpy(&oh, &ih, sizeof(oh));
312 oh.imagesize = osize;
313 oh.entry = orig_entry;
314 oh.relocsize = relocnum * 4;
315
316 const unsigned overlay = file_size - adam_offset - ih.imagesize - ih.relocsize - sizeof(ih);
317 checkOverlay(overlay);
318
319 // write decompressed file
320 if (fo) {
321 fo->write(&oh, sizeof(oh));
322 fo->write(obuf, osize);
323 fo->write(relocs, relocnum * 4);
324 }
325
326 // copy the overlay
327 copyOverlay(fo, overlay, obuf);
328}

32#define SW16(f, cond, get, set) \
33 byte *b = f->buf; \
34 byte *b_end = b + f->buf_len - 3; \
35 do { \
36 if (cond) { \
37 b += 1; \
38 unsigned a = (unsigned) (b - f->buf); \
39 f->lastcall = a; \
40 set(b, get(b)); \
41 f->calls++; \
42 b += 2 - 1; \
43 } \
44 } while (++b < b_end); \
45 if (f->lastcall) \
46 f->lastcall += 2; \
47 return 0;

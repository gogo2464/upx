40template <class TElfClass>
41class PackVmlinuxBase : public Packer
42{
43 typedef Packer super;
44protected:
45 typedef TElfClass ElfClass;
46 typedef typename ElfClass::BeLePolicy BeLePolicy;
47 typedef typename ElfClass::ElfITypes ElfITypes;
48 // integral types
49 typedef typename ElfClass::TE16 TE16;
50 typedef typename ElfClass::TE32 TE32;
51 typedef typename ElfClass::TE64 TE64;
52 typedef typename ElfITypes::Addr Addr;
53 // ELF types
54 typedef typename ElfClass::Ehdr Ehdr;
55 typedef typename ElfClass::Shdr Shdr;
56 typedef typename ElfClass::Phdr Phdr;
57 typedef typename ElfClass::Dyn Dyn;
58 typedef typename ElfClass::Sym Sym;
59
60public:
61 PackVmlinuxBase(InputFile *, unsigned, unsigned, unsigned, char const *);
62 virtual ~PackVmlinuxBase();
63 virtual int getVersion() const override { return 13; }
64
65protected:
66 unsigned int const my_e_machine;
67 unsigned char const my_elfclass;
68 unsigned char const my_elfdata;
69 char const *const my_boot_label;
70
71 int n_ptload;
72 unsigned sz_ptload;
73 unsigned paddr_min;
74 Phdr *phdri; // from input file
75 Shdr *shdri; // from input file
76 char *shstrtab; // from input file
77 Shdr *p_text;
78 Shdr *p_note0;
79 Shdr *p_note1;
80 Ehdr ehdri; // from input file
81
82 virtual Shdr const *getElfSections();
83 virtual int getStrategy(Filter &/*ft*/);
84 virtual bool is_valid_e_entry(Addr) = 0;
85 virtual bool has_valid_vmlinux_head() = 0;
86 virtual tribool canPack() override;
87 virtual void pack(OutputFile *fo) override;
88 virtual tribool canUnpack() override; // bool, except -1: format known, but not packed
89 virtual void unpack(OutputFile *fo) override;
90 virtual unsigned write_vmlinux_head(
91 OutputFile *fo,
92 Shdr *stxt
93 ) = 0;
94 static int __acc_cdecl_qsort compare_Phdr(void const *aa, void const *bb);
95};

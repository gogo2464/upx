122bool PackPs1::readBkupHeader() {
123 fi->seek(sizeof(ps1_exe_t) + 8, SEEK_SET);
124 fi->readx(&bh, sizeof(bh));
125
126 if (bh.ih_csum != upx_adler32(&bh, SZ_IH_BKUP)) {
127 byte buf[sizeof(bh)];
128 fi->seek(sizeof(ps1_exe_t), SEEK_SET);
129 fi->readx(buf, sizeof(bh));
130 if (!getBkupHeader(buf, (byte *) &bh))
131 return false;
132 }
133 return true;
134}

136#define INIT_BH_BKUP(p, l) \
137 do { \
138 (p)->id = '1'; \
139 (p)->len = l; \
140 } while (0)
141#define ADLER16(a) (((a) >> 16) ^ ((a) &0xffff))

143void PackPs1::putBkupHeader(const byte *src, byte *dst, unsigned *len) {
144 unsigned sz_cbh = MemBuffer::getSizeForCompression(SZ_IH_BKUP);
145
146 if (src && dst) {
147 ByteArray(cpr_bh, sz_cbh);
148
149 memset(cpr_bh, 0, sizeof(bh));
150 ps1_exe_chb_t *p = (ps1_exe_chb_t *) cpr_bh;
151
152 int r = upx_compress(src, SZ_IH_BKUP, &p->ih_bkup, &sz_cbh, nullptr, M_NRV2E_8, 10, nullptr,
153 nullptr);
154 if (r != UPX_E_OK || sz_cbh >= SZ_IH_BKUP)
155 throwInternalError("header compression failed");
156 INIT_BH_BKUP(p, sz_cbh);
157 *len = ALIGN_UP(sz_cbh + usizeof(ps1_exe_chb_t) - 1, 4u);
158 p->ih_csum = ADLER16(upx_adler32(&ih.epc, SZ_IH_BKUP));
159 memcpy(dst, cpr_bh, SZ_IH_BKUP);
160 } else
161 throwInternalError("header compression failed");
162}
163
164#define ADLER16_HI(a, b) ((((a) &0xffff) ^ (b)) << 16)
165#define ADLER16_LO(a, b) (((a) >> 16) ^ (b))
166#define RE_ADLER16(a, b) (ADLER16_HI(a, b) | ADLER16_LO(a, b))
167
168bool PackPs1::getBkupHeader(byte *p, byte *dst) {
169 ps1_exe_chb_t *src = (ps1_exe_chb_t *) p;
170
171 if (src && (src->id == '1' && src->len < SZ_IH_BKUP) && dst) {
172 ByteArray(unc_bh, MemBuffer::getSizeForDecompression(SZ_IH_BKUP));
173
174 unsigned sz_bh = SZ_IH_BKUP;
175 int r = upx_decompress((const byte *) &src->ih_bkup, src->len, unc_bh, &sz_bh, M_NRV2E_8,
176 nullptr);
177 if (r == UPX_E_OUT_OF_MEMORY)
178 throwOutOfMemoryException();
179 if (r != UPX_E_OK || sz_bh != SZ_IH_BKUP)
180 throwInternalError("header decompression failed");
181 unsigned ad = upx_adler32(unc_bh, SZ_IH_BKUP);
182 unsigned ch = src->ih_csum;
183 if (ad != RE_ADLER16(ad, ch))
184 throwInternalError("backup header damaged");
185 memcpy(dst, unc_bh, SZ_IH_BKUP);
186 } else
187 return false;
188 return true;
189}

191bool PackPs1::checkFileHeader() {
192 if (fdata_size != ih.tx_len || (ih.tx_len & 3)) {
193 if (!opt->force)
194 throwCantPack("file size entry damaged (try --force)");
195 else {
196 opt->info_mode += !opt->info_mode ? 1 : 0;
197 infoWarning("fixing damaged header, keeping backup file");
198 opt->backup = 1;
199 ih.tx_len = fdata_size;
200 }
201 }
202 if (!opt->force && (ih.da_ptr != 0 || ih.da_len != 0 || ih.bs_ptr != 0 || ih.bs_len != 0)) {
203 infoWarning("unsupported header field entry");
204 return false;
205 }
206 if (ih.is_ptr < (EXE_BS | (PS_RAM_SIZE - PS_STACK_SIZE))) {
207 if (!opt->force)
208 return false;
209 else
210 infoWarning("%s: stack pointer offset low", fi->getName());
211 }
212 return true;
213}

219tribool PackPs1::canPack() {
220 byte buf[PS_HDR_SIZE - sizeof(ps1_exe_t)];
221
222 if (!readFileHeader())
223 return false;
224
225 fi->readx(buf, sizeof(buf));
226 checkAlreadyPacked(buf, sizeof(buf));
227
228 for (size_t i = 0; i < sizeof(buf); i++)
229 if (buf[i] != 0) {
230 if (!opt->force)
231 throwCantPack("unknown data in header (try --force)");
232 else {
233 opt->info_mode += !opt->info_mode ? 1 : 0;
234 infoWarning("clearing header, keeping backup file");
235 opt->backup = 1;
236 break;
237 }
238 }
239 if (!checkFileHeader())
240 throwCantPack("unsupported header flags (try --force)");
241 if (!opt->force && file_size < PS_MIN_SIZE)
242 throwCantPack("file is too small (try --force)");
243 if (!opt->force && file_size_u > PS_MAX_SIZE)
244 throwCantPack("file is too big (try --force)");
245 return true;
246}

252void PackPs1::buildLoader(const Filter *) {
253 const char *method = nullptr;
254
255 if (ph.method == M_NRV2B_8)
256 method = isCon ? "nrv2b.small,8bit.sub,nrv.done" : "nrv2b.8bit,nrv.done";
257 else if (ph.method == M_NRV2D_8)
258 method = isCon ? "nrv2d.small,8bit.sub,nrv.done" : "nrv2d.8bit,nrv.done";
259 else if (ph.method == M_NRV2E_8)
260 method = isCon ? "nrv2e.small,8bit.sub,nrv.done" : "nrv2e.8bit,nrv.done";
261 else if (ph.method == M_NRV2B_LE32)
262 method = isCon ? "nrv2b.small,32bit.sub,nrv.done" : "nrv2b.32bit,nrv.done";
263 else if (ph.method == M_NRV2D_LE32)
264 method = isCon ? "nrv2d.small,32bit.sub,nrv.done" : "nrv2d.32bit,nrv.done";
265 else if (ph.method == M_NRV2E_LE32)
266 method = isCon ? "nrv2e.small,32bit.sub,nrv.done" : "nrv2e.32bit,nrv.done";
267 else if (M_IS_LZMA(ph.method))
268 method = "nrv2b.small,8bit.sub,nrv.done,lzma.prep";
269 else
270 throwInternalError("unknown compression method");
271
272 unsigned sa_tmp = sa_cnt;
273 if (ph.overlap_overhead > sa_cnt) {
274 if (!opt->force) {
275 infoWarning("not in-place decompressible");
276 throwCantPack("packed data overlap (try --force)");
277 } else
278 sa_tmp += overlap = ALIGN_UP((ph.overlap_overhead - sa_tmp), 4u);
279 }
280
281 if (isCon || M_IS_LZMA(ph.method))
282 foundBss = findBssSection();
283
284 if (M_IS_LZMA(ph.method) && !buildPart2) {
285 initLoader(stub_mipsel_r3000_ps1, sizeof(stub_mipsel_r3000_ps1));
286 addLoader("decompressor.start", isCon ? "LZMA_DEC20" : "LZMA_DEC10", "lzma.init", nullptr);
287 addLoader(sa_tmp > (0x10000 << 2) ? "memset.long" : "memset.short",
288 !foundBss ? "con.exit" : "bss.exit", nullptr);
289 } else {
290 if (M_IS_LZMA(ph.method) && buildPart2) {
291 sz_lcpr = MemBuffer::getSizeForCompression(sz_lunc);
292 byte *cprLoader = New(byte, sz_lcpr); // FIXME: does this leak? => should put into class
293 int r = upx_compress(getLoader(), sz_lunc, cprLoader, &sz_lcpr, nullptr, M_NRV2B_8, 10,
294 nullptr, nullptr);
295 if (r != UPX_E_OK || sz_lcpr >= sz_lunc)
296 throwInternalError("loader compression failed");
297 initLoader(stub_mipsel_r3000_ps1, sizeof(stub_mipsel_r3000_ps1),
298 isCon || !M_IS_LZMA(ph.method) ? 0 : 1);
299 linker->addSection("lzma.exec", cprLoader, sz_lcpr, 0);
300 } else
301 initLoader(stub_mipsel_r3000_ps1, sizeof(stub_mipsel_r3000_ps1));
302
303 pad_code = ALIGN_UP_GAP((ph.c_len + (isCon ? sz_lcpr : 0)), 4u);
304 assert(pad_code < 4);
305 static const byte pad_buffer[4] = {0, 0, 0, 0};
306 linker->addSection("pad.code", pad_buffer, pad_code, 0);
307
308 if (isCon) {
309 if (M_IS_LZMA(ph.method))
310 addLoader(!foundBss ? "con.start" : "bss.con.start", method,
311 ih.tx_ptr & 0xffff ? "dec.ptr" : "dec.ptr.hi", "con.entry", "pad.code",
312 "lzma.exec", nullptr);
313 else
314 addLoader(!foundBss ? "con.start" : "bss.con.start", "con.mcpy",
315 ph.c_len & 3 ? "con.padcd" : "",
316 ih.tx_ptr & 0xffff ? "dec.ptr" : "dec.ptr.hi", "con.entry", method,
317 sa_cnt ? sa_cnt > (0x10000 << 2) ? "memset.long" : "memset.short" : "",
318 !foundBss ? "con.exit" : "bss.exit", "pad.code", nullptr);
319 } else {
320 if (M_IS_LZMA(ph.method))
321 addLoader(!foundBss ? "cdb.start.lzma" : "bss.cdb.start.lzma", "pad.code",
322 !foundBss ? "cdb.entry.lzma" : "bss.cdb.entry.lzma", method,
323 "cdb.lzma.cpr", ih.tx_ptr & 0xffff ? "dec.ptr" : "dec.ptr.hi",
324 "lzma.exec", nullptr);
325 else {
326 assert(foundBss != true);
327 addLoader("cdb.start", "pad.code", "cdb.entry",
328 ih.tx_ptr & 0xffff ? "cdb.dec.ptr" : "cdb.dec.ptr.hi", method,
329 sa_cnt ? sa_cnt > (0x10000 << 2) ? "memset.long" : "memset.short" : "",
330 "cdb.exit", nullptr);
331 }
332 }
333 addLoader("UPX1HEAD", "IDENTSTR", nullptr);
334 }

337#define OPTYPE(x) (((x) >> 13) & 0x7)
338#define OPCODE(x) (((x) >> 10) & 0x7)
339#define REG1(x) (((x) >> 5) & 0x1f)
340#define REG2(x) ((x) &0x1f)
341
342#define MIPS_IMM(a, b) ((((a) - (((b) &0x8000) >> 15)) << 16) | (b))
343
344// Type
345#define REGIMM 1
346#define STORE 5
347// Op
348#define LUI 7
349#define ADDIU 1
350#define SW 3
351
352#define IS_LUI(a) ((OPTYPE(a) == REGIMM && OPCODE(a) == LUI))
353#define IS_ADDIU(a) ((OPTYPE(a) == REGIMM && OPCODE(a) == ADDIU))
354#define IS_SW_ZERO(a) ((OPTYPE(a) == STORE && OPCODE(a) == SW) && REG2(a) == 0)
355
356#define BSS_CHK_LIMIT (18)
357
358bool PackPs1::findBssSection() {
359 byte reg;
360 const LE32 *const p1 = ACC_CCAST(const LE32 *, ibuf + (ih.epc - ih.tx_ptr));
361
362 if ((ih.epc - ih.tx_ptr + (BSS_CHK_LIMIT * 4)) > fdata_size)
363 return false;
364
365 // check 18 opcodes for sw zero,0(x)
366 for (signed i = BSS_CHK_LIMIT; i >= 0; i--) {
367 upx_uint16_t op = p1[i] >> 16;
368 if (IS_SW_ZERO(op)) {
369 // found! get reg (x) for bss_start
370 reg = REG1(op);
371 for (; i >= 0; i--) {
372 const bss_nfo *const p = ACC_CCAST(const bss_nfo *, &p1[i]);
373 upx_uint16_t op1 = p->op1, op2 = p->op2;
374
375 // check for la (x),bss_start
376 if ((IS_LUI(op1) && REG2(op1) == reg) && (IS_ADDIU(op2) && REG1(op2) == reg)) {
377 op1 = p->op3, op2 = p->op4;
378
379 // check for la (y),bss_end
380 if (IS_LUI(op1) && IS_ADDIU(op2)) {
381 // bss section info found!
382 bss_start = MIPS_IMM(p->hi1, p->lo1);
383 bss_end = MIPS_IMM(p->hi2, p->lo2);
384
385 if (0 < ALIGN_DOWN(bss_end - bss_start, 4u)) {
386 unsigned wkmem_sz = M_IS_LZMA(ph.method) ? 32768 : 800;
387 unsigned end_offs = ih.tx_ptr + fdata_size + overlap;
388 if (bss_end > (end_offs + wkmem_sz))
389 return isCon || (!isCon && M_IS_LZMA(ph.method));
390 else
391 return false;
392 }
393 } else
394 return false;
395 }
396 }
397 }
398 }
399 return false;
400}

406void PackPs1::pack(OutputFile *fo) {
407 ibuf.alloc(fdata_size);
408 obuf.allocForCompression(fdata_size);
409 const byte *p_scan = ibuf + fdata_size;
410
411 // read file
412 fi->seek(PS_HDR_SIZE, SEEK_SET);
413 fi->readx(ibuf, fdata_size);
414
415 // scan EOF for 2048 bytes sector alignment
416 // the removed space will secure in-place decompression
417 while (!(*--p_scan)) {
418 if (sa_cnt++ > (0x10000 << 5) || sa_cnt >= fdata_size - 1024)
419 break;
420 }
421
422 if (sa_cnt > (0x10000 << 2))
423 sa_cnt = ALIGN_DOWN(sa_cnt, 32u);
424 else
425 sa_cnt = ALIGN_DOWN(sa_cnt, 4u);
426
427 // prepare packheader
428 ph.u_len = (fdata_size - sa_cnt);
429 ph.filter = 0;
430 Filter ft(ph.level);
431
432 // compress (max_match = 65535)
433 upx_compress_config_t cconf;
434 cconf.reset();
435 cconf.conf_ucl.max_match = 65535;
436 cconf.conf_lzma.max_num_probs = 1846 + (768 << 4); // ushort: ~28 KiB stack
437 compressWithFilters(&ft, sa_cnt, &cconf);
438
439 if (overlap) {
440 opt->info_mode += !opt->info_mode ? 1 : 0;
441 infoWarning("overlap - relocating load address (+%d bytes)", overlap);
442 sa_cnt += overlap;
443 }
444
445 /*
446 if (bss_start && bss_end && !foundBss)
447 infoWarning("%s: .bss section too small - use stack", fi->getName());
448 */
449
450 unsigned lzma_init = 0;
451
452 if (M_IS_LZMA(ph.method)) {
453 sz_lunc = getLoaderSize();
454
455 lzma_init = 0u - (sz_lunc - linker->getSymbolOffset("lzma.init"));
456 defineDecompressorSymbols();
457 linker->defineSymbol("entry", ih.epc);
458 linker->defineSymbol("SC", sa_cnt > (0x10000 << 2) ? sa_cnt >> 5 : sa_cnt >> 2);
459 relocateLoader();
460
461 buildPart2 = true;
462 buildLoader(&ft);
463 }
464
465 memcpy(&oh, &ih, sizeof(ih));
466
467 unsigned sz_cbh;
468 putBkupHeader((const byte *) &ih.epc, (byte *) &bh, &sz_cbh);
469
470 if (ih.is_ptr < (EXE_BS | (PS_RAM_SIZE - PS_STACK_SIZE)))
471 oh.is_ptr = (EXE_BS | (PS_RAM_SIZE - 16));
472
473 if (ih.da_ptr != 0 || ih.da_len != 0 || ih.bs_ptr != 0 || ih.bs_len != 0)
474 oh.da_ptr = oh.da_len = oh.bs_ptr = oh.bs_len = 0;
475
476 const int lsize = getLoaderSize();
477
478 unsigned filelen = ALIGN_UP(ih.tx_len, 4u);
479
480 const unsigned decomp_data_start = ih.tx_ptr;
481 const unsigned comp_data_start = (decomp_data_start + filelen + overlap) - ph.c_len;
482
483 const int h_len = lsize - getLoaderSectionStart("UPX1HEAD");
484 int d_len = 0;
485 int e_len = 0;
486
487 if (isCon) {
488 e_len = lsize - h_len;
489 d_len = e_len - getLoaderSectionStart("con.entry");
490 } else {
491 const char *entry_lzma = !foundBss ? "cdb.entry.lzma" : "bss.cdb.entry.lzma";
492
493 d_len = (lsize - h_len) -
494 getLoaderSectionStart(M_IS_LZMA(ph.method) ? entry_lzma : "cdb.entry");
495 e_len = (lsize - d_len) - h_len;
496 }
497
498 linker->defineSymbol("entry", ih.epc);
499 linker->defineSymbol("SC", MIPS_LO(sa_cnt > (0x10000 << 2) ? sa_cnt >> 5 : sa_cnt >> 2));
500 linker->defineSymbol("DECO", decomp_data_start);
501 linker->defineSymbol("ldr_sz", M_IS_LZMA(ph.method) ? sz_lunc + 16 : (d_len - pad_code));
502
503 if (foundBss) {
504 if (M_IS_LZMA(ph.method))
505 linker->defineSymbol("wrkmem",
506 bss_end - 160 - getDecompressorWrkmemSize() - (sz_lunc + 16));
507 else
508 linker->defineSymbol("wrkmem", bss_end - 16 - (d_len - pad_code));
509 }
510
511 const unsigned entry = comp_data_start - e_len;
512 oh.epc = oh.tx_ptr = entry;
513 oh.tx_len = ph.c_len + e_len;
514
515 unsigned pad = 0;
516
517 if (!opt->ps1_exe.no_align || !isCon) {
518 pad = oh.tx_len;
519 oh.tx_len = ALIGN_UP(oh.tx_len, CD_SEC + 0u);
520 pad = oh.tx_len - pad;
521 oh.tx_ptr -= pad;
522 }
523
524 ibuf.clear(0, fdata_size);
525 byte *paddata = ibuf;
526
527 if (M_IS_LZMA(ph.method)) {
528 linker->defineSymbol("lzma_init_off", lzma_init);
529 linker->defineSymbol("gb_e", linker->getSymbolOffset("gb8_e"));
530 } else if (isCon)
531 linker->defineSymbol("gb_e", linker->getSymbolOffset(is32Bit ? "gb32_e" : "gb8_e"));
532
533 if (isCon) {
534 linker->defineSymbol("PAD", pad_code);
535 if (M_IS_LZMA(ph.method))
536 linker->defineSymbol("DCRT", (entry + getLoaderSectionStart("lzma.exec")));
537 else
538 linker->defineSymbol("DCRT", (entry + (e_len - d_len)));
539 } else {
540 linker->defineSymbol("PSVR", FIX_PSVR);
541 linker->defineSymbol("CPDO", comp_data_start);
542 if (M_IS_LZMA(ph.method)) {
543 unsigned entry_lzma =
544 getLoaderSectionStart(!foundBss ? "cdb.entry.lzma" : "bss.cdb.entry.lzma");
545
546 linker->defineSymbol("lzma_cpr", getLoaderSectionStart("lzma.exec") - entry_lzma);
547 }
548 }
549
550 relocateLoader();
551 // linker->dumpSymbols();
552 MemBuffer loader(lsize);
553 assert(lsize == getLoaderSize());
554 memcpy(loader, getLoader(), lsize);
555 patchPackHeader(loader, lsize);
556
557 if (!isCon && M_IS_LZMA(ph.method) && (HD_CODE_OFS + d_len + h_len) > CD_SEC)
558 throwInternalError("lzma --boot-only loader > 2048");
559
560 // ps1_exe_t structure
561 fo->write(&oh, sizeof(oh));
562 fo->write(&bh, sz_cbh);
563 // decompressor
564 fo->write(loader + e_len, isCon ? h_len : (d_len + h_len));
565
566 // header size is 2048 bytes + sector alignment
567 fo->write(paddata, (pad + PS_HDR_SIZE) - fo->getBytesWritten());
568 // entry
569 fo->write(loader, e_len);
570 // compressed body
571 fo->write(obuf, ph.c_len);
572
573 verifyOverlappingDecompression();
574 if (!checkFinalCompressionRatio(fo))
575 throwNotCompressible();
576
577#if 0
578 printf("%-13s: uncompressed : %8d bytes\n", getName(), (int) ph.u_len);

631void PackPs1::unpack(OutputFile *fo) {
632 // restore orig exec hdr
633 memcpy(&oh, &ih, sizeof(ih));
634 memcpy((void *) &oh.epc, &bh, SZ_IH_BKUP);
635
636 // check for removed sector alignment
637 assert(oh.tx_len >= ph.u_len);
638 const unsigned pad = oh.tx_len - ph.u_len;
639
640 ibuf.alloc(fdata_size > PS_HDR_SIZE ? fdata_size : PS_HDR_SIZE);
641 obuf.allocForDecompression(ph.u_len, pad);
642
643 fi->seek(PS_HDR_SIZE, SEEK_SET);
644 fi->readx(ibuf, fdata_size);
645
646 // decompress
647 decompress(ibuf + (fdata_size - ph.c_len), obuf);
648
649 // write decompressed file
650 if (fo) {
651 // write header
652 fo->write(&oh, sizeof(oh));
653 // align the ps exe header (mode 2 sector data size)
654 ibuf.clear();
655 fo->write(ibuf, PS_HDR_SIZE - fo->getBytesWritten());
656 // write uncompressed data + pad
657 obuf.clear(ph.u_len, pad);
658 fo->write(obuf, ph.u_len + pad);
659 }
660}

282// a static_cast that does not trigger -Wcast-align warnings
283template <class Result, class From>
284forceinline constexpr Result ptr_static_cast(From *ptr) noexcept {
285 static_assert(std::is_pointer_v<Result>);
286 // don't cast through "void *" if type is convertible
287 typedef std::conditional_t<std::is_convertible_v<decltype(ptr), Result>, Result, void *>
288 VoidPtr;
289 // NOLINTNEXTLINE(bugprone-multi-level-implicit-pointer-conversion)
290 return static_cast<Result>(static_cast<VoidPtr>(ptr));
291}
292template <class Result, class From>
293forceinline constexpr Result ptr_static_cast(const From *ptr) noexcept {
294 static_assert(std::is_pointer_v<Result>);
295 static_assert(std::is_const_v<std::remove_pointer_t<Result> >); // required
296 // don't cast through "const void *" if type is convertible
297 typedef std::conditional_t<std::is_convertible_v<decltype(ptr), Result>, Result, const void *>
298 VoidPtr;
299 // NOLINTNEXTLINE(bugprone-multi-level-implicit-pointer-conversion)
300 return static_cast<Result>(static_cast<VoidPtr>(ptr));
301}

315// atomic_exchange
316template <class T>
317forceinline T atomic_exchange(T *ptr, T new_value) noexcept {
318#if 1
319 static_assert(sizeof(T) == sizeof(void *)); // UPX convention: restrict to pointer-size for now
320#endif
321 static_assert(std::is_standard_layout_v<T>);
322 static_assert(std::is_trivially_copyable_v<T>);
323#if !(WITH_THREADS)
324 T old_value = *ptr;
325 *ptr = new_value;
326 return old_value;
327#else
328 static_assert(sizeof(T) <= sizeof(void *)); // UPX convention: restrict to fundamental types
329 static_assert(alignof(T) == sizeof(T)); // UPX convention: require proper alignment
330#if __has_builtin(__atomic_exchange_n) && defined(__ATOMIC_SEQ_CST)
331 return __atomic_exchange_n(ptr, new_value, __ATOMIC_SEQ_CST);
332#elif __has_builtin(__sync_swap)
333 return __sync_swap(ptr, new_value);
334#else
335 return std::atomic_exchange(ptr_std_atomic_cast(ptr), new_value);
336#endif
337#endif
338}

341template <class T>
342struct ObjectDeleter final {
343 T **items; // public
344 std::size_t count; // public
345 explicit ObjectDeleter(T **p, std::size_t n) noexcept : items(p), count(n) {}
346 ~ObjectDeleter() noexcept { delete_items(); }
347 void delete_items() noexcept {
348 for (std::size_t i = 0; i < count; i++) {
349 T *item = atomic_exchange(&items[i], (T *) nullptr);
350 delete item; // single object delete
351 }
352 }
353};
354template <class T>
355struct ArrayDeleter final {
356 T **items; // public
357 std::size_t count; // public
358 explicit ArrayDeleter(T **p, std::size_t n) noexcept : items(p), count(n) {}
359 ~ArrayDeleter() noexcept { delete_items(); }
360 void delete_items() noexcept {
361 for (std::size_t i = 0; i < count; i++) {
362 T *item = atomic_exchange(&items[i], (T *) nullptr);
363 delete[] item; // array delete
364 }
365 }
366};
367template <class T>
368struct MallocDeleter final {
369 T **items; // public
370 std::size_t count; // public
371 explicit MallocDeleter(T **p, std::size_t n) noexcept : items(p), count(n) {}
372 ~MallocDeleter() noexcept { delete_items(); }
373 void delete_items() noexcept {
374 for (std::size_t i = 0; i < count; i++) {
375 T *item = atomic_exchange(&items[i], (T *) nullptr);
376 ::free(item); // free memory from malloc()
377 }
378 }
379};

603template <class T = int, bool IsThirdTrue = false> // Third is false by default
604struct TriBool final {
605 static constexpr bool is_third_true = IsThirdTrue;
606 // types
607 typedef T underlying_type;
608 static_assert(std::is_integral_v<underlying_type>);
609 typedef decltype(T(0) + T(0)) promoted_type;
610 static_assert(std::is_integral_v<promoted_type>);
611 enum value_type : underlying_type { False = 0, True = 1, Third = 2 };
612 static_assert(sizeof(value_type) == sizeof(underlying_type));
613 static_assert(sizeof(underlying_type) <= sizeof(promoted_type));
614 // constructors
615 forceinline constexpr TriBool() noexcept {}
616 forceinline constexpr TriBool(value_type x) noexcept : value(x) {}
617 // IMPORTANT NOTE: permissive, converts all other values to Third!
618 constexpr TriBool(promoted_type x) noexcept : value(x == 0 ? False : (x == 1 ? True : Third)) {}
619#if __cplusplus >= 202002L
620 forceinline constexpr ~TriBool() noexcept = default;
621#else
622 forceinline ~TriBool() noexcept = default;
623#endif
624 // explicit conversion to bool
625 explicit constexpr operator bool() const noexcept {
626 return IsThirdTrue ? (value != False) : (value == True);
627 }
628 // query; this is NOT the same as operator bool()
629 constexpr bool isStrictFalse() const noexcept { return value == False; }
630 constexpr bool isStrictTrue() const noexcept { return value == True; }
631 constexpr bool isStrictBool() const noexcept { return value == False || value == True; }
632 constexpr bool isThird() const noexcept { return value != False && value != True; }
633 // access
634 constexpr value_type getValue() const noexcept { return value; }
635 // equality
636 constexpr bool operator==(TriBool other) const noexcept { return value == other.value; }
637 constexpr bool operator==(value_type other) const noexcept { return value == other; }
638 constexpr bool operator==(promoted_type other) const noexcept {
639 return value == TriBool(other).value;
640 }
641
642 // "Third" can mean many things - depending on usage context, so provide some alternate names:
643#if 0
644 // constexpr bool isDefault() const noexcept { return isThird(); } // might be misleading
645 constexpr bool isIndeterminate() const noexcept { return isThird(); }
646 constexpr bool isNone() const noexcept { return isThird(); }
647 constexpr bool isOther() const noexcept { return isThird(); }
648 constexpr bool isUndecided() const noexcept { return isThird(); }
649 // constexpr bool isUnset() const noexcept { return isThird(); } // might be misleading
650#endif
651
652private:
653 value_type value = False; // the actual value of this type
654 UPX_CXX_DISABLE_NEW_DELETE_NO_VIRTUAL(TriBool) // UPX convention
655};

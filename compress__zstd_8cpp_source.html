41static int convert_errno_from_zstd(size_t zr) {
42 const ZSTD_ErrorCode ze = ZSTD_getErrorCode(zr);
43 switch (ze) {
44 case ZSTD_error_memory_allocation:
45 return UPX_E_OUT_OF_MEMORY;
46 case ZSTD_error_srcSize_wrong:
47 return UPX_E_INPUT_OVERRUN;
48 case ZSTD_error_dstSize_tooSmall:
49 return UPX_E_OUTPUT_OVERRUN;
50 default:
51 break;
52 }
53 return UPX_E_ERROR;
54}

60int upx_zstd_compress(const upx_bytep src, unsigned src_len, upx_bytep dst, unsigned *dst_len,
61 upx_callback_t *cb_parm, int method, int level,
62 const upx_compress_config_t *cconf_parm, upx_compress_result_t *cresult) {
63 assert(method == M_ZSTD);
64 assert(level > 0);
65 assert(cresult != nullptr);
66 UNUSED(cb_parm);
67 int r = UPX_E_ERROR;
68 size_t zr;
69 const zstd_compress_config_t *const lcconf = cconf_parm ? &cconf_parm->conf_zstd : nullptr;
70 zstd_compress_result_t *const res = &cresult->result_zstd;
71 res->reset();
72
73 // TODO later: map level 1..10 to zstd-level 1..22
74 if (level == 10)
75 level = 22;
76
77 // cconf overrides
78 if (lcconf) {
79 UNUSED(lcconf);
80 }
81
82 zr = ZSTD_compress(dst, *dst_len, src, src_len, level);
83 if (ZSTD_isError(zr)) {
84 *dst_len = 0; // TODO ???
85 r = convert_errno_from_zstd(zr);
86 assert(r != UPX_E_OK);
87 } else {
88 assert(zr <= *dst_len);
89 *dst_len = (unsigned) zr;
90 r = UPX_E_OK;
91 }
92
93 return r;
94}

102 assert(method == M_ZSTD);

120}

126int upx_zstd_test_overlap(const upx_bytep buf, const upx_bytep tbuf, unsigned src_off,
127 unsigned src_len, unsigned *dst_len, int method,
128 const upx_compress_result_t *cresult) {
129 assert(method == M_ZSTD);
130
131 MemBuffer b(src_off + src_len);
132 memcpy(b + src_off, buf + src_off, src_len);
133 unsigned saved_dst_len = *dst_len;
134 int r = upx_zstd_decompress(raw_index_bytes(b, src_off, src_len), src_len,
135 raw_bytes(b, *dst_len), dst_len, method, cresult);
136 if (r != UPX_E_OK)
137 return r;
138 if (*dst_len != saved_dst_len)
139 return UPX_E_ERROR;
140 // NOTE: there is a very tiny possibility that decompression has
141 // succeeded but the data is not restored correctly because of
142 // in-place buffer overlapping, so we use an extra memcmp().
143 if (tbuf != nullptr && memcmp(tbuf, b, *dst_len) != 0)
144 return UPX_E_ERROR;
145 return UPX_E_OK;
146}

166static bool check_zstd(const int method, const int level, const unsigned expected_c_len) {
167 const unsigned u_len = 16384;
168 const unsigned c_extra = 4096;
169 MemBuffer u_buf, c_buf, d_buf;
170 unsigned c_len, d_len;
171 upx_compress_result_t cresult;
172 int r;
173
174 u_buf.alloc(u_len);
175 memset(u_buf, 0, u_len);
176 c_buf.allocForCompression(u_len, c_extra);
177 d_buf.allocForDecompression(u_len);
178
179 c_len = c_buf.getSize() - c_extra;
180 r = upx_zstd_compress(raw_bytes(u_buf, u_len), u_len, raw_index_bytes(c_buf, c_extra, c_len),
181 &c_len, nullptr, method, level, NULL_cconf, &cresult);
182 if (r != 0 || c_len != expected_c_len)
183 return false;
184
185 d_len = d_buf.getSize();
186 r = upx_zstd_decompress(raw_index_bytes(c_buf, c_extra, c_len), c_len, raw_bytes(d_buf, d_len),
187 &d_len, method, nullptr);
188 if (r != 0 || d_len != u_len || memcmp(u_buf, d_buf, u_len) != 0)
189 return false;
190
191 d_len = u_len - 1;
192 r = upx_zstd_decompress(raw_index_bytes(c_buf, c_extra, c_len), c_len, raw_bytes(d_buf, d_len),
193 &d_len, method, nullptr);
194 if (r == 0)
195 return false;
196
197 // TODO: rewrite Packer::findOverlapOverhead() so that we can test it here
198 // unsigned x_len = d_len;
199 // r = upx_zstd_test_overlap(c_buf, u_buf, c_extra, c_len, &x_len, method, nullptr);
200 return true;
201}

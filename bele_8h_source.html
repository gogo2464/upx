414forceinline constexpr int sign_extend(unsigned v, unsigned bits) noexcept {
415#if (ACC_ARCH_M68K) // no barrel shifter
416 const unsigned sign_bit = 1u << (bits - 1);
417 return ACC_ICAST(int, (v & (sign_bit - 1)) - (v & sign_bit));
418#else
419 return ACC_ICAST(int, v << (32 - bits)) >> (32 - bits);
420#endif
421}
422
423forceinline constexpr upx_int64_t sign_extend(upx_uint64_t v, unsigned bits) noexcept {
424#if (ACC_ARCH_M68K) // no barrel shifter
425 const upx_uint64_t sign_bit = upx_uint64_t(1) << (bits - 1);
426 return ACC_ICAST(upx_int64_t, (v & (sign_bit - 1)) - (v & sign_bit));
427#else
428 return ACC_ICAST(upx_int64_t, v << (64 - bits)) >> (64 - bits);
429#endif
430}

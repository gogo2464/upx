69// common part of canPack(), enhanced by subclasses
70tribool PackUnix::canPack()
71{
72 if (exetype == 0)
73 return false;
74
75#if defined(__unix__)
76 // must be executable by owner
77 if ((fi->st.st_mode & S_IXUSR) == 0)
78 throwCantPack("file not executable; try 'chmod +x'");
79#endif
80 if (file_size < 4096)
81 throwCantPack("file is too small");
82
83 // info: currently the header is 36 (32+4) bytes before EOF
84 unsigned char buf[256];
85 fi->seek(-(off_t)sizeof(buf), SEEK_END);
86 fi->readx(buf, sizeof(buf));
87 checkAlreadyPacked(buf, sizeof(buf));
88
89 return true;
90}

136int PackUnix::getStrategy(Filter &/*ft*/)
137{
138 // Called just before reading and compressing each block.
139 // Might want to adjust blocksize, etc.
140
141 // If user specified the filter, then use it (-2==filter_strategy).
142 // Else try the first two filters, and pick the better (2==filter_strategy).
143 return (opt->no_filter ? -3 : ((opt->filter > 0) ? -2 : 2));
144}

146int PackUnix::pack2(OutputFile *fo, Filter &ft)
147{
148 // compress blocks
149 total_in = 0;
150 total_out = 0;
151
152// FIXME: ui_total_passes is not correct with multiple blocks...
153// ui_total_passes = (file_size + blocksize - 1) / blocksize;
154// if (ui_total_passes == 1)
155// ui_total_passes = 0;
156
157 unsigned remaining = file_size;
158 unsigned n_block = 0;
159 while (remaining > 0)
160 {
161 // FIXME: disable filters if we have more than one block.
162 // FIXME: There is only 1 un-filter in the stub [as of 2002-11-10].
163 // So the next block really has no choice!
164 // This merely prevents an assert() in compressWithFilters(),
165 // which assumes it has free choice on each call [block].
166 // And if the choices aren't the same on each block,
167 // then un-filtering will give incorrect results.
168 int filter_strategy = getStrategy(ft);
169 if (file_size > (off_t)blocksize)
170 filter_strategy = -3; // no filters
171
172 int l = fi->readx(ibuf, UPX_MIN(blocksize, remaining));
173 remaining -= l;
174
175 // Note: compression for a block can fail if the
176 // file is e.g. blocksize + 1 bytes long
177
178 // compress
179 ph.overlap_overhead = 0;
180 ph.c_len = ph.u_len = l;
181 ft.buf_len = l;
182
183 // compressWithFilters() updates u_adler _inside_ compress();
184 // that is, AFTER filtering. We want BEFORE filtering,
185 // so that decompression checks the end-to-end checksum.
186 unsigned const end_u_adler = upx_adler32(ibuf, ph.u_len, ph.u_adler);
187 compressWithFilters(&ft, OVERHEAD, NULL_cconf, filter_strategy,
188 !!n_block++); // check compression ratio only on first block
189
190 if (ph.c_len < ph.u_len) {
191 const upx_bytep tbuf = nullptr;
192 if (ft.id == 0) tbuf = ibuf;
193 ph.overlap_overhead = OVERHEAD;
194 if (!testOverlappingDecompression(obuf, tbuf, ph.overlap_overhead)) {
195 // not in-place compressible
196 ph.c_len = ph.u_len;
197 }
198 }
199 if (ph.c_len >= ph.u_len) {
200 // block is not compressible
201 ph.c_len = ph.u_len;
202 // must manually update checksum of compressed data
203 ph.c_adler = upx_adler32(ibuf, ph.u_len, ph.saved_c_adler);
204 }
205
206 // write block header
207 b_info blk_info;
208 memset(&blk_info, 0, sizeof(blk_info));
209 set_te32(&blk_info.sz_unc, ph.u_len);
210 set_te32(&blk_info.sz_cpr, ph.c_len);
211 if (ph.c_len < ph.u_len) {
212 blk_info.b_method = (unsigned char) ph.method;
213 blk_info.b_ftid = (unsigned char) ph.filter;
214 blk_info.b_cto8 = (unsigned char) ph.filter_cto;
215 }
216 fo->write(&blk_info, sizeof(blk_info));
217 b_len += sizeof(b_info);
218
219 // write compressed data
220 if (ph.c_len < ph.u_len) {
221 fo->write(obuf, ph.c_len);
222 verifyOverlappingDecompression(); // uses ph.u_adler
223 }
224 else {
225 fo->write(ibuf, ph.u_len);
226 }
227 ph.u_adler = end_u_adler;
228
229 total_in += ph.u_len;
230 total_out += ph.c_len;
231 }
232
233 // update header with totals
234 ph.u_len = total_in;
235 ph.c_len = total_out;
236
237 if ((off_t)total_in != file_size) {
238 throwEOFException();
239 }
240
241 return 1; // default: write end-of-compression bhdr next
242}

258off_t PackUnix::pack3(OutputFile *fo, Filter &ft)
259{
260 if (nullptr==linker) {
261 // If no filter, then linker is not constructed by side effect
262 // of packExtent calling compressWithFilters.
263 // This is typical after "/usr/bin/patchelf --set-rpath".
264 buildLoader(&ft);

267 lsize = getLoaderSize();
268 updateLoader(fo);
269 patchLoaderChecksum();
270 fo->write(p, lsize);

283void PackUnix::pack(OutputFile *fo)
284{
285 Filter ft(ph.level);
286 ft.addvalue = 0;
287 b_len = 0;
288 progid = 0;
289
290 // set options
291 blocksize = opt->o_unix.blocksize;
292 if (blocksize <= 0)
293 blocksize = BLOCKSIZE;
294 if ((off_t)blocksize > file_size)
295 blocksize = file_size;
296
297 // init compression buffers
298 ibuf.alloc(blocksize);
299 obuf.allocForCompression(blocksize);
300
301 fi->seek(0, SEEK_SET);
302 pack1(fo, ft); // generate Elf header, etc.
303
304 // Shlib probably did not generate Elf header yet.
305 if (fo->st_size()) { // Only append if pack1 actually wrote something.
306 p_info hbuf;
307 set_te32(&hbuf.p_progid, progid);
308 set_te32(&hbuf.p_filesize, file_size);
309 set_te32(&hbuf.p_blocksize, blocksize);
310 fo->write(&hbuf, sizeof(hbuf));
311 }
312
313 // append the compressed body
314 if (pack2(fo, ft)) {
315 // write block end marker (uncompressed size 0)
316 b_info hdr; memset(&hdr, 0, sizeof(hdr));
317 set_le32(&hdr.sz_cpr, UPX_MAGIC_LE32);
318 fo->write(&hdr, sizeof(hdr));
319 }
320
321 pack3(fo, ft); // append loader
322
323 pack4(fo, ft); // append PackHeader and overlay_offset; update Elf header
324
325 // finally check the compression ratio
326 if (!checkFinalCompressionRatio(fo))
327 throwNotCompressible();
328}

331void PackUnix::packExtent(
332 const Extent &x,
333 Filter *ft,
334 OutputFile *fo,
335 unsigned hdr_u_len,
336 unsigned b_extra,
337 bool inhibit_compression_check
338)
339{
340 unsigned const init_u_adler = ph.u_adler;
341 unsigned const init_c_adler = ph.c_adler;
342 MemBuffer hdr_ibuf;
343 if (hdr_u_len) {
344 hdr_ibuf.alloc(hdr_u_len);
345 fi->seek(0, SEEK_SET);
346 int l = fi->readx(hdr_ibuf, hdr_u_len);
347 (void)l;
348 }
349 fi->seek(x.offset, SEEK_SET);
350 for (off_t rest = x.size; 0 != rest; ) {
351 int const filter_strategy = ft ? getStrategy(*ft) : 0;
352 int l = fi->readx(ibuf, UPX_MIN(rest, (off_t)blocksize));
353 if (l == 0) {
354 break;
355 }
356 rest -= l;
357
358 // Note: compression for a block can fail if the
359 // file is e.g. blocksize + 1 bytes long
360
361 // compress
362 ph.c_len = ph.u_len = l;
363 ph.overlap_overhead = 0;
364 unsigned end_u_adler = 0;
365 if (ft) {
366 // compressWithFilters() updates u_adler _inside_ compress();
367 // that is, AFTER filtering. We want BEFORE filtering,
368 // so that decompression checks the end-to-end checksum.
369 end_u_adler = upx_adler32(ibuf, ph.u_len, ph.u_adler);
370 ft->buf_len = l;
371
372 // compressWithFilters() requirements?
373 ph.filter = 0;
374 ph.filter_cto = 0;
375 ft->id = 0;
376 ft->cto = 0;
377
378 compressWithFilters(ft, OVERHEAD, NULL_cconf, filter_strategy,
379 0, 0, 0, hdr_ibuf, hdr_u_len, inhibit_compression_check);
380 }
381 else {
382 (void) compress(ibuf, ph.u_len, obuf); // ignore return value
383 }
384
385 if (ph.c_len < ph.u_len) {
386 const upx_bytep tbuf = nullptr;
387 if (ft == nullptr || ft->id == 0) tbuf = ibuf;
388 ph.overlap_overhead = OVERHEAD;
389 if (!testOverlappingDecompression(obuf, tbuf, ph.overlap_overhead)) {
390 // not in-place compressible
391 ph.c_len = ph.u_len;
392 }
393 }
394 if (ph.c_len >= ph.u_len) {
395 // block is not compressible
396 ph.c_len = ph.u_len;
397 memcpy(obuf, ibuf, ph.c_len);
398 // must update checksum of compressed data
399 ph.c_adler = upx_adler32(ibuf, ph.u_len, ph.c_adler);
400 }
401
402 // write block sizes
403 b_info tmp;
404 if (hdr_u_len) {
405 unsigned hdr_c_len = 0;
406 MemBuffer hdr_obuf;
407 hdr_obuf.allocForCompression(hdr_u_len);
408 int r = upx_compress(hdr_ibuf, hdr_u_len, hdr_obuf, &hdr_c_len,
409 /* &progress callback */ nullptr,
410 ph_forced_method(ph.method), 10,
411 /* &config_t */ nullptr, /* &result_t */ nullptr);
412 if (r != UPX_E_OK)
413 throwInternalError("header compression failed");
414 if (hdr_c_len >= hdr_u_len)
415 throwInternalError("header compression size increase");
416 ph.saved_u_adler = upx_adler32(hdr_ibuf, hdr_u_len, init_u_adler);
417 ph.saved_c_adler = upx_adler32(hdr_obuf, hdr_c_len, init_c_adler);
418 ph.u_adler = upx_adler32(ibuf, ph.u_len, ph.saved_u_adler);
419 ph.c_adler = upx_adler32(obuf, ph.c_len, ph.saved_c_adler);
420 end_u_adler = ph.u_adler;
421 memset(&tmp, 0, sizeof(tmp));
422 set_te32(&tmp.sz_unc, hdr_u_len);
423 set_te32(&tmp.sz_cpr, hdr_c_len);
424 tmp.b_method = (unsigned char) ph_forced_method(ph.method);
425 tmp.b_extra = b_extra;
426 fo->write(&tmp, sizeof(tmp));
427 total_out += sizeof(tmp);
428 b_len += sizeof(b_info);
429 fo->write(hdr_obuf, hdr_c_len);
430 total_out += hdr_c_len;
431 total_in += hdr_u_len;
432 hdr_u_len = 0; // compress hdr one time only
433 }
434 memset(&tmp, 0, sizeof(tmp));
435 set_te32(&tmp.sz_unc, ph.u_len);
436 set_te32(&tmp.sz_cpr, ph.c_len);
437 if (ph.c_len < ph.u_len) {
438 tmp.b_method = (unsigned char) ph.method;
439 if (ft) {
440 tmp.b_ftid = (unsigned char) ft->id;
441 tmp.b_cto8 = ft->cto;
442 }
443 }
444 tmp.b_extra = b_extra;
445 fo->write(&tmp, sizeof(tmp));
446 total_out += sizeof(tmp);
447 b_len += sizeof(b_info);
448
449 if (ft) {
450 ph.u_adler = end_u_adler;
451 }
452 // write compressed data
453 if (ph.c_len < ph.u_len) {
454 fo->write(obuf, ph.c_len);
455 total_out += ph.c_len;
456 // Checks ph.u_adler after decompression, after unfiltering
457 verifyOverlappingDecompression(ft);
458 }
459 else {
460 fo->write(ibuf, ph.u_len);
461 total_out += ph.u_len;
462 }
463
464 total_in += ph.u_len;
465 }
466}

468// Consumes b_info header block and sz_cpr data block from input file 'fi'.
469// De-compresses; appends to output file 'fo' unless rewrite or peeking.
470// For "peeking" without writing: set (fo = nullptr), (is_rewrite = -1)
471// Return actual length when peeking; else 0.
472unsigned PackUnix::unpackExtent(unsigned wanted, OutputFile *fo,
473 unsigned &c_adler, unsigned &u_adler,
474 bool first_PF_X,
475 int is_rewrite // 0(false): write; 1(true): rewrite; -1: no write
476)
477{
478 b_info hdr; memset(&hdr, 0, sizeof(hdr));
479 unsigned inlen = 0; // output index (if-and-only-if peeking)
480 while (wanted) {
481 fi->readx(&hdr, szb_info);
482 int const sz_unc = ph.u_len = get_te32(&hdr.sz_unc);
483 int const sz_cpr = ph.c_len = get_te32(&hdr.sz_cpr);
484 ph.filter_cto = hdr.b_cto8;
485
486 if (sz_unc == 0 || M_LZMA < hdr.b_method) {
487 throwCantUnpack("corrupt b_info");
488 break;
489 }
490 if (sz_unc <= 0 || sz_cpr <= 0)
491 throwCantUnpack("corrupt b_info");
492 if (sz_cpr > sz_unc || sz_unc > (int)blocksize)
493 throwCantUnpack("corrupt b_info");
494
495 // place the input for overlapping de-compression
496 int j = inlen + sz_unc + OVERHEAD - sz_cpr;
497 if (ibuf.getSize() < (unsigned)(j + sz_cpr)) {
498 throwCantUnpack("corrupt b_info");
499 }
500 fi->readx(ibuf+j, sz_cpr);
501 total_in += sz_cpr;
502 // update checksum of compressed data
503 c_adler = upx_adler32(ibuf + j, sz_cpr, c_adler);
504
505 if (sz_cpr < sz_unc) { // block was compressed
506 decompress(ibuf+j, ibuf+inlen, false);
507 if (12==szb_info) { // modern per-block filter
508 if (hdr.b_ftid) {
509 Filter ft(ph.level); // FIXME: ph.level for b_info?
510 ft.init(hdr.b_ftid, 0);
511 ft.cto = hdr.b_cto8;
512 ft.unfilter(ibuf+inlen, sz_unc);
513 }
514 }
515 else { // ancient per-file filter
516 if (first_PF_X) { // Elf32_Ehdr is never filtered
517 first_PF_X = false; // but everything else might be
518 }
519 else if (ph.filter) {
520 Filter ft(ph.level);
521 ft.init(ph.filter, 0);
522 ft.cto = (unsigned char) ph.filter_cto;
523 ft.unfilter(ibuf+inlen, sz_unc);
524 }
525 }
526 }
527 else if (sz_cpr == sz_unc) { // slide literal (non-compressible) block
528 memmove(&ibuf[inlen], &ibuf[j], sz_unc);
529 }
530 // update checksum of uncompressed data
531 u_adler = upx_adler32(ibuf + inlen, sz_unc, u_adler);
532 // write block
533 if (fo) {
534 if (is_rewrite) {
535 fo->rewrite(ibuf, sz_unc);
536 }
537 else {
538 fo->write(ibuf, sz_unc);
539 total_out += sz_unc;
540 }
541 }
542 else if (is_rewrite < 0) { // append to &ibuf[inlen]
543 inlen += sz_unc; // accounting; data is already there
544 if (wanted <= (unsigned)sz_unc) // done
545 break;
546 }
547 else if (wanted < (unsigned)sz_unc) // mismatched end-of-block
548 throwCantUnpack("corrupt b_info");
549 else { // "upx -t": (!fo && !(is_rewrite < 0))
550 // No output.
551 }
552 wanted -= sz_unc;
553 }
554 return inlen;
555}

561// The prize is the value of overlay_offset: the offset of compressed data
562tribool PackUnix::canUnpack()
563{
564 int const small = 32 + sizeof(overlay_offset);
565 // Allow zero-filled last page, for Mac OS X code signing.
566 int bufsize = 2*4096 + 2*small +1;
567 if (bufsize > fi->st_size())
568 bufsize = fi->st_size();
569 MemBuffer buf(bufsize);
570
571 fi->seek(-(off_t)bufsize, SEEK_END);
572 fi->readx(buf, bufsize);
573 return find_overlay_offset(buf);
574}
575
576int PackUnix::find_overlay_offset(MemBuffer const &buf)
577{
578 int const small = 32 + sizeof(overlay_offset);
579 int const bufsize = buf.getSize();
580 int i = bufsize;
581 while (i > small && 0 == buf[--i]) { }
582 i -= small;
583 // allow incompressible extents
584 if (i < 0 || !getPackHeader(buf + i, bufsize - i, true))
585 return false;
586
587 int l = ph.buf_offset + ph.getPackHeaderSize();
588 if (l < 0 || i + l + 4 > bufsize)
589 throwCantUnpack("file corrupted");
590 overlay_offset = get_te32(buf + i + l);
591 if ((off_t)overlay_offset >= file_size)
592 throwCantUnpack("file corrupted");
593
594 return true;
595}

604void PackUnix::unpack(OutputFile *fo)
605{
606 b_info bhdr;
607 unsigned c_adler = upx_adler32(nullptr, 0);
608 unsigned u_adler = upx_adler32(nullptr, 0);
609
610 if (ph.version <= 11) {
611 szb_info = sizeof(bhdr.sz_unc) + sizeof(bhdr.sz_cpr); // old style
612 }
613 // defaults for ph.version == 8
614 unsigned orig_file_size = 0;
615 blocksize = 512 * 1024;
616
617 fi->seek(overlay_offset, SEEK_SET);
618 if (ph.version > 8)
619 {
620 p_info hbuf;
621 fi->readx(&hbuf, sizeof(hbuf));
622 orig_file_size = get_te32(&hbuf.p_filesize);
623 blocksize = get_te32(&hbuf.p_blocksize);
624 off_t max_inflated = file_size * 273; // zlib limit (256 + 16 + 1)
625
626 if (max_inflated < orig_file_size
627 || max_inflated < blocksize
628 || file_size > (off_t)orig_file_size
629 || blocksize > orig_file_size) {
630 throwCantUnpack("file header corrupted");
631 }
632 }
633 else
634 {
635 // skip 4 bytes (program id)
636 fi->seek(4, SEEK_CUR);
637 }
638
639 if ((int)(blocksize + OVERHEAD) < 0)
640 throwCantUnpack("blocksize corrupted");
641 ibuf.alloc(blocksize + OVERHEAD);
642
643 // decompress blocks
644 total_in = 0;
645 total_out = 0;
646 memset(&bhdr, 0, sizeof(bhdr));
647 for (;;)
648 {
649#define buf ibuf
650 int i;
651 unsigned sz_unc, sz_cpr;
652
653 fi->readx(&bhdr, szb_info);
654 ph.u_len = sz_unc = get_te32(&bhdr.sz_unc);
655 ph.c_len = sz_cpr = get_te32(&bhdr.sz_cpr);
656
657 if (sz_unc == 0) // uncompressed size 0 -> EOF
658 {
659 // note: must reload sz_cpr as magic is always stored le32
660 sz_cpr = get_le32(&bhdr.sz_cpr);
661 if (sz_cpr != UPX_MAGIC_LE32) // sz_cpr must be h->magic
662 throwCompressedDataViolation();
663 break;
664 }
665 // Minimum compressed length: LZMA has 5-byte info header.
666 // NRV_d8 has 1-byte initial 8 flag bits, plus end-of-block marker
667 // (32 bit look-back offset of all 1s: encoded as 24 pairs of bits
668 // {not last, 1} then low 8-bits of 0xff; total: 8 + 2*24 + 8 bits
669 // ==> 8 bytes)
670 if (sz_unc <= 0 || sz_cpr <= 5u
671 || sz_cpr > sz_unc || sz_unc > blocksize)
672 throwCantUnpack("corrupt b_info %#x %#x", sz_unc, sz_cpr);
673
674 // Compressed output has control bytes such as the 32-bit
675 // first flag bits of NRV_d32, the 5-byte info of LZMA, etc.
676 // Fuzzers may try sz_cpr shorter than possible.
677 // Use some OVERHEAD for safety.
678 i = blocksize + OVERHEAD - upx::umax(12u, sz_cpr);
679 if (i < 0)
680 throwCantUnpack("corrupt b_info %#x %#x", sz_cpr, blocksize);
681 fi->readx(buf+i, sz_cpr);
682 // update checksum of compressed data

694 }

698 total_out += sz_unc;

702#undef buf
703 }

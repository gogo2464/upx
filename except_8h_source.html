34class Throwable : public std::exception {

36
37protected:

39
40public:
41 Throwable(const Throwable &) noexcept; // copy constructor
42 virtual ~Throwable() noexcept;
43 const char *getMsg() const noexcept { return msg; }
44 int getErrno() const noexcept { return err; }
45 bool isWarning() const noexcept { return is_warning; }
46
47private:
48 OwningPointer(char) msg = nullptr;
49 int err = 0;
50protected:
51 bool is_warning = false; // can be set by subclasses
52
53private:
54 // disable copy assignment and move
55 Throwable &operator=(const Throwable &) noexcept DELETED_FUNCTION; // copy assignment
56 UPX_CXX_DISABLE_MOVE(Throwable)
57 // disable dynamic allocation => force throwing by value
58 UPX_CXX_DISABLE_NEW_DELETE(Throwable)
59 // disable taking the address => force passing by reference
60 // [I'm not too sure about this design decision, but we can always allow it if needed]
61 UPX_CXX_DISABLE_ADDRESS(Throwable)

63private:
64 // static debug stats
65 struct Stats {
66 upx_std_atomic(size_t) counter_total;
67 upx_std_atomic(size_t) counter_current;
68 };
69 static Stats stats;

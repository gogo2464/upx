88#ifndef xspan_mem_size_impl
89template <class T>
90static inline size_t xspan_mem_size_impl(size_t n) {
91#ifdef UPX_VERSION_HEX
92 // check for overflow and sane limits
93 return mem_size(sizeof(T), n);
94#else
95 return sizeof(T) * n;
96#endif
97}

100template <class T>
101static inline size_t xspan_mem_size(size_t n) {
102 return xspan_mem_size_impl<typename TypeForSizeOf<T>::type>(n);
103}

105template <class T>
106static inline void xspan_mem_size_assert_ptrdiff(ptrdiff_t n) {
107 if (n >= 0)
108 (void) xspan_mem_size<T>((size_t) n);
109 else
110 (void) xspan_mem_size<T>((size_t) -n);
111}

32#define CT16(f, cond, addvalue, get, set) \
33 byte *b = f->buf; \
34 byte *b_end = b + f->buf_len - 3; \
35 do { \
36 if (cond) { \
37 b += 1; \
38 unsigned a = (unsigned) (b - f->buf); \
39 f->lastcall = a; \
40 set(b, get(b) + (addvalue)); \
41 f->calls++; \
42 b += 2 - 1; \
43 } \
44 } while (++b < b_end); \
45 if (f->lastcall) \
46 f->lastcall += 2; \
47 return 0;

391static int CTarm64(Filter *f, int dir) { // dir: 1, 0, -1
392 byte *b = f->buf; // will be incremented
393 byte *const b_end = b + f->buf_len - 4;
394 do {
395 unsigned const a = b - f->buf;
396 int const d = dir * (f->addvalue + (a >> 2));
397 unsigned const v = get_le32(f->buf); // the 32-bit instruction
398 if (0x05 == (0x1f & (v >> 26))) { // b, bl
399 f->lastcall = a;
400 if (dir)
401 set_le26(b, v + d);
402 f->calls++;
403 } else if ((0x54 == (v >> 24)) // b.cond
404 || (0x1a == ((v >> 25) & 0x3f)) // cb{z,nz}
405 ) {
406 f->lastcall = a;
407 if (dir)
408 set_le19_5(b, (v >> 5) + d);
409 f->calls++;
410 } else if (0x1b == ((v >> 25) & 0x3f)) { // tb{z,nz}
411 f->lastcall = a;
412 if (dir)
413 set_le14_5(b, (v >> 5) + d);
414 f->calls++;
415 }
416 b += 4;
417 } while (b < b_end);
418 if (f->lastcall)
419 f->lastcall += 4;
420 return 0;
421}

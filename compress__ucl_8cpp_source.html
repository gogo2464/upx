34static int convert_errno_from_ucl(int r) {
35 switch (r) {
36 case UCL_E_OK:
37 return UPX_E_OK;
38 case UCL_E_ERROR:
39 return UPX_E_ERROR;
40 case UCL_E_OUT_OF_MEMORY:
41 return UPX_E_OUT_OF_MEMORY;
42 case UCL_E_NOT_COMPRESSIBLE:
43 return UPX_E_NOT_COMPRESSIBLE;
44 case UCL_E_INPUT_OVERRUN:
45 return UPX_E_INPUT_OVERRUN;
46 case UCL_E_OUTPUT_OVERRUN:
47 return UPX_E_OUTPUT_OVERRUN;
48 case UCL_E_LOOKBEHIND_OVERRUN:
49 return UPX_E_LOOKBEHIND_OVERRUN;

79int upx_ucl_compress(const upx_bytep src, unsigned src_len, upx_bytep dst, unsigned *dst_len,
80 upx_callback_t *cb_parm, int method, int level,
81 const upx_compress_config_t *cconf_parm, upx_compress_result_t *cresult) {
82 int r;
83 assert(level > 0);
84 assert(cresult != nullptr);
85
86 COMPILE_TIME_ASSERT(sizeof(ucl_compress_config_t) == sizeof(REAL_ucl_compress_config_t))
87
88 ucl_progress_callback_t cb;
89 cb.callback = nullptr;
90 cb.user = nullptr;
91 if (cb_parm && cb_parm->nprogress) {
92 cb.callback = wrap_nprogress_ucl;
93 cb.user = cb_parm;
94 }
95
96 ucl_compress_config_t cconf;
97 cconf.reset();
98 if (cconf_parm) {
99 // cconf = cconf_parm->conf_ucl; // struct copy
100 memcpy(&cconf, &cconf_parm->conf_ucl, sizeof(cconf));
101 }
102
103 ucl_uint *const res = cresult->result_ucl.result;
104 // assume no info available - fill in worst case results
105 // res[0] = 1; // min_offset_found - NOT USED
106 res[1] = src_len - 1; // max_offset_found
107 // res[2] = 2; // min_match_found - NOT USED
108 res[3] = src_len - 1; // max_match_found
109 // res[4] = 1; // min_run_found - NOT USED
110 res[5] = src_len; // max_run_found
111 res[6] = 1; // first_offset_found
112 // res[7] = 999999; // same_match_offsets_found - NOT USED
113
114 // prepare bit-buffer settings
115 cconf.bb_endian = 0;
116 cconf.bb_size = 0;
117 if (method >= M_NRV2B_LE32 && method <= M_NRV2E_LE16) {
118 static const upx_uint8_t sizes[3] = {32, 8, 16};
119 cconf.bb_size = sizes[(method - M_NRV2B_LE32) % 3];
120 } else {
121 throwInternalError("unknown compression method");
122 return UPX_E_ERROR;
123 }
124
125 // optimize compression params
126 if (level <= 3 && cconf.max_offset == UCL_UINT_MAX)
127 cconf.max_offset = 8 * 1024 - 1;
128 else if (level == 4 && cconf.max_offset == UCL_UINT_MAX)
129 cconf.max_offset = 32 * 1024 - 1;
130
131 if M_IS_NRV2B (method)
132 r = ucl_nrv2b_99_compress(src, src_len, dst, dst_len, &cb, level, &cconf, res);
133 else if M_IS_NRV2D (method)
134 r = ucl_nrv2d_99_compress(src, src_len, dst, dst_len, &cb, level, &cconf, res);
135 else if M_IS_NRV2E (method)
136 r = ucl_nrv2e_99_compress(src, src_len, dst, dst_len, &cb, level, &cconf, res);
137 else {
138 throwInternalError("unknown compression method");
139 return UPX_E_ERROR;
140 }
141
142 // make sure first_offset_found is set
143 if (res[6] == 0)
144 res[6] = 1;
145
146 return convert_errno_from_ucl(r);
147}

153int upx_ucl_decompress(const upx_bytep src, unsigned src_len, upx_bytep dst, unsigned *dst_len,
154 int method, const upx_compress_result_t *cresult) {
155 int r;
156
157 switch (method) {
158 case M_NRV2B_8:
159 r = ucl_nrv2b_decompress_safe_8(src, src_len, dst, dst_len, nullptr);
160 break;
161 case M_NRV2B_LE16:
162 r = ucl_nrv2b_decompress_safe_le16(src, src_len, dst, dst_len, nullptr);
163 break;
164 case M_NRV2B_LE32:
165 r = ucl_nrv2b_decompress_safe_le32(src, src_len, dst, dst_len, nullptr);
166 break;
167 case M_NRV2D_8:
168 r = ucl_nrv2d_decompress_safe_8(src, src_len, dst, dst_len, nullptr);
169 break;
170 case M_NRV2D_LE16:
171 r = ucl_nrv2d_decompress_safe_le16(src, src_len, dst, dst_len, nullptr);
172 break;
173 case M_NRV2D_LE32:
174 r = ucl_nrv2d_decompress_safe_le32(src, src_len, dst, dst_len, nullptr);
175 break;
176 case M_NRV2E_8:
177 r = ucl_nrv2e_decompress_safe_8(src, src_len, dst, dst_len, nullptr);
178 break;
179 case M_NRV2E_LE16:
180 r = ucl_nrv2e_decompress_safe_le16(src, src_len, dst, dst_len, nullptr);
181 break;
182 case M_NRV2E_LE32:
183 r = ucl_nrv2e_decompress_safe_le32(src, src_len, dst, dst_len, nullptr);
184 break;
185 default:
186 throwInternalError("unknown decompression method");
187 return UPX_E_ERROR;
188 }
189
190 UNUSED(cresult);
191 return convert_errno_from_ucl(r);
192}

283static bool check_ucl(const int method, const unsigned expected_c_len) {
284 const unsigned u_len = 16384;
285 const unsigned c_extra = 4096;
286 MemBuffer u_buf, c_buf, d_buf;
287 unsigned c_len, d_len;
288 upx_compress_result_t cresult;
289 int r;
290 const int level = 3; // don't waste time
291
292 u_buf.alloc(u_len);
293 memset(u_buf, 0, u_len);
294 c_buf.allocForCompression(u_len, c_extra);
295 d_buf.allocForDecompression(u_len);
296
297 c_len = c_buf.getSize() - c_extra;
298 r = upx_ucl_compress(raw_bytes(u_buf, u_len), u_len, raw_index_bytes(c_buf, c_extra, c_len),
299 &c_len, nullptr, method, level, NULL_cconf, &cresult);
300 if (r != 0 || c_len != expected_c_len)
301 return false;
302
303 d_len = d_buf.getSize();
304 r = upx_ucl_decompress(raw_index_bytes(c_buf, c_extra, c_len), c_len, raw_bytes(d_buf, d_len),
305 &d_len, method, nullptr);
306 if (r != 0 || d_len != u_len || memcmp(u_buf, d_buf, u_len) != 0)
307 return false;
308
309 d_len = u_len - 1;
310 r = upx_ucl_decompress(raw_index_bytes(c_buf, c_extra, c_len), c_len, raw_bytes(d_buf, d_len),
311 &d_len, method, nullptr);
312 if (r == 0)
313 return false;
314
315 // TODO: rewrite Packer::findOverlapOverhead() so that we can test it here
316 // unsigned x_len = d_len;
317 // r = upx_ucl_test_overlap(c_buf, u_buf, c_extra, c_len, &x_len, method, nullptr);
318 return true;
319}

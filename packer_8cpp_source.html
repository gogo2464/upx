154bool Packer::compress(SPAN_P(byte) i_ptr, unsigned i_len, SPAN_P(byte) o_ptr,
155 const upx_compress_config_t *cconf_parm) {
156 ph.u_len = i_len;
157 ph.c_len = 0;
158 assert(ph.level >= 1);
159 assert(ph.level <= 10);
160
161 // Avoid too many progress bar updates. 64 is s->bar_len in ui.cpp.
162 unsigned step = (ph.u_len < 64 * 1024) ? 0 : ph.u_len / 64;
163
164 // save current checksums
165 ph.saved_u_adler = ph.u_adler;
166 ph.saved_c_adler = ph.c_adler;
167 // update checksum of uncompressed data
168 ph.u_adler = upx_adler32(raw_bytes(i_ptr, ph.u_len), ph.u_len, ph.u_adler);
169
170 // set compression parameters
171 upx_compress_config_t cconf;
172 cconf.reset();
173 if (cconf_parm)
174 cconf = *cconf_parm;
175 // cconf options
176 int method = ph_forced_method(ph.method);
177 if (M_IS_NRV2B(method) || M_IS_NRV2D(method) || M_IS_NRV2E(method)) {
178 if (opt->crp.crp_ucl.c_flags != -1)
179 cconf.conf_ucl.c_flags = opt->crp.crp_ucl.c_flags;
180 if (opt->crp.crp_ucl.p_level != -1)
181 cconf.conf_ucl.p_level = opt->crp.crp_ucl.p_level;
182 if (opt->crp.crp_ucl.h_level != -1)
183 cconf.conf_ucl.h_level = opt->crp.crp_ucl.h_level;
184 if (opt->crp.crp_ucl.max_offset != UINT_MAX &&
185 opt->crp.crp_ucl.max_offset < cconf.conf_ucl.max_offset)
186 cconf.conf_ucl.max_offset = opt->crp.crp_ucl.max_offset;
187 if (opt->crp.crp_ucl.max_match != UINT_MAX &&
188 opt->crp.crp_ucl.max_match < cconf.conf_ucl.max_match)
189 cconf.conf_ucl.max_match = opt->crp.crp_ucl.max_match;
190#if (WITH_NRV)
191 if ((ph.level >= 7 || (ph.level >= 4 && ph.u_len >= 512 * 1024)) && !opt->prefer_ucl)
192 step = 0;
193#endif
194 }
195 if (M_IS_LZMA(method)) {
196 upx::oassign(cconf.conf_lzma.pos_bits, opt->crp.crp_lzma.pos_bits);
197 upx::oassign(cconf.conf_lzma.lit_pos_bits, opt->crp.crp_lzma.lit_pos_bits);
198 upx::oassign(cconf.conf_lzma.lit_context_bits, opt->crp.crp_lzma.lit_context_bits);
199 upx::oassign(cconf.conf_lzma.dict_size, opt->crp.crp_lzma.dict_size);
200 upx::oassign(cconf.conf_lzma.num_fast_bytes, opt->crp.crp_lzma.num_fast_bytes);
201 }
202 if (M_IS_DEFLATE(method)) {
203 upx::oassign(cconf.conf_zlib.mem_level, opt->crp.crp_zlib.mem_level);
204 upx::oassign(cconf.conf_zlib.window_bits, opt->crp.crp_zlib.window_bits);
205 upx::oassign(cconf.conf_zlib.strategy, opt->crp.crp_zlib.strategy);
206 }
207 if (uip->ui_pass >= 0)
208 uip->ui_pass++;
209 uip->startCallback(ph.u_len, step, uip->ui_pass, uip->ui_total_passes);
210 uip->firstCallback();
211
212 // OutputFile::dump("data.raw", in, ph.u_len);
213
214 // compress
215 int r = upx_compress(raw_bytes(i_ptr, ph.u_len), ph.u_len, raw_bytes(o_ptr, 0), &ph.c_len,
216 uip->getCallback(), method, ph.level, &cconf, &ph.compress_result);
217
218 // uip->finalCallback(ph.u_len, ph.c_len);
219 uip->endCallback();
220
221 if (r == UPX_E_OUT_OF_MEMORY)
222 throwOutOfMemoryException();
223 if (r != UPX_E_OK)
224 throwInternalError("compression failed");
225
226 if (M_IS_NRV2B(method) || M_IS_NRV2D(method) || M_IS_NRV2E(method)) {
227 const ucl_uint *res = ph.compress_result.result_ucl.result;
228 // ph.min_offset_found = res[0];
229 ph.max_offset_found = res[1];
230 // ph.min_match_found = res[2];
231 ph.max_match_found = res[3];
232 // ph.min_run_found = res[4];
233 ph.max_run_found = res[5];
234 ph.first_offset_found = res[6];
235 // ph.same_match_offsets_found = res[7];
236 if (cconf_parm) {
237 assert(cconf.conf_ucl.max_offset == 0 ||
238 cconf.conf_ucl.max_offset >= ph.max_offset_found);
239 assert(cconf.conf_ucl.max_match == 0 || cconf.conf_ucl.max_match >= ph.max_match_found);
240 }
241 }
242
243 NO_printf("\nPacker::compress: %d/%d: %7d -> %7d\n", method, ph.level, ph.u_len, ph.c_len);
244 if (!checkCompressionRatio(ph.u_len, ph.c_len))
245 return false;
246 // return in any case if not compressible
247 if (ph.c_len >= ph.u_len)
248 return false;
249
250 // update checksum of compressed data
251 ph.c_adler = upx_adler32(raw_bytes(o_ptr, ph.c_len), ph.c_len, ph.c_adler);
252 // Decompress and verify. Skip this when using the fastest level.
253 if (!ph_skipVerify(ph)) {
254 // decompress
255 unsigned new_len = ph.u_len;
256 r = upx_decompress(raw_bytes(o_ptr, ph.c_len), ph.c_len, raw_bytes(i_ptr, ph.u_len),
257 &new_len, method, &ph.compress_result);
258 if (r == UPX_E_OUT_OF_MEMORY)
259 throwOutOfMemoryException();
260 // printf("%d %d: %d %d %d\n", method, r, ph.c_len, ph.u_len, new_len);
261 if (r != UPX_E_OK)
262 throwInternalError("decompression failed");
263 if (new_len != ph.u_len)
264 throwInternalError("decompression failed (size error)");
265
266 // verify decompression
267 if (ph.u_adler != upx_adler32(raw_bytes(i_ptr, ph.u_len), ph.u_len, ph.saved_u_adler))
268 throwInternalError("decompression failed (checksum error)");
269 }
270 return true;
271}

273bool Packer::checkDefaultCompressionRatio(unsigned u_len, unsigned c_len) const {
274 assert((int) u_len > 0);
275 assert((int) c_len > 0);
276 if (c_len >= u_len)
277 return false;
278 unsigned gain = u_len - c_len;
279
280 if (gain < 512) // need at least 512 bytes gain
281 return false;
282#if 1
283 if (gain >= 4096) // ok if we have at least 4096 bytes gain
284 return true;
285#endif
286 if (gain >= u_len / 16) // ok if we have at least 6.25% gain
287 return true;
288 return false;
289}

318void Packer::verifyOverlappingDecompression(Filter *ft) {
319 assert(ph.c_len < ph.u_len);
320 assert((int) ph.overlap_overhead > 0);
321 // Idea:
322 // obuf[] was allocated with MemBuffer::allocForCompression(), and
323 // its contents are no longer needed, i.e. the compressed data
324 // must have been already written.
325 // We now can perform a real overlapping decompression and
326 // verify the checksum.
327 //
328 // Note:
329 // This verify is just because of complete paranoia that there
330 // could be a hidden bug in the upx_test_overlap implementation,
331 // and it should not be necessary at all.
332 //
333 // See also:
334 // Filter::verifyUnfilter()
335
336 if (ph_skipVerify(ph))
337 return;
338 unsigned offset = (ph.u_len + ph.overlap_overhead) - ph.c_len;
339 if (offset + ph.c_len > obuf.getSize())
340 return;
341 memmove(obuf + offset, obuf, ph.c_len);
342 decompress(obuf + offset, obuf, true, ft);
343 obuf.checkState();
344}

368unsigned Packer::findOverlapOverhead(const byte *buf, const byte *tbuf, unsigned range,
369 unsigned upper_limit) const {
370 assert((int) range >= 0);
371
372 // prepare to deal with very pessimistic values
373 unsigned low = 1;
374 unsigned high = UPX_MIN(ph.u_len + 512, upper_limit);
375 // but be optimistic for first try (speedup)
376 unsigned m = UPX_MIN(16u, high);
377 //
378 unsigned overhead = 0;
379 unsigned nr = 0; // statistics
380
381 while (high >= low) {
382 assert(m >= low);
383 assert(m <= high);
384 assert(m < overhead || overhead == 0);
385 nr++;
386 bool success = testOverlappingDecompression(buf, tbuf, m);
387 // printf("testOverlapOverhead(%d): %d %d: %d -> %d\n", nr, low, high, m, (int)success);
388 if (success) {
389 overhead = m;
390 // Succeed early if m lies in [low .. low+range-1], i.e. if
391 // if the range of the current interval is <= range.
392 // if (m <= low + range - 1)
393 // if (m < low + range)
394 if (m - low < range) // avoid underflow
395 break;
396 high = m - 1;
397 } else
398 low = m + 1;
400 m = (low & high) + ((low ^ high) >> 1); // avoid overflow
401 }
402
403 // printf("findOverlapOverhead: %d (%d tries)\n", overhead, nr);
404 if (overhead == 0)
405 throwInternalError("this is an oo bug");
406
407 UNUSED(nr);
408 return overhead;
409}

440void Packer::copyOverlay(OutputFile *fo, unsigned overlay, MemBuffer &buf, bool do_seek) {
441 assert((int) overlay >= 0);
442 assert(overlay < file_size_u);
443 buf.checkState();
444 if (!fo || overlay == 0)
445 return;
446 if (opt->overlay != opt->COPY_OVERLAY) {
447 assert(opt->overlay == opt->STRIP_OVERLAY);
448 infoWarning("stripping overlay: %d bytes", overlay);
449 return;
450 }
451 info("Copying overlay: %d bytes", overlay);
452 if (do_seek)
453 fi->seek(-(upx_off_t) overlay, SEEK_END);
454
455 // get buffer size, align to improve i/o speed
456 unsigned buf_size = buf.getSize();
457 if (buf_size > 65536)
458 buf_size = ALIGN_DOWN(buf_size, 4096u);
459 assert((int) buf_size > 0);
460
461 do {
462 unsigned len = overlay < buf_size ? overlay : buf_size;
463 fi->readx(buf, len);
464 fo->write(buf, len);
465 overlay -= len;
466 } while (overlay > 0);
467 buf.checkState();
468}

622}

908int Packer::prepareMethods(int *methods, int ph_method, const int *all_methods) const {
909 int nmethods = 0;
910 if (!opt->all_methods || all_methods == nullptr || (-0x80 == (ph_method >> 24))) {
911 methods[nmethods++] = ph_forced_method(ph_method);
912 return nmethods;
913 }
914 for (int mm = 0; all_methods[mm] != M_END; ++mm) {
915 int method = all_methods[mm];
916 if (method == M_ULTRA_BRUTE && !opt->ultra_brute)
917 break;
918 if (method == M_SKIP || method == M_ULTRA_BRUTE)
919 continue;
920 if (opt->all_methods && opt->all_methods_use_lzma != 1 && M_IS_LZMA(method))
921 continue;
922 // check duplicate
923 assert(Packer::isValidCompressionMethod(method));
924 for (int i = 0; i < nmethods; i++)
925 assert(method != methods[i]);
926 // use this method
927 methods[nmethods++] = method;
928 }
929 // debug
930 if (opt->debug.use_random_method && nmethods >= 2) {
931 int method = methods[upx_rand() % nmethods];
932 NO_printf("\nuse_random_method = %#x (%d)\n", method, nmethods);
933 methods[0] = method;
934 nmethods = 1;
935 }
936 return nmethods;
937}

939static int prepareFilters(int *filters, int &filter_strategy, const int *all_filters) {
940 int nfilters = 0;
941
942 // setup filter filter_strategy
943 if (filter_strategy == 0) {
944 if (opt->all_filters)
945 // choose best from all available filters
946 filter_strategy = INT_MAX;
947 else if (opt->filter >= 0 && Filter::isValidFilter(opt->filter, all_filters))
948 // try opt->filter
949 filter_strategy = -2;
950 else
951 // try the first working filter
952 filter_strategy = -1;
953 }
954 assert(filter_strategy != 0);
955
956 if (filter_strategy <= -3)
957 goto done;
958 if (filter_strategy == -2) {
959 if (opt->filter >= 0 && Filter::isValidFilter(opt->filter, all_filters)) {
960 filters[nfilters++] = opt->filter;
961 goto done;
962 }
963 filter_strategy = -1;
964 }
965 assert(filter_strategy >= -1);
966
967 while (all_filters && *all_filters != FT_END) {
968 int filter_id = *all_filters++;
969 if (filter_id == FT_ULTRA_BRUTE && !opt->ultra_brute)
970 break;
971 if (filter_id == FT_SKIP || filter_id == FT_ULTRA_BRUTE)
972 continue;
973 if (filter_id == 0)
974 continue;
975 // check duplicate
976 assert(Filter::isValidFilter(filter_id));
977 for (int i = 0; i < nfilters; i++)
978 assert(filter_id != filters[i]);
979 // use this filter
980 filters[nfilters++] = filter_id;
981 if (filter_strategy >= 0 && nfilters >= filter_strategy)
982 break;
983 }
984
985done:
986 // filter_strategy now only means "stop after first successful filter"
987 filter_strategy = (filter_strategy < 0) ? -1 : 0;
988 // make sure that we have a "no filter" fallback
989 bool have_filter0 = false;
990 for (int i = 0; i < nfilters; i++)
991 if (filters[i] == 0) {
992 have_filter0 = true;
993 break;
994 }
995 if (!have_filter0)
996 filters[nfilters++] = 0;
997 // debug
998 if (opt->debug.use_random_filter && nfilters >= 3 && filters[nfilters - 1] == 0) {
999 int filter_id = filters[upx_rand() % (nfilters - 1)];
1000 if (filter_id > 0) {
1001 NO_printf("\nuse_random_filter = %#x (%d)\n", filter_id, nfilters - 1);
1002 filters[0] = filter_id;
1003 filters[1] = 0;
1004 nfilters = 2;
1005 }
1006 }
1007 return nfilters;
1008}

1010void Packer::compressWithFilters(byte *i_ptr,
1011 const unsigned i_len, // written and restored by filters
1012 byte *const o_ptr, // where to put compressed output
1013 byte *f_ptr,
1014 const unsigned f_len, // subset of [*i_ptr, +i_len)
1015 byte *const hdr_ptr, const unsigned hdr_len,
1016 Filter *const parm_ft, // updated
1017 const unsigned overlap_range,
1018 upx_compress_config_t const *const cconf,
1019 int filter_strategy, // in+out for prepareFilters
1020 bool const inhibit_compression_check) {
1021 parm_ft->buf_len = f_len;
1022 // struct copies
1023 const PackHeader orig_ph = this->ph;
1024 PackHeader best_ph = this->ph;
1025 const Filter orig_ft = *parm_ft;
1026 Filter best_ft = *parm_ft;
1027 //
1028 best_ph.c_len = i_len;
1029 best_ph.overlap_overhead = 0;
1030 unsigned best_ph_lsize = 0;
1031 unsigned best_hdr_c_len = 0;
1032
1033 // preconditions
1034 assert(orig_ph.filter == 0);
1035 assert(orig_ft.id == 0);
1036
1037 // prepare methods and filters
1038 int methods[MAX_METHODS];
1039 int nmethods = prepareMethods(methods, ph.method, getCompressionMethods(M_ALL, ph.level));
1040 assert_noexcept(nmethods > 0);
1041 assert_noexcept(nmethods < (int) MAX_METHODS);
1042 int filters[MAX_FILTERS];
1043 int nfilters = prepareFilters(filters, filter_strategy, getFilters());
1044 assert_noexcept(nfilters > 0);
1045 assert_noexcept(nfilters < (int) MAX_FILTERS);
1046#if 0
1047 printf("compressWithFilters: m(%d):", nmethods);
1048 for (int i = 0; i < nmethods; i++)
1049 printf(" %#x", methods[i]);
1050 printf(" f(%d):", nfilters);
1051 for (int i = 0; i < nfilters; i++)
1052 printf(" %#x", filters[i]);
1053 printf("\n");
1054#endif
1055
1056 // update total_passes; previous (ui_total_passes > 0) means incremental
1057 if (!ph_is_forced_method(ph.method)) {
1058 if (uip->ui_total_passes > 0)
1059 uip->ui_total_passes -= 1;
1060 if (filter_strategy < 0)
1061 uip->ui_total_passes += nmethods;
1062 else
1063 uip->ui_total_passes += nfilters * nmethods;
1064 }
1065
1066 // Working buffer for compressed data. Don't waste memory and allocate as needed.
1067 byte *o_tmp = o_ptr;
1068 MemBuffer o_tmp_buf;
1069
1070 // compress using all methods/filters
1071 int nfilters_success_total = 0;
1072 for (int mm = 0; mm < nmethods; mm++) // for all methods
1073 {
1074 NO_printf("\nmethod %d (%d of %d)\n", methods[mm], 1 + mm, nmethods);
1075 assert(isValidCompressionMethod(methods[mm]));
1076 unsigned hdr_c_len = 0;
1077 if (hdr_ptr != nullptr && hdr_len) {
1078 if (nfilters_success_total != 0 && o_tmp == o_ptr) {
1079 // do not overwrite o_ptr
1080 o_tmp_buf.allocForCompression(UPX_MAX(hdr_len, i_len));
1081 o_tmp = o_tmp_buf;
1082 }
1083 int r = upx_compress(hdr_ptr, hdr_len, o_tmp, &hdr_c_len, nullptr, methods[mm], 10,
1084 nullptr, nullptr);
1085 if (r != UPX_E_OK)
1086 throwInternalError("header compression failed");
1087 if (hdr_c_len >= hdr_len)
1088 throwInternalError("header compression size increase");
1089 }
1090 int nfilters_success_mm = 0;
1091 for (int ff = 0; ff < nfilters; ff++) // for all filters
1092 {
1093 assert(isValidFilter(filters[ff]));
1094 // get fresh packheader
1095 ph = orig_ph;
1096 ph.method = methods[mm];
1097 ph.filter = filters[ff];
1098 ph.overlap_overhead = 0;
1099 // get fresh filter
1100 Filter ft = orig_ft;
1101 ft.init(ph.filter, orig_ft.addvalue);
1102 // filter
1103 optimizeFilter(&ft, f_ptr, f_len);
1104 bool success = ft.filter(f_ptr, f_len);
1105 if (ft.id != 0 && ft.calls == 0) {
1106 // filter did not do anything - no need to call ft.unfilter()
1107 success = false;
1108 }
1109 if (!success) {
1110 // filter failed or was useless
1111 if (filter_strategy >= 0) {
1112 // adjust ui passes
1113 if (uip->ui_pass >= 0)
1114 uip->ui_pass++;
1115 }
1116 continue;
1117 }
1118 // filter success
1119 NO_printf("\nfilter: id 0x%02x size %6d, calls %5d/%5d/%3d/%5d/%5d, cto 0x%02x\n",
1120 ft.id, ft.buf_len, ft.calls, ft.noncalls, ft.wrongcalls, ft.firstcall,
1121 ft.lastcall, ft.cto);
1122 if (nfilters_success_total != 0 && o_tmp == o_ptr) {
1123 o_tmp_buf.allocForCompression(i_len);
1124 o_tmp = o_tmp_buf;
1125 }
1126 nfilters_success_total++;
1127 nfilters_success_mm++;
1128 ph.filter_cto = ft.cto;
1129 ph.n_mru = ft.n_mru;
1130 // compress
1131 if (compress(i_ptr, i_len, o_tmp, cconf)) {
1132 unsigned lsize = 0;
1133 // findOverlapOperhead() might be slow; omit if already too big.
1134 if (ph.c_len + lsize + hdr_c_len <=
1135 best_ph.c_len + best_ph_lsize + best_hdr_c_len) {
1136 // get results
1137 ph.overlap_overhead = findOverlapOverhead(o_tmp, i_ptr, overlap_range);
1138 if (-4 < filter_strategy) {
1139 buildLoader(&ft);
1140 lsize = getLoaderSize();
1141 assert(lsize > 0);
1142 }
1143 }
1144 NO_printf("\n%2d %02x: %d +%4d +%3d = %d (best: %d +%4d +%3d = %d)\n", ph.method,
1145 ph.filter, ph.c_len, lsize, hdr_c_len, ph.c_len + lsize + hdr_c_len,
1146 best_ph.c_len, best_ph_lsize, best_hdr_c_len,
1147 best_ph.c_len + best_ph_lsize + best_hdr_c_len);
1148 bool update = false;
1149 if (ph.c_len + lsize + hdr_c_len < best_ph.c_len + best_ph_lsize + best_hdr_c_len)
1150 update = true;
1151 else if (ph.c_len + lsize + hdr_c_len ==
1152 best_ph.c_len + best_ph_lsize + best_hdr_c_len) {
1153 // prefer smaller loaders
1154 if (lsize + hdr_c_len < best_ph_lsize + best_hdr_c_len)
1155 update = true;
1156 else if (lsize + hdr_c_len == best_ph_lsize + best_hdr_c_len) {
1157 // prefer less overlap_overhead
1158 if (ph.overlap_overhead < best_ph.overlap_overhead)
1159 update = true;
1160 }
1161 }
1162 if (update) {
1163 assert((int) ph.overlap_overhead > 0);
1164 // update o_ptr[] with best version
1165 if (o_tmp != o_ptr)
1166 memcpy(o_ptr, o_tmp, ph.c_len);
1167 // save compression results
1168 best_ph = ph;
1169 best_ph_lsize = lsize;
1170 best_hdr_c_len = hdr_c_len;
1171 best_ft = ft;
1172 }
1173 }
1174 // restore - unfilter with verify
1175 ft.unfilter(f_ptr, f_len, true);
1176 if (filter_strategy < 0)
1177 break;
1178 }
1179 assert(nfilters_success_mm > 0);
1180 }
1181
1182 // postconditions 1)
1183 assert(nfilters_success_total > 0);
1184 assert(best_ph.u_len == orig_ph.u_len);
1185 assert(best_ph.filter == best_ft.id);
1186 assert(best_ph.filter_cto == best_ft.cto);
1187 // FIXME assert(best_ph.n_mru == best_ft.n_mru);
1188
1189 // copy back results
1190 this->ph = best_ph;
1191 *parm_ft = best_ft;
1192
1193 // Finally, check compression ratio.
1194 // Might be inhibited when blocksize < file_size, for instance.
1195 if (!inhibit_compression_check) {
1196 if (best_ph.c_len + best_ph_lsize >= best_ph.u_len)
1197 throwNotCompressible();
1198 if (!checkCompressionRatio(best_ph.u_len, best_ph.c_len))
1199 throwNotCompressible();
1200
1201 // postconditions 2)
1202 assert(best_ph.overlap_overhead > 0);
1203 }
1204
1205 // convenience
1206 buildLoader(&best_ft);
1207}

78void PackLinuxElf32x86interp::pack1(OutputFile *fo, Filter &)
79{
80 fi->seek(0, SEEK_SET);
81 fi->readx(&ehdri, sizeof(ehdri));
82 assert(ehdri.e_phoff == sizeof(Elf32_Ehdr)); // checked by canPack()
83 sz_phdrs = ehdri.e_phnum * ehdri.e_phentsize;
84
85 phdri = New(Elf32_Phdr, ehdri.e_phnum);
86 fi->seek(ehdri.e_phoff, SEEK_SET);
87 fi->readx(phdri, sz_phdrs);
88
89#define E Elf32_Ehdr
90 cprElfHdr3 h3;
91 memset(&h3, 0, sizeof(h3));
92 memcpy(h3.ehdr.e_ident, "\177ELF", 4);
93 h3.ehdr.e_ident[E::EI_CLASS] = E::ELFCLASS32;
94 h3.ehdr.e_ident[E::EI_DATA] = E::ELFDATA2LSB;
95 h3.ehdr.e_ident[E::EI_VERSION] = E::EV_CURRENT;
96 h3.ehdr.e_ident[E::EI_OSABI] = E::ELFOSABI_LINUX;
97 h3.ehdr.e_ident[E::EI_ABIVERSION] = E::EV_CURRENT;
98 h3.ehdr.e_type = E::ET_EXEC;
99 h3.ehdr.e_machine = E::EM_386;
100 h3.ehdr.e_version = 1;
101 h3.ehdr.e_phoff = sizeof(Elf32_Ehdr);
102 h3.ehdr.e_ehsize = sizeof(Elf32_Ehdr);
103 h3.ehdr.e_phentsize = sizeof(Elf32_Phdr);
104 h3.ehdr.e_phnum = 3;
105 h3.phdr[0].p_type = PT_LOAD;
106 h3.phdr[0].p_flags = Elf32_Phdr::PF_X | Elf32_Phdr::PF_R;
107 h3.phdr[0].p_align = 0x1000;
108 h3.phdr[1].p_type = PT_LOAD;
109 h3.phdr[1].p_flags = Elf32_Phdr::PF_W | Elf32_Phdr::PF_R;
110 h3.phdr[1].p_align = 1;
111 h3.phdr[2].p_type = PT_INTERP;
112 h3.phdr[2].p_offset = (char *)&h3.phdr[2].p_vaddr - (char *)&h3;
113 memcpy(&h3.phdr[2].p_vaddr, "/upxrun", h3.phdr[2].p_filesz = 8);
114 h3.phdr[2].p_align = 1;
115
116 if (opt->o_unix.make_ptinterp) { // unusual "once per release"
117 *(cprElfHdr3 *)(void *)&elfout = h3;
118 elfout.ehdr.e_phnum = 1;
119 fo->write(&elfout, elfout.ehdr.e_ehsize + elfout.ehdr.e_phentsize);
120 }
121 else { // usual case
122 generateElfHdr(fo, &h3, getbrk(phdri, ehdri.e_phnum));
123 }
124#undef E
125}

198void PackLinuxElf32x86interp::unpack(OutputFile *fo)
199{
200#define MAX_INTERP_HDR 512
201 union {
202 unsigned char buf[MAX_INTERP_HDR];
203 //struct { Elf32_Ehdr ehdr; Elf32_Phdr phdr; } e;
204 } u;
205 Elf32_Ehdr *const ehdr = (Elf32_Ehdr *) u.buf;
206 Elf32_Phdr const *phdr = (Elf32_Phdr *) (u.buf + sizeof(*ehdr));
207
208 {
209 fi->seek(0, SEEK_SET);
210 fi->readx(u.buf, MAX_INTERP_HDR);
211 unsigned const e_entry = get_te32(&ehdr->e_entry);
212 if (e_entry < 0x401180) { /* old style, 8-byte b_info */
213 szb_info = 2*sizeof(unsigned);
214 }
215 }
216
217 fi->seek(overlay_offset, SEEK_SET);
218 p_info hbuf;
219 fi->readx(&hbuf, sizeof(hbuf));
220 unsigned orig_file_size = get_te32(&hbuf.p_filesize);
221 blocksize = get_te32(&hbuf.p_blocksize);
222 if (file_size > (off_t)orig_file_size || blocksize > orig_file_size)
223 throwCantUnpack("file header corrupted");
224
225 ibuf.alloc(blocksize + OVERHEAD);
226 b_info bhdr; memset(&bhdr, 0, sizeof(bhdr));
227 fi->readx(&bhdr, szb_info);
228 ph.u_len = get_te32(&bhdr.sz_unc);
229 ph.c_len = get_te32(&bhdr.sz_cpr);
230 ph.filter_cto = bhdr.b_cto8;
231
232 // Uncompress Ehdr and Phdrs.
233 fi->readx(ibuf, ph.c_len);
234 decompress(ibuf, (upx_byte *)ehdr, false);
235
236 total_in = 0;
237 total_out = 0;
238 unsigned c_adler = upx_adler32(nullptr, 0);
239 unsigned u_adler = upx_adler32(nullptr, 0);
240 off_t ptload0hi=0, ptload1lo=0, ptload1sz=0;
241
242 // decompress PT_LOAD
243 bool first_PF_X = true;
244 fi->seek(- (off_t) (szb_info + ph.c_len), SEEK_CUR);
245 for (unsigned j=0; j < ehdr->e_phnum; ++phdr, ++j) {
246 if (PT_LOAD==phdr->p_type) {
247 if (0==ptload0hi) {
248 ptload0hi = phdr->p_filesz + phdr->p_offset;
249 }
250 else if (0==ptload1lo) {
251 ptload1lo = phdr->p_offset;
252 ptload1sz = phdr->p_filesz;
253 }
254 if (fo)
255 fo->seek(phdr->p_offset, SEEK_SET);
256 if (Elf32_Phdr::PF_X & phdr->p_flags) {
257 unpackExtent(phdr->p_filesz, fo,
258 c_adler, u_adler, first_PF_X, szb_info);
259 first_PF_X = false;
260 }
261 else {
262 unpackExtent(phdr->p_filesz, fo,
263 c_adler, u_adler, false, szb_info);
264 }
265 }
266 }
267
268 if (0!=ptload1sz && ptload0hi < ptload1lo) { // alignment hole?
269 if (fo)
270 fo->seek(ptload0hi, SEEK_SET);
271 unpackExtent(ptload1lo - ptload0hi, fo,
272 c_adler, u_adler, false, szb_info);
273 }
274 if (total_out != orig_file_size) { // non-PT_LOAD stuff
275 if (fo)
276 fo->seek(0, SEEK_END);
277 unpackExtent(orig_file_size - total_out, fo,
278 c_adler, u_adler, false, szb_info);
279 }
280
281 // check for end-of-file
282 fi->readx(&bhdr, szb_info);
283 unsigned const sz_unc = ph.u_len = get_te32(&bhdr.sz_unc);
284
285 if (sz_unc == 0) { // uncompressed size 0 -> EOF
286 // note: magic is always stored le32
287 unsigned const sz_cpr = get_le32(&bhdr.sz_cpr);
288 if (sz_cpr != UPX_MAGIC_LE32) // sz_cpr must be h->magic
289 throwCompressedDataViolation();
290 }
291 else { // extra bytes after end?
292 throwCompressedDataViolation();
293 }
294
295 // update header with totals
296 ph.c_len = total_in;
297 ph.u_len = total_out;
298
299 // all bytes must be written
300 if (total_out != orig_file_size)
301 throwEOFException();
302
303 // finally test the checksums
304 if (ph.c_adler != c_adler || ph.u_adler != u_adler)
305 throwChecksumError();
306#undef MAX_INTERP_HDR
307}

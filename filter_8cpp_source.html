47/*static*/ const FilterImpl::FilterEntry *FilterImpl::getFilter(int id) {
48 static upx_uint8_t filter_map[256];
49 static upx_std_once_flag init_done;
50
51 upx_std_call_once(init_done, []() noexcept {
52 // init the filter_map[] (using a lambda function)
53 assert_noexcept(n_filters <= 254); // as 0xff means "empty slot"
54 memset(filter_map, 0xff, sizeof(filter_map));
55 for (int i = 0; i < n_filters; i++) {
56 int filter_id = filters[i].id;
57 assert_noexcept(filter_id >= 0 && filter_id <= 255);
58 assert_noexcept(filter_map[filter_id] == 0xff);
59 filter_map[filter_id] = (upx_uint8_t) i;
60 }
61 });
62
63 if (id < 0 || id > 255)
64 return nullptr;
65 unsigned index = filter_map[id];
66 if (index == 0xff) // empty slot
67 return nullptr;
68 assert_noexcept(filters[index].id == id);
69 return &filters[index];
70}

77/*static*/ bool Filter::isValidFilter(int filter_id, const int *allowed_filters) {
78 if (!isValidFilter(filter_id))
79 return false;
80 if (filter_id == 0)
81 return true;
82 if (allowed_filters == nullptr)
83 return false;
84 while (*allowed_filters != FT_END)
85 if (*allowed_filters++ == filter_id)
86 return true;
87 return false;
88}

107bool Filter::filter(SPAN_0(byte) xbuf, unsigned buf_len_) {
108 byte *const buf_ = raw_bytes(xbuf, buf_len_);
109 initFilter(this, buf_, buf_len_);
110
111 const FilterImpl::FilterEntry *const fe = FilterImpl::getFilter(id);
112 if (fe == nullptr)
113 throwInternalError("filter-1");
114 if (fe->id == 0)
115 return true;
116 if (buf_len < fe->min_buf_len)
117 return false;
118 if (fe->max_buf_len && buf_len > fe->max_buf_len)
119 return false;
120 if (!fe->do_filter)
121 throwInternalError("filter-2");
122
123 // save checksum
124 this->adler = 0;
125 if (clevel != 1)
126 this->adler = upx_adler32(this->buf, this->buf_len);
127
128 NO_printf("filter: %02x %p %d\n", this->id, this->buf, this->buf_len);
129 // OutputFile::dump("filter.dat", buf, buf_len);
130 int r = (*fe->do_filter)(this);
131 NO_printf("filter: %02x %d\n", fe->id, r);
132 if (r > 0)
133 throwFilterException();
134 if (r == 0)
135 return true;
136 return false;
137}
